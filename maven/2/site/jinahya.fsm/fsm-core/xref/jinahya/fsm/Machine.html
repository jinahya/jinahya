27  /**
28   * Represents a finite state machine.
29   *
30   * @author <a href="mailto:jinahya@gmail.com">Jin Kwon</a>

32  public class Machine {

34  
35      /**
36       * The state that all new instances of machines has.
37       */
38      public static final int UNKNOWN = 0xFFFFFFFF;

41      /**
42       * Creates a new instance with a spec. The maximumHistorySize is 0x0A.
43       *
44       * @param spec machine spec
45       */
46      public Machine(final MachineSpec spec) {
47          super();
48  
49          if (spec == null) {
50              throw new NullPointerException("spec");
51          }
52  
53          this.spec = spec;
54      }

58       * Returns the current state value.

60       * @return the current state

62      public final synchronized int getState() {

67      /**
68       * Sets the current state. Identical to transit().
69       *
70       * @param state new state
71       * @throws MachineException if any error occurs.

75          throws MachineException {

81      /**
82       * Transit the state.
83       *
84       * @param state new state
85       * @throws MachineException if any error occurs.
86       * @see #setState(int)
87       */
88      public final synchronized void transit(final int state)
89          throws MachineException {
90  
91          // ------------------------------------------------------ CHECK FINISHED
92          if (finished) {
93              return;
94          }
95  
96          // --------------------------------------------------- CREATE TRANSITION
97          final int sourceState = this.state;
98          final int targetState = state;
99          final int[] previousStates = new int[states.size()];
100         for (int i = 0; i < previousStates.length; i++) {
101             previousStates[i] = ((Integer) states.elementAt(i)).intValue();
102         }
103 
104         final Transition transition = new Transition() {
105 
106             private static final long serialVersionUID = -3638658218680618735L;
107 
108             //@Override
109             public int getSourceState() {
110                 return sourceState;
111             }
112 
113             //@Override
114             public int getTargetState() {
115                 return targetState;
116             }
117 
118             //@Override
119             public int getPreviousStateCount() {
120                 return previousStates.length;
121             }
122 
123             //@Override
124             public int getPreviousStateAt(final int index) {
125                 return previousStates[index];
126             }
127 
128             //@Override
129             public final String toString() {
130                 return "[TRANSITION: " + sourceState + " -> " +
131                         targetState + "]";
132             }
133         };
134 
135 
136         // -------------------------------------------- CHECK TRANSITION ALLOWED
137         if (!spec.isTransitionAllowed(transition)) {
138             throw new MachineException("not allowed by spec: " + transition);
139         }
140 
141 
142         // ------------------------------------------------------ CHECK STARTING
143         if (!started && spec.isStartingTransition(transition)) {
144             started = Boolean.TRUE.booleanValue();
145         }
146 
147 
148         // ----------------------------------------------------- CHECK FINISHING
149         if (!finished && spec.isFinishingTransition(transition)) {
150             finished = Boolean.TRUE.booleanValue();
151         }
152 
153 
154         // ------------------------------------ PUT CURRENT STATE TO THE HISTORY
155         states.insertElementAt(new Integer(this.state), 0);
156         states.setSize(Math.min(states.size(), maximumHistorySize));
157 
158 
159         // ------------------------------------------------ CHANGE CURRENT STATE
160         this.state = state;
161 
162 
163         // ---------------------------------------------------------- FIRE EVENT
164         final TransitionEvent event = new TransitionEvent(this, transition);
165         final Object[] listeners = els.getListeners(TransitionListener.class);
166         for (int i = 0; i < listeners.length; i++) {
167             ((TransitionListener) listeners[i]).transited(event);
168         }
169 
170     }
171

193     /**

195      *
196      * @return true if this machine has been stated, false otherwise.

198     public synchronized final boolean isStarted() {
199         return started;

203     /**

208     public synchronized final boolean isFinished() {

213     /**
214      * Sets the value of maximumHistorySize.
215      *
216      * @param maximumHistorySize maximum history size

218     protected final void setMaximumHistorySize(int maximumHistorySize) {
219         if (maximumHistorySize < 0) {
220             throw new IllegalArgumentException("maximumHistorySize < 0");
221         }
222 
223         synchronized (this) {
224             this.maximumHistorySize = maximumHistorySize;
225         }
226     }
