28      /**
29       * Creates a new instance.
30       *
31       * @param spec spec
32       * @param factory task factory
33       * @param state initial state
34       * @throws StateMachineException if any error occurs.
35       */
36      public StateMachineSupport(final StateMachineSpec spec,
37                                 final TaskFactory factory,
38                                 final State state)
39          throws StateMachineException {
40  
41          super();
42  
43          if (spec == null) {
44              throw new IllegalArgumentException("spec is null");
45          }
46  
47          if (factory == null) {
48              throw new IllegalArgumentException("factory is null");
49          }
50  
51          if (state == null) {
52              throw new IllegalArgumentException("state is null");
53          }
54  
55          this.spec = spec;
56          this.factory = factory;
57  
58          transit(state);
59      }

85      /**
86       * Transit the state.
87       *
88       * @param newState new state
89       * @throws StateMachineException if any error occurs.
90       */
91      public synchronized void transit(final State newState)
92          throws StateMachineException {
93  
94          if (newState == null) {
95              throw new IllegalArgumentException("new state is null!");
96          }
97  
98          if (finished) {
99              throw new StateMachineException("already finished!");
100         }
101 
102         final Transition transition = new Transition(state, newState);
103 
104         if (!spec.isTransitionAllowed(transition)) {
105             throw new StateMachineException
106                 ("transition is not allowed: " + transition);
107         }
108 
109         state = newState;
110 
111         // ------------------------------------------------------------- STARTED
112         if (!started && spec.isStartingTransition(transition)) {
113             started = true;
114 
115             tasks = factory.createTasks();
116             for (int i = 0; i < tasks.length; i++) {
117                 tasks[i].initialize();
118             }
119         }
120 
121 
122         // ------------------------------------------------------------ FINISHED
123         if (!finished && spec.isFinishingTransition(transition)) {
124             finished = true;
125 
126             for (int i = 0; i < tasks.length; i++) {
127                 tasks[i].destroy();
128             }
129         }
130 
131 
132         TaskExecutorService parent = null;
133         for (int priority = 9; priority >= 0; priority--) {
134             TaskExecutorService child = new TaskExecutorService
135                 (parent, tasks, transition, priority, threadCount);
136             child.start();
137             parent = child;
138         }
139 
140         try {
141             parent.join();
142         } catch (InterruptedException ie) {
143             ie.printStackTrace();
144         }
145     }

148     /**

150      *
151      * @param threadCount new thread count
152      */
153     public synchronized void setThreadCount(final int threadCount) {
154         if (threadCount <= 0) {
155             throw new IllegalArgumentException
156                 ("illegal thread count: " + threadCount);
157         }
158         this.threadCount = threadCount;
159     }
