87      /**
88       * Transit the state.
89       *
90       * @param state new state
91       * @throws StateMachineException if any error occurs.
92       */
93      public synchronized void transit(final State state)
94          throws StateMachineException {
95  
96          if (state == null) {
97              throw new IllegalArgumentException("new state is null!");
98          }
99  
100         if (finished) {
101             throw new IllegalStateException("already finished!");
102         }
103 
104 
105         // -------------------------------------- JOIN PREVIOUS EXECUTOR SERVICE
106         if (previousExecutorService != null) {
107             try {
108                 previousExecutorService.join();
109             } catch (InterruptedException ie) {
110                 ie.printStackTrace();
111             }
112         }
113 
114 
115         // --------------------------------------------------- CREATE TRANSITION
116         final State[] transitionHistory = new State[history.size()];
117         history.toArray(transitionHistory);
118         final Transition transition =
119             new Transition(this.state, state, transitionHistory);
120 
121 
122         // -------------------------------------------- CHECK TRANSITION ALLOWED
123         if (!spec.isTransitionAllowed(transition)) {
124             throw new StateMachineException
125                 ("transition is not allowed: " + transition);
126         }
127 
128 
129         // -------------------------------------------------------- CHANGE STATE
130         State oldState = this.state;
131         this.state = state;
132 
133 
134         // ------------------------------------------------------ CHECK STARTING
135         if (!started && spec.isStartingTransition(transition)) {
136             started = true;
137 
138             final Task[] created = factory.createTasks();
139             for (int i = 0; i < created.length; i++) {
140                 if (created[i] == null) {
141                     throw new NullPointerException("null task @ " + i);
142                 }
143             }
144             tasks = new Task[created.length];
145             System.arraycopy(created, 0, tasks, 0, tasks.length);
146 
147             // ------------------------------------------------ INITIALIZE TASKS
148             for (int i = 0; i < tasks.length; i++) {
149                 tasks[i].initialize();
150             }
151         }
152 
153 
154         if (started) {
155 
156             // --------------------------------------------------------- PERFORM
157             if (threadCount > 0) {
158                 ExecutorService parent = null;
159                 for (int priority = 9; priority >= 0; priority--) {
160                     ExecutorService child =
161                         new ExecutorService(parent, tasks, transition, priority,
162                                             threadCount);
163                     child.start();
164                     parent = child;
165                 }
166                 previousExecutorService = parent;
167             } else {
168                 for (int priority = 9; priority >= 0; priority--) {
169                     for (int i = 0; i < tasks.length; i++) {
170                         tasks[i].perform(transition, priority);
171                     }
172                 }
173             }
174 
175             /*
176             // ------------------------------------------------------------ JOIN
177             try {
178                 parent.join();
179             } catch (InterruptedException ie) {
180                 ie.printStackTrace();
181             }
182              */
183 
184             // ------------------------------------------------- CHECK FINISHING
185             if (!finished && spec.isFinishingTransition(transition)) {
186                 finished = true;
187 
188                 // ----------------------------------------------- DESTROY TASKS
189                 for (int i = 0; i < tasks.length; i++) {
190                     tasks[i].destroy();
191                 }
192             }
193         }
194 
195         // ------------------------------------------------------------- HISTORY
196         history.addElement(oldState);
197         while (history.size() > historyCount) {
198             history.removeElementAt(0);
199         }
200     }
