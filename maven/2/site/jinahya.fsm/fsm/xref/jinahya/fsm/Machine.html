71      /**
72       * Transit the state.
73       *
74       * @param state new state
75       * @throws MachineException if any error occurs.
76       * @see #setState(int)
77       */
78      public synchronized final void transit(final int state)
79          throws MachineException {
80  
81          // ------------------------------------------------------ CHECK FINISHED
82          if (finished) {
83              throw new IllegalStateException("already finished");
84          }
85  
86          // --------------------------------------------------- CREATE TRANSITION
87          final int sourceState = this.state;
88          final int targetState = state;
89          final int[] previousStates = new int[states.size()];
90          for (int i = 0; i < previousStates.length; i++) {
91              previousStates[i] = ((Integer) states.elementAt(i)).intValue();
92          }
93          final Transition transition = new Transition() {
94              //@Override
95              public int getSourceState() {
96                  return sourceState;
97              }
98              //@Override
99              public int getTargetState() {
100                 return targetState;
101             }
102             //@Override
103             public int getPreviousState(final int depth) {
104                 if (depth < previousStates.length) {
105                     return previousStates[depth];
106                 } else {
107                     return State.UNKNOWN;
108                 }
109             }
110             //@Override
111             public String toString() {
112                 return "[TRANSITION: " + sourceState + " -> " +
113                         targetState + "]";
114             }
115         };
116 
117 
118         // -------------------------------------------- CHECK TRANSITION ALLOWED
119         if (!spec.isTransitionAllowed(transition)) {
120             throw new MachineException("not allowed: " + transition);
121         }
122 
123 
124         // ------------------------------------------------------ CHECK STARTING
125         if (!started && spec.isStartingTransition(transition)) {
126             started = Boolean.TRUE.booleanValue();
127         }
128 
129 
130         if (started) {
131 
132 
133             // --------------------------------------------------------- PERFORM
134 
135             final int minimumPrecedence =
136                 Math.max(0x00, spec.getMinimumPrecedence(transition));
137 
138             Thread[] threads = null;
139 
140             for (int p = 0; p <= minimumPrecedence; p++) {
141                 threads = perform(threads, transition, p);
142             }
143 
144             // wait for all threads in last group end
145             // minimumPrecedence is always positive(>=0)
146             //assert threads != null;
147             for (int i = 0; i < threads.length; i++) {
148                 while (threads[i].isAlive()) {
149                     try {
150                         threads[i].join();
151                     } catch (InterruptedException ie) {
152                         ie.printStackTrace();
153                     }
154                 }
155             }
156         }
157 
158 
159         // ----------------------------------------------------- CHECK FINISHING
160         if (!finished && spec.isFinishingTransition(transition)) {
161             finished = Boolean.TRUE.booleanValue();
162         }
163 
164 
165         // ------------------------------------------------------------- HISTORY
166         states.insertElementAt(new Integer(this.state), 0);
167         final int maximumHistorySize =
168             Math.max(0x00, spec.getMaximumHistorySize());
169         states.setSize
170             (Math.min(states.size(), maximumHistorySize));
171 
172 
173         this.state = state;
174     }

177     private Thread[] perform(final Thread[] parents,
178                              final Transition transition,
179                              final int precedence) {
180 
181         // wait for all previous threads end
182         if (parents != null) {
183             for (int i = 0; i < parents.length; i++) {
184                 while (parents[i].isAlive()) {
185                     try {
186                         parents[i].join();
187                     } catch (InterruptedException ie) {
188                         ie.printStackTrace();
189                     }
190                 }
191             }
192         }
193 
194         // prepare a copy of task list
195         final Vector tmp = new Vector();
196         for (int i = 0; i < tasks.size(); i++) {
197             tmp.addElement(tasks.elementAt(i));
198         }
199 
200         final int maximumPoolSize =
201             Math.max(0x01, spec.getMaximumPoolSize(transition, precedence));
202 
203         Thread[] children = new Thread[maximumPoolSize];
204         for (int i = 0; i < children.length; i++) {
205             children[i] = new Thread() {
206                 //@Override
207                 public void run() {
208                     for (Task task = null; true; task = null) {
209 
210                         // locate next task
211                         synchronized (tmp) {
212                             if (!tmp.isEmpty()) {
213                                 task = (Task) tmp.firstElement();
214                                 tmp.removeElementAt(0x00);
215                             }
216                         }
217 
218                         if (task == null) {
219                             break;
220                         }
221 
222                         // perform the task
223                         try {
224                             task.perform(transition, precedence);
225                         } catch (MachineException me) {
226                             me.printStackTrace();
227                         } catch (Exception e) {
228                             e.printStackTrace();
229                         }
230                     }
231                 }
232             };
233         }
234 
235         for (int i = 0; i < children.length; i++) {
236             children[i].start();
237         }
238 
239         return children;
240     }

243     public synchronized final void submit(final Task task) {
244 
245         if (task == null) {
246             throw new NullPointerException("task");
247         }
248 
249         synchronized (this) {
250             if (started) {
251                 throw new IllegalStateException("already started");
252             }
253 
254             if (finished) {
255                 throw new IllegalStateException("already finished");
256             }
257 
258             tasks.addElement(task);
259         }
260     }
