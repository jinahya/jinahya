76      /**
77       * Transit the state.
78       *
79       * @param state new state
80       * @throws MachineException if any error occurs.
81       * @see #setState(jinahya.fsm.State)
82       */
83      public synchronized void transit(final State state)
84          throws MachineException {
85  
86          if (state == null) {
87              throw new NullPointerException("state");
88          }
89  
90  
91          // ------------------------------------------------------ CHECK FINISHED
92          if (isFinished()) {
93              return;
94          }
95  
96  
97          // --------------------------------------------------- CREATE TRANSITION
98          final State sourceState = this.state;
99          final State targetState = state;
100         final Transition transition = new Transition() {
101             //@Override
102             public State getSourceState() {
103                 return sourceState;
104             }
105             //@Override
106             public State getTargetState() {
107                 return targetState;
108             }
109             //@Override
110             public String toString() {
111                 return "[TRANSITION: " + sourceState + " -> " +
112                         targetState + "]";
113             }
114         };
115 
116 
117         // -------------------------------------------- CHECK TRANSITION ALLOWED
118         if (!spec.isTransitionAllowed(transition)) {
119             throw new MachineException("not allowed: " + transition);
120         }
121 
122 
123         // ------------------------------------------------------ CHECK STARTING
124         if (spec.isStartingTransition(transition)) {
125             start();
126         }
127 
128 
129         if (isStarted()) {
130 
131             // --------------------------------------------------------- PERFORM
132             if (threadPoolSize == 0) {
133                 for (int precedence = 0; precedence <= minimumPrecedence;
134                      precedence++) {
135                     for (int i = 0; i < tasks.length; i++) {
136                         tasks[i].perform(transition, precedence);
137                     }
138                 }
139             } else {
140 
141                 Thread parent = null;
142 
143                 for (int precedence = 0; precedence <= minimumPrecedence;
144                      precedence++) {
145                     Thread child = new Thread
146                         (new Executor(parent, tasks, transition, precedence,
147                                       threadPoolSize));
148                     child.start();
149                     parent = child;
150                 }
151 
152                 while (parent.isAlive()) {
153                     try {
154                         parent.join();
155                     } catch (InterruptedException ie) {
156                         ie.printStackTrace();
157                     }
158                 }
159             }
160         }
161 
162 
163         // ----------------------------------------------------- CHECK FINISHING
164         if (spec.isFinishingTransition(transition)) {
165             finish();
166         }
167 
168 
169         this.state = state;
170     }

173     public synchronized boolean isStarted() {
174         return started;
175     }
176

178     public synchronized void start() throws MachineException {
179         if (started) {
180             return;
181         }
182 
183         //tasks = factory.createTasks();
184 
185         for (int i = 0; i < tasks.length; i++) {
186             tasks[i].initialize();
187         }
188 
189         started = Boolean.TRUE.booleanValue();
190     }
191 
192 
193     public synchronized boolean isFinished() {
194         return finished;
195     }
196 
197 
198     public synchronized void finish() throws MachineException {
199         if (finished) {
200             return;
201         }
202 
203         if (isStarted()) {
204             for (int i = 0; i < tasks.length; i++) {
205                 tasks[i].destroy();
206             }
207         }
208 
209         finished = Boolean.TRUE.booleanValue();
210     }
