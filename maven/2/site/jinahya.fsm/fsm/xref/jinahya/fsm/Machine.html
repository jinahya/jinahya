96      /**
97       * Transit the state.
98       *
99       * @param state new state
100      * @throws MachineException if any error occurs.
101      */
102     public synchronized void transit(final State state)
103         throws MachineException {
104 
105         if (state == null) {
106             throw new NullPointerException("state");
107         }
108 
109         if (finished) {
110             throw new IllegalStateException("finished");
111         }
112 
113 
114         // -------------------------------------- JOIN PREVIOUS EXECUTOR SERVICE
115         if (pool != null) {
116             try {
117                 pool.join();
118             } catch (InterruptedException ie) {
119                 ie.printStackTrace();
120             }
121         }
122 
123 
124         // --------------------------------------------------- CREATE TRANSITION
125         final State sourceState = this.state;
126         final State targetState = state;
127         final State[] transitionHistory = new State[history.size()];
128         history.toArray(transitionHistory);
129         final Transition transition = new Transition() {
130             //@Override
131             public State getSourceState() {
132                 return sourceState;
133             }
134             //@Override
135             public State getTargetState() {
136                 return targetState;
137             }
138             //@Override
139             public State[] getTransitionHistory() {
140                 return transitionHistory;
141             }
142         };
143 
144 
145         // -------------------------------------------- CHECK TRANSITION ALLOWED
146         if (!spec.isTransitionAllowed(transition)) {
147             throw new MachineException("not allowed: " + transition);
148         }
149 
150         log("Transiting from " + sourceState + " to " + targetState);
151 
152         // ------------------------------------------------------ CHECK STARTING
153         if (!started && spec.isStartingTransition(transition)) {
154             log("Checked as a starting transition");
155 
156             // ---------------------------------------------------- CREATE TASKS
157             log("Creating tasks");
158             final Task[] _tasks = factory.createTasks();
159             tasks = new Task[_tasks.length];
160             System.arraycopy(_tasks, 0, tasks, 0, tasks.length);
161             log("Tasks are created");
162 
163             // ------------------------------------------------ INITIALIZE TASKS
164             log("Initializing tasks");
165             for (int i = 0; i < tasks.length; i++) {
166                 tasks[i].initialize();
167             }
168             log("Tasks are initalized");
169 
170             started = true;
171             log("Machine started");
172         }
173 
174 
175         if (started) {
176 
177             final int _poolSize = poolSize;
178             log("Pool size: " + _poolSize);
179 
180             // --------------------------------------------------------- PERFORM
181             if (_poolSize == 0) {
182                 log("Perform without threads");
183                 // ------------------------------------------------ NO THREADING
184                 for (int precedence = 0; precedence <= minimumPrecedence;
185                      precedence++) {
186                     for (int i = 0; i < tasks.length; i++) {
187                         tasks[i].perform(transition, precedence);
188                     }
189                 }
190             } else {
191                 log("Perform with threads");
192                 // --------------------------------------------------- THREADING
193                 final long _poolSleep = poolSleep;
194                 Thread parent = null;
195                 for (int precedence = 0; precedence <= minimumPrecedence;
196                      precedence++) {
197                     Thread child = new Thread(new ExecutorService
198                         (parent, tasks, transition, precedence, _poolSize,
199                          _poolSleep));
200                     child.start();
201                     parent = child;
202                 }
203                 pool = parent;
204             }
205 
206             // ------------------------------------------------- CHECK FINISHING
207             if (spec.isFinishingTransition(transition)) {
208                 log("Checked as a finishing transition");
209 
210                 if (_poolSize > 0) {
211                     while (pool.isAlive()) {
212                         try {
213                             pool.join();
214                         } catch (InterruptedException ie) {
215                             ie.printStackTrace();
216                         }
217                     }
218                 }
219 
220                 // ----------------------------------------------- DESTROY TASKS
221                 log("Destroying tasks");
222                 for (int i = 0; i < tasks.length; i++) {
223                     tasks[i].destroy();
224                 }
225                 log("Tasks are destroyed");
226 
227                 finished = true;
228                 log("Machine finished");
229             }
230         }
231 
232         // ------------------------------------------------------------- HISTORY
233         history.insertElementAt(this.state, 0);
234         int _historySize = historySize;
235         while (history.size() > _historySize) {
236             history.removeElementAt(history.size() - 1);
237         }
238 
239         this.state = state;
240     }
