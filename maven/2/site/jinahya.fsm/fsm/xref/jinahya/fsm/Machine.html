71      /**
72       * Transit the state.
73       *
74       * @param state new state
75       * @throws MachineException if any error occurs.
76       * @see #setState(jinahya.fsm.int)
77       */
78      public synchronized final void transit(final int state)
79          throws MachineException {
80  
81  
82          // ------------------------------------------------------ CHECK FINISHED
83          if (isFinished()) {
84              return;
85          }
86  
87  
88          // --------------------------------------------------- CREATE TRANSITION
89          final int sourceState = this.state;
90          final int targetState = state;
91          final int[] previousStates = new int[history.size()];
92          for (int i = 0; i < previousStates.length; i++) {
93              previousStates[i] = ((Integer) history.elementAt(i)).intValue();
94          }
95          final Transition transition = new Transition() {
96              //@Override
97              public int getSourceState() {
98                  return sourceState;
99              }
100             //@Override
101             public int getTargetState() {
102                 return targetState;
103             }
104             //@Override
105             public int[] getPreviousStates() {
106                 return previousStates;
107             }
108             //@Override
109             public String toString() {
110                 return "[TRANSITION: " + sourceState + " -> " +
111                         targetState + "]";
112             }
113         };
114 
115 
116         // -------------------------------------------- CHECK TRANSITION ALLOWED
117         if (!spec.isTransitionAllowed(transition)) {
118             throw new MachineException("not allowed: " + transition);
119         }
120 
121 
122         // ------------------------------------------------------ CHECK STARTING
123         if (spec.isStartingTransition(transition)) {
124             start();
125         }
126 
127 
128         if (isStarted()) {
129 
130             // --------------------------------------------------------- PERFORM
131             if (poolSize == 0x00) {
132                 for (int p = 0; p <= minimumPrecedence; p++) {
133                     for (int i = 0; i < tasks.size(); i++) {
134                         ((Task)tasks.elementAt(i)).perform(transition, p);
135                     }
136                 }
137             } else {
138 
139                 Thread[] threads = null;
140 
141                 for (int p = 0; p <= minimumPrecedence; p++) {
142                     threads = perform(threads, transition, p);
143                 }
144 
145                 // wait for all threads in last group end
146                 // minimumPrecedence is always positive(>=0)
147                 //assert threads != null;
148                 for (int i = 0; i < threads.length; i++) {
149                     while (threads[i].isAlive()) {
150                         try {
151                             threads[i].join();
152                         } catch (InterruptedException ie) {
153                             ie.printStackTrace();
154                         }
155                     }
156                 }
157             }
158         }
159 
160 
161         // ----------------------------------------------------- CHECK FINISHING
162         if (spec.isFinishingTransition(transition)) {
163             finish();
164         }
165 
166 
167         // ------------------------------------------------------------- HISTORY
168         history.insertElementAt(new Integer(this.state), 0);
169         history.setSize(Math.min(history.size(), historySize));
170         //history.trimToSize();
171 
172 
173         this.state = state;
174     }

177     private Thread[] perform(final Thread[] parents,
178                              final Transition transition,
179                              final int precedence) {
180 
181         // wait for all previous threads end
182         if (parents != null) {
183             for (int i = 0; i < parents.length; i++) {
184                 while (parents[i].isAlive()) {
185                     try {
186                         parents[i].join();
187                     } catch (InterruptedException ie) {
188                         ie.printStackTrace();
189                     }
190                 }
191             }
192         }
193 
194         // prepare a copy of task list
195         final Vector tmp = new Vector();
196         for (int i = 0; i < tasks.size(); i++) {
197             tmp.addElement(tasks.elementAt(i));
198         }
199 
200         Thread[] children = new Thread[poolSize];
201         for (int i = 0; i < children.length; i++) {
202             children[i] = new Thread() {
203                 //@Override
204                 public void run() {
205                     while (Boolean.TRUE.booleanValue()) {
206 
207                         // locate next task
208                         Task task = null;
209                         synchronized (tmp) {
210                             if (tmp.isEmpty()) {
211                                 break;
212                             }
213                             task = (Task) tmp.firstElement();
214                             tmp.removeElementAt(0x00);
215                         }
216 
217                         // perform the task
218                         try {
219                             task.perform(transition, precedence);
220                         } catch (MachineException me) {
221                             me.printStackTrace();
222                         } catch (Exception e) {
223                             e.printStackTrace();
224                         }
225                     }
226                 }
227             };
228         }
229 
230         for (int i = 0; i < children.length; i++) {
231             children[i].start();
232         }
233 
234         return children;
235     }

243     public final synchronized void start() throws MachineException {
244         if (isStarted()) {
245             return;
246         }
247 
248         try {
249             synchronized (tasks) {
250                 for (int i = 0; i < tasks.size(); i++) {
251                     ((Task) tasks.elementAt(i)).initialize();
252                 }
253             }
254         } finally {
255             started = Boolean.TRUE.booleanValue();
256         }
257     }

265     public final synchronized void finish() throws MachineException {
266         if (isFinished()) {
267             return;
268         }
269 
270         try {
271             synchronized (tasks) {
272                 for (int i = tasks.size() - 1; i >= 0; i--) {
273                     ((Task) tasks.elementAt(i)).destroy();
274                 }
275             }
276         } finally {
277             finished = Boolean.TRUE.booleanValue();
278         }
279     }

287     public final void setMinimumPrecedence(int minimumPrecedence) {
288 
289         if (minimumPrecedence < 0) {
290             throw new IllegalArgumentException
291                 ("minimumPrecedence(" + minimumPrecedence + ") < 0");
292         }
293 
294         synchronized (this) {
295             if (isStarted()) {
296                 throw new IllegalStateException("already started");
297             }
298 
299             if (isFinished()) {
300                 throw new IllegalStateException("already finished");
301             }
302 
303             this.minimumPrecedence = minimumPrecedence;
304         }
305     }
