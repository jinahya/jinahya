17  package jinahya.net;
18  
19  
20  import java.io.EOFException;
21  import java.io.FilterInputStream;
22  import java.io.IOException;
23  import java.io.InputStream;
24  
25  
26  /**
27   *
28   * @author <a href="mailto:jinahya@gmail.com">Jin Kwon</a>
29   */
30  public class URLDecodingInputStream extends FilterInputStream {
31  
32  
33      public URLDecodingInputStream(final InputStream in) {
34          super(in);
35      }
36  
37  
38      @Override
39      public int read() throws IOException {
40  
41          final int b = super.read();
42          if (b == -1) {
43              return b;
44          }
45                  
46          if ((b >= 0x30 && b <= 0x39) // digit
47              || (b >= 0x41 && b <= 0x5A) // upper case alpha
48              || (b >= 0x61 && b <= 0x7A) // lower case alpha
49              || (b == 0x2E || b == 0x2D || b == 0x2A || b == 0x5F)) { // . - * _
50              return b;
51          }
52  
53          if (b == 0x2B) { // +
54              return 0x20; // SPACE
55          }
56  
57          if (b == 0x25) { // %
58              return ((atoi(super.read()) << 4) | (atoi(super.read()) & 0x0F));
59          }
60  
61          throw new IOException("illegal octet: " + b);
62      }
63  
64  
65      private int atoi(final int i) throws EOFException {
66          if (i == -1) {
67              throw new EOFException("eof");
68          }
69          return i - (i >= 0x41 ? 0x37 : 0x30);
70      }
71  
72  
73      @Override
74      public int read(byte[] b) throws IOException {
75          return read(b, 0, b.length);
76      }
77  
78  
79      @Override
80      public int read(byte[] b, int off, int len) throws IOException {
81          for (int i = 0; i < len; i++) {
82              if ((b[off + i] = (byte)read()) == -1) {
83                  return i == 0 ? -1 : i;
84              }
85          }
86          return len;
87      }
88  
89  
90      @Override
91      public boolean markSupported() {
92          return false;
93      }
94  
95  
96      @Override
97      public long skip(long n) throws IOException {
98          long l = 0;
99          for (; l < n; l++) {
100             int b = read();
101             if (b == -1) {
102                 return l;
103             }
104         }
105         return l;
106     }
107 }
