50      public void loadTasks(Vector tasks) throws StateMachineException {
51          String name = "/META-INF/jinahya/util/state/" +
52                  hex(machine.getIdentifier()).toUpperCase() + "." +
53                  hex(spec.getIdentifier()).toUpperCase();
54          InputStream stream = machine.getClass().getResourceAsStream(name);
55          if (stream == null) {
56              throw new StateMachineException("RESOURCE NOT FOUND: " + name);
57          }
58          try {
59              try {
60                  BufferedReader reader =
61                      new BufferedReader(new InputStreamReader(stream, "UTF8"));
62                  try {
63                      String line = null;
64                      while ((line = reader.readLine()) != null) {
65                          line = line.trim();
66                          if (line.length() == 0) {
67                              continue;
68                          }
69                          try {
70                              Class clazz = Class.forName(line);
71                              if (!StateMachineTask.class.isAssignableFrom(clazz)) {
72                                  throw new StateMachineException
73                                      (clazz.getName() +
74                                       " is not assignable to " +
75                                       StateMachineTask.class);
76                              }
77                              try {
78                                  tasks.addElement(clazz.newInstance());
79                              } catch (InstantiationException ie) {
80                                  throw new StateMachineException(ie);
81                              } catch (IllegalAccessException iae) {
82                                  throw new StateMachineException(iae);
83                              }
84                          } catch (ClassNotFoundException cnfe) {
85                              throw new StateMachineException(cnfe);
86                          }
87                      }
88                  } finally {
89                      reader.close();
90                  }
91              } finally {
92                  stream.close();
93              }
94          } catch (IOException ioe) {
95              throw new StateMachineException(ioe);
96          }
97      }

100     private String hex(int val) {
101         return hex(bytes(val));
102     }

105     private String hex(byte[] bytes) {
106         StringBuffer buffer = new StringBuffer();
107         for (int i = 0; i < bytes.length; i++) {
108             int val = bytes[i] & 0xFF;
109             if (val < 16) {
110                 buffer.append('0');
111             }
112             buffer.append(Integer.toHexString(val));
113         }
114         return buffer.toString();
115     }
116 
117 
118     private byte[] bytes(int val) {
119         byte[] bytes = new byte[4];
120         for (int i = 0; i < bytes.length; i++) {
121             bytes[bytes.length - (i + 1)] = (byte)((val >>> (8 * i)) & 0xFF);
122         }
123         return bytes;
124     }
