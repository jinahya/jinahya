45      /**
46       * Generates a string encoded in Modified UTF-8.
47       *
48       * @param length the maximum length in bytes between 0 (inclusive) and 65536
49       *        (exclusive)
50       * @return a generated string
51       * @throws IOException if an I/O error occurs.
52       */
53      public static String generateString(final int length) throws IOException {
54          final CharArrayWriter caw = new CharArrayWriter();
55          try {
56              generateString(length, caw);
57              caw.flush();
58              return caw.toString();
59          } finally {
60              caw.close();
61          }
62      }

65      /**
66       * Generates a string encoded in Modified UTF-8.
67       *
68       * @param length the maximum number of bytes between 0 (inclusive) and 65536
69       *        (exclusive)
70       * @param target a writer to which generated character sequence is write
71       * @throws IOException if an I/O error occurs.
72       */
73      public static void generateString(final int length, final Writer target)
74          throws IOException {
75  
76          int _length = Math.max(0, length);
77          _length = Math.min(_length, 65535);
78  
79          int first;
80          int second;
81          int third;
82  
83          int count = 0;
84          while (count < _length - 2) {
85              first = (int) (Math.random() * 256);
86  
87              switch ((first >> 4)) {
88                  case 0:
89                  case 1:
90                  case 2:
91                  case 3:
92                  case 4:
93                  case 5:
94                  case 6:
95                  case 7:
96                      /* 0xxx xxxx */
97                      char ch1 = (char) first;
98                      if (Character.isISOControl(ch1)) {
99                          break;
100                     }
101                     target.write(ch1);
102                     count++;
103                     break;
104 
105                 case 12:
106                 case 13:
107                     /* 110x xxxx */
108                     second = ((int) (Math.random() * 64)) + 128;
109                     char ch2 = (char) (((first & 0x1F) << 6)
110                         | (second & 0x3F));
111                     if (Character.isISOControl(ch2)) {
112                         break;
113                     }
114                     target.write(ch2);
115                     count += 2;
116                     break;
117 
118                 case 14:
119                     /* 1110 xxxx */
120                     second = ((int) (Math.random() * 64)) + 128;
121                     third = ((int) (Math.random() * 64)) + 128;
122                     char ch3 = (char) (((first & 0x0F) << 12)
123                         | ((second & 0x3F) << 6)
124                         | (third & 0x3F));
125                     if (Character.isISOControl(ch3)) {
126                         break;
127                     }
128                     target.write(ch3);
129                     count += 3;
130                     break;
131 
132                 default:
133                     /* 10xx xxxx, 1111 xxxx */
134                     break;
135             }
136         }
137 
138         // zero padding
139         while (count < _length) {
140             target.write(0);
141             count++;
142         }
143     }

146     /**
147      * Reads a string from <code>in</code>.
148      *
149      * @param in input stream
150      * @return decoded string
151      * @throws IOException if an I/O error occur.
152      */
153     public static String readString(final InputStream in) throws IOException {
154 
155         final int length = ((in.read() << 8) | in.read());
156         if (length == -1) {
157             throw new EOFException("EOF while reading length");
158         }
159 
160         final byte[] encoded = new byte[length];
161 
162         // read fully
163         for (int offset = 0; offset < encoded.length;) {
164             int read = in.read(encoded, offset, encoded.length - offset);
165             if (read == -1) {
166                 throw new EOFException("EOF while reading bytes at " + offset);
167             }
168             offset += read;
169         }
170 
171         return new String(decode(encoded));
172     }

175     /**
176      * Writes the given <code>s</code> to specified <code>out</code>.
177      *
178      * @param s string
179      * @param out outputs stream
180      * @throws IOException if an I/O error occurs.
181      */
182     public static void writeString(final String s, final OutputStream out)
183         throws IOException {
184 
185         final byte[] encoded = encode(s.toCharArray());
186 
187         out.write(encoded.length >> 8);
188         out.write(encoded.length & 0xFF);
189 
190         out.write(encoded);
191     }

194     /**
195      * Decodes an array of bytes into an array of chars.
196      *
197      * @param encoded encoded byte array
198      * @return decoded char array
199      * @throws IOException if an I/O error occurs
200      */
201     public static char[] decode(final byte[] encoded) throws IOException {
202 
203         final CharArrayWriter caw = new CharArrayWriter();
204         try {
205             int byte1;
206             int byte2;
207             int byte3;
208             for (int i = 0; i < encoded.length;) {
209                 byte1 = encoded[i++] & 0xFF;
210                 switch ((byte1 >> 4)) {
211                     case 0:
212                     case 1:
213                     case 2:
214                     case 3:
215                     case 4:
216                     case 5:
217                     case 6:
218                     case 7:
219                         /* 0xxx xxxx */
220 
221                         caw.write(byte1);
222 
223                         break;
224 
225                     case 12:
226                     case 13:
227                         /* 110x xxxx */
228 
229                         try {
230                             byte2 = encoded[i++] & 0xFF;
231                         } catch (ArrayIndexOutOfBoundsException aioobe) {
232                             throw new EOFException("EOF at " + i);
233                         }
234                         if (byte2 >> 6 != 0x02) { // !(10xx xxxx)
235                             throw new UTFDataFormatException(
236                                 "illegal byte2 byte("
237                                 + Integer.toBinaryString(byte2) + ")");
238                         }
239 
240                         char ch2 = (char) (((byte1 & 0x1F) << 6) | (byte2 & 0x3F));
241                         caw.write(ch2);
242 
243                         break;
244 
245                     case 14:
246                         /* 1110 xxxx */
247 
248                         try {
249                             byte2 = encoded[i++] & 0xFF;
250                         } catch (ArrayIndexOutOfBoundsException aioobe) {
251                             throw new EOFException("EOF at " + i);
252                         }
253                         if (byte2 >> 6 != 0x02) { // !(10xx xxxx)
254                             throw new UTFDataFormatException(
255                                 "illegal byte2(" + Integer.toBinaryString(byte2)
256                                 + ")");
257                         }
258 
259                         try {
260                             byte3 = encoded[i++] & 0xFF;
261                         } catch (ArrayIndexOutOfBoundsException aioobe) {
262                             throw new EOFException("EOF at " + i);
263                         }
264                         if (byte3 >> 6 != 0x02) { // !(10xx xxxx)
265                             throw new UTFDataFormatException(
266                                 "illegal byte3(" + Integer.toBinaryString(byte3)
267                                 + ")");
268                         }
269 
270                         char ch3 = (char) (((byte1 & 0x0F) << 12)
271                                            | ((byte2 & 0x3F) << 6)
272                                            | (byte3 & 0x3F));
273                         caw.write(ch3);
274                         break;
275 
276                     default:
277                         /* 10xx xxxx, 1111 xxxx */
278                         throw new UTFDataFormatException(
279                             "illegal byte1(" + Integer.toBinaryString(byte1)
280                             + ")");
281                 }
282             }
283 
284             caw.flush();
285             return caw.toCharArray();
286 
287         } finally {
288             caw.close();
289         }
290     }

296     /**
297      * Encodes a sequence of characters into a byte array in Modified UTF-8
298      * format.
299      *
300      * @param decoded decoded char array
301      * @return encoded byte array
302      * @throws IOException if an I/O error occurs.
303      */
304     public static byte[] encode(final char[] decoded) throws IOException {
305         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
306         try {
307             for (int i = 0; i < decoded.length; i++) {
308                 char c = decoded[i];
309                 if (c >= '\u0001' && c <= '\u007F') {
310                     baos.write(c);
311                 } else if (c == '\u0000' || c <= '\u07FF') {
312                     baos.write(0xC0 | (0x1F & (c >> 6)));
313                     baos.write(0x80 | (0x3F & c));
314                 } else { // if (c <= '\uFFFF') {
315                     baos.write(0xE0 | (0x0F & (c >> 12)));
316                     baos.write(0x80 | (0x3F & (c >> 6)));
317                     baos.write(0x80 | (0x3F & c));
318                 }
319                 if (baos.size() > 65535) {
320                     throw new UTFDataFormatException(
321                         "length(" + baos.size() + ") > 65535");
322                 }
323             }
324 
325             baos.flush();
326             return baos.toByteArray();
327         } finally {
328             baos.close();
329         }
330     }
