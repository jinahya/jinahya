123     /**
124      * Generates a single char.
125      *
126      * @param acceptor the char acceptor
127      * @return a generated char
128      */
129     public static char generate(final Random random) {
130 
131         if (random == null) {
132             throw new IllegalArgumentException(
133                 "param:0:" + Random.class + " is null");
134         }
135 
136         final int n = random.nextInt(3);
137 
138         if (n == 0) {
139             final int a = random.nextInt(128);
140             return (char) a;
141         } else if (n == 1) {
142             final int a = random.nextInt(32) + 192;
143             final int b = random.nextInt(64) + 128;
144             return (char) (((a & 0x1F) << 6) | (b & 0x3F));
145         } else { // n == 2
146             final int a = random.nextInt(16) + 224;
147             final int b = random.nextInt(64) + 128;
148             final int c = random.nextInt(64) + 128;
149             return (char) (((a & 0x0F) << 12) | ((b & 0x3F) << 6) | (c & 0x3F));
150         }
151     }

170     /**
171      *
172      * @param in
173      * @param out
174      * @throws IOException
175      */
176     public static void encode(final Reader in, final OutputStream out)
177         throws IOException {
178 
179         for (int c = -1; (c = in.read()) != -1;) {
180             if (c >= '\u0001' && c <= '\u007F') {
181                 out.write(c);
182             } else if (c == '\u0000' || c <= '\u07FF') {
183                 out.write(0xC0 | (0x1F & (c >> 6)));
184                 out.write(0x80 | (0x3F & c));
185             } else { // if (c <= '\uFFFF') {
186                 out.write(0xE0 | (0x0F & (c >> 12)));
187                 out.write(0x80 | (0x3F & (c >> 6)));
188                 out.write(0x80 | (0x3F & c));
189             }
190         }
191     }

194     public static void decode(final InputStream in, final OutputStream out)
195         throws IOException {
196 
197         for (int a = -1; (a = in.read()) != -1;) {
198             if (a >> 7 == 0) { // 0xxxxxxx
199                 out.write(a);
200             } else if (a >> 5 == 6) { // 011xxxxx
201                 int b = in.read();
202                 if (b == -1) {
203                     throw new UTFDataFormatException("eof");
204                 }
205                 if (b >> 6 != 2) { // !10xxxxxx
206                     throw new UTFDataFormatException(
207                         "illeal b: " + Integer.toHexString(b));
208                 }
209                 out.write(((a & 0x1F) << 6) | (b & 0x3F));
210             } else if (a >> 4 == 14) { // 1110xxxx
211                 int b = in.read();
212                 if (b == -1) {
213                     throw new UTFDataFormatException("eof");
214                 }
215                 if (b >> 6 != 2) { // !10xxxxxx
216                     throw new UTFDataFormatException(
217                         "illeal b: " + Integer.toHexString(b));
218                 }
219                 int c = in.read();
220                 if (c == -1) {
221                     throw new UTFDataFormatException("eof");
222                 }
223                 if (c >> 6 != 2) { // !10xxxxxx
224                     throw new UTFDataFormatException(
225                         "illeal c: " + Integer.toHexString(c));
226                 }
227 
228                 out.write(((a & 0x0F) << 12) | ((b & 0x3F) << 6) | (c & 0x3F));
229             } else {
230                 throw new UTFDataFormatException(
231                     "illeal a: " + Integer.toHexString(a));
232             }
233         }
234     }
