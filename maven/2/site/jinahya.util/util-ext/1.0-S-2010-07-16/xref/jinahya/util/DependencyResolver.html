102     public void addDependency(final T source, final T target) {
103 
104         if (source == null) {
105             throw new IllegalArgumentException("param:0:: is null");
106         }
107 
108         if (target == null) {
109             throw new IllegalArgumentException("param:1:: is null");
110         }
111 
112         if (source.equals(target)) {
113             throw new IllegalArgumentException(
114                 "self dependency: " + source + " -> " + target);
115         }
116 
117         synchronized (dependencies) {
118 
119             Vector<T> targets = dependencies.get(source);
120             if (targets == null) {
121                 targets = new Vector<T>();
122                 dependencies.put(source, targets);
123             }
124 
125             if (!targets.contains(target)) {
126                 targets.addElement(target);
127             }
128         }
129     }

138     public Vector<Vector<T>> getDependencyGroups(final T source,
139                                                  final T target) {
140 
141         if (source == null) {
142             throw new IllegalArgumentException("param:0:: is null");
143         }
144 
145         if (target == null) {
146             throw new IllegalArgumentException("param:1:: is null");
147         }
148 
149         if (source.equals(target)) {
150             throw new IllegalArgumentException(
151                 "self dependency: " + source + " -> " + target);
152         }
153 
154         synchronized (dependencies) {
155 
156             if (!dependencies.containsKey(source)) {
157                 return new Vector();
158             }
159 
160             final Vector<Vector<T>> groups = new Vector<Vector<T>>();
161 
162             final Vector<T> targets = dependencies.get(source);
163 
164             for (int i = 0; i < targets.size(); i++) {
165 
166                 final T auxiliary = targets.elementAt(i);
167 
168                 if (auxiliary.equals(target)) {
169                     final Vector<T> group = new Vector<T>();
170                     group.addElement(source);
171                     group.addElement(auxiliary);
172                     groups.addElement(group);
173                     continue;
174                 }
175 
176                 final Vector<Vector<T>> dependencyGroups =
177                     getDependencyGroups(auxiliary, target);
178                 for (int j = 0; j <dependencyGroups.size(); j++) {
179                     final Vector<T> dependencyGroup =
180                         dependencyGroups.elementAt(j);
181                     groups.addElement(dependencyGroup);
182                     dependencyGroup.insertElementAt(source, 0);
183                 }
184             }
185 
186             return groups;
187         }

212     public boolean hasDependency(final T source, final T target) {
213 
214         if (source == null) {
215             throw new IllegalArgumentException("param:0:: is null");
216         }
217 
218         if (target == null) {
219             throw new IllegalArgumentException("param:1:: is null");
220         }
221 
222         if (source.equals(target)) {
223             throw new IllegalArgumentException(
224                 "self dependency: " + source + " -> " + target);
225         }
226 
227         synchronized (dependencies) {
228 
229             if (!dependencies.containsKey(source)) {
230                 return false;
231             }
232 
233             final Vector<T> targets = dependencies.get(source);
234 
235             if (targets.contains(target)) {
236                 return true;
237             }
238 
239             for (T auxiliary : targets) {
240                 if (hasDependency(auxiliary, target)) {
241                     return true;
242                 }
243             }
244 
245             return false;
246         }

327     private void getFlatten(final T source, final Vector<T> flatten) {
328 
329         synchronized (dependencies) {
330 
331             if (flatten.contains(source)) {
332                 return;
333             }
334 
335             final Vector<T> targets = dependencies.get(source);
336             if (targets != null) {

341 
342             flatten.addElement(source);

374     /**
375      *
376      * @param maximum
377      */
378     public Vector<Vector<T>> getHorizontalGroups(final int maximum) {
379 
380         if (maximum <= 0) {
381             throw new IllegalArgumentException(
382                 "param:0:" + Integer.TYPE + ":" + maximum + " <= 0");
383         }
384 
385         final Vector<T> remains = getFlatten();
386 
387         final Vector<Vector<T>> groups = new Vector<Vector<T>>();
388 
389         while (!remains.isEmpty() && (groups.size() < (maximum - 1))) {
390 
391             final Vector<T> group = new Vector<T>();
392 
393             groups.addElement(group);
394 
395             group.addElement(remains.firstElement());
396             remains.removeElementAt(0);
397 
398             for (int i = 0; i < remains.size();) {
399 
400                 boolean added = false;
401 
402                 for (T p : group) {
403                     if (hasDependency(remains.elementAt(i), p)) {
404                         group.addElement(remains.elementAt(i));
405                         remains.removeElementAt(i);
406                         added = true;
407                         break;
408                     }
409                 }
410 
411                 if (!added) {
412                     outer:
413                     for (int j = i + 1; j < remains.size(); j++) {
414                         if (!hasDependency(remains.elementAt(j),
415                                            remains.elementAt(i))) {
416                             continue;
417                         }
418                         for (T p : group) {
419                             if (hasDependency(remains.elementAt(j), p)) {
420                                 group.addElement(remains.elementAt(i));
421                                 remains.removeElementAt(i);
422                                 added = true;
423                                 break outer;
424                             }
425                         }
426                     }
427                 }
428 
429                 if (!added) {
430                     i++;
431                 }
432             }
433         }
434 
435         if (!remains.isEmpty()) {
436             groups.addElement(remains);
437         }
438 
439         return groups;
440     }

454      */
455     public Vector<Vector<T>> getVerticalGroups(final int maximum) {
456 
457         if (maximum <= 0) {
458             throw new IllegalArgumentException(
459                 "param:0:" + Integer.TYPE + ":" + maximum + " <= 0");
460         }
461 
462         final Vector<T> remains = getFlatten();
463 
464         final Vector<Vector<T>> groups = new Vector<Vector<T>>();
465 
466         while (!remains.isEmpty() && (groups.size() < (maximum - 1)))  {
467 
468             final Vector<T> group = new Vector<T>();
469 
470             groups.addElement(group);
471 
472             outer:
473             for (int i = remains.size() - 1; i >= 0; i--) {
474                 for (int j = 0; j < i; j++) {
475                     if (hasDependency(remains.elementAt(i),
476                                       remains.elementAt(j))) {
477                         continue outer;
478                     }
479                 }
480                 group.addElement(remains.elementAt(i));
481                 remains.removeElementAt(i);
482             }
483         }
484 
485         if (!remains.isEmpty()) {
486             groups.addElement(remains);
487         }
488 
489         return groups;
490     }
