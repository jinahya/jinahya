145     private static void splitText(FontMetrics metrics, int width, char[] line,
146                                   Vector vector) {
147 
148         int offset = 0;
149         while (offset < line.length) {
150             int length = splitText(metrics, width, line, offset, 1,
151                                    line.length - offset);
152             vector.addElement(new String(line, offset, length));
153             offset += length;
154         }
155     }

158     private static int splitText(FontMetrics metrics, int width, char[] line,
159                                  int offset, int low, int high) {
160 
161         if (metrics.charsWidth(line, offset, high) <= width) {
162             return high;
163         }
164         int half = (low + high) / 2;
165         if (metrics.charsWidth(line, offset, half) <= width) { // fit
166             return splitText(metrics, width, line, offset, half + 1, high);
167         } else { // not fit
168             return splitText(metrics, width, line, offset, low, half - 1);
169         }
170     }

173     protected void drawString(Graphics g, String string, Rectangle boundary,
174                               boolean clip, int halign, int valign) {
175 
176         if (string == null) {
177             return;
178         }
179         FontMetrics metrics = g.getFontMetrics();
180 
181         int stringWidth = metrics.stringWidth(string);
182         int x = boundary.x;
183         switch (halign) {
184             case 0x01: // LEFT
185                 break;
186             case 0x02: // CENTER
187                 x += (boundary.width - stringWidth) / 2;
188                 break;
189             case 0x03: // RIGHT
190                 x += (boundary.width - stringWidth);
191                 break;
192             default:
193                 break;
194         }
195 
196         int stringHeight = metrics.getAscent() + metrics.getDescent();
197         int y = boundary.y + metrics.getAscent(); // TOP
198         switch (valign) {
199             case 0x01: // TOP
200                 break;
201             case 0x02: // CENTER
202                 y += (boundary.height - stringHeight) / 2;
203                 break;
204             case 0x03: // BOTTOM
205                 y = boundary.y + boundary.height - metrics.getDescent();
206                 break;
207             default:
208                 break;
209         }
210         Shape previousClip = g.getClip();
211         if (clip) {
212             g.setClip(boundary);
213         }
214         g.drawString(string, x, y);
215         if (clip) {
216             g.setClip(previousClip);
217         }
218     }

284     protected Image getImage(Image image) {
285         synchronized (table) {
286             if (!table.containsKey(image)) { // NOT EVEN ADDED YET
287                 addImage(image);
288             }
289             if (!((Boolean) table.get(image)).booleanValue()) { // NOT LOADED
290                 int id = image.hashCode();
291                 tracker.addImage(image, id);
292                 try {
293                     tracker.waitForID(id);
294                 } catch (InterruptedException ie) {
295                     ie.printStackTrace();
296                 }
297                 if (tracker.isErrorID(id)) {
298                     System.out.println
299                         (getClass() + ": IMAGE LOADING ERROR: " + image);
300                 }
301                 tracker.removeImage(image, id);
302                 table.put(image, Boolean.TRUE);
303             }
304             return image;
305         }
306     }

317     public void removeAllImages() {
318         synchronized (table) {
319             Component[] components = getComponents();
320             for (int i = 0; i < components.length; i++) {
321                 if (components[i] instanceof View) {
322                     ((View) components[i]).removeAllImages();
323                 }
324             }
325             for (Enumeration e = table.keys(); e.hasMoreElements(); ) {
326                 Image image = (Image) e.nextElement();
327                 image.flush();
328             }
329             table.clear();
330         }
331     }
