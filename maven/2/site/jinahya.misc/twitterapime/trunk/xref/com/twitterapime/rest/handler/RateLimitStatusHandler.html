8   package com.twitterapime.rest.handler;
9   
10  import java.util.Hashtable;
11  
12  import com.twitterapime.model.MetadataSet;
13  import com.twitterapime.parser.DefaultXMLHandler;
14  import com.twitterapime.parser.ParserException;
15  import com.twitterapime.rest.RateLimitStatus;
16  import com.twitterapime.util.StringUtil;
17  
18  /**
19   * <p>
20   * Handler class for parsing the XML rate limit status from Twitter API. 
21   * </p>
22   * 
23   * @author Ernandes Mourao Junior (ernandes@gmail.com)
24   * @version 1.0
25   * @since 1.1
26   */
27  public class RateLimitStatusHandler extends DefaultXMLHandler {
28  	/**
29  	 * <p>
30  	 * Hash with rate limit status values.
31  	 * </p>
32  	 */
33  	private Hashtable limitStatusValues = new Hashtable(4);
34  	
35  	/**
36  	 * @see com.twitterapime.parser.DefaultXMLHandler#text(java.lang.String)
37  	 */
38  	public void text(String text) throws ParserException {
39  		text = text.trim();
40  		//
41  		if (xmlPath.equals("/hash/remaining-hits")) {
42  			limitStatusValues.put(
43  				MetadataSet.RATELIMITSTATUS_REMAINING_HITS, text);
44  		} else if (xmlPath.equals("/hash/hourly-limit")) {
45  			limitStatusValues.put(
46  				MetadataSet.RATELIMITSTATUS_HOURLY_LIMIT, text);
47  		} else if (xmlPath.equals("/hash/reset-time")) {
48  			limitStatusValues.put(
49  				MetadataSet.RATELIMITSTATUS_RESET_TIME,
50  				"" + StringUtil.convertTweetDateToLong(text));
51  		}
52  	}
53  
54  	/**
55  	 * <p>
56  	 * Return the parsed rate limit status info.
57  	 * </p>
58  	 * @return Rate limit status.
59  	 */
60  	public RateLimitStatus getParsedRateLimitStatus() {
61  		return new RateLimitStatus(limitStatusValues);
62  	}
63  }
