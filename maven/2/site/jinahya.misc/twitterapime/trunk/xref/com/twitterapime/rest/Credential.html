8   package com.twitterapime.rest;
9   
10  import java.util.Hashtable;
11  
12  import com.twitterapime.model.DefaultEntity;
13  import com.twitterapime.model.MetadataSet;
14  
15  /**
16   * <p>
17   * This class defines an entity that represents a user's credential.
18   * </p>
19   * 
20   * @author Ernandes Mourao Junior (ernandes@gmail.com)
21   * @version 1.0
22   * @since 1.1
23   * @see UserAccountManager
24   */
25  public final class Credential extends DefaultEntity {
26  	/**
27  	 * <p>
28  	 * Create an instance of Credential class.
29  	 * </p>
30  	 * @param username Username.
31  	 * @param password Password.
32  	 * @throws IllegalArgumentException If username/password is empty or null.
33  	 */
34  	public Credential(String username, String password) {
35  		if (username == null || (username = username.trim()).length() == 0) {
36  			throw new IllegalArgumentException(
37  				"Username must not be empty/null");
38  		}
39  		if (password == null || (password = password.trim()).length() == 0) {
40  			throw new IllegalArgumentException(
41  				"Password must not be empty/null");
42  		}
43  		//
44  		Hashtable credtls = new Hashtable(2);
45  		credtls.put(MetadataSet.CREDENTIAL_USERNAME, username);
46  		credtls.put(MetadataSet.CREDENTIAL_PASSWORD, password);
47  		setData(credtls);
48  	}
49  	
50  	/**
51  	 * <p>
52  	 * Get the credential in HttpAuth format (e.g. username:password).
53  	 * </p>
54  	 * @return Credential.
55  	 */
56  	String getBasicHttpAuthCredential() {
57  		return getString(MetadataSet.CREDENTIAL_USERNAME) +
58  		       ':' +
59  		       getString(MetadataSet.CREDENTIAL_PASSWORD);
60  	}
61  }
