175 	 * <p>
176 	 * Create a Http connection to the given URL.
177 	 * </p>
178 	 * @param url URL.
179 	 * @param c User's credential for log in purposes.
180 	 * @return Http connection.
181 	 * @throws IOException If an I/O error occurs.
182 	 */
183 	static synchronized HttpConnection getHttpConn(String url, Credential c)
184 		throws IOException {
185 		if (url == null || (url = url.trim()).length() == 0) {
186 			throw new IllegalArgumentException("URL must not be empty/null.");
187 		}
188 		//
189 		HttpConnection conn = HttpConnector.open(url);
190 		boolean hasException = true;
191 		//
192 		try {
193 			if (c != null) {
194 				String crdntls = c.getBasicHttpAuthCredential();
195 				crdntls = HttpConnector.encodeBase64(crdntls);
196 				conn.setRequestProperty("Authorization", "Basic " + crdntls);
197 			}
198 			//
199 			hasException = false;
200 		} finally {
201 			if (hasException) {
202 				conn.close();
203 			}
204 		}
205 		//
206 		return conn;
207 	}

275 	 * <p>
276 	 * Verify whether the given user's credential are valid. This method
277 	 * authenticates the API to Twitter REST API.
278 	 * </p>
279 	 * @return Valid credentials (true).
280 	 * @throws IOException If an I/O error occurs.
281 	 * @throws LimitExceededException If limit has been hit.
282 	 */
283 	public boolean verifyCredential() throws IOException,
284 		LimitExceededException {
285 		checkValid();
286 		//
287 		if (verified) {
288 			return true; //already verified.
289 		}
290 		//
291 		HttpConnection conn =
292 			getHttpConn(TWITTER_URL_VERIFY_CREDENTIALS, credential);
293 		//
294 		try {
295 			final int respCode = conn.getResponseCode();
296 			//
297 			if (respCode == HttpConnection.HTTP_OK) {
298 				verified = true;
299 				//
300 				try {
301 					Parser parser =	ParserFactory.getDefaultParser();
302 					AccountHandler handler = new AccountHandler();
303 					parser.parse(conn.openInputStream(), handler);
304 					//
305 					account = handler.getParsedUserAccount();
306 				} catch (ParserException e) {
307 					throw new IOException(e.getMessage());
308 				}
309 				//
310 				saveSelfOnPool();
311 			} else if (respCode == HttpConnection.HTTP_UNAUTHORIZED) {
312 				verified = false;
313 			} else {
314 				HttpResponseCodeInterpreter.perform(conn);
315 			}
316 		} finally {
317 			if (conn != null) {
318 				conn.close();
319 			}
320 		}
321 		//
322 		return verified;
323 	}
