101 	 * <p>
102 	 * Release the objects which account is no longer authenticated.
103 	 * </p>
104 	 */
105 	static synchronized void cleanPool() {
106 		if (tweetERPool != null) {
107 			Enumeration keys = tweetERPool.keys();
108 			Object key;
109 			TweetER value;
110 			//
111 			while (keys.hasMoreElements()) {
112 				key = keys.nextElement();
113 				value = (TweetER)tweetERPool.get(key);
114 				//
115 				if (!value.userAccountMngr.isVerified()) {
116 					tweetERPool.remove(key);
117 				}
118 			}
119 		}
120 	}

133 		if (uam == null) {
134 			throw new IllegalArgumentException(
135 				"UserAccountManager must not be null.");
136 		}
137 		//
138 		if (!uam.isVerified()) {
139 			throw new SecurityException("User's credential must be verified.");
140 		}
141 		//
142 		if (tweetERPool == null) {
143 			tweetERPool = new Hashtable();
144 		}
145 		//
146 		TweetER ter = (TweetER)tweetERPool.get(uam);
147 		if (ter == null) {
148 			ter = new TweetER(uam);
149 			tweetERPool.put(uam, ter);
150 		}
151 		//
152 		return ter;
153 	}

201 	/**
202 	 * <p>
203 	 * Find the tweet associated to the given ID. This method does not require
204 	 * to be logged in to Twitter to use it. In other words, you can use the
205 	 * TweetER single instance ({@link TweetER#getInstance()}). If the given ID
206 	 * does not exist, <code>null</code> will be returned.
207 	 * </p>
208 	 * @param id Tweet's ID.
209 	 * @return Tweet.
210 	 * @throws LimitExceededException If the limit of access is exceeded.
211 	 * @throws IOException If an I/O error occurs.
212 	 * @throws SecurityException If the requested tweet is protected.
213 	 * @throws IllegalArgumentException If the given ID is empty/null.
214 	 */
215 	public Tweet findByID(String id) throws LimitExceededException,
216 		IOException {
217 		if (id == null || (id = id.trim()).length() == 0) {
218 			throw new IllegalArgumentException("ID must not be empty/null.");
219 		}
220 		//
221 		final String url = TWITTER_URL_SHOW_STATUS + id + ".xml";
222 		Credential credential = null;
223 		//
224 		if (userAccountMngr != null) {
225 			checkUserAuth();
226 			credential = userAccountMngr.getCredential();
227 		}
228 		//
229 		HttpConnection conn = UserAccountManager.getHttpConn(url, credential);
230 		//
231 		try {
232 			final int repCode = conn.getResponseCode();
233 			//
234 			if (repCode == HttpConnection.HTTP_FORBIDDEN) {
235 				throw new SecurityException(); //the refereed tweet id is protected.
236 			} else if (repCode == HttpConnection.HTTP_NOT_FOUND) {
237 				return (Tweet)null; //tweet id not found.
238 			}
239 			//
240 			HttpResponseCodeInterpreter.perform(conn);
241 			//
242 			Parser parser = ParserFactory.getDefaultParser();
243 			StatusHandler handler = new StatusHandler();
244 			parser.parse(conn.openInputStream(), handler);
245 			//
246 			return handler.getParsedTweet();
247 		} catch (ParserException e) {
248 			throw new IOException(e.getMessage());
249 		} finally {
250 			if (conn != null) {
251 				conn.close();
252 			}
253 		}
254 	}

256 	/**
257 	 * <p>
258 	 * Post a given tweet to Twitter. The tweet's content must be up to 140
259 	 * characters and must be properly logged in ({@link UserAccountManager}).
260 	 * </p>
261 	 * @param tweet Tweet to be posted.
262 	 * @return Tweet post with some additional data.
263 	 * @throws IOException If an I/O error occurs.
264 	 * @throws LimitExceededException If limit has been hit.
265 	 * @throws SecurityException If it is not properly logged in.
266 	 * @throws IllegalArgumentException If the given tweet is null/empty, etc.
267 	 */
268 	public Tweet post(Tweet tweet) throws IOException, LimitExceededException {
269 		if (tweet == null) {
270 			throw new IllegalArgumentException("Tweet must not be null.");
271 		}
272 		//
273 		tweet.validateContent();
274 		//
275 		checkUserAuth();
276 		//
277 		HttpConnection conn = getConn(TWITTER_URL_UPDATE_STATUS);
278 		//
279 		try {
280 			final String content = tweet.getString(MetadataSet.TWEET_CONTENT);
281 			//
282 			conn.setRequestMethod(HttpConnection.POST);
283 			//
284 			OutputStream out = conn.openOutputStream();
285 			out.write(("status=" + content).getBytes());
286 			out.flush();
287 			out.close();
288 			//
289 			HttpResponseCodeInterpreter.perform(conn);
290 			//
291 			Parser parser = ParserFactory.getDefaultParser();
292 			StatusHandler handler = new StatusHandler();
293 			parser.parse(conn.openInputStream(), handler);
294 			handler.loadParsedTweet(tweet);
295 			//
296 			return tweet;
297 		} catch (ParserException e) {
298 			throw new IOException(e.getMessage());
299 		} finally {
300 			if (conn != null) {
301 				conn.close();
302 			}
303 		}
304 	}

353 	/**
354 	 * <p>
355 	 * Send a given Direct Message to a given recipient. The DM's content must
356 	 * be up to 140 characters, addressed to a proper user (he/she must also
357 	 * follow you) and must be properly logged in ({@link UserAccountManager}).
358 	 * </p>
359 	 * @param dm Direct Message to be sent.
360 	 * @return Sent Direct Message.
361 	 * @throws IOException If an I/O error occurs.
362 	 * @throws LimitExceededException If limit has been hit.
363 	 * @throws InvalidQueryException If recipient does not exist or the sender
364 	 *         does not follow the recipient and vice versa.
365 	 * @throws SecurityException If it is not properly logged in.
366 	 * @throws IllegalArgumentException If the given DM is null/empty, etc.
367 	 */
368 	public Tweet send(Tweet dm) throws IOException, LimitExceededException {
369 		if (dm == null) {
370 			throw new IllegalArgumentException("DM must not be null.");
371 		}
372 		//
373 		dm.validateRecipient();
374 		dm.validateContent();
375 		//
376 		checkUserAuth();
377 		//
378 		HttpConnection conn = getConn(TWITTER_URL_SEND_DIRECT_MESSAGE);
379 		//
380 		try {
381 			final String content = dm.getString(MetadataSet.TWEET_CONTENT);
382 			String recipient = dm.getString(MetadataSet.TWEET_AUTHOR_USERNAME);
383 			if (recipient == null) {
384 				UserAccount ua = dm.getUserAccount();
385 				recipient = ua.getString(MetadataSet.USERACCOUNT_ID);
386 				if (recipient == null) {
387 					recipient =	ua.getString(MetadataSet.USERACCOUNT_USER_NAME);
388 				}
389 			}
390 			//
391 			conn.setRequestMethod(HttpConnection.POST);
392 			//
393 			OutputStream out = conn.openOutputStream();
394 			out.write(("user=" + recipient).getBytes());
395 			out.write(("&text=" + content).getBytes());
396 			out.flush();
397 			out.close();
398 			//
399 			HttpResponseCodeInterpreter.perform(conn);
400 			//
401 			Parser parser = ParserFactory.getDefaultParser();
402 			DirectMessageHandler handler = new DirectMessageHandler();
403 			parser.parse(conn.openInputStream(), handler);
404 			handler.loadParsedTweet(dm, 0);
405 			//
406 			return dm;
407 		} catch (ParserException e) {
408 			throw new IOException(e.getMessage());
409 		} finally {
410 			if (conn != null) {
411 				conn.close();
412 			}
413 		}
414 	}
