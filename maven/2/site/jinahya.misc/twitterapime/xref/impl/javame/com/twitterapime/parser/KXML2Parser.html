23  /**
24   * <p>
25   * This class implements a parser based on KXML2 library.
26   * </p>
27   * 
28   * @author Ernandes Mourao Junior (ernandes@gmail.com)
29   * @version 1.0
30   * @since 1.1
31   */
32  public final class KXML2Parser extends Parser {
33  	/**
34  	 * @see com.twitterapime.parser.Parser#parse(java.io.InputStream, com.twitterapime.parser.Handler)
35  	 */
36  	public void parse(InputStream stream, Handler handler) throws IOException,
37  		ParserException {
38  		if (stream == null) {
39  			throw new NullPointerException("stream must not be null.");
40  		}
41  		if (handler == null) {
42  			throw new NullPointerException("handler must not be null.");
43  		} else if (!(handler instanceof XMLHandler)) {
44  			throw new ClassCastException(
45  				"handler must be an instace of XMLHandler.");
46  		}
47  		//
48  		KXmlParser parser = new KXmlParser();
49  		XMLHandler xmlHandler = (XMLHandler)handler;
50  		KXML2Attributes attrs = new KXML2Attributes();
51  		//
52  		try {
53  			parser.setInput(new InputStreamReader(stream));
54  			int etype;
55  			//
56  			while (true) {
57  				etype = parser.next();
58  				//
59  				if (etype == XmlPullParser.START_DOCUMENT) {
60  					xmlHandler.startDocument();
61  				} else if (etype == XmlPullParser.START_TAG) {
62  					attrs.loadAttributes(parser);
63  					//
64  					xmlHandler.startElement(
65  						parser.getNamespace(),
66  						parser.getName(),
67  						getQName(parser),
68  						attrs);
69  				} else if (etype == XmlPullParser.END_TAG) {
70  					xmlHandler.endElement(
71  						parser.getNamespace(),
72  						parser.getName(),
73  						getQName(parser));
74  				} else if (etype == XmlPullParser.TEXT) {
75  					xmlHandler.text(parser.getText().trim());
76  				} else if (etype == XmlPullParser.END_DOCUMENT) {
77  					xmlHandler.endDocument();
78  					break;
79  				}
80  			}
81  		} catch (XmlPullParserException e) {
82  			throw new ParserException(e.getMessage());
83  		}
84  	}
85  	
86  	/**
87  	 * Get the qualified name of the current tag beig parsed.
88  	 * @param p Parser.
89  	 * @return Qualified name.
90  	 */
91  	private String getQName(KXmlParser p) {
92  		String qName = p.getPrefix();
93  		if (qName != null && !qName.equals("")) {
94  			qName += ":" + p.getName();
95  		}
96  		//
97  		return qName;
98  	}
