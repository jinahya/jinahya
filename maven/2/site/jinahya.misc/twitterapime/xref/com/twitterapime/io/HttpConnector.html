15  /**
16   * <p>
17   * This is factory class for creating new HttpConnection objects.
18   * </p>
19   * <p>
20   * The creation of HttpConnection is performed dynamically by looking up the
21   * current platform provider from PlatformProviderSelector class. For each
22   * supported platform, there is a specific implementation class.
23   * </p>
24   * <p>
25   * The parameter string that describes the target should conform to the Http URL
26   * format as described in RFC 1738.
27   * </p>
28   * 
29   * @author Ernandes Mourao Junior (ernandes@gmail.com)
30   * @version 1.1
31   * @since 1.0
32   * @see HttpConnection
33   */
34  public final class HttpConnector {
35  	/**
36  	 * <p>
37  	 * Create and open a HttpConnection.
38  	 * </p>
39  	 * 
40  	 * @param url The URL for the connection.
41  	 * @return A new HttpConnection object.
42  	 * @throws IOException If an I/O error occurs.
43  	 * @throws IllegalArgumentException If url is null or empty.
44  	 */
45  	public static HttpConnection open(String url) throws IOException {
46  		if (url == null || url.trim().length() == 0) {
47  			throw new IllegalArgumentException("URL must not be null/empty.");
48  		}
49  		//
50  		final String JAVA_ME_HTTP_IMPL_CLASS =
51  			"impl.javame.com.twitterapime.io.HttpConnectionImpl";
52  		final String JAVA_ME_HTTP_USER_AGENT =
53  			"Twitter API ME/1.1 (compatible; Java ME; MIDP-2.0; CLDC-1.0)";
54  		final String ANDROID_HTTP_IMPL_CLASS =
55  			"impl.android.com.twitterapime.io.HttpConnectionImpl";
56  		final String ANDROID_HTTP_USER_AGENT =
57  			"Twitter API ME/1.1 (compatible; Android 1.1)";
58  		//
59  		final long PPID = PlatformProviderSelector.getCurrentProvider().getID();
60  		//
61  		HttpConnection conn = null;
62  		String userAgent = null;
63  		//
64  		if (PPID == PlatformProvider.PPID_JAVA_ME) {
65  			conn = newInstance(JAVA_ME_HTTP_IMPL_CLASS);
66  			userAgent = JAVA_ME_HTTP_USER_AGENT;
67  		} else if (PPID == PlatformProvider.PPID_ANDROID) {
68  			conn = newInstance(ANDROID_HTTP_IMPL_CLASS);
69  			userAgent = ANDROID_HTTP_USER_AGENT;
70  		} else {
71  			throw new IllegalArgumentException("Unknown platform ID: " + PPID);
72  		}
73  		//
74  		// #ifdef DEBUG
75  		System.out.println(url);
76  		// #endif
77  		conn.open(url);
78  		conn.setRequestProperty("User-Agent", userAgent);
79  		//
80  		return conn;
81  	}

174 	/**
175 	 * <p>
176 	 * Encode a given string to Base 64.
177 	 * </p>
178 	 * <p>
179 	 * <i>This implementation is adopted from Kenneth Ballard's HttpClient
180 	 * package. Released under LGPL.</i>
181 	 * </p>
182 	 * @param str String to encode.
183 	 * @return Base64 encoded string.
184 	 * @throws IllegalArgumentException If url is null or empty.
185 	 */
186 	public static String encodeBase64(String str) {
187 		if (str == null) {
188 			throw new IllegalArgumentException("Str must not be null.");
189 		}
190 		//
191 		final String c =
192 			"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
193 		//
194 		byte[] code = c.getBytes();
195 		byte[] s = str.getBytes();
196 		int x;
197 		int y = str.length() - (str.length() % 3);
198 		byte[] coded = new byte[4];
199 		StringBuffer dest = new StringBuffer();
200 		//
201 		for (x = 0; x < y; x += 3) {
202 			coded[3] = code[s[x + 2] % 64];
203 			coded[0] = code[s[x] >> 2];
204 			coded[1] = new Integer((s[x] % 4) << 4).byteValue();
205 			coded[1] += s[x + 1] >> 4;
206 			coded[1] = code[coded[1]];
207 			coded[2] = new Integer((s[x + 1] % 16) << 2).byteValue();
208 			coded[2] += s[x + 2] / 64;
209 			coded[2] = code[coded[2]];
210 			//
211 			dest.append(new String(coded));
212 		}
213 		//
214 		x = y;
215 		//
216 		if (s.length % 3 == 0) {
217 			return dest.toString();
218 		}
219 		if (s.length % 3 == 1) {
220 			coded[2] = '=';
221 			coded[3] = '=';
222 			coded[0] = code[s[x] >> 2];
223 			coded[1] = code[new Integer((s[x] % 4) << 4).byteValue()];
224 			//
225 			dest.append(new String(coded));
226 		}
227 		if (s.length % 3 == 2) {
228 			coded[3] = '=';
229 			coded[0] = code[s[x] >> 2];
230 			coded[1] = new Integer((s[x] % 4) << 4).byteValue();
231 			coded[1] += s[x + 1] >> 4;
232 			coded[1] = code[coded[1]];
233 			coded[2] = code[new Integer((s[x + 1] % 16) << 2).byteValue()];
234 			//
235 			dest.append(new String(coded));
236 		}
237 		//
238 		return dest.toString();
239 	}
