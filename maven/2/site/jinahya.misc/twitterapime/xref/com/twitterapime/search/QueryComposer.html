181 	/**
182 	 * <p>
183 	 * Append a query to another one.
184 	 * </p>
185 	 * @param q1 Query 1.
186 	 * @param q2 Query 2.
187 	 * @return A new query object with the content from both queries
188 	 *         concatenated.
189 	 * @throws IllegalArgumentException If q1/q2 is null.
190 	 */
191 	public static Query append(Query q1, Query q2) {
192 		if (q1 == null || q2 == null) {
193 			throw new IllegalArgumentException("Q1/Q2 must not be null.");
194 		}
195 		//
196 		return new Query(q1.toString() + q2.toString());
197 	}

524 	 * <p>

526 	 * </p>
527 	 * @param date The date.
528 	 * @return The date in string format.
529 	 */
530 	private static String convertDate(Date date) {
531 		Calendar c = Calendar.getInstance();
532 		c.setTime(date);
533 		//
534 		return c.get(Calendar.YEAR) + "-" +	(c.get(Calendar.MONTH) + 1) + "-" +
535 			c.get(Calendar.DAY_OF_MONTH);
536 	}
