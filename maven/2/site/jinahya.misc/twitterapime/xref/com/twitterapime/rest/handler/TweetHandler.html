8   package com.twitterapime.rest.handler;

10  import java.util.Hashtable;
11  
12  import com.twitterapime.model.MetadataSet;
13  import com.twitterapime.parser.DefaultXMLHandler;
14  import com.twitterapime.util.StringUtil;
15  
16  /**
17   * <p>
18   * Handler class for parsing the XML tweet from Twitter API. 
19   * </p>
20   * 
21   * @author Ernandes Mourao Junior (ernandes@gmail.com)
22   * @version 1.1
23   * @since 1.1
24   */
25  public final class TweetHandler extends DefaultXMLHandler {
26  	/**
27  	 * <p>
28  	 * Populate the given hash according to the tags and their values
29  	 * </p>
30  	 * @param data Hash to be populated.
31  	 * @param path XML path.
32  	 * @param text Tag's text.
33  	 */
34  	public void populate(Hashtable data, String path, String text) {
35  		if (path.endsWith("/created_at")) {
36  			data.put(
37  				MetadataSet.TWEET_PUBLISH_DATE,
38  				"" + StringUtil.convertTweetDateToLong(text));
39  		} else if (path.endsWith("/id")) {
40  			data.put(MetadataSet.TWEET_ID, text);
41  		} else if (path.endsWith("/text")) {
42  			data.put(MetadataSet.TWEET_CONTENT, text);
43  		} else if (path.endsWith("/source")) {
44  			data.put(MetadataSet.TWEET_SOURCE, StringUtil.removeTags(text));
45  		} else if (path.endsWith("/favorited")) {
46  			data.put(MetadataSet.TWEET_FAVOURITE, text);
47  		}
48  	}
49  }
