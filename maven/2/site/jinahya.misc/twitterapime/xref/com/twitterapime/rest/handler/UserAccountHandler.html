10  import java.util.Hashtable;
11  
12  import com.twitterapime.model.MetadataSet;
13  import com.twitterapime.parser.DefaultXMLHandler;
14  import com.twitterapime.util.StringUtil;

16  /**
17   * <p>
18   * Handler class for parsing the XML user account from Twitter API. 
19   * </p>
20   * 
21   * @author Ernandes Mourao Junior (ernandes@gmail.com)
22   * @version 1.1
23   * @since 1.1
24   */
25  public final class UserAccountHandler extends DefaultXMLHandler {
26  	/**
27  	 * <p>
28  	 * Populate the given hash according to the tags and their values
29  	 * </p>
30  	 * @param data Hash to be populated.
31  	 * @param path XML path.
32  	 * @param text Tag's text.
33  	 */
34  	public void populate(Hashtable data, String path, String text) {
35  		if (path.endsWith("/id")) {
36  			data.put(MetadataSet.USERACCOUNT_ID, text);
37  		} else if (path.endsWith("/name")) {
38  			data.put(MetadataSet.USERACCOUNT_NAME, text);
39  		} else if (path.endsWith("/screen_name")) {
40  			data.put(MetadataSet.USERACCOUNT_USER_NAME, text);
41  		} else if (path.endsWith("/location")) {
42  			data.put(MetadataSet.USERACCOUNT_LOCATION, text);
43  		} else if (path.endsWith("/description")) {
44  			data.put(MetadataSet.USERACCOUNT_DESCRIPTION, text);
45  		} else if (path.endsWith("/profile_image_url")) {
46  			data.put(MetadataSet.USERACCOUNT_PICTURE_URI, text);
47  		} else if (path.endsWith("/url")) {
48  			data.put(MetadataSet.USERACCOUNT_URL, text);
49  		} else if (path.endsWith("/protected")) {
50  			data.put(MetadataSet.USERACCOUNT_PROTECTED, text);
51  		} else if (path.endsWith("/followers_count")) {
52  			data.put(MetadataSet.USERACCOUNT_FOLLOWERS_COUNT, text);
53  		} else if (path.endsWith("/profile_background_color")) {
54  			data.put(MetadataSet.USERACCOUNT_PROFILE_BACKGROUND_COLOR, text);
55  		} else if (path.endsWith("/profile_text_color")){
56  			data.put(MetadataSet.USERACCOUNT_PROFILE_TEXT_COLOR, text);
57  		} else if (path.endsWith("/profile_link_color")){
58  			data.put(MetadataSet.USERACCOUNT_PROFILE_LINK_COLOR, text);
59  		} else if (path.endsWith("/friends_count")) {
60  			data.put(MetadataSet.USERACCOUNT_FRIENDS_COUNT, text);
61  		} else if (path.endsWith("/created_at")) {
62  			data.put(
63  				MetadataSet.USERACCOUNT_CREATE_DATE,
64  				"" + StringUtil.convertTweetDateToLong(text));
65  		} else if (path.endsWith("/favourites_count")) {
66  			data.put(MetadataSet.USERACCOUNT_FAVOURITES_COUNT, text);
67  		} else if (path.endsWith("/utc_offset")) {
68  			data.put(MetadataSet.USERACCOUNT_UTC_OFFSET, text);
69  		} else if (path.endsWith("/time_zone")) {
70  			data.put(MetadataSet.USERACCOUNT_TIME_ZONE, text);
71  		} else if (path.endsWith("/profile_background_image_url")){
72  			data.put(MetadataSet.USERACCOUNT_PROFILE_BACKGROUND_IMAGE_URI,text);
73  		} else if (path.endsWith("/statuses_count")) {
74  			data.put(MetadataSet.USERACCOUNT_TWEETS_COUNT, text);
75  		} else if (path.endsWith("/notifications")) {
76  			data.put(MetadataSet.USERACCOUNT_NOTIFICATIONS, text);
77  		} else if (path.endsWith("/verified")) {
78  			data.put(MetadataSet.USERACCOUNT_VERIFIED, text);
79  		}
80  	}
81  }
