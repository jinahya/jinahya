282 	private Tweet[] searchTweets(Query query, final SearchDeviceListener l)
283 		throws IOException, LimitExceededException {
284 		updateAPIInfo();
285 		//
286 		HttpConnection conn = getHttpConn(query.toString());
287 		Parser parser = ParserFactory.getDefaultParser();
288 		SearchResultHandler handler = new SearchResultHandler();
289 		handler.setSearchDeviceListener(l);
290 		//
291 		try {
292 			parser.parse(conn.openInputStream(), handler);
293 			//
294 			return handler.getParsedTweets();
295 		} catch (ParserException e) {
296 			throw new IOException(e.getMessage());
297 		} finally {
298 			conn.close();
299 		}

302 	/**
303 	 * <p>
304 	 * Get a Http connection to the given query string.
305 	 * </p>
306 	 * @param queryStr The query string.
307 	 * @return The Http connection object.
308 	 * @throws IOException If an I/O error occurs.
309 	 * @throws LimitExceededException If the limit of access is exceeded.
310 	 */
311 	private HttpConnection getHttpConn(String queryStr) throws IOException,
312 		LimitExceededException {
313 		if (queryStr == null || (queryStr = queryStr.trim()).length() == 0) {
314 			throw new IllegalArgumentException(
315 				"Query String must not be empty/null.");
316 		}
317 		//
318 		if (!queryStr.startsWith("?")) {
319 			if (!queryStr.startsWith(TWITTER_QUERY_STRING_PREFIX)) {
320 				queryStr = '?' + TWITTER_QUERY_STRING_PREFIX + queryStr;
321 			} else {
322 				queryStr = '?' + queryStr;
323 			}
324 		}
325 		//
326 		final String url = TWITTER_URL_ATOM + HttpConnector.encodeURL(queryStr);
327 		HttpConnection c = HttpConnector.open(url);
328 		boolean hasException = true;
329 		//
330 		try {
331 			c.setRequestMethod(HttpConnection.GET);
332 			//
333 			if (c.getResponseCode() == HttpConnection.HTTP_FORBIDDEN) {
334 				throw new InvalidQueryException(
335 					HttpResponseCodeInterpreter.getErrorMessage(c));
336 			}
337 			//
338 			//verify whether there is an error in the request.
339 			HttpResponseCodeInterpreter.perform(c);
340 			hasException = false;
341 		} finally {
342 			if (hasException) {
343 				c.close();
344 			}
345 		}
346 		//
347 		return c;
348 	}
