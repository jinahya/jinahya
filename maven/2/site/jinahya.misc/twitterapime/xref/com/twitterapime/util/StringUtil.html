33  	/**
34  	 * <p>
35  	 * Convert a given date according to a format used by Twitter to long. 
36  	 * </p>
37  	 * <p>
38  	 * Formats:
39  	 * <lu>
40  	 * <li>2009-11-28 21:43:12</li>
41  	 * <li>2009-12-01T01:25:00+00:00</li>
42  	 * <li>Sat Nov 07 21:30:03 +0000 2009</li>
43  	 * </lu>
44  	 * </p>
45  	 * @param date Tweet date value.
46  	 * @return Long date.
47  	 * @throws IllegalArgumentException If date is null/empty.
48  	 */
49  	public static long convertTweetDateToLong(String date) {
50  		if (date == null || (date = date.trim()).length() == 0) {
51  			throw new IllegalArgumentException("Date must not be null/empty.");
52  		}
53  		//
54  		char fc = date.charAt(0);
55  		//)
56  		if (Character.isDigit(fc)) { //is date format 1?
57  			date = formatTweetDate1(date);
58  		} else { //date format 2.
59  			date = formatTweetDate2(date);
60  		}
61  		//
62  		String[] parts = split(date, ' ');
63  		String[] dparts = split(parts[0], '-');
64  		String[] tparts = split(parts[1], ':');
65  		//
66  		Calendar c = Calendar.getInstance();
67  		c.set(Calendar.YEAR, Integer.parseInt(dparts[0]));
68  		c.set(Calendar.MONTH, Integer.parseInt(dparts[1]) -1);
69  		c.set(Calendar.DAY_OF_MONTH, Integer.parseInt(dparts[2]));
70  		c.set(Calendar.HOUR_OF_DAY, Integer.parseInt(tparts[0]));
71  		c.set(Calendar.MINUTE, Integer.parseInt(tparts[1]));
72  		c.set(Calendar.SECOND, Integer.parseInt(tparts[2]));
73  		c.set(Calendar.MILLISECOND, 0);
74  		//
75  		String dim = Long.toString(c.getTime().getTime());
76  		dim = dim.substring(0, dim.length() -3);
77  		//
78  		return Long.parseLong(dim); 
79  	}

91      	if (str == null) {
92      		throw new IllegalArgumentException("Str must not be null.");
93      	}
94      	//
95          Vector v = new Vector();
96          int start = 0;
97          int iof;
98          //
99          while ((iof = str.indexOf(delimiter, start)) != -1) {
100             v.addElement(str.substring(start, iof).trim());
101             start = iof +1;
102         }
103         //
104         v.addElement(str.substring(start, str.length()).trim());

106         v.copyInto(codes);
107         //

142 		if (str == null) {
143 			throw new IllegalArgumentException("Str must not be null.");
144 		}
145 		//
146 		StringBuffer out = new StringBuffer();
147 		char cs[] = str.toCharArray();
148 		boolean tagFound = false;
149 		int i1 = 0;
150 		int l = 0;
151 		//
152 		for (int i = 0; i < cs.length; i++) {
153 			if (cs[i] == '<' && !tagFound) {
154 				out.append(cs, i1, l);
155 				//
156 				i1 = i;
157 				l = 0;
158 				tagFound = true;
159 				l++;
160 			} else if (cs[i] == '>' && tagFound) {
161 				i1 = i +1;
162 				l = 0;
163 				tagFound = false;
164 			} else {
165 				l++;
166 			}
167 		}
168 		if (l > 0) {
169 			out.append(cs, i1, l);
170 		}
171 		//
172 		return out.toString().trim();
173 	}
