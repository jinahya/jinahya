8   package com.twitterapime.rest.handler;
9   
10  import java.util.Hashtable;
11  
12  import com.twitterapime.model.MetadataSet;
13  import com.twitterapime.parser.DefaultXMLHandler;
14  import com.twitterapime.parser.ParserException;
15  import com.twitterapime.rest.UserAccount;
16  import com.twitterapime.search.Tweet;
17  
18  /**
19   * <p>
20   * Handler class for parsing the status' XML results from Twitter API. 
21   * </p>
22   * 
23   * @author Ernandes Mourao Junior (ernandes@gmail.com)
24   * @version 1.0
25   * @since 1.2
26   */
27  public final class StatusHandler extends DefaultXMLHandler {
28  	/**
29  	 * <p>
30  	 * User Account XML handler object.
31  	 * </p>
32  	 */
33  	private UserAccountHandler uaHandler = new UserAccountHandler();
34  	
35  	/**
36  	 * <p>
37  	 * Tweet XML handler object.
38  	 * </p>
39  	 */
40  	private TweetHandler tHandler = new TweetHandler();
41  	
42  	/**
43  	 * <p>
44  	 * Hash with user account values.
45  	 * </p>
46  	 */
47  	private Hashtable userAccountValues = new Hashtable(25);
48  	
49  	/**
50  	 * <p>
51  	 * Hash with user account values from reposted tweet.
52  	 * </p>
53  	 */
54  	private Hashtable reuserAccountValues = new Hashtable(25);
55  
56  	/**
57  	 * <p>
58  	 * Hash with tweet values.
59  	 * </p>
60  	 */
61  	private Hashtable tweetValues = new Hashtable(10);
62  	
63  	/**
64  	 * <p>
65  	 * Hash with reposted tweet values.
66  	 * </p>
67  	 */
68  	private Hashtable retweetValues = new Hashtable(10);
69  	
70  	/**
71  	 * @see com.twitterapime.parser.DefaultXMLHandler#text(java.lang.String)
72  	 */
73  	public void text(String text) throws ParserException {
74  		text = text.trim();
75  		//
76  		if (xmlPath.startsWith("/status/retweeted_status/user/")) {
77  			uaHandler.populate(reuserAccountValues, xmlPath, text);
78  		} else if (xmlPath.startsWith("/status/retweeted_status/")) {
79  			tHandler.populate(retweetValues, xmlPath, text);
80  		} else if (xmlPath.startsWith("/status/user/")) {
81  			uaHandler.populate(userAccountValues, xmlPath, text);
82  		} else if (xmlPath.startsWith("/status/")) {
83  			tHandler.populate(tweetValues, xmlPath, text);
84  		}
85  	}
86  	
87  	/**
88  	 * @see com.twitterapime.parser.DefaultXMLHandler#endDocument()
89  	 */
90  	public void endDocument() throws ParserException {
91  		tweetValues.put(
92  			MetadataSet.TWEET_USER_ACCOUNT, new UserAccount(userAccountValues));
93  		if (retweetValues.size() > 0) { // is it a retweet?
94  			retweetValues.put(
95  				MetadataSet.TWEET_USER_ACCOUNT,
96  				new UserAccount(reuserAccountValues));
97  			tweetValues.put(MetadataSet.TWEET_REPOSTED_TWEET, retweetValues);
98  		}
99  	}
100 	
101 	/**
102 	 * <p>
103 	 * Return the parsed tweet.
104 	 * </p>
105 	 * @return Tweet.
106 	 */
107 	public Tweet getParsedTweet() {
108 		return new Tweet(tweetValues);
109 	}
110 	
111 	/**
112 	 * <p>
113 	 * Load the parsed values into the given tweet.
114 	 * </p>
115 	 * @param tweet Tweet to be loaded.
116 	 */
117 	public void loadParsedTweet(Tweet tweet) {
118 		tweet.setData(tweetValues);
119 	}
120 }
