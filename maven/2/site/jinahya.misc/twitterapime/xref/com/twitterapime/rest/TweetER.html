99  	 * Release the objects which account is no longer authenticated.
100 	 * </p>

102 	private static void cleanPool() {
103 		Enumeration keys = tweetERPool.keys();
104 		Object key;
105 		TweetER value;
106 		//
107 		while (keys.hasMoreElements()) {
108 			key = keys.nextElement();
109 			value = (TweetER)tweetERPool.get(key);
110 			//
111 			if (!value.userAccountMngr.isVerified()) {
112 				tweetERPool.remove(key);
113 			}
114 		}
115 	}

127 	public synchronized static TweetER getInstance(UserAccountManager uam) {
128 		if (uam == null) {
129 			throw new IllegalArgumentException(
130 				"UserAccountManager must not be null.");
131 		}
132 		//
133 		if (!uam.isVerified()) {
134 			throw new SecurityException("User's credential must be verified.");
135 		}
136 		//
137 		if (tweetERPool == null) {
138 			tweetERPool = new Hashtable();
139 		}
140 		//
141 		cleanPool();
142 		//
143 		TweetER ter = (TweetER)tweetERPool.get(uam);
144 		if (ter == null) {
145 			ter = new TweetER(uam);
146 			tweetERPool.put(uam, ter);
147 		}
148 		//
149 		return ter;
150 	}

198 	/**
199 	 * <p>
200 	 * Find the tweet associated to the given ID. This method does not require
201 	 * to be logged in to Twitter to use it. In other words, you can use the
202 	 * TweetER single instance ({@link TweetER#getInstance()}). If the given ID
203 	 * does not exist, <code>null</code> will be returned.
204 	 * </p>
205 	 * @param id Tweet's ID.
206 	 * @return Tweet.
207 	 * @throws LimitExceededException If the limit of access is exceeded.
208 	 * @throws IOException If an I/O error occurs.
209 	 * @throws SecurityException If the request tweet is protected.
210 	 * @throws IllegalArgumentException If the given ID is empty/null.
211 	 */
212 	public Tweet findByID(String id) throws LimitExceededException,
213 		IOException {
214 		if (id == null || (id = id.trim()).length() == 0) {
215 			throw new IllegalArgumentException("ID must not be empty/null.");
216 		}
217 		//
218 		final String url = TWITTER_URL_SHOW_STATUS + id + ".xml";
219 		Credential credential = null;
220 		//
221 		if (userAccountMngr != null) {
222 			checkUserAuth();
223 			credential = userAccountMngr.getCredential();
224 		}
225 		//
226 		HttpConnection conn = UserAccountManager.getHttpConn(url, credential);
227 		//
228 		try {
229 			HttpResponseCodeInterpreter.perform(conn);
230 			//
231 			Parser parser = ParserFactory.getDefaultParser();
232 			StatusHandler handler = new StatusHandler();
233 			parser.parse(conn.openInputStream(), handler);
234 			//
235 			return handler.getParsedTweet();
236 		} catch (ParserException e) {
237 			throw new IOException(e.getMessage());
238 		} catch (InvalidQueryException e) {
239 			final int repCode = conn.getResponseCode();
240 			//
241 			if (repCode == HttpConnection.HTTP_FORBIDDEN) {
242 				//the refered tweet id is protected.
243 				throw new SecurityException(e.getMessage());
244 			} else if (repCode == HttpConnection.HTTP_NOT_FOUND) {
245 				//tweet id not found.
246 				return (Tweet)null;
247 			} else {
248 				throw e;
249 			}
250 		} finally {
251 			if (conn != null) {
252 				conn.close();
253 			}
254 		}
255 	}

257 	/**
258 	 * <p>
259 	 * Post a given tweet to Twitter. The tweet's content must be up to 140
260 	 * characters and must be properly logged in ({@link UserAccountManager}).
261 	 * </p>
262 	 * @param tweet Tweet to be posted.
263 	 * @return Tweet post with some additional data.
264 	 * @throws IOException If an I/O error occurs.
265 	 * @throws LimitExceededException If limit has been hit.
266 	 * @throws SecurityException If it is not properly logged in.
267 	 * @throws IllegalArgumentException If the given tweet is null/empty, etc.
268 	 */
269 	public Tweet post(Tweet tweet) throws IOException, LimitExceededException {
270 		if (tweet == null) {
271 			throw new IllegalArgumentException("Tweet must not be null.");
272 		}
273 		//
274 		tweet.validateContent();
275 		//
276 		checkUserAuth();
277 		//
278 		HttpConnection conn = getConn(TWITTER_URL_UPDATE_STATUS);
279 		//
280 		try {
281 			final String content = tweet.getString(MetadataSet.TWEET_CONTENT);
282 			//
283 			conn.setRequestMethod(HttpConnection.POST);
284 			//
285 			OutputStream out = conn.openOutputStream();
286 			out.write(("status=" + content).getBytes());
287 			out.flush();
288 			out.close();
289 			//
290 			HttpResponseCodeInterpreter.perform(conn);
291 			//
292 			Parser parser = ParserFactory.getDefaultParser();
293 			StatusHandler handler = new StatusHandler();
294 			parser.parse(conn.openInputStream(), handler);
295 			handler.loadParsedTweet(tweet);
296 			//
297 			return tweet;
298 		} catch (ParserException e) {
299 			throw new IOException(e.getMessage());
300 		} finally {
301 			if (conn != null) {
302 				conn.close();
303 			}
304 		}
305 	}

354 	/**
355 	 * <p>
356 	 * Send a given Direct Message to a given recipient. The DM's content must
357 	 * be up to 140 characters, addressed to a proper user (he/she must also
358 	 * follow you) and must be properly logged in ({@link UserAccountManager}).
359 	 * </p>
360 	 * @param dm Direct Message to be sent.
361 	 * @return Sent Direct Message.
362 	 * @throws IOException If an I/O error occurs.
363 	 * @throws LimitExceededException If limit has been hit.
364 	 * @throws InvalidQueryException If recipient does not exist or the sender
365 	 *         does not follow the recipient and vice versa.
366 	 * @throws SecurityException If it is not properly logged in.
367 	 * @throws IllegalArgumentException If the given DM is null/empty, etc.
368 	 */
369 	public Tweet send(Tweet dm) throws IOException, LimitExceededException {
370 		if (dm == null) {
371 			throw new IllegalArgumentException("DM must not be null.");
372 		}
373 		//
374 		dm.validateRecipient();
375 		dm.validateContent();
376 		//
377 		checkUserAuth();
378 		//
379 		HttpConnection conn = getConn(TWITTER_URL_SEND_DIRECT_MESSAGE);
380 		//
381 		try {
382 			final String content = dm.getString(MetadataSet.TWEET_CONTENT);
383 			String recipient = dm.getString(MetadataSet.USERACCOUNT_ID);
384 			//
385 			if (recipient == null) {
386 				recipient = dm.getString(MetadataSet.USERACCOUNT_USER_NAME);
387 				if (recipient == null) {
388 					recipient =	dm.getString(MetadataSet.TWEET_AUTHOR_USERNAME);
389 				}
390 			}
391 			//
392 			conn.setRequestMethod(HttpConnection.POST);
393 			//
394 			OutputStream out = conn.openOutputStream();
395 			out.write(("user=" + recipient).getBytes());
396 			out.write(("&text=" + content).getBytes());
397 			out.flush();
398 			out.close();
399 			//
400 			HttpResponseCodeInterpreter.perform(conn);
401 			//
402 			Parser parser = ParserFactory.getDefaultParser();
403 			DirectMessageHandler handler = new DirectMessageHandler();
404 			parser.parse(conn.openInputStream(), handler);
405 			handler.loadParsedTweet(dm, 0);
406 			//
407 			return dm;
408 		} catch (ParserException e) {
409 			throw new IOException(e.getMessage());
410 		} finally {
411 			if (conn != null) {
412 				conn.close();
413 			}
414 		}
415 	}
