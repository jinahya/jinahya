146 		Enumeration keys = userAccountMngrPoll.keys();
147 		Object key;
148 		UserAccountManager value;
149 		//

151 			key = keys.nextElement();

155 				userAccountMngrPoll.remove(key);

157 		}

170 		if (c == null) {
171 			throw new IllegalArgumentException("Credential must not be null.");
172 		}
173 		//
174 		UserAccountManager uam = null;
175 		//
176 		if (userAccountMngrPoll == null) {
177 			userAccountMngrPoll = new Hashtable();
178 		} else {
179 			synchronized (userAccountMngrPoll) {
180 				cleanPool();
181 				//
182 				uam = (UserAccountManager)userAccountMngrPoll.get(c);
183 			}
184 		}
185 		//
186 		return uam != null ? uam : new UserAccountManager(c);
187 	}

198 	static synchronized HttpConnection getHttpConn(String url, Credential c)
199 		throws IOException {
200 		if (url == null || (url = url.trim()).length() == 0) {
201 			throw new IllegalArgumentException("URL must not be empty/null.");
202 		}
203 		//
204 		HttpConnection conn = HttpConnector.open(url);
205 		boolean hasException = true;
206 		//
207 		try {
208 			if (c != null) {
209 				String crdntls = c.getBasicHttpAuthCredential();
210 				crdntls = HttpConnector.encodeBase64(crdntls);
211 				conn.setRequestProperty("Authorization", "Basic " + crdntls);
212 			}
213 			//
214 			hasException = false;
215 		} finally {
216 			if (hasException) {
217 				conn.close();
218 			}
219 		}
220 		//
221 		return conn;
222 	}

288 	 * <p>
289 	 * Verify whether the given user's credential are valid. This method
290 	 * authenticates the API to Twitter REST API.
291 	 * </p>
292 	 * @return Valid credentials (true).
293 	 * @throws IOException If an I/O error occurs.
294 	 */
295 	public boolean verifyCredential() throws IOException {
296 		if (verified) {
297 			return true; //already verified.
298 		}
299 		//
300 		HttpConnection conn =
301 			getHttpConn(TWITTER_URL_VERIFY_CREDENTIALS, credential);
302 		//
303 		try {
304 			final int respCode = conn.getResponseCode();
305 			//
306 			if (respCode == HttpConnection.HTTP_OK) {
307 				verified = true;
308 				//
309 				try {
310 					Parser parser =	ParserFactory.getDefaultParser();
311 					AccountHandler handler = new AccountHandler();
312 					parser.parse(conn.openInputStream(), handler);
313 					//
314 					account = handler.getParsedUserAccount();
315 				} catch (ParserException e) {
316 					throw new IOException(e.getMessage());
317 				}
318 				//
319 				saveSelfOnPool();
320 			} else if (respCode == HttpConnection.HTTP_UNAUTHORIZED) {
321 				verified = false;
322 			} else {
323 				try {
324 					HttpResponseCodeInterpreter.perform(conn);
325 				} catch (LimitExceededException e) {
326 					//Twitter API specs states this operation is not API rate
327 					//limited. That's why this exception is suppressed.
328 				}
329 			}
330 		} finally {
331 			if (conn != null) {
332 				conn.close();
333 			}
334 		}
335 		//
336 		return verified;
337 	}

410 	public boolean isFollowing(UserAccount ua) throws IOException,
411 		LimitExceededException {
412 		if (ua == null) {
413 			throw new IllegalArgumentException(
414 				"UserAccount object must not me null.");
415 		}
416 		String id = ua.getString(MetadataSet.USERACCOUNT_ID);
417 		if (id == null || (id = id.trim()).length() == 0) {
418 			id = ua.getString(MetadataSet.USERACCOUNT_USER_NAME);
419 			if (id == null || (id = id.trim()).length() == 0) {
420 				throw new IllegalArgumentException(
421 					"Username or ID must not be empty/null.");
422 			}
423 		}
424 		//
425 		checkVerified();
426 		//
427 		final String qryStr =
428 			"?user_a=" + account.getString(MetadataSet.USERACCOUNT_USER_NAME) +
429 			"&user_b=" + id;
430 		//
431 		HttpConnection conn =
432 			getHttpConn(TWITTER_URL_IS_FOLLOWING_USER + qryStr, credential);
433 		//
434 		try {
435 			HttpResponseCodeInterpreter.perform(conn);
436 			//
437 			InputStream dis = conn.openInputStream();
438 			byte[] bs = new byte[dis.available()];
439 			dis.read(bs);
440 			final String result = new String(bs).trim().toLowerCase();
441 			//
442 			return result.equals("true");

579 		//
580 		HttpConnection conn = getHttpConn(actionUrl, credential);
581 		//
