127 	public synchronized static Timeline getInstance(UserAccountManager uam) {
128 		if (uam == null) {
129 			throw new IllegalArgumentException(
130 				"UserAccountManager must not be null.");
131 		}
132 		//
133 		if (!uam.isVerified()) {
134 			throw new SecurityException("User's credential must be verified.");
135 		}
136 		//
137 		if (timelinePool == null) {
138 			timelinePool = new Hashtable();
139 		}
140 		//
141 		Timeline ter = (Timeline)timelinePool.get(uam);
142 		if (ter == null) {
143 			ter = new Timeline(uam);
144 			timelinePool.put(uam, ter);
145 		}
146 		//
147 		return ter;
148 	}

360 	/**
361 	 * <p>
362 	 * Start a retrieval on a given URL according to a filter.
363 	 * </p>
364 	 * @param url The URL.
365 	 * @param q The filter query.
366 	 * @param l Listener object to be notified about the search's result.
367 	 * @param h Content handler.
368 	 * @param checkAuth Authentication required (true).
369 	 */
370 	private void startGet(final String url, final Query q,
371 		final SearchDeviceListener l, final Handler h, final boolean checkAuth){
372 		if (url == null || url.trim().length() == 0) {
373 			throw new IllegalArgumentException("Url must not be empty/null.");
374 		}
375 		if (l == null) {
376 			throw new IllegalArgumentException("Listener must not be null.");
377 		}
378 		if (h == null) {
379 			throw new IllegalArgumentException("Handler must not be null.");
380 		}
381 		//
382 		if (checkAuth) {
383 			checkUserAuth();
384 		}
385 		//
386 		Runnable r = new Runnable() {
387 			public void run() {
388 				HttpConnection conn = null;
389 				//
390 				try {
391 					conn = getConn(q != null ? url + '?' + q.toString() : url);
392 					HttpResponseCodeInterpreter.perform(conn);
393 					//
394 					Parser parser = ParserFactory.getDefaultParser();
395 					parser.parse(conn.openInputStream(), h);
396 					//
397 					l.searchCompleted();
398 				} catch (Exception e) {
399 					l.searchFailed(e);
400 				} finally {
401 					if (conn != null) {
402 						try {
403 							conn.close();
404 						} catch (IOException e) {}
405 					}
406 				}
407 			}
408 		};
409 		//
410 		Thread t = new Thread(r);
411 		t.start();
412 	}
