8   package com.twitterapime.parser;
9   
10  import com.twitterapime.platform.PlatformProvider;
11  import com.twitterapime.platform.PlatformProviderSelector;
12  
13  /**
14   * <p>
15   * This factory class is responsible for creating new parser objects. 
16   * </p>
17   * <p>
18   * The creation of a parser object is performed dynamically by looking up the
19   * current platform provider from PlatformProviderSelector class. For each
20   * supported platform, there is a specific implementation class.
21   * </p>
22   * 
23   * @author Ernandes Mourao Junior (ernandes@gmail.com)
24   * @version 1.1
25   * @since 1.0
26   */
27  public final class ParserFactory {
28  	/**
29  	 * <p>
30  	 * Create the default parser instance according to the underlying platform.
31  	 * </p>
32  	 * @return Parser instance.
33  	 */
34  	public static Parser getDefaultParser() {
35  		final String JAVA_ME_PARSER_IMPL_CLASS =
36  			"impl.javame.com.twitterapime.parser.KXML2Parser";
37  		final String ANDROID_PARSER_IMPL_CLASS =
38  			"impl.android.com.twitterapime.parser.SAXParser";
39  		//
40  		final long PPID = PlatformProviderSelector.getCurrentProvider().getID();
41  		//
42  		//if JAVA ME PLATFORM
43  		if (PPID == PlatformProvider.PPID_JAVA_ME) {
44  			return newInstance(JAVA_ME_PARSER_IMPL_CLASS);
45  		} else if (PPID == PlatformProvider.PPID_ANDROID) {
46  			return newInstance(ANDROID_PARSER_IMPL_CLASS);
47  		} else {
48  			throw new IllegalArgumentException("Unknown platform ID: " + PPID);
49  		}
50  	}
51  	
52  	/**
53  	 * <p>
54  	 * Create an instance of the given class.
55  	 * </p>
56  	 * @param className Class name.
57  	 * @return Parser instance.
58  	 */
59  	private static Parser newInstance(String className) {
60  		try {
61  			return (Parser)Class.forName(className).newInstance();
62  		} catch (IllegalAccessException e) {
63  		} catch (InstantiationException e) {
64  		} catch (ClassNotFoundException e) {
65  		}
66  		//
67  		return null;
68  	}
69  	
70  	/**
71  	 * <p>
72  	 * Private constructor to avoid object instantiation.
73  	 * </p>
74  	 */
75  	private ParserFactory() {
76  	}
77  }
