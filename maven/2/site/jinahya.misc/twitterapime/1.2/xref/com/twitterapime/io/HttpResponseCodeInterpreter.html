8   package com.twitterapime.io;

10  import java.io.IOException;
11  
12  import com.twitterapime.io.handler.HttpResponseCodeErrorHandler;
13  import com.twitterapime.parser.Parser;
14  import com.twitterapime.parser.ParserException;
15  import com.twitterapime.parser.ParserFactory;
16  import com.twitterapime.search.InvalidQueryException;
17  import com.twitterapime.search.LimitExceededException;

19  /**
20   * <p>
21   * This class is responsible for interpreting the Http response-codes from
22   * Twitter API, in order to check the status of a request: success or failure.
23   * </p>
24   * 
25   * @author Ernandes Mourao Junior (ernandes@gmail.com)
26   * @version 1.1
27   * @since 1.1
28   * @see LimitExceededException
29   * @see InvalidQueryException
30   */
31  public final class HttpResponseCodeInterpreter {
32  	/**
33  	 * <p>
34  	 * 420: Custom Http code from Twitter API which means to wait a little bit
35  	 * for a new request.
36  	 * </p> 
37  	 */
38  	public static final int CUSTOM_HTTP_CODE_ENHANCE_YOUR_CALM = 420;
39  	
40  	/**
41  	 * <p>
42  	 * Perform an analyze of a given HttpConnection object's response-code in
43  	 * order to interpret whether the requests to Twitter API went well.
44  	 * Otherwise, an exception is thrown describing the problem.
45  	 * </p>
46  	 * @param conn HttpConnection object to be interpreted.
47  	 * @throws IOException If an I/O or service error occurs.
48  	 * @throws LimitExceededException If a request limit exceeded error occurs.
49  	 * @throws InvalidQueryException If an invalid query error occurs.
50  	 * @throws SecurityException If a security error occurs.
51  	 * @throws IllegalArgumentException If conn is null.
52  	 */
53  	public static void perform(HttpConnection conn) throws IOException,
54  		LimitExceededException {
55  		if (conn == null) {
56  			throw new IllegalArgumentException("Connection must not be null.");
57  		}
58  		//
59  		final int respCode = conn.getResponseCode();
60  		//
61  		if (respCode != HttpConnection.HTTP_OK
62  				&& respCode != HttpConnection.HTTP_NOT_MODIFIED) {
63  			if (isInvalidQueryError(respCode)) {
64  				throw new InvalidQueryException(getErrorMessage(conn));
65  			} else if (isLimitExceededError(respCode)) {
66  				throw new LimitExceededException(getErrorMessage(conn));
67  			} else if (isSecurityError(respCode)) {
68  				throw new SecurityException(getErrorMessage(conn));
69  			} else {
70  				throw new IOException(getErrorMessage(conn));
71  			}
72  		}
73  	}
74  	
75  	/**
76  	 * <p>
77  	 * Get error messages from connection's response.
78  	 * </p>
79  	 * @param conn Http connection.
80  	 * @return Message.
81  	 * @throws IOException If any I/O error occurs.
82  	 */
83  	public static String getErrorMessage(HttpConnection conn)
84  		throws IOException {
85  		String errorMsg = null;
86  		//
87  		Parser parser = ParserFactory.getDefaultParser();
88  		HttpResponseCodeErrorHandler handler =
89  			new HttpResponseCodeErrorHandler();
90  		//
91  		try {
92  			parser.parse(conn.openInputStream(), handler);
93  			//
94  			errorMsg = handler.getParsedErrorMessage();
95  		} catch (ParserException e) {
96  			errorMsg = "HTTP ERROR CODE: " + conn.getResponseCode();
97  		}
98  		//
99  		return errorMsg;
100 	}
101 
102 	/**
103 	 * <p>
104 	 * Check if the response-code reports a request limit exceeded error.
105 	 * </p>
106 	 * @param code The response-code.
107 	 * @return true if the response-code represents a limit exceeded error.
108 	 */
109 	static boolean isLimitExceededError(int code) {
110 		return code == HttpConnection.HTTP_BAD_REQUEST
111 				|| code == HttpConnection.HTTP_FORBIDDEN
112 				|| code == CUSTOM_HTTP_CODE_ENHANCE_YOUR_CALM;
113 	}
114 	
115 	/**
116 	 * <p>
117 	 * Check if the response-code reports an invalid query error.
118 	 * </p>
119 	 * @param code The response-code.
120 	 * @return true if the response-code represents an invalid query error.
121 	 */
122 	static boolean isInvalidQueryError(int code) {
123 		return code == HttpConnection.HTTP_NOT_FOUND
124 				|| code == HttpConnection.HTTP_NOT_ACCEPTABLE;
125 	}
126 	
127 	/**
128 	 * <p>
129 	 * Check if the response-code reports a service error.
130 	 * </p>
131 	 * @param code The response-code.
132 	 * @return true if the response-code represents a service error.
133 	 */
134 	static boolean isServiceError(int code) {
135 		return code == HttpConnection.HTTP_BAD_GATEWAY
136 				|| code == HttpConnection.HTTP_INTERNAL_ERROR
137 				|| code == HttpConnection.HTTP_UNAVAILABLE;
138 	}
139 	
140 	/**
141 	 * <p>
142 	 * Check if the response-code reports a security error, e.g., Authentication
143 	 * credentials were missing or incorrect.
144 	 * </p>
145 	 * @param code The response-code.
146 	 * @return true if the response-code represents a security error.
147 	 */
148 	static boolean isSecurityError(int code) {
149 		return code == HttpConnection.HTTP_UNAUTHORIZED;
150 	}
151 	
152 	/**
153 	 * <p>
154 	 * Private constructor to avoid object instantiation.
155 	 * </p>
156 	 */
157 	private HttpResponseCodeInterpreter() {
158 	}
