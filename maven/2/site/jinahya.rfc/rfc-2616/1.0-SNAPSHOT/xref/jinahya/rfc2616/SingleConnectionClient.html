126         /**
127          *
128          * @param content
129          * @throws IOException
130          */
131         public void setContent(final InputStream content) throws IOException {
132 
133             if (content == null) {
134                 throw new IllegalArgumentException("content is null");
135             }
136 
137             final ByteArrayOutputStream baos = new ByteArrayOutputStream();
138             try {
139                 final byte[] buffer = new byte[1024];
140                 for (int read = -1; (read = content.read(buffer)) != -1;) {
141                     baos.write(buffer, 0, read);
142                 }
143                 baos.flush();
144                 this.content = baos.toByteArray();
145             } finally {
146                 baos.close();
147             }
148         }
149

152         public String toString() {
153             return new String(content);
154         }

161     /**
162      *
163      * @param address
164      * @param request
165      * @param response
166      * @throws IOException
167      */
168     public void request(final SocketAddress address,
169                         final RequestMessage request,
170                         final ResponseMessage response)
171         throws IOException {
172 
173         final MessageHeaders requestMessageHeaders =
174             request.getMessageHeaders();
175 
176         /*
177         if (!requestMessageHeaders.containsField("Host") &&
178             address instanceof InetSocketAddress) {
179 
180             final InetSocketAddress addr = (InetSocketAddress) address;
181             requestMessageHeaders.getFieldValues("Host").
182                 add(addr.getHostName() + ":" + addr.getPort());
183         }
184          */
185 
186         final Set connectionFieldValues =
187             requestMessageHeaders.getFieldValues("Connection");
188         if (connectionFieldValues.isEmpty()) {
189             connectionFieldValues.add("close");
190         }
191 
192 
193         final Socket socket = socketConfigurator.configure(new Socket());
194 
195         try {
196             socket.connect(address, connectTimeout);
197 
198             // ------------------------------------------------------------ SEND
199             final OutputStream output = socket.getOutputStream();
200             request.write(output);
201             output.flush();
202 
203             try {
204                 socket.shutdownOutput();
205             } catch (IOException ioe) {
206                 ioe.printStackTrace();
207             }
208 
209             // --------------------------------------------------------- RECEIVE
210             final InputStream input = socket.getInputStream();
211             response.read(input);
212 
213             try {
214                 socket.shutdownInput();
215             } catch (IOException ioe) {
216                 ioe.printStackTrace();
217             }
218 
219         } finally {
220             try {
221                 socket.close();
222             } catch (IOException ioe) {
223                 ioe.printStackTrace();
224             }
225         }
226     }
