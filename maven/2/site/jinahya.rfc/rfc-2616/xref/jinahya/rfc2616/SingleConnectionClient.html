95          /**
96           *
97           * @param content

99          public void setContent(final byte[] content) {
100             if (content == null) {
101                 throw new NullPointerException("content");
102             }
103 
104             this.content.reset();
105             try {
106                 this.content.write(content);
107             } catch (IOException ioe) {
108                 // no gonna happen
109             }
110         }

113         public void setContent(final InputStream content) throws IOException {
114             this.content.reset();
115             final byte[] buffer = new byte[1024];
116             int read = -1;
117             while ((read = content.read(buffer)) != -1) {
118                 this.content.write(buffer, 0, read);
119             }
120         }

124             content.reset();

134     /**
135      *
136      * @param address
137      * @param request
138      * @param response
139      * @throws IOException
140      */
141     public void request(final SocketAddress address,
142                         final RequestMessage request,
143                         final ResponseMessage response)
144         throws IOException {
145 
146         final MessageHeaders requestMessageHeaders =
147             request.getMessageHeaders();
148 
149         if (!requestMessageHeaders.containsField("Host") &&
150             address instanceof InetSocketAddress) {
151 
152             final InetSocketAddress addr = (InetSocketAddress) address;
153             requestMessageHeaders.getFieldValues("Host").
154                 add(addr.getHostName() + ":" + addr.getPort());
155         }
156 
157         final Set connectionFieldValues =
158             requestMessageHeaders.getFieldValues("Connection");
159         if (connectionFieldValues.isEmpty()) {
160             connectionFieldValues.add("close");
161         }
162 
163         Socket socket = new Socket();
164         if (handler != null) {
165             socket = handler.handleOptions(socket);
166         }
167         try {
168             socket.connect(address, timeout);
169 
170             if (request.getMessageBody() == null) {
171                 request.setMessageBody(new BufferedMessageBody());
172             }
173             final OutputStream output = socket.getOutputStream();
174             request.write(output);
175             output.flush();
176             socket.shutdownOutput();
177 
178             if (response.getMessageBody() == null) {
179                 response.setMessageBody(new BufferedMessageBody());
180             }
181             final InputStream input = socket.getInputStream();
182             response.read(input);
183             socket.shutdownInput();
184 
185         } finally {
186             socket.close();
187         }
188     }
