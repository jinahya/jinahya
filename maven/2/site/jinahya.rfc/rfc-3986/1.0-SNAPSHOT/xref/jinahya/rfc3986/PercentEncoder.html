33  public class PercentEncoder {
34  
35  
36      /**
37       *
38       * @param s
39       * @return
40       * @throws IOException
41       */
42      public static String encode(final String s) throws IOException {
43  
44          if (s == null) {
45              throw new IllegalArgumentException(
46                  "param:0:" + String.class + ": is null");
47          }
48  
49          final StringWriter out = new StringWriter();
50          try {
51              encode(new ByteArrayInputStream(s.getBytes("UTF-8")), out);
52              out.flush();
53              return out.toString();
54          } finally {
55              out.close();
56          }
57      }
58  
59  
60      /**
61       *
62       * @param in
63       * @param out
64       * @throws IOException
65       */
66      public static void encode(final InputStream in, final OutputStream out)
67          throws IOException {
68  
69          if (in == null) {
70              throw new IllegalArgumentException(
71                  "param:0:" + InputStream.class + ": is null");
72          }
73  
74          if (out == null) {
75              throw new IllegalArgumentException(
76                  "param:1:" + OutputStream.class + ": is null");
77          }
78  
79          final Writer writer = new OutputStreamWriter(out, "UTF-8");
80          try {
81              encode(in, writer);
82              writer.flush();
83          } finally {
84              //writer.close();
85          }
86      }

88  
89      /**
90       * 
91       * @param in
92       * @param out
93       * @throws IOException
94       */
95      public static void encode(final InputStream in, final Writer out)
96          throws IOException {
97  
98          if (in == null) {
99              throw new IllegalArgumentException(
100                 "param:0:" + InputStream.class + ": is null");
101         }
102 
103         if (out == null) {
104             throw new IllegalArgumentException(
105                 "param:1:" + Writer.class + ": is null");
106         }
107 
108         for (int b = -1; (b = in.read()) != -1; ) {
109 
110             if (b >= 0x30 && b <= 0x39) { // digit
111                 out.write(b);
112                 continue;
113             }
114 
115             if (b >= 0x41 && b <= 0x5A) { // upper case alpha
116                 out.write(b);
117                 continue;
118             }
119 
120             if (b >= 0x61 && b <= 0x7A) { // lower case alpha
121                 out.write(b);
122                 continue;
123             }
124 
125             if (b == 0x2D || b == 0x5F || b == 0x2E || b == 0x7E) { // - _ . ~
126                 out.write(b);
127                 continue;
128             }
129 
130             out.write(0x25);     // '%'
131             if (b < 0x10) {
132                 out.write(0x30); // '0'
133             }
134             out.write(Integer.toHexString(b).toUpperCase());
135         }
136     }
