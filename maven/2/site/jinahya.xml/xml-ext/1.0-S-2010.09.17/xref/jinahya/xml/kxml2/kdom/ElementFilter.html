17  package jinahya.xml.kxml2.kdom;
18  
19  
20  import org.kxml2.kdom.Document;
21  import org.kxml2.kdom.Element;
22  import org.kxml2.kdom.Node;
23  
24  
25  /**
26   *
27   * @author <a href="mailto:jinahya@gmail.com">Jin Kwon</a>
28   */
29  public abstract class ElementFilter {
30  
31  
32      /**
33       *
34       * @param document
35       */
36      public void filter(final Document document) {
37  
38          if (document == null) {
39              throw new IllegalArgumentException(
40                  "param:0:" + Document.class + ": is null");
41          }
42  
43          filter(document.getRootElement());
44      }
45  
46  
47      /**
48       * Pass given <code>element</code> through given <code>filter</code>.
49       *
50       * @param element element to be filtered
51       */
52      public void filter(final Element element) {
53  
54          if (element == null) {
55              throw new IllegalArgumentException(
56                  "param:0:" + Element.class + ": is null");
57          }
58  
59          if (startFiltering(element)) {
60              final int childCount = element.getChildCount();
61              for (int i = childCount - 1; i >= 0; i--) {
62                  if (Node.ELEMENT == element.getType(i)) {
63                      filter(element.getElement(i));
64                  }
65              }
66          }
67  
68          finishFiltering(element);
69      }
70  
71  
72      /**
73       * Starts filtering for given <code>element</code>.
74       *
75       * @param element element to be filtered.
76       * @return true if child elements have to be filtered. false otherwise.
77       */
78      protected abstract boolean startFiltering(Element element);
79  
80  
81      /**
82       * Finishes filtering for given <code>element</code>.
83       *
84       * @param element the element to be filtered.
85       */
86      protected abstract void finishFiltering(Element element);
87  }
