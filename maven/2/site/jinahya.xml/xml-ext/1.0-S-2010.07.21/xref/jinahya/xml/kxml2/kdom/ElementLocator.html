52      public static ElementLocator newInstance(final File file)
53          throws XmlPullParserException, IOException {
54  
55          if (file == null) {
56              throw new IllegalArgumentException(
57                  "param:0:" + File.class + ": is null");
58          }
59  
60          final InputStream in = new FileInputStream(file);
61          try {
62              return newInstance(in, null);
63          } finally {
64              in.close();
65          }
66      }

86      /**
87       *
88       * @param in
89       * @param enc
90       * @return
91       * @throws XmlPullParserException
92       * @throws IOException
93       */
94      public static ElementLocator newInstance(final InputStream in,
95                                               final String enc)
96          throws XmlPullParserException, IOException {
97  
98          if (in == null) {
99              throw new IllegalArgumentException(
100                 "param:0:" + InputStream.class + ": is null");
101         }
102 
103         if (enc != null) {
104             return newInstance(new InputStreamReader(in, enc));
105         }
106 
107         final XmlPullParser parser = new KXmlParser();
108         parser.setInput(in, enc);
109 
110         final Document document = new Document();
111         document.parse(parser);
112 
113         return new ElementLocator(document);
114     }
