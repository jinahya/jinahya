43      public static LinkedElement newInstance(final Document document) {
44  
45          if (document == null) {
46              throw new IllegalArgumentException(
47                  "param:0:" + Document.class + ": is null");
48          }
49  
50          final int childCount = document.getChildCount();
51          for (int i = 0; i < childCount; i++) {
52              if (Node.ELEMENT == document.getType(i)) {
53                  return new LinkedElement(null, document.getElement(i), i);
54              }
55          }
56  
57          throw new IllegalArgumentException(
58              "param:0:" + Document.class + ":" + document +
59              " has no child elements");
60      }

92       *
93       * @param attributes
94       * @return given <code>attributes</code>
95       */
96      public Hashtable<String, Vector<String>> attributes(
97          final Hashtable<String, Vector<String>> attributes) {
98  
99          final int attributeCount = element.getAttributeCount();
100         for (int i = 0; i < attributeCount; i++) {
101 
102             final String namespace = element.getAttributeNamespace(i);
103             final String name = element.getAttributeName(i);
104 
105             Vector<String> names = attributes.get(namespace);
106             if (names == null) {
107                 names = new Vector<String>();
108                 attributes.put(namespace, names);
109             }
110             names.addElement(name);
111         }
112 
113         return attributes;
114     }

182     Vector<LinkedElement> children() {
183 
184         final Vector<LinkedElement> children = new Vector<LinkedElement>();
185 
186         final int childCount = element.getChildCount();
187         for (int i = 0; i < childCount; i++) {
188             if (Node.ELEMENT != element.getType(i)) {

199     /**
200      *
201      * @param namespace
202      * @param name
203      * @return
204      */
205     private Vector<LinkedElement> children(final String namespace,
206                                            final String name) {
207 
208         if (namespace == null) {
209             throw new IllegalArgumentException(
210                 "param:0:namespace:" + String.class + ": is null");
211         }
212 
213         if (name == null) {
214             throw new IllegalArgumentException(
215                 "param:1:name:" + String.class + ": is null");
216         }
217 
218         final String key = "{" + namespace + "}" + name;
219 
220         Vector<LinkedElement> children = classes.get(key);
221 
222         if (children == null) {
223             children = new Vector<LinkedElement>();
224 
225             final int childCount = element.getChildCount();
226             for (int i = 0; i < childCount; i++) {
227                 if (Node.ELEMENT != element.getType(i)) {
228                     continue;
229                 }
230                 final Element child = element.getElement(i);
231                 if (!(namespace.equals(child.getNamespace()))
232                     || !(name.equals(child.getName()))) {
233                     continue;
234                 }
235 
236                 children.addElement(new LinkedElement(this, child, i));
237             }
238 
239             classes.put(key, children);
240         }
241 
242         return children;
243     }

328     /**
329      * 
330      * @param index
331      */
332     private void remove(final int index) {
333 
334         element.removeChild(index);
335 
336         final Enumeration<Vector<LinkedElement>> elements =
337             classes.elements();
338 
339         while (elements.hasMoreElements()) {
340 
341             final Vector<LinkedElement> children = elements.nextElement();
342 
343             for (int i = children.size() - 1; i >= 0; i--) {
344                 if (children.elementAt(i).index >= index) {
345                     if (children.elementAt(i).index == index) {
346                         children.removeElementAt(i);
347                     } else {
348                         children.elementAt(i).index--;
349                     }
350                 }
351             }
352         }
353     }

384     /**
385      *
386      * @param buffer a buffer to which text values are appended.
387      * @return true if any text value appended to given <code>buffer</code>,
388      *         false otherwise.
389      */
390     public boolean text(final StringBuffer buffer) {
391 
392         if (buffer == null) {
393             throw new IllegalArgumentException(
394                 "param:0:" + StringBuffer.class + ": is null");
395         }
396 
397         boolean appended = false;
398         final int childCount = element.getChildCount();
399         for (int i = 0; i < childCount; i++) {
400             if (element.isText(i)
401                 && (Node.IGNORABLE_WHITESPACE != element.getType(i))) {
402 
403                 buffer.append(element.getText(i));
404                 appended = true;
405             }
406         }
407 
408         return appended;
409     }

412     /**
413      *
414      * @param text
415      */
416     public void text(final String text) {
417 
418         final int childCount = element.getChildCount();
419         for (int i = childCount - 1; i >= 0; i--) {
420             if (Node.TEXT == element.getType(i)
421                 || Node.IGNORABLE_WHITESPACE == element.getType(i)) {
422 
423                 remove(i);
424             }
425         }
426 
427         if (text != null) {
428             element.addChild(Node.TEXT, text);
429         }
430     }
