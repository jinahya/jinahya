208     /**
209      *
210      * @param expression
211      * @param returnType
212      * @return
213      * @throws XPathExpressionException
214      */
215     public Object evaluate(final String expression, final QName returnType)
216         throws XPathExpressionException {
217 
218         if (expression == null) {
219             throw new IllegalArgumentException(
220                 "param:0:" + String.class + ": is null");
221         }
222 
223         if (returnType == null) {
224             throw new IllegalArgumentException(
225                 "param:1:" + QName.class + ": is null");
226         }
227 
228         if (cacheExpressions) {
229             synchronized (compiledExpressions) {
230                 XPathExpression compiled = compiledExpressions.get(expression);
231                 if (compiled == null) {
232                     compiled = path.compile(expression);
233                     compiledExpressions.put(expression, compiled);
234                 }
235                 return compiled.evaluate(node, returnType);
236             }
237         }
238 
239         return path.evaluate(expression, node, returnType);
240     }

256     /**
257      *
258      * @param node
259      * @return new child node path
260      * @throws IllegalArguementException if given <code>node</code> is null
261      *         or not a descendent of current node.
262      */
263     public NodePath child(final Node node) {
264 
265         if (node == null) {
266             throw new IllegalArgumentException(
267                 "param:0:" + Node.class + ": is null");
268         }
269 
270         for (Node parent = node.getParentNode(); parent != null;
271              parent = parent.getParentNode()) {
272 
273             if (parent.equals(this.node)) {
274                 return new NodePath(node, path);
275             }
276         }
277 
278         /*
279          * Billie Jean is not my lover
280          * She's just a girl who claims that I am the one
281          * But the kid is not my son
282          * She says I am the one, but the kid is not my son
283          * (Don't think twice)
284          */
285 
286         throw new IllegalArgumentException(
287             "param:0:" + Node.class + ":" + node
288             + " is not a descendent of current node");
289     }
