49          Vector<LinkedElement> children(final String namespace,
50                                         final String name) {
51  
52              if (namespace == null) {
53                  throw new IllegalArgumentException(
54                      "param:0:namespace:" + String.class + ": is null");
55              }
56  
57              if (name == null) {
58                  throw new IllegalArgumentException(
59                      "param:1:name:" + String.class + ": is null");
60              }
61  
62              final String key = "{" + namespace + "}" + name;
63  
64              Vector<LinkedElement> children = classes().get(key);
65  
66              if (children == null) {
67  
68                  children = new Vector<LinkedElement>();
69                  classes().put(key, children);
70  
71                  final int childCount = element.getChildCount();
72                  for (int i = 0; i < childCount; i++) {
73                      if (Node.ELEMENT != element.getType(i)) {
74                          continue;
75                      }
76                      final Element childElement = element.getElement(i);
77                      if (namespace.equals(childElement.getNamespace())
78                          && name.equals(childElement.getName())) {
79                          final LinkedElement child = new LinkedElement();
80                          child.parent = this;
81                          child.element = childElement;
82                          child.index = i;
83                          children.addElement(child);
84                      }
85                  }
86              }
87  
88              return children;
89          }
90  
91  
92  
93          LinkedElement remove(final int index) {
94  
95              element.removeChild(index);
96  
97              final Enumeration<Vector<LinkedElement>> e = classes().elements();
98  
99              while (e.hasMoreElements()) {
100 
101                 final Vector<LinkedElement> children = e.nextElement();
102 
103                 for (int i = children.size() - 1; i >= 0; i--) {
104                     if (children.elementAt(i).index < index) {
105                         continue;
106                     }
107                     if (children.elementAt(i).index == index) {
108                         children.removeElementAt(i);
109                     } else {
110                         children.elementAt(i).index--;
111                     }
112                 }
113             }
114 
115             return this;
116         }
117

119         Hashtable<String, Vector<LinkedElement>> classes() {
120             if (classes == null) {
121                 classes = new Hashtable<String, Vector<LinkedElement>>();
122             }
123 
124             return classes;
125         }

142     public ElementLocator(final Document document) {
143 
144         super();
145 
146         if (document == null) {
147             throw new IllegalArgumentException(
148                 "param:0:" + Document.class + ": is null");
149         }
150 
151         final int childCount = document.getChildCount();
152         for (int i = 0; i < childCount; i++) {
153             if (Node.ELEMENT == document.getType(i)) {
154                 current = new LinkedElement();
155                 current.parent = null;
156                 current.element = document.getElement(i);
157                 current.index = i;
158                 break;
159             }
160         }
161 
162         if (current == null) {
163             throw new IllegalArgumentException("no root element");
164         }
165     }

564     /**
565      *
566      * @param element
567      * @param buffer
568      * @return given <code>buffer</code>
569      */
570     StringBuffer print(final Element element, final StringBuffer buffer) {
571 
572         buffer.append(("<" + element.getName()));
573         final int attributeCount = element.getAttributeCount();
574         for (int i = 0; i < attributeCount; i++) {
575             buffer.append((" " + element.getAttributeName(i) + "=\""
576                            + element.getAttributeValue(i) + "\""));
577         }
578 
579         final Vector<Element> children = new Vector<Element>();
580         final int childCount = element.getChildCount();
581         for (int i = 0; i < childCount; i++) {
582             if (Node.ELEMENT == element.getType(i)) {
583                 children.addElement(element.getElement(i));
584             }
585         }
586 
587         if (children.isEmpty()) {
588             final int offset = buffer.length();
589             boolean appended = false;
590             for (int i = 0; i < childCount; i++) {
591                 if (element.getType(i) == Node.TEXT) {
592                     buffer.append(element.getText(i));
593                     appended = true;
594                 }
595             }
596             if (appended) {
597                 buffer.insert(offset, '>');
598                 buffer.append(("</" + element.getName() + ">"));
599             } else {
600                 buffer.append("/>");
601             }
602         } else {
603             buffer.append('>');
604             for (int i = 0; i < children.size(); i++) {
605                 print(children.elementAt(i), buffer);
606             }
607             buffer.append(("</" + element.getName() + ">"));
608         }
609 
610         return buffer;
611     }
