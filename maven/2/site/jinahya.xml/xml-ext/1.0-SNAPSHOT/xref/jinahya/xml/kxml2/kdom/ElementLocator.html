44          String key(final Element element) {

49          String key(final String namespace, final String name) {
50              return "{" + namespace + "}" + name;

55              return children(element.getNamespace(), element.getName());

58  
59          Vector<LinkedElement> children(final String namespace,
60                                         final String name) {
61  
62              if (namespace == null) {
63                  throw new IllegalArgumentException(
64                      "param:0:namespace:" + String.class + ": is null");
65              }
66  
67              if (name == null) {
68                  throw new IllegalArgumentException(
69                      "param:1:name:" + String.class + ": is null");
70              }
71  
72              final String key = key(namespace, name);
73  
74              Vector<LinkedElement> children = classes().get(key);
75  
76              if (children == null) {
77  
78                  children = new Vector<LinkedElement>();
79                  classes().put(key, children);
80  
81                  final int childCount = element.getChildCount();
82                  for (int i = 0; i < childCount; i++) {
83                      if (Node.ELEMENT != element.getType(i)) {
84                          continue;
85                      }
86                      final Element childElement = element.getElement(i);
87                      if (namespace.equals(childElement.getNamespace())
88                          && name.equals(childElement.getName())) {
89                          final LinkedElement child = new LinkedElement();
90                          child.parent = this;
91                          child.element = childElement;
92                          child.index = i;
93                          children.addElement(child);
94                      }
95                  }
96              }
97  
98              return children;
99          }

102 
103         LinkedElement remove(final int index) {
104 
105             String key = null;
106             if (element.getType(index) == Node.ELEMENT) {
107                 key = key(element.getElement(index));
108             }
109 
110             element.removeChild(index);
111 
112             final Enumeration<Vector<LinkedElement>> e = classes().elements();
113 
114             while (e.hasMoreElements()) {
115 
116                 final Vector<LinkedElement> children = e.nextElement();
117 
118                 for (int i = children.size() - 1; i >= 0; i--) {
119                     if (children.elementAt(i).index < index) {
120                         continue;
121                     }
122                     if (children.elementAt(i).index == index) {
123                         children.removeElementAt(i);
124                     } else {
125                         children.elementAt(i).index--;
126                     }
127                 }
128             }
129 
130             if (key != null) {
131                 Vector<LinkedElement> children = classes.get(key);
132                 if (children.isEmpty()) {
133                     classes.remove(key);
134                 }
135             }
136 
137             return this;
138         }
139 
140 
141         Hashtable<String, Vector<LinkedElement>> classes() {
142             if (classes == null) {
143                 classes = new Hashtable<String, Vector<LinkedElement>>();
144             }
145 
146             return classes;
147         }

160      * thrown if specified <code>document</code> is null.

163      */
164     public ElementLocator(final Document document) {
165 
166         super();
167 
168         if (document == null) {
169             throw new IllegalArgumentException(
170                 "param:0:" + Document.class + ": is null");
171         }
172 
173         final int childCount = document.getChildCount();
174         for (int i = 0; i < childCount; i++) {
175             if (Node.ELEMENT == document.getType(i)) {
176                 current = new LinkedElement();
177                 current.parent = null;
178                 current.element = document.getElement(i);
179                 current.index = i;
180                 break;
181             }
182         }
183 
184         if (current == null) {
185             throw new IllegalArgumentException("no root element");
186         }
187     }

233     public ElementLocator childNS(final String namespace, final String name) {
234

236             throw new IllegalArgumentException(

238         }
239 
240         if (name == null) {
241             throw new IllegalArgumentException(
242                 "param:1:" + String.class + ": is null");
243         }
244 
245         final Vector<LinkedElement> children =
246             current.children(namespace, name);
247 
248         final int childCount = current.element.getChildCount();
249 
250         current.element.addChild(
251             Node.ELEMENT, current.element.createElement(namespace, name));
252 
253         final LinkedElement child = new LinkedElement();
254         child.parent = current;
255         child.element = current.element.getElement(childCount);
256         child.index = childCount;
257 
258         children.addElement(child);
259 
260         current = child;
261 
262         return this;
263     }

574         }

586     /**
587      *
588      * @param element
589      * @param buffer
590      * @return given <code>buffer</code>
591      */
592     StringBuffer print(final Element element, final StringBuffer buffer) {
593 
594         buffer.append(("<" + element.getName()));
595         final int attributeCount = element.getAttributeCount();
596         for (int i = 0; i < attributeCount; i++) {
597             buffer.append((" " + element.getAttributeName(i) + "=\""
598                            + element.getAttributeValue(i) + "\""));
599         }
600 
601         final Vector<Element> children = new Vector<Element>();
602         final int childCount = element.getChildCount();
603         for (int i = 0; i < childCount; i++) {
604             if (Node.ELEMENT == element.getType(i)) {
605                 children.addElement(element.getElement(i));
606             }
607         }
608 
609         if (children.isEmpty()) {
610             final int offset = buffer.length();
611             boolean appended = false;
612             for (int i = 0; i < childCount; i++) {
613                 if (element.getType(i) == Node.TEXT) {
614                     buffer.append(element.getText(i));
615                     appended = true;
616                 }
617             }
618             if (appended) {
619                 buffer.insert(offset, '>');
620                 buffer.append(("</" + element.getName() + ">"));
621             } else {
622                 buffer.append("/>");
623             }
624         } else {
625             buffer.append('>');
626             for (int i = 0; i < children.size(); i++) {
627                 print(children.elementAt(i), buffer);
628             }
629             buffer.append(("</" + element.getName() + ">"));
630         }
631 
632         return buffer;
633     }
