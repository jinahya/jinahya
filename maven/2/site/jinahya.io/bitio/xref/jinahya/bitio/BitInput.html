105     /**
106      * Reads a sequence of bytes.
107      *
108      * @param value value to be read
109      * @param offset offset to start
110      * @param length length to be read
111      * @throws IOException if an I/O error occurs.
112      */
113     public void readBytes(byte[] value, int offset, int length)
114         throws IOException {
115 
116         if (offset < 0 || offset >= value.length) {
117             throw new IllegalArgumentException("offset: " + offset);
118         }
119 
120         if (length < 0 || (offset + length) > value.length) {
121             throw new IllegalArgumentException("length: " + length);
122         }
123 
124         for (int i = 0; i < length; i++) {
125             value[offset + i] = (byte) readUnsignedShort(8);
126         }
127     }

130     private int readUnsignedByte(int length) throws IOException {
131 
132         if (length < 1 || length >= 8) {
133             throw new IllegalArgumentException("illegal length: " + length);
134         }
135 
136         if (avail == 0x00) {
137             octet = input.readOctet();
138             if (octet == -1) {
139                 throw new EOFException(":)");
140             }
141             avail = 0x08;
142         }
143 
144         if (avail >= length) {
145             int value = octet >> (avail - length);
146             avail -= length;
147             count += length;
148             octet &= (0xFF >> (8 - avail));
149             return value;
150         } else {
151             int requi = length - avail;
152             return ((readUnsignedByte(avail) << requi) |
153                     readUnsignedByte(requi));
154         }
155     }

158     private int readUnsignedShort(int length) throws IOException {
159 
160         if (length < 1 || length >= 16) {
161             throw new IllegalArgumentException("illegal length: " + length);
162         }
163 
164         int value = 0x00;
165 
166         int quotient = length / 0x07;
167         for (int i = 0; i < quotient; i++) {
168             value <<= 0x07;
169             value |= readUnsignedByte(0x07);
170         }
171 
172         int remainder = length % 7;
173         if (remainder > 0x00) {
174             value <<= remainder;
175             value |= readUnsignedByte(remainder);
176         }
177 
178         return value;
179     }

186      * @return an unsigned integer

189     public int readUnsignedInt(int length) throws IOException {
190 
191         if (length < 0x01 || length >= 0x20) {
192             throw new IllegalArgumentException("illegal length: " + length);
193         }
194 
195         int value = 0x00;
196 
197         int quotient = length / 0x0F;
198         for (int i = 0; i < quotient; i++) {
199             value <<= 15;
200             value |= readUnsignedShort(0x0F);
201         }
202 
203         int remainder = length % 0x0F;
204         if (remainder > 0) {
205             value <<= remainder;
206             value |= readUnsignedShort(remainder);
207         }
208 
209         return value;
210     }

221 
222         if (length <= 0x01 || length > 0x20) {
223             throw new IllegalArgumentException("length: " + length);
224         }
225 
226         int value = (0x00 - readUnsignedByte(0x01)) << (length - 1);
227 
228         value |= readUnsignedInt(length -1);
229 
230         return value;

383      * Reads some (if required) dummy bits for alignment.
384      *
385      * @param length number of bits to be aligned
386      * @throws IOException if an I/O error occurs
387      */
388     public final void alignOctets(final int length) throws IOException {
389 
390         if (length <= 0x00) {
391             throw new IllegalArgumentException("illegal length: " + length);
392         }
393 
394         int required = (int) (length - (count % length));
395 
396         int quotient = required / 7;
397         for (int i = 0; i < quotient; i++) {
398             readUnsignedByte(7);
399         }
400 
401         int remainder = required % 7;
402         if (remainder > 0) {
403             readUnsignedByte(remainder);
404         }
405     }

426     /**
427      * Reads a sequence of characters in modified UTF-8 encoding.
428      *
429      * @param writer character source
430      * @throws IOException if an I/O error occurs.
431      */
432     public void readUTF(Writer writer) throws IOException {
433 
434         byte[] bytes = new byte[readUnsignedInt(16)];
435         readBytes(bytes);
436 
437         ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
438         try {
439             for (int a = -1; (a = bais.read()) != -1;) {
440                 if ((a >> 4) == 0x0F || // 1111xxxx
441                     (a >> 6) == 0x02) { // 10xxxxxx
442                     throw new UTFDataFormatException("a: " + a);
443                 }
444                 if ((a >> 7) == 0x00) { // 0xxxxxxxx
445                     writer.append((char) a);
446                 } else if ((a >> 5) == 0x06) { // 110xxxxx
447                     int b = bais.read();
448                     if (b == -1 || (b >> 6) != 0x02) { // NOT(10xxxxxx)
449                         throw new UTFDataFormatException("2b: " + b);
450                     }
451                     writer.append((char) (((a & 0x1F) << 6) | (b & 0x3F)));
452                 } else if (a >> 4 == 0x0E) { // 1110xxxx
453                     int b = bais.read();
454                     if (b == -1 || (b >> 6) != 0x02) { // NOT(10xxxxxx)
455                         throw new UTFDataFormatException("3b: " + b);
456                     }
457                     int c = bais.read();
458                     if (c == -1 || (c >> 6) != 0x02) { // NOT(10xxxxxx)
459                         throw new UTFDataFormatException("3c: " + b);
460                     }
461                     writer.write((char) (((a & 0x0F) << 12) |
462                                          ((b & 0x3F) << 6) |
463                                           (c & 0x3F)));
464                 }
465             }
466         } finally {
467             bais.close();
468         }
469     }

473      *
474      * @param m
475      * @param n
476      * @return an unsigned fixed point number denoted Q(m, n)
477      * @throws IOException if an I/O error occurs.
478      */
479     public float readUnsignedFixedPoint(int m, int n) throws IOException {
480 
481         if (m < 0) {
482             throw new IllegalArgumentException("m(" + m + ") < 0");
483         }
484 
485         if (n < 0) {
486             throw new IllegalArgumentException("n(" + n + ") < 0");
487         }
488 
489         if (m == 0 && n == 0) {
490             throw new IllegalArgumentException("m == 0 && n == 0");
491         }
492 
493         if (m + n > 0x20) {
494             throw new IllegalArgumentException
495                 ("m(" + m + ") + n(" + n + ") > 0x20");
496         }
497 
498         return readUnsignedInt(m + n) / new Double(Math.pow(2, n)).floatValue();
499     }
