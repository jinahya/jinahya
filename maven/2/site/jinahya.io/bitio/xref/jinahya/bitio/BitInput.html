103     /**
104      * See comemnts from {@link java.io.DataInput#readFully(byte[], int, int)}.
105      *
106      * @param value byte array to which octets are stored
107      * @param offset the offset to start
108      * @param length the number of octets to read.
109      * @throws IOException if an I/O error occurs.
110      * @see java.io.DataInput#readFully(byte[], int, int)
111      */
112     public final void readBytes(final byte[] value, final int offset,
113                                 final int length)
114         throws IOException {
115 
116         if (value == null) {
117             throw new NullPointerException("value");
118         }
119 
120         if (offset < 0) {
121             throw new IndexOutOfBoundsException("offset(" + offset + ") < 0");
122         }
123 
124         if (length < 0) {
125             throw new IndexOutOfBoundsException("length(" + length + ") < 0");
126         }
127 
128         if ((offset + length) > value.length) {
129             throw new IndexOutOfBoundsException(
130                 "offset(" + offset + ") + length(" + length
131                 + ") > value.length(" + value.length + ")");
132         }
133 
134         for (int i = 0; i < length; i++) {
135             value[offset + i] = (byte) readUnsignedByte(8);
136         }
137     }

140     /**
141      * Reads an unsigned <code>length</code> bit byte.
142      *
143      * @param length bit length between 0 (exclusive) and 8 (inclusive).
144      * @return an unsigned <code>length</code> bit integer.
145      * @throws IOException if an I/O error occurs.
146      */
147     private final int readUnsignedByte(final int length) throws IOException {
148 
149         //System.out.println("readUnsignedByte(" + length + ")");
150 
151         if (length <= 0) {
152             throw new IllegalArgumentException("length(" + length + ") <= 0");
153         }
154 
155         if (length > 8) {
156             throw new IllegalArgumentException("length(" + length + " > 8");
157         }
158 
159         if (avail == 0) {
160             octet = input.readOctet();
161             //System.out.println("OctetInput.readOctet: " + octet);
162             if (octet == -1) {
163                 throw new EOFException("EOF");
164             }
165             avail = 8;
166         }
167 
168         if (avail >= length) {
169             int value = octet >> (avail - length);
170             avail -= length;
171             count += length;
172             octet ^= (value << avail);
173             return value;
174         } else {
175             final int requi = length - avail;
176             return ((readUnsignedByte(avail) << requi)
177                     | readUnsignedByte(requi));
178         }
179     }

182     /**
183      * Reads an unsigned <code>length</code> bit short.
184      *
185      * @param length bit length between 0 (exclusive) and 16 (inclusive).
186      * @return an unsigned <code>length</code> bit integer.
187      * @throws IOException if an I/O error occurs
188      */
189     private final int readUnsignedShort(final int length) throws IOException {
190 
191         if (length <= 0) {
192             throw new IllegalArgumentException("length(" + length + ") <= 0");
193         }
194 
195         if (length > 16) {
196             throw new IllegalArgumentException("length(" + length + ") > 16");
197         }
198 
199         int value = 0x00;
200 
201         int quotient = length / 8;
202         for (int i = 0; i < quotient; i++) {
203             value <<= 8;
204             value |= readUnsignedByte(8);
205         }
206 
207         int remainder = length % 8;
208         if (remainder > 0) {
209             value <<= remainder;
210             value |= readUnsignedByte(remainder);
211         }
212 
213         return value;
214     }

449     /**
450      * Align to <code>length</code> bits by reading some dummy bits if required.
451      *
452      * @param length bit length greater than 0
453      * @throws IOException if an I/O error occurs
454      * @return this object
455      */
456     public final BitInput align(final int length) throws IOException {
457 
458         if (length <= 0) {
459             throw new IllegalArgumentException("length(" + length + ") <= 0");
460         }
461 
462         final int mod = (int) (count % length);
463         if (mod > 0) {
464             final int required = length - mod;
465 
466             final int quotient = required / 8;
467             for (int i = 0; i < quotient; i++) {
468                 readUnsignedByte(8);
469             }
470 
471             final int remainder = required % 8;
472             if (remainder > 0) {
473                 readUnsignedByte(remainder);
474             }
475         }
476 
477         return this;
478     }

510     /**
511      * Reads a sequence of characters in modified UTF-8 encoding and write to
512      * given <code>writer</code>.
513      *
514      * @param writer character output
515      * @throws IOException if an I/O error occurs.
516      */
517     public final void readUTF(final Writer writer) throws IOException {
518 
519         final byte[] bytes = new byte[readUnsignedShort(16)];
520         readBytes(bytes);
521 
522         final ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
523         try {
524             for (int a = -1; (a = bais.read()) != -1;) {
525                 if ((a >> 4) == 0x0F || // 1111xxxx
526                     (a >> 6) == 0x02) { // 10xxxxxx
527                     throw new UTFDataFormatException("a: " + a);
528                 }
529                 if ((a >> 7) == 0x00) { // 0xxxxxxxx
530                     writer.append((char) a);
531                 } else if ((a >> 5) == 0x06) { // 110xxxxx
532                     int b = bais.read();
533                     if (b == -1 || (b >> 6) != 0x02) { // NOT(10xxxxxx)
534                         throw new UTFDataFormatException("2b: " + b);
535                     }
536                     writer.append((char) (((a & 0x1F) << 6) | (b & 0x3F)));
537                 } else if (a >> 4 == 0x0E) { // 1110xxxx
538                     int b = bais.read();
539                     if (b == -1 || (b >> 6) != 0x02) { // NOT(10xxxxxx)
540                         throw new UTFDataFormatException("3b: " + b);
541                     }
542                     int c = bais.read();
543                     if (c == -1 || (c >> 6) != 0x02) { // NOT(10xxxxxx)
544                         throw new UTFDataFormatException("3c: " + b);
545                     }
546                     writer.write((char) (((a & 0x0F) << 12) |
547                                          ((b & 0x3F) << 6) |
548                                           (c & 0x3F)));
549                 }
550             }
551         } finally {
552             bais.close();
553         }
554     }
