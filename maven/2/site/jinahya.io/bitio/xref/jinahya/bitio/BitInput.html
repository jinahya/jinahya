86      /**
87       * Reads a sequence of bytes.
88       *
89       * @param value value to be read
90       * @param offset offset to start
91       * @param length length to be read
92       * @throws IOException if an I/O error occurs.
93       */
94      public void readBytes(byte[] value, int offset, int length)
95          throws IOException {
96  
97          if (offset < 0 || offset >= value.length) {
98              throw new IllegalArgumentException("illegal offset: " + offset);
99          }
100 
101         if (length <= 0 || (offset + length) > value.length) {
102             throw new IllegalArgumentException("illegal length: " + length);
103         }
104 
105         for (int i = 0; i < length; i++) {
106             value[offset + i] = (byte) readUnsignedShort(8);
107         }
108     }

111     private int readUnsignedByte(int length) throws IOException {
112 
113         if (length < 1 || length >= 8) {
114             throw new IllegalArgumentException("illegal length: " + length);
115         }
116 
117         if (avail == 0x00) {
118             octet = input.readOctet();
119             if (octet == -1) {
120                 throw new EOFException(":)");
121             }
122             avail = 0x08;
123         }
124 
125         if (avail >= length) {
126             int value = octet >> (avail - length);
127             avail -= length;
128             count += length;
129             octet &= (0xFF >> (8 - avail));
130             return value;
131         } else {
132             int requi = length - avail;
133             return ((readUnsignedByte(avail) << requi) |
134                     readUnsignedByte(requi));
135         }
136     }

139     private int readUnsignedShort(int length) throws IOException {
140 
141         if (length < 1 || length >= 16) {
142             throw new IllegalArgumentException("illegal length: " + length);
143         }
144 
145         int value = 0x00;
146 
147         int quotient = length / 0x07;
148         for (int i = 0; i < quotient; i++) {
149             value <<= 0x07;
150             value |= readUnsignedByte(0x07);
151         }
152 
153         int remainder = length % 7;
154         if (remainder > 0x00) {
155             value <<= remainder;
156             value |= readUnsignedByte(remainder);
157         }
158 
159         return value;
160     }

164      * Reads specifed number of bits and returns an unsigned integer.
165      *
166      * @param length number of bits to read
167      * @return an unsigned integer
168      * @throws IOException if an I/O error occurs
169      */
170     public int readUnsignedInt(int length) throws IOException {
171 
172         if (length < 0x01 || length >= 0x20) {
173             throw new IllegalArgumentException("illegal length: " + length);
174         }
175 
176         int value = 0x00;
177 
178         int quotient = length / 0x0F;
179         for (int i = 0; i < quotient; i++) {
180             value <<= 15;
181             value |= readUnsignedShort(0x0F);
182         }
183 
184         int remainder = length % 0x0F;
185         if (remainder > 0) {
186             value <<= remainder;
187             value |= readUnsignedShort(remainder);
188         }
189 
190         return value;
191     }

201     public int readInt(int length) throws IOException {
202 
203         if (length <= 0x01 || length > 0x20) {
204             throw new IllegalArgumentException("illegal length: " + length);
205         }
206 
207         int value = (0x00 - readUnsignedByte(0x01)) << (length - 1);
208 
209         value |= readUnsignedInt(length -1);
210 
211         return value;

364      * Reads some (if required) dummy bits for octet alignment.
365      *
366      * @param octetLength number of octets to be aligned
367      * @throws IOException if an I/O error occurs
368      */
369     public void alignOctets(int octetLength) throws IOException {
370         if (octetLength <= 0x00) {
371             throw new IllegalArgumentException
372                 ("illegal octet length: " + octetLength);
373         }
374 
375         int length = 8 - ((int) (count % (octetLength * 8)));
376 
377         int quotient = length / 7;
378         for (int i = 0; i < quotient; i++) {
379             readUnsignedByte(7);
380         }
381 
382         int remainder = length % 7;
383         if (remainder > 0) {
384             readUnsignedByte(remainder);
385         }
386     }

407     /**
408      * Reads a sequence of characters in modified UTF-8 encoding.
409      *
410      * @param writer character source
411      * @throws IOException if an I/O error occurs.
412      */
413     public void readUTF(Writer writer) throws IOException {
414 
415         byte[] bytes = new byte[readUnsignedInt(16)];
416         readBytes(bytes);
417 
418         ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
419         try {
420             for (int a = -1; (a = bais.read()) != -1;) {
421                 if ((a >> 4) == 0x0F || // 1111xxxx
422                     (a >> 6) == 0x02) { // 10xxxxxx
423                     throw new UTFDataFormatException("a: " + a);
424                 }
425                 if ((a >> 7) == 0x00) { // 0xxxxxxxx
426                     writer.append((char) a);
427                 } else if ((a >> 5) == 0x06) { // 110xxxxx
428                     int b = bais.read();
429                     if (b == -1 || (b >> 6) != 0x02) { // NOT(10xxxxxx)
430                         throw new UTFDataFormatException("2b: " + b);
431                     }
432                     writer.append((char) (((a & 0x1F) << 6) | (b & 0x3F)));
433                 } else if (a >> 4 == 0x0E) { // 1110xxxx
434                     int b = bais.read();
435                     if (b == -1 || (b >> 6) != 0x02) { // NOT(10xxxxxx)
436                         throw new UTFDataFormatException("3b: " + b);
437                     }
438                     int c = bais.read();
439                     if (c == -1 || (c >> 6) != 0x02) { // NOT(10xxxxxx)
440                         throw new UTFDataFormatException("3c: " + b);
441                     }
442                     writer.write((char) (((a & 0x0F) << 12) |
443                                          ((b & 0x3F) << 6) |
444                                           (c & 0x3F)));
445                 }
446             }
447         } finally {
448             bais.close();
449         }
450     }
