102     /**
103      * Writes an array of bytes.
104      *
105      * @param value byte array
106      * @param offset offset
107      * @param length length
108      * @throws IOException if an I/O error occurs.
109      */
110     public final void writeBytes(final byte[] value, final int offset,
111                                  final int length)
112         throws IOException {
113 
114         /*
115         System.out.println(
116             "writeBytes(" + value + ", " + offset + ", " + length + ")");
117          */
118 
119         if (value == null) {
120             throw new NullPointerException("value");
121         }
122 
123         if (offset < 0) {
124             throw new IndexOutOfBoundsException("offset(" + offset + ") < 0");
125         }
126 
127         if (length < 0) {
128             throw new IndexOutOfBoundsException("length(" + length + ") < 0");
129         }
130 
131         if ((offset + length) > value.length) {
132             throw new IndexOutOfBoundsException(
133                 "offset(" + offset + ") + length(" + length +
134                 ") > value.length(" + value.length + ")");
135         }
136 
137         for (int i = 0; i < length; i++) {
138             writeUnsignedByte(8, value[offset + i] & 0xFF);
139         }
140     }

143     /**
144      * Writes an <code>length</code> bit unsigned byte.
145      *
146      * @param length bit length between 0 (exclusive) and 8 (inclusive).
147      * @param value value
148      * @throws IOException if an I/O error occurs
149      */
150     private void writeUnsignedByte(final int length, final int value)
151         throws IOException {
152 
153         //System.out.println("writeUnsignedByte(" + length + ", " + value + ")");
154 
155         if (length <= 0) {
156             throw new IllegalArgumentException("length(" + length + ") <= 0");
157         }
158 
159         if (length > 8) {
160             throw new IllegalArgumentException("length(" + length + ") > " + 8);
161         }
162 
163         if (avail >= length) {
164             octet <<= length;
165             octet |= (value & (0xFF >> (8 - length)));
166             count += length;
167             avail -= length;
168             if (avail == 0) {
169                 output.writeOctet(octet);
170                 //System.out.println("OctetOutput.writeOctet(" + octet + ")");
171                 octet = 0;
172                 avail = 8;
173             }
174         } else {
175             final int requi = length - avail;
176             writeUnsignedByte(avail, (value >> (requi)));
177             writeUnsignedByte(requi, value);
178         }
179     }

183      * Writes an unsigned <code>length</code> bit short.
184      *
185      * @param length bit length between 0 (exclusive) and 16 (inclusive)
186      * @param value value
187      * @throws IOException if an I/O error occurs
188      */
189     private void writeUnsignedShort(final int length, final int value)
190         throws IOException {
191 
192         if (length <= 0) {
193             throw new IllegalArgumentException("length(" + length + ") <= 0");
194         }
195 
196         if (length > 16) {
197             throw new IllegalArgumentException("length(" + length + ") > 16");
198         }
199 
200         int quotient = length / 8;
201         int remainder = length % 8;
202         if (remainder > 0) {
203             writeUnsignedByte(remainder, (value >> (quotient * 8)));
204         }
205         for (int i = quotient - 1; i >= 0; i--) {
206             writeUnsignedByte(8, (value >> (i * 8)));
207         }
208     }

411     /**
412      * Align to <code>length</code> bits by writing some dummy bits if required.
413      *
414      * @param length number of bits to be aligned
415      * @throws IOException if an I/O error occurs.
416      * @return this object
417      */
418     public final BitOutput align(final int length) throws IOException {
419 
420         if (length <= 0) {
421             throw new IllegalArgumentException("length(" + length + ") < 0");
422         }
423 
424         final int mod = (int) (count % length);
425         if (mod > 0) {
426             final int required = length - mod;
427 
428             final int quotient = required / 8;
429             for (int i = 0; i < quotient; i++) {
430                 writeUnsignedByte(8, 0x00);
431             }
432 
433             final int remainder = required % 8;
434             if (remainder > 0) {
435                 writeUnsignedByte(remainder, 0x00);
436             }
437         }
438 
439         return this;
440     }

470     /**
471      * Writes a sequence of characters in modifed UTF-8 encoding.
472      *
473      * @param reader character source
474      * @throws IOException if an I/O error occurs
475      */
476     public void writeUTF(final Reader reader) throws IOException {
477         //System.out.println("writeUTF(" + reader + ")");
478         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
479         try {
480             for (int c = -1; (c = reader.read()) != -1;) {
481                 if (c >= '\u0001' && c <= '\u007F') {
482                     baos.write(c);
483                 } else if (c == '\u0000' || c <= '\u07FF') {
484                     baos.write(0xC0 | (0x1F & (c >> 6)));
485                     baos.write(0x80 | (0x3F & c));
486                 } else { // if (c <= '\uFFFF') {
487                     baos.write(0xE0 | (0x0F & (c >> 12)));
488                     baos.write(0x80 | (0x3F & (c >> 6)));
489                     baos.write(0x80 | (0x3F & c));
490                 }
491                 if (baos.size() > 65535) {
492                     throw new UTFDataFormatException(
493                         "length(" + baos.size() + ") > 65535");
494                 }
495             }
496             baos.flush();
497 
498             writeUnsignedInt(16, baos.size());
499             writeBytes(baos.toByteArray());
500 
501         } finally {
502             baos.close();
503         }
504     }
