103     /**
104      * Writes an array of bytes.
105      *
106      * @param value byte array
107      * @param offset offset
108      * @param length length
109      * @throws IOException if an I/O error occurs.
110      */
111     public void writeBytes(byte[] value, int offset, int length)
112         throws IOException {
113 
114         if (offset < 0 || offset >= value.length) {
115             throw new IllegalArgumentException("illegal offset: " + offset);
116         }
117 
118         if (length <= 0 || (offset + length) > value.length) {
119             throw new IllegalArgumentException("illegal length: " + length);
120         }
121 
122         for (int i = 0; i < length; i++) {
123             writeUnsignedShort(8, value[offset + i]);
124         }
125     }

128     private void writeUnsignedByte(final int length, final int value)
129         throws IOException {
130 
131         if (avail >= length) {
132             octet <<= length;
133             octet |= (value & (0xFF >> (0x08 - length)));
134             count += length;
135             if ((avail -= length) == 0x00) {
136                 output.writeOctet(octet);
137                 octet = 0x00;
138                 avail = 0x08;
139             }
140         } else {
141             int requi = length - avail;
142             writeUnsignedByte(avail, (value >> (requi)));
143             writeUnsignedByte(requi, value);
144         }
145     }

148     private void writeUnsignedShort(final int length, final int value)
149         throws IOException {
150 
151         int quotient = length / 0x07;
152         int remainder = length % 0x07;
153         if (remainder > 0) {
154             writeUnsignedByte(remainder, (value >> (quotient * 0x07)));
155         }
156         for (int i = quotient - 1; i >= 0; i--) {
157             writeUnsignedByte(7, (value >> (i * 0x07)));
158         }
159     }

343     /**
344      * Writes some (if required) dummy bits for alignemnt.
345      *
346      * @param length number of bits to be aligned
347      * @throws IOException if an I/O error occurs.
348      */
349     public final void align(final int length) throws IOException {
350 
351         if (length <= 0x00) {
352             throw new IllegalArgumentException("illegal length: " + length);
353         }
354 
355         int required = (int) (length - (count % length));
356 
357         int quotient = required / 7;
358         for (int i = 0; i < quotient; i++) {
359             writeUnsignedByte(7, 0x00);
360         }
361 
362         int remainder = required % 7;
363         if (remainder > 0) {
364             writeUnsignedByte(remainder, 0x00);
365         }
366     }

385     /**
386      * Writes a sequence of characters in modifed UTF-8 encoding.
387      *
388      * @param reader character source
389      * @throws IOException if an I/O error occurs
390      */
391     public void writeUTF(Reader reader) throws IOException {
392         ByteArrayOutputStream baos = new ByteArrayOutputStream();
393         try {
394             for (int c = -1; (c = reader.read()) != -1;) {
395                 if (c >= '\u0001' && c <= '\u007F') {
396                     baos.write(c);
397                 } else if (c == '\u0000' || c <= '\u07FF') {
398                     baos.write(0xC0 | (0x1F & (c >> 6)));
399                     baos.write(0x80 | (0x3F & c));
400                 } else { // if (c <= '\uFFFF') {
401                     baos.write(0xE0 | (0x0F & (c >> 12)));
402                     baos.write(0x80 | (0x3F & (c >> 6)));
403                     baos.write(0x80 | (0x3F & c));
404                 }
405                 if (baos.size() > 65535) {
406                     throw new UTFDataFormatException("length: " + baos.size());
407                 }
408             }
409             baos.flush();
410 
411             writeUnsignedInt(16, baos.size());
412             writeBytes(baos.toByteArray());
413 
414         } finally {
415             baos.close();
416         }
417     }

420     public void writeUnsignedFixedPoint(final int m, final int n,
421                                         final float value)
422         throws IOException {
423 
424         if (m < 0) {
425             throw new IllegalArgumentException("m(" + m + ") < 0");
426         }

437             throw new IllegalArgumentException
438                 ("m(" + m + ") + n(" + n + ") > 0x20");
439         }
440

442     }
