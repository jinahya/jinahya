85      /**
86       * Writes an array of bytes.
87       *
88       * @param value byte array
89       * @param offset offset
90       * @param length length
91       * @throws IOException if an I/O error occurs.
92       */
93      public void writeBytes(byte[] value, int offset, int length)
94          throws IOException {
95  
96          if (offset < 0 || offset >= value.length) {
97              throw new IllegalArgumentException("illegal offset: " + offset);
98          }
99  
100         if (length <= 0 || (offset + length) > value.length) {
101             throw new IllegalArgumentException("illegal length: " + length);
102         }
103 
104         for (int i = 0; i < length; i++) {
105             writeUnsignedShort(8, value[offset + i]);
106         }
107     }

110     private void writeUnsignedByte(final int length, final int value)
111         throws IOException {
112 
113         if (avail >= length) {
114             octet <<= length;
115             octet |= (value & (0xFF >> (0x08 - length)));
116             count += length;
117             if ((avail -= length) == 0x00) {
118                 output.writeOctet(octet);
119                 octet = 0x00;
120                 avail = 0x08;
121             }
122         } else {
123             int requi = length - avail;
124             writeUnsignedByte(avail, (value >> (requi)));
125             writeUnsignedByte(requi, value);
126         }
127     }

130     private void writeUnsignedShort(final int length, final int value)
131         throws IOException {
132 
133         int quotient = length / 0x07;
134         int remainder = length % 0x07;
135         if (remainder > 0) {
136             writeUnsignedByte(remainder, (value >> (quotient * 0x07)));
137         }
138         for (int i = quotient - 1; i >= 0; i--) {
139             writeUnsignedByte(7, (value >> (i * 0x07)));
140         }
141     }

151     public void writeUnsignedInt(final int length, final int value)
152         throws IOException {
153 
154         if (length < 0x1 || length >= 0x20) {
155             throw new IllegalArgumentException("illegal length: " + length);
156         }
157 
158         int quotient = length / 0x0F;
159         int remainder = length % 0x0F;
160         if (remainder > 0) {
161             writeUnsignedShort(remainder, (value >> (quotient * 0x0F)));
162         }
163         for (int i = quotient -1; i >= 0; i--) {
164             writeUnsignedShort(0x0F, (value >> (i * 0x0F)));
165         }

325     /**
326      * Writes some (if required) dummy bits for octet alignemnt.
327      *
328      * @param octetLength number of bytes to be aligned
329      * @throws IOException if an I/O error occurs.
330      */
331     public void alignOctets(int octetLength) throws IOException {
332         if (octetLength <= 0x00) {
333             throw new IllegalArgumentException
334                 ("illegal octet length: " + octetLength);
335         }
336         int length = (octetLength * 8) - (int) (count % 8);
337 
338         int quotient = length / 7;
339         for (int i = 0; i < quotient; i++) {
340             writeUnsignedByte(7, 0x00);
341         }
342 
343         int remainder = length % 7;
344         if (remainder > 0) {
345             writeUnsignedByte(remainder, 0x00);
346         }
347     }

366     /**
367      * Writes a sequence of characters in modifed UTF-8 encoding.
368      *
369      * @param reader character source
370      * @throws IOException if an I/O error occurs
371      */
372     public void writeUTF(Reader reader) throws IOException {
373         ByteArrayOutputStream baos = new ByteArrayOutputStream();
374         try {
375             for (int c = -1; (c = reader.read()) != -1;) {
376                 if (c >= '\u0001' && c <= '\u007F') {
377                     baos.write(c);
378                 } else if (c == '\u0000' || c <= '\u07FF') {
379                     baos.write(0xC0 | (0x1F & (c >> 6)));
380                     baos.write(0x80 | (0x3F & c));
381                 } else { // if (c <= '\uFFFF') {
382                     baos.write(0xE0 | (0x0F & (c >> 12)));
383                     baos.write(0x80 | (0x3F & (c >> 6)));
384                     baos.write(0x80 | (0x3F & c));
385                 }
386                 if (baos.size() > 65535) {
387                     throw new UTFDataFormatException("length: " + baos.size());
388                 }
389             }
390             baos.flush();
391 
392             writeUnsignedInt(16, baos.size());
393             writeBytes(baos.toByteArray());
394 
395         } finally {
396             baos.close();
397         }
398     }
