116     /**
117      * Writes an array of bytes.
118      *
119      * @param value byte array
120      * @param offset offset
121      * @param length length
122      * @throws IOException if an I/O error occurs.
123      */
124     public final void writeBytes(final byte[] value, final int offset,
125                                  final int length)
126         throws IOException {
127 
128         /*
129         System.out.println(
130             "writeBytes(" + value + ", " + offset + ", " + length + ")");
131          */
132 
133         if (value == null) {
134             throw new NullPointerException("value");
135         }
136 
137         if (offset < 0) {
138             throw new IndexOutOfBoundsException("offset(" + offset + ") < 0");
139         }
140 
141         if (length < 0) {
142             throw new IndexOutOfBoundsException("length(" + length + ") < 0");
143         }
144 
145         if ((offset + length) > value.length) {
146             throw new IndexOutOfBoundsException(
147                 "offset(" + offset + ") + length(" + length +
148                 ") > value.length(" + value.length + ")");
149         }
150 
151         for (int i = 0; i < length; i++) {
152             writeUnsignedByte(8, value[offset + i]);
153         }
154     }

157     /**
158      * Writes the <code>length</code> low-order bits is of the argument
159      * <code>value</code>. The 32 - <code>length</code> high-order bits of
160      * <code>value</code> are ignored.
161      *
162      * @param length bit length between 1 (inclusive) and 8 (inclusive).
163      * @param value value
164      * @throws IOException if an I/O error occurs
165      */
166     private void writeUnsignedByte(final int length, final int value)
167         throws IOException {
168 
169         if (length < 1) {
170             throw new IllegalArgumentException("length(" + length + ") < 1");
171         }
172 
173         if (length > 8) {
174             throw new IllegalArgumentException("length(" + length + ") > " + 8);
175         }
176 
177         if (avail >= length) {
178             octet <<= length;
179             octet |= (value & (0xFF >> (8 - length)));
180             count += length;
181             avail -= length;
182             if (avail == 0) {
183                 output.writeOctet(octet);
184                 //System.out.println("write octet: " + octet);
185                 octet = 0;
186                 avail = 8;
187             }
188         } else {
189             final int requi = length - avail;
190             writeUnsignedByte(avail, (value >> (requi)));
191             writeUnsignedByte(requi, value);
192         }
193     }

196     /**
197      * Writes an unsigned <code>length</code> bit short.
198      *
199      * @param length bit length between 1 (inclusive) and 16 (inclusive)
200      * @param value value
201      * @throws IOException if an I/O error occurs
202      */
203     private void writeUnsignedShort(final int length, final int value)
204         throws IOException {
205 
206         if (length < 1) {
207             throw new IllegalArgumentException("length(" + length + ") < 1");
208         }
209 
210         if (length > 16) {
211             throw new IllegalArgumentException("length(" + length + ") > 16");
212         }
213 
214         int quotient = length / 8;
215         int remainder = length % 8;
216         if (remainder > 0) {
217             writeUnsignedByte(remainder, (value >> (quotient * 8)));
218         }
219         for (int i = quotient - 1; i >= 0; i--) {
220             writeUnsignedByte(8, (value >> (i * 8)));
221         }
222     }

423     /**
424      * Align to <code>length</code> bits by writing some dummy bits if required.
425      *
426      * @param length number of bits to be aligned; non-zero positive
427      * @throws IOException if an I/O error occurs.
428      * @return this object
429      */
430     public final int align(final int length) throws IOException {
431 
432         if (length <= 0) {
433             throw new IllegalArgumentException("length(" + length + ") <= 0");
434         }
435 
436         final int mod = (int) (count % length);
437 
438         if (mod == 0) {
439             return mod;
440         }
441 
442         final int required = length - mod;
443 
444         final int quotient = required / 8;
445         for (int i = 0; i < quotient; i++) {
446             writeUnsignedByte(8, 0x00);
447         }
448 
449         final int remainder = required % 8;
450         if (remainder > 0) {
451             writeUnsignedByte(remainder, 0x00);
452         }
453         return required;
454     }
