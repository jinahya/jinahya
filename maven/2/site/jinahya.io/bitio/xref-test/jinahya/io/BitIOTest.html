74      @Test
75      public void testInt() throws IOException {
76  
77          for (int i = 0; i < COUNT; i++) {
78  
79              int expected = RANDOM.nextInt();
80  
81              final int length = RANDOM.nextInt(32) + 1; // 1 - 32
82  
83              if (length == 1) { // unsigned int
84                  expected &= 0x01;
85  
86                  output.writeUnsignedInt(length, expected);
87                  alignAndFlush(); // 8
88                  assertEquals(expected, input.readUnsignedInt(length));
89                  input.align();
90  
91              } else if (length == 32) { // signed int
92                  dos.writeInt(expected);
93                  dos.flush();
94                  assertEquals(expected, input.readInt());
95  
96                  output.writeInt(expected);
97                  alignAndFlush();
98                  assertEquals(expected, input.readInt());
99  
100                 output.writeInt(expected);
101                 alignAndFlush();
102                 assertEquals(expected, dis.readInt());
103 
104             } else {
105                 final int upper = 32 - length;
106                 expected <<= upper;
107                 expected >>= upper;
108 
109                 if (expected >= 0) { // unsigned
110                     output.writeUnsignedInt(length, expected);
111                     alignAndFlush();
112                     assertEquals(expected, input.readUnsignedInt(length));
113                     input.align();
114 
115                 } else { // sigend
116                     output.writeInt(length, expected);
117                     alignAndFlush();
118                     assertEquals(expected, input.readInt(length));
119                     input.align();
120                 }
121             }
122         }
