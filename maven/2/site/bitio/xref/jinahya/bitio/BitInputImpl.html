62          0x03, // 3   0000 0011
63          0x07, // 7   0000 0111
64          0x0F, // 15  0000 1111

72      protected int readUnsignedByte(int length) throws IOException {
73  
74          if (length < 1 || length >= 8) {
75              throw new IllegalArgumentException("illegal length: " + length);
76          }
77  
78          if (index == 0x08) {
79              octet = input.readByte();
80              if (octet == -1) {
81                  throw new EOFException("EOF");
82              }
83              index = 0x00;
84          }
85  
86          int available = 0x08 - index;
87  
88          if (available >= length) {
89              index += length;
90              count += length;
91              return (octet >>> (available - length)) & powers[length -1];
92          }
93  
94          int required = length - available;
95          return ((readUnsignedByte(available) << required) |
96                  readUnsignedByte(required));
97      }

100     protected int readUnsignedShort(int length) throws IOException {
101 
102         if (length < 1 || length >= 16) {
103             throw new IllegalArgumentException("illegal length: " + length);
104         }
105 
106         int value = 0x00;
107 
108         int quotient = length / 7;
109         for (int i = 0; i < quotient; i++) {
110             value <<= 7;
111             value |= readUnsignedByte(7);
112         }
113 
114         int remainder = length % 7;
115         if (remainder > 0) {
116             value <<= remainder;
117             value |= readUnsignedByte(remainder);
118         }
119 
120         return value;
121     }

152         if (length <= 1 || length > 32) {
153             throw new IllegalArgumentException("illegal length: " + length);
154         }
155 
156         int value = readUnsignedByte(1);
157         value <<= (length - 1);
158         for (int i = 0; i < 32 - (length - 1); i++) {

194 
195         if (length <= 1 || length > 64) {
196             throw new IllegalArgumentException("illegal length: " + length);
197         }
198 
199         long value = readUnsignedByte(1); // sign bit
200         value <<= (length - 1); // zero padding
201         for (int i = 0; i < 64 - (length - 1); i++) {
202             value |= (value <<= 1); // copy shift
203         }
