34  package jinahya.xlet.model;

37  import java.beans.PropertyChangeEvent;
38  import java.beans.PropertyChangeListener;
39  import java.lang.reflect.InvocationTargetException;
40  import java.lang.reflect.Method;
41  
42  
43  /**
44   *
45   * @author <a href="mailto:jinahya@gmail.com">Jin Kwon</a>
46   */
47  public abstract class BoundModel extends Model
48      implements PropertyChangeListener {
49  
50  
51      public BoundModel(Class clazz) {
52          super();
53  
54          try {
55              setBean(bean = (this.clazz = clazz).newInstance());
56          } catch (InstantiationException ie) {
57              ie.printStackTrace();
58          } catch (IllegalAccessException iae) {
59              iae.printStackTrace();
60          }
61      }
62  
63  
64      public void propertyChange(PropertyChangeEvent pce) {
65          // empty
66      }
67  
68  
69      public synchronized Object getBean() {
70          return bean;
71      }
72  
73  
74      public synchronized void setBean(Object bean) {
75          if (!clazz.isInstance(bean)) {
76              throw new IllegalArgumentException("not an instance of " + clazz);
77          }
78  
79          if (this.bean != null) { // in case of construction time
80              try {
81                  Method method = clazz.getMethod
82                      ("removePropertyChangeListener",
83                       new Class[]{PropertyChangeListener.class});
84                  try {
85                      method.invoke(this.bean, new Object[]{this});
86                  } catch (IllegalAccessException iae) {
87                      iae.printStackTrace();
88                  } catch (InvocationTargetException ite) {
89                      ite.printStackTrace();
90                  }
91              } catch (NoSuchMethodException nsme) {
92                  nsme.printStackTrace();
93              }
94          }
95  
96          Object oldBean = this.bean;
97          this.bean = bean;
98          firePropertyChange("bean", oldBean, this.bean);
99  
100         try {
101             Method method = clazz.getMethod
102                 ("addPropertyChangeListener",
103                  new Class[]{PropertyChangeListener.class});
104             try {
105                 method.invoke(this.bean, new Object[]{this});
106             } catch (IllegalAccessException iae) {
107                 iae.printStackTrace();
108             } catch (InvocationTargetException ite) {
109                 ite.printStackTrace();
110             }
111         } catch (NoSuchMethodException nsme) {
112             nsme.printStackTrace();
113         }
114     }
115 
116 
117     private Class clazz;
118     private Object bean;
119 }
