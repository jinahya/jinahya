54      /**
55       * Writes an unsigned byte value. This method doesn't check the validity of
56       * the value and writes the lower <code>length</code> bits in
57       * <code>value</code>.
58       *
59       * @param length bit length between 0x00 (exclusive) and 0x08 (inclusive).
60       * @param value the value to write
61       * @throws IOException if an I/O error occurs.
62       */
63      protected void writeUnsignedByte(final int length, final int value)
64          throws IOException {
65  
66          if (length <= 0x00) {
67              throw new IllegalArgumentException(
68                  "length(" + length + ") <= 0x00");
69          }
70  
71          if (length > 0x08) {
72              throw new IllegalArgumentException("length(" + length + ") > 0x08");
73          }
74  
75          final int required = length - (0x08 - index);
76          if (required > 0x00) {
77              writeUnsignedByte(length - required, value >> required);
78              writeUnsignedByte(required, value);
79              return;
80          }
81  
82          index += length;
83          for (int i = 0; i < length; i++) {
84              index--;
85              set.set(index, ((value >>> i) & 0x01) == 0x01);
86          }
87  
88          index += length;
89          if (index == 0x08) {
90              int octet = 0x00;
91              for (int i = 0; i < 8; i++) {
92                  octet <<= 1;
93                  octet |= (set.get(i) ? 0x01 : 0x00);
94              }
95              out.write(octet);
96              count++;
97              index = 0;
98          }
99      }

162     /**
163      * Writes an unsigned short value. This method doesn't check the validity of
164      * the value and writes the lower <code>length</code> bits in
165      * <code>value</code>.
166      *
167      * @param length bit length between 0x01 (inclusive) and 0x10 (inclusive)
168      * @param value value to write
169      * @throws IOException if an I/O error occurs
170      */
171     protected void writeUnsignedShort(final int length, final int value)
172         throws IOException {
173 
174         if (length < 0x01) {
175             throw new IllegalArgumentException("length(" + length + ") < 0x01");
176         }
177 
178         if (length > 0x20) {
179             throw new IllegalArgumentException("length(" + length + ") > 0x20");
180         }
181 
182         final int quotient = length / 0x08;
183         final int remainder = length % 0x08;
184 
185         if (remainder > 0x00) {
186             writeUnsignedByte(remainder, value >> (quotient * 0x08));
187         }
188 
189         for (int i = quotient - 1; i >= 0; i--) {
190             writeUnsignedByte(0x08, value >> (0x08 * i));
191         }
192     }

544      * Writes a 7-bit ASCII string. First, a 1-bit boolean is written for a null
545      * flag. And, if the <code>value</code> is not null, a 31-bit unsigned
546      * integer is written for the character count following all characters which

568     /**
569      * Writes a UTF string. Identical to {@link DataOutput#writeUTF(String)}.
570      *
571      * @param value string to write
572      * @throws IOException if an I/O error occurs
573      * @see DataOutput#writeUTF(String)
574      * @deprecated Not for Bit I/O. Use {@link #writeSTRING(String, String)}
575      */
576     public void writeUTF(final String value) throws IOException {
577 
578         if (value == null) {
579             throw new NullPointerException("null value");
580         }
581 
582         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
583 
584         for (int i = 0; i < value.length(); i++) {
585 
586             final char c = value.charAt(i);
587 
588             if (baos.size() > 65535) {
589                 throw new UTFDataFormatException("too long");
590             }
591 
592             if (c >= '\u0001' && c <= '\u007F') {
593                 baos.write(c);
594                 continue;
595             }
596 
597             if (c == '\u0000' || (c >= '\u0080' && c <= '\u07FF')) {
598                 baos.write((0xC0 | (0x1F & (c >> 6))));
599                 baos.write((0x80 | (0x3F & c)));
600                 continue;
601             }
602 
603             if (c >= '\u0800' && c <= '\uFFFF') {
604                 baos.write((0xE0 | (0x0F & (c >> 12))));
605                 baos.write((0x80 | (0x3F & (c >> 6))));
606                 baos.write((0x80 | (0x3F & c)));
607                 continue;
608             }
609         }
610 
611         final byte[] bytes = baos.toByteArray();
612         writeUnsignedInt(0x10, bytes.length);
613         for (int i = 0; i < bytes.length; i++) {
614             writeUnsignedByte(0x08, bytes[i]);
615         }
616     }

658     /**
659      * Aligns to given <code>length</code> as bytes.
660      *
661      * @param length the number of bytes to align
662      * @return the bits padded to align
663      * @throws IOException if an I/O error occurs.
664      */
665     public int align(final int length) throws IOException {
666 
667         if (length <= 0) {
668             throw new IllegalArgumentException("length(" + length + ") <= 0");
669         }
670 
671         int bits = 0;
672 
673         if (index > 0) { // bit index to write
674             bits = (8 - index);
675             writeUnsignedByte(8 - index, 0);
676         }
677 
678         int octets = count % length;
679 
680         if (octets == 0) {
681             return bits;
682         }
683 
684         if (octets > 0) {
685             octets = length - octets;
686         } else { // mod < 0
687             octets = 0 - octets;
688         }
689 
690         for (; octets > 0; octets--) {
691             writeUnsignedByte(8, 0);
692             bits += 8;
693         }
694 
695         return bits;
696     }
