26  package javax.xml.stream;
27  
28  /**
29   * An error class for reporting factory configuration errors.
30   *
31   * @version 1.0
32   * @author Copyright (c) 2003 by BEA Systems. All Rights Reserved.
33   */
34  public class FactoryConfigurationError extends Error {
35  
36    Exception nested;
37  
38    /**
39     * Default constructor
40     */
41    public FactoryConfigurationError(){}
42  
43    /**
44     * Construct an exception with a nested inner exception
45     *
46     * @param e the exception to nest
47     */
48    public FactoryConfigurationError(java.lang.Exception e){
49      nested = e;
50    }
51  
52    /**
53     * Construct an exception with a nested inner exception
54     * and a message
55     *
56     * @param e the exception to nest
57     * @param msg the message to report
58     */
59    public FactoryConfigurationError(java.lang.Exception e, java.lang.String msg){
60      super(msg);
61      nested = e;
62    }
63  
64    /**
65     * Construct an exception with a nested inner exception
66     * and a message
67     *
68     * @param msg the message to report
69     * @param e the exception to nest
70     */
71    public FactoryConfigurationError(java.lang.String msg, java.lang.Exception e){
72      super(msg);
73      nested = e;
74    }
75  
76    /**
77     * Construct an exception with associated message
78     *
79     * @param msg the message to report
80     */
81    public FactoryConfigurationError(java.lang.String msg){
82      super(msg);
83    }
84    
85    /**
86     * Return the nested exception (if any)
87     *
88     * @return the nested exception or null
89     */
90    public Exception getException() {
91      return nested;
92    }
93    
94  
95    /**
96     * Report the message associated with this error
97     * 
98     * @return the string value of the message
99     */
100   public String getMessage() {
101     String msg = super.getMessage();
102     if(msg != null)
103       return msg;
104     if(nested != null){
105       msg = nested.getMessage();
106       if(msg == null)
107         msg = nested.getClass().toString();
108     }
109     return msg;  
110   }
111   
112 
113   
114 }
