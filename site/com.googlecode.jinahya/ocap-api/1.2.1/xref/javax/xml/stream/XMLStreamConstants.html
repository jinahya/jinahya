33  /**
34   * This interface declares the constants used in this API.
35   * Numbers in the range 0 to 256 are reserved for the specification,
36   * user defined events must use event codes outside that range.
37   */
38  
39  public interface XMLStreamConstants {
40    /**
41     * Indicates an event is a start element
42     */
43    public static final int START_ELEMENT=1;
44    /**
45     * Indicates an event is an end element
46     */
47    public static final int END_ELEMENT=2;
48    /**
49     * Indicates an event is a processing instruction
50     */
51    public static final int PROCESSING_INSTRUCTION=3;
52  
53    /**
54     * Indicates an event is characters
55     */
56    public static final int CHARACTERS=4;
57  
58    /**
59     * Indicates an event is a comment
60     */
61    public static final int COMMENT=5;
62  
63    /**
64     * The characters are white space
65     * (see [XML], 2.10 "White Space Handling").
66     * Events are only reported as SPACE if they are ignorable white
67     * space.  Otherwise they are reported as CHARACTERS.
68     */
69    public static final int SPACE=6;
70  
71    /**
72     * Indicates an event is a start document
73     */
74    public static final int START_DOCUMENT=7;
75  
76    /**
77     * Indicates an event is an end document
78     */
79    public static final int END_DOCUMENT=8;
80  
81    /**
82     * Indicates an event is an entity reference
83     */
84    public static final int ENTITY_REFERENCE=9;
85  
86    /**
87     * Indicates an event is an attribute
88     */
89    public static final int ATTRIBUTE=10;
90  
91    /**
92     * Indicates an event is a DTD
93     */
94    public static final int DTD=11;
95  
96    /**
97     * Indicates an event is a CDATA section
98     */
99    public static final int CDATA=12;
100 
101   /**
102    * Indicates the event is a namespace declaration
103    */
104   public static final int NAMESPACE=13;
105 
106   /**
107    * Indicates a Notation 
108    */
109   public static final int NOTATION_DECLARATION=14;
110 
111   /**
112    * Indicates a Entity Declaration
113    */
114   public static final int ENTITY_DECLARATION=15;
115 }
