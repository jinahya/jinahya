26  package javax.xml.stream;
27  
28  /**
29   * This interface is used to resolve resources during an XML parse.  If an application wishes to 
30   * perform custom entity resolution it must register an instance of this interface with
31   * the XMLInputFactory using the setXMLResolver method.  
32   *
33   * @version 1.0
34   * @author Copyright (c) 2003 by BEA Systems. All Rights Reserved.
35   */
36  public interface XMLResolver {
37  
38    /**
39     * Retrieves a resource.  This resource can be of the following two return types:
40     * (1) java.io.InputStream (2) javax.xml.stream.XMLStreamReader 
41     * If this method returns null the processor will attempt to resolve the entity using its
42     * default mechanism.
43     *  
44     * @param publicID The public identifier of the external entity being referenced, or null if none was supplied.
45     * @param systemID The system identifier of the external entity being referenced.
46     * @param baseURI  Absolute base URI associated with systemId.
47     * @param namespace The namespace of the entity to resolve. 
48     * @return The resource requested or null.
49     * @throws XMLStreamException if there was a failure attempting to resolve the resource.
50     */
51    public Object resolveEntity(
52        String publicID, String systemID, String baseURI, String namespace)
53      throws XMLStreamException;
54  }
