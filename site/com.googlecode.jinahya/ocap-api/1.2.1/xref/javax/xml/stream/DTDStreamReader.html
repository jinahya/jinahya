35  /**
36   * This interface extends the Streaming API for XML (StAX) Cursor API. 
37   * It enables forward, read-only access to
38   * general internal and external parsed and unparsed entity declarations, 
39   * notation declarations, processing instructions and comments contained in 
40   * a DTD.
41   * <p>
42   * The DTDStreamReader instance has very specific lifecycle constraints.
43   * An application can only obtain an implementation of DTDStreamReader from 
44   * {@link XMLStreamReader XMLStreamReader} immediately after the method 
45   * {@link XMLStreamReader#next() next()} returns a
46   * {@link XMLStreamReader#DTD DTD} event, and prior to invoking any other
47   * methods on {@link XMLStreamReader XMLStreamReader}. The 
48   * DTDStreamReader instance is obtained by calling method 
49   * {@link XMLStreamReader#getProperty getProperty} with 
50   * <code>"javax.xml.stream.DTDStreamReader"</code> as the property name 
51   * parameter. 
52   * Multiple calls to {@link XMLStreamReader#getProperty getProperty} with 
53   * a <code>"javax.xml.stream.DTDStreamReader"</code> argument will return the 
54   * same instance of DTDStreamReader.
55   * Once the DTDStreamReader object is instantiated by the 
56   * application, its various methods can be used to access the DTD data. Its
57   * processing capability extends only until the {@link #END_DTD END_DTD} event 
58   * is reached;
59   * at this point, any further document processing must be performed via the
60   * {@link XMLStreamReader XMLStreamReader} instance.
61   * </p>
62   * <p>
63   * If an application chooses to instantiate a DTDStreamReader,
64   * the  {@link XMLStreamReader#getText() getText()} method on 
65   * {@link XMLStreamReader XMLStreamReader} is invalidated
66   * for the {@link XMLStreamReader#DTD DTD} event: any attempt to invoke
67   * it will generate a <code>java.lang.IllegalStateException</code>. In this case, 
68   * the data for the DTD event must be obtained via the DTDStreamReader
69   * instance.
70   * </p><p>
71   *
72   * After DTDStreamReader is instantiated, invocation of any of
73   * the following {@link XMLStreamReader XMLStreamReader} methods
74   * will change the state of 
75   * DTDStreamReader implementation to {@link #END_DTD END_DTD}:
76   * <ul><li>{@link XMLStreamReader#next() next()},</li> 
77   * <li>{@link XMLStreamReader#hasNext() hasNext()},</li>
78   * <li>{@link XMLStreamReader#nextTag() nextTag()} and</li>
79   * <li>{@link XMLStreamReader#close() close()}.</li></ul>
80   * Once in the {@link #END_DTD END_DTD} state, 
81   * no further processing of the DTD data is possible.  
82   * </p>
83   * <p>
84   * If no instance of DTDStreamReader is created (either the
85   * {@link XMLStreamReader#getProperty getProperty} method with parameter
86   * <code>"javax.xml.stream.DTDStreamReader"</code> is never invoked, 
87   * or it returns <code>null</code> because it is not invoked
88   * immediately after a {@link XMLStreamReader#DTD DTD} event is received),
89   * the behavior of
90   * {@link XMLStreamReader XMLStreamReader} is unchanged and follows the
91   * specification in
92   * <a href="http://www.jcp.org/en/jsr/detail?id=173&showPrint">JSR173</a>
93   * for the JSR 280 subset of the API.
94   * </p>
95   * <p>
96   * DTDStreamReader intentionally mimics 
97   * {@link XMLStreamReader XMLStreamReader} in order to provide developers
98   * with a familiar programming model.
99   * DTDStreamReader includes several methods identical in signature 
100  * and semantics to methods defined on {@link XMLStreamReader XMLStreamReader}. 
101  * </p><p>
102  * An object which implements DTDStreamReader uses methods {@link #next next} 
103  * and {@link #getEventType getEventType} to provide the current event type. 
104  * The first event is always {@link #START_DTD START_DTD}. 
105  * Once an {@link #END_DTD END_DTD} event is returned by methods 
106  * {@link #next next} or {@link #getEventType getEventType}, the application 
107  * must continue further document processing using the
108  * {@link XMLStreamReader XMLStreamReader} interface.
109  * The application can use the {@link #close close} method at any time to 
110  * terminate processing of any remaining DTD content and skip to the 
111  * {@link #END_DTD END_DTD} event.
112  * </p><p>
113  * The following table describes which methods are valid in what state.
114  * If a method is called in an invalid state the method will throw a
115  * java.lang.IllegalStateException. Method {@link #next() next()} is an 
116  * exception from this rule: it is defined to throw 
117  * NoSuchElementException if it is called when {@link #hasNext() hasNext()} 
118  * returns <code>false</code>, so in the case where {@link #next() next()}
119  * is called in the END_DTD state it will return NoSuchElementException rather
120  * than java.lang.IllegalStateException. 
121  * </p>

176  *            {@link #getSystemIdentifier() getSystemIdentifier}
177  *     </tr>
178  *     </tr>
179  *     <tr>
180  *       <td> {@link #COMMENT COMMENT}  </td>
181  *       <td> {@link #next() next}, 
182  *            {@link #getText() getText}, 
183  *            {@link #getTextCharacters getTextCharacters}, 
184  *            {@link #getTextStart() getTextStart}, 
185  *            {@link #getTextLength() getTextLength} 
186  *       </td>
187  *     </tr>
188  *     <tr>
189  *       <td> {@link #PROCESSING_INSTRUCTION PROCESSING_INSTRUCTION}  </td>
190  *       <td> {@link #next() next}, 
191  *            {@link #getPITarget() getPITarget}, 
192  *            {@link #getPIData() getPIData} 
193  *       </td>
194  *     </tr>
195  *  </table>
196  *
197  * @version 1.0
198  * @author Copyright (c) 2006 by BEA Systems. All Rights Reserved.
199  * @see XMLStreamConstants
200  * @see XMLStreamReader
201  */

203 public interface DTDStreamReader
204 {
205 	/** 
206 	 * Indicates an event is the start of a DTD. On this event, method 
207 	 * {@link #getName() getName} returns the root element's 
208 	 * qualified name, and methods 
209 	 * {@link #getPublicIdentifier() getPublicIdentifier} and 
210 	 * {@link #getSystemIdentifier() getSystemIdentifier} return the document's
211 	 * public and system identifiers.
212 	 */
213 	public static final int START_DTD = 1001;
214 
215 	/** 
216 	 * Indicates an event is the end of a DTD. On this event, method 
217 	 * {@link #hasNext() hasNext} returns <code>false</code>, 
218 	 * and method {@link #getEventType() getEventType} returns 
219 	 * {@link #END_DTD END_DTD}. This event marks the end of the valid
220 	 * lifecycle of the DTDStreamReader instance, so any 
221 	 * method call other than {@link #next() next}, {@link #hasNext() hasNext},
222 	 * {@link #getEventType() getEventType}, {@link #close() close}, 
223 	 * or {@link #getLocation() getLocation} on this interface will
224 	 * throw java.lang.IllegalStateException.
225 	 */
226 	public static final int END_DTD = 1002;
227 
228 	/** 
229 	 * Indicates an event is a parsed entity declaration.
230 	 * <p>
231 	 * If 
232 	 * the {@link #getTextLength getTextLength} method returns a negative 
233 	 * value, the event represents an external parsed entity: 
234 	 * method {@link #getName() getName} returns the 
235 	 * entity name, methods {@link #getPublicIdentifier getPublicIdentifier}  
236 	 * and {@link #getSystemIdentifier getSystemIdentifier} return the 
237 	 * entity's public and system identifiers, and methods
238 	 * {@link #getText getText} and {@link #getTextCharacters getTextCharacters} 
239 	 * return null and empty array, respectively.
240 	 * </p><p>
241 	 * If the {@link #getTextLength getTextLength} method returns a positive
242 	 * value, then the event represents an internal parsed entity:
243 	 * method {@link #getName() getName} returns the 
244 	 * entity name, methods {@link #getText getText} or 
245 	 * {@link #getTextCharacters getTextCharacters} return the replacement 
246 	 * text of an internal parsed entity, and methods 
247 	 * {@link #getPublicIdentifier getPublicIdentifier} and  
248 	 * {@link #getSystemIdentifier getSystemIdentifier} return null.
249 	 */
250 	public static final int ENTITY_DECLARATION = 
251 		XMLStreamConstants.ENTITY_DECLARATION;
252 
253 	/** 
254 	 * Indicates an event is an unparsed entity declaration. On this event,
255 	 * method {@link #getName() getName} returns the unparsed 
256 	 * entity name, methods {@link #getPublicIdentifier getPublicIdentifier} 
257 	 * and {@link #getSystemIdentifier getSystemIdentifier} return 
258 	 * the public identifier and the system identifier of an unparsed entity,
259 	 * and the 
260 	 * method {@link #getNotationName() getNotationName} returns the name of 
261 	 * the notation which identifies the format of the unparsed entity.
262 	 */
263 	public static final int UNPARSED_ENTITY_DECLARATION = 1003;
264 
265 	/** 
266 	 * Indicates an event is a notation declaration. On this event the method 
267 	 * {@link #getName() getName} returns the notation name, 
268 	 * the methods {@link #getPublicIdentifier getPublicIdentifier} 
269 	 * and {@link #getSystemIdentifier getSystemIdentifier} return the
270 	 * public identifier and the system identifier of the notation.
271 	 */
272 	public static final int NOTATION_DECLARATION = 
273 		XMLStreamConstants.NOTATION_DECLARATION;
274 
275 	/** 
276 	 * Indicates an event is a processing instruction. On this event,
277 	 * the method 
278 	 * {@link #getPITarget() getPITarget} returns the processing instruction 
279 	 * target and the method {@link #getPIData() getPIData} returns the
280 	 * processing instruction data section.
281 	 */
282 	public static final int PROCESSING_INSTRUCTION = 
283 		XMLStreamConstants.PROCESSING_INSTRUCTION;
284 
285 	/** 
286 	 * Indicates an event is a comment. On this event, the methods 
287 	 * {@link #getText() getText} or 
288 	 * {@link #getTextCharacters getTextCharacters} return the character 
289 	 * data of the comment, and
290 	 * the methods {@link #getTextStart() getTextStart} and 
291 	 * {@link #getTextLength() getTextLength} return the index of the 
292 	 * first character 
293 	 * and the number of characters in the character array returned by 
294 	 * the {@link #getTextCharacters getTextCharacters} method.
295 	 */
296 	public static final int COMMENT = 
297 		XMLStreamConstants.COMMENT;
298 
299 	/**
300 	 * Returns next DTD parsing event.
301 	 *
302 	 * @return the integer code corresponding to the current parse event
303 	 * @throws NoSuchElementException if this is called when 
304 	 *   {@link #hasNext() hasNext()} returns <code>false</code>
305 	 * @throws XMLStreamException if there is an error processing the 
306 	 *   underlying DTD source
307 	 */
308 	public int next() throws XMLStreamException;
309 
310 	/**
311 	 * Returns <code>true</code> if there are more parsing events and 
312 	 * <code>false</code> if there are no more events.  This method will return
313 	 * <code>false</code> if the current state of the DTDStreamReader is
314 	 * {@link #END_DTD END_DTD}
315 	 *
316 	 * @return <code>true</code> if there are more events, <code>false</code> 
317 	 *   otherwise
318 	 * @throws XMLStreamException if there is a fatal error detecting the next 
319 	 *   state
320 	 */
321 	public boolean hasNext() throws XMLStreamException;
322 
323 	/**
324 	 * Returns an integer code that indicates the type of the event at
325 	 * the current cursor location.
326 	 *
327 	 * @return the integer code corresponding to the current parse event
328 	 */
329 	public int getEventType();
330 
331 	/**
332 	 * Returns the current value of the parse event as a string. 
333 	 * This returns the value of a {@link #COMMENT COMMENT} or the replacement 
334 	 * value for an {@link #ENTITY_DECLARATION ENTITY_DECLARATION}. This 
335 	 * method returns <code>null</code> if there is no text available.
336 	 *
337 	 * @return the current text or <code>null</code> if there is no text available
338 	 * @throws java.lang.IllegalStateException if this state is not a valid
339 	 *   text state.
340 	 */
341 	public String getText();
342 
343 	/**
344 	 * Returns an array which contains the characters from this event.
345 	 * This array should be treated as read-only and transient: the array 
346 	 * will contain the text characters only until the DTDStreamReader
347 	 * moves on to the next event.
348 	 * Attempts to hold onto the character array beyond that time or 
349 	 * modify the contents of the array are breaches of the contract for this 
350 	 * interface.
351 	 *
352 	 * @return the current text or an empty array
353 	 * @throws java.lang.IllegalStateException if this state is not a valid 
354 	 *   text state.
355 	 */
356 	public char[] getTextCharacters();
357 
358 	/**
359 	 * Returns the offset into the text character array at which the first
360 	 * character of the data for the current event is located.
361 	 *
362 	 * @return the offset of the first character
363 	 * @throws java.lang.IllegalStateException if this state is not a valid
364 	 *   text state.
365 	 */
366 	public int getTextStart();
367 
368 	/**
369 	 * Returns the length of the sequence of characters for the current event 
370 	 * within the text character array. This method returns <code>-1</code> 
371 	 * if there is no text available.
372 	 *
373 	 * @return the length of the character sequence for current event or 
374 	 *   <code>-1</code>
375 	 * @throws java.lang.IllegalStateException if this state is not a valid
376 	 *   text state.
377 	 */
378 	public int getTextLength();
379 
380 	/**
381 	 * Returns the current location of the processor.
382 	 * If the location is unknown the processor should return an 
383 	 * implementation 
384 	 * of {@link Location Location} that returns <code>-1</code> for
385 	 * each of the lineNumber, columnNumber, and characterOffset,
386 	 * and <code>null</code> for each of the publicId and systemId. 
387 	 * The location information is only valid until {@link #next() next()}
388 	 * is called.
389 	 *
390 	 * @return the {@link Location Location} object
391 	 */
392 	public Location getLocation();
393 
394 	/**
395 	 * Returns the target of a processing instruction.
396 	 *
397 	 * @return the target or <code>null</code>
398 	 * @throws java.lang.IllegalStateException if this method is not valid in
399 	 * the current state.
400 	 */
401 	public String getPITarget();
402 
403 	/**
404 	 * Returns the data section of a processing instruction.
405 	 *
406 	 * @return the data or <code>null</code>
407 	 * @throws java.lang.IllegalStateException if this method is not valid in
408 	 * the current state.
409 	 */
410 	public String getPIData();
411 
412 	/**
413 	 * Returns the name as a String.
414 	 *
415 	 * @return the name
416 	 * @throws java.lang.IllegalStateException if this state is not a valid
417 	 *   text state.
418 	 */
419 	public String getName();
420 
421 	/**
422 	 * Returns the public identifier.
423 	 *
424 	 * @return the public identifier, or null if not available
425 	 * @throws java.lang.IllegalStateException if this method is not valid in
426 	 * the current state.
427 	 */
428 	public String getPublicIdentifier();
429 
430 	/**
431 	 * Returns the system identifier.
432 	 *
433 	 * @return the system identifier, or null if not available
434 	 * @throws java.lang.IllegalStateException if this method is not valid in
435 	 * the current state.
436 	 */
437 	public String getSystemIdentifier();
438 
439 	/**
440 	 * Returns the notation name.
441 	 *
442 	 * @return the notation name
443 	 * @throws java.lang.IllegalStateException if this method is not valid in
444 	 * the current state.
445 	 */
446 	public String getNotationName();
447 
448 	/**
449 	 * Terminates DTD processing, skipping all DTD related events up 
450 	 * to {@link #END_DTD END_DTD}. This method does not close the
451 	 * underlying input source.
452 	 * If this method is called when current state is already 
453 	 * {@link #END_DTD END_DTD}, this method does nothing. 
454 	 * Once this method has been invoked,
455 	 * method {@link #hasNext() hasNext()} returns <code>false</code>, 
456 	 * method {@link #getEventType() getEventType()} returns 
457 	 * {@link #END_DTD END_DTD} and any other method call except 
458 	 * {@link #next() next()} or {@link #getLocation() getLocation()}
459 	 * on this interface generates an
460 	 * <code>java.lang.IllegalStateException</code>.
461 	 *
462 	 * @throws XMLStreamException if there is an error processing the
463 	 *   underlying DTD source
464 	 */
465 	public void close() throws XMLStreamException;
466 }
