31  /**
32   * Defines a factory API that enables applications to configure and
33   * obtain a SAX based parser to parse XML documents.<p>
34   * An implementation of the <code>SAXParserFactory</code> class is
35   * <em>NOT</em> guaranteed to be thread safe. It is up to the user application 
36   * to make sure about the use of the <code>SAXParserFactory</code> from 
37   * more than one thread. Alternatively the application can have one instance 
38   * of the <code>SAXParserFactory</code> per thread.
39   * An application can use the same instance of the factory to obtain one or 
40   * more instances of the <code>SAXParser</code> provided the instance
41   * of the factory isn't being used in more than one thread at a time.
42   * <p>
43   *
44   * The static <code>newInstance</code> method returns a new concrete 
45   * implementation of this class.
46   *
47   * @since JAXP 1.0
48   */
49  
50  public abstract class SAXParserFactory
51  {
52  
53      private boolean namespaceAware = false;
54  
55      private boolean validating = false;
56  
57      protected SAXParserFactory()
58      {
59      }
60  
61      /**
62       * Obtain a new instance of a <code>SAXParserFactory</code>. This
63       * static method creates a new factory instance
64       * This method uses the following ordered lookup procedure to determine
65       * the <code>SAXParserFactory</code> implementation class to
66       * load:
67       * <ul>
68       * <li>
69       * Use the <code>javax.xml.parsers.SAXParserFactory</code> system
70       * property.
71       * </li>
72       * <li>
73       * Platform default <code>SAXParserFactory</code> instance.
74       * </li>
75       * </ul>
76       *
77       * Once an application has obtained a reference to a
78       * <code>SAXParserFactory</code> it can use the factory to
79       * configure and obtain parser instances.
80       *
81       * @return A new instance of a SAXParserFactory.
82       *
83       * @exception FactoryConfigurationError if the implementation is
84       * not available or cannot be instantiated.
85       */
86  
87      public static SAXParserFactory newInstance()
88          throws FactoryConfigurationError
89      {
90          try {
91              String fname = System.getProperty("javax.xml.parsers.SAXParserFactory");
92              if(fname == null)
93                  fname = "com.sun.ukit.xml.SAXParserFactoryImp";
94              return (SAXParserFactory) Class.forName(fname).newInstance();
95          } catch (java.lang.Throwable ex) {
96              throw new FactoryConfigurationError("");
97          }
98      }
