25  package javax.xml.parsers;

27  import java.io.IOException;
28  import java.io.InputStream;
29  
30  import org.xml.sax.*;
31  import org.xml.sax.helpers.DefaultHandler;

34  /**
35   * Defines the API that represents a simple SAX parser.
36   *
37   * An instance of this class can be obtained from the
38   * {@link javax.xml.parsers.SAXParserFactory#newSAXParser()} method.
39   * Once an instance of this class is obtained, XML can be parsed from
40   * an InputStream<p>
41   *
42   * As the content is parsed by the underlying parser, methods of the
43   * given {@link org.xml.sax.helpers.DefaultHandler} are called.<p>
44   *
45   * An implementation of <code>SAXParser</code> is <em>NOT</em> 
46   * guaranteed to behave as per the specification if it is used concurrently by 
47   * two or more threads. It is recommended to have one instance of the
48   * <code>SAXParser</code> per thread or it is upto the application to 
49   * make sure about the use of <code>SAXParser</code> from more than one
50   * thread.
51   *
52   * @since JAXP 1.0
53   */
54  
55  public abstract class SAXParser {
56  
57      protected SAXParser()
58      {
59      }
60  
61  
62      /**
63       * Parse the content of the given {@link java.io.InputStream}
64       * instance as XML using the specified
65       * {@link org.xml.sax.helpers.DefaultHandler}.
66       *
67       * @param is InputStream containing the content to be parsed.
68       * @param dh The SAX DefaultHandler to use.
69       * @exception IOException If any IO errors occur.
70       * @exception IllegalArgumentException If the given InputStream is null.
71       * @exception SAXException If the underlying parser throws a
72       * SAXException while parsing.
73       */
74  
75      public abstract void parse(InputStream is, DefaultHandler dh)
76          throws SAXException, IOException;
77  
78  
79      /**
80       * Parse the content given {@link org.xml.sax.InputSource}
81       * as XML using the specified
82       * {@link org.xml.sax.helpers.DefaultHandler}.
83       *
84       * @param is The InputSource containing the content to be parsed.
85       * @param dh The SAX DefaultHandler to use.
86       * @exception IOException If any IO errors occur.
87       * @exception IllegalArgumentException If the InputSource is null.
88       * @exception SAXException If the underlying parser throws a
89       * SAXException while parsing.
90       * @see org.xml.sax.helpers.DefaultHandler
91       */
92  
93      public abstract void parse(InputSource is, DefaultHandler dh)
94          throws SAXException, IOException;
95  
96  
97      /**
98       * Returns the {@link org.xml.sax.XMLReader} that is encapsulated by the
99       * implementation of this class.
100      *
101      * @return The XMLReader that is encapsulated by the
102      *         implementation of this class.
103      * 
104      * @throws SAXException If any SAX errors occur during processing.
105      */
106 
107     public abstract org.xml.sax.XMLReader getXMLReader()
108         throws SAXException;
109 
110 
111     /**
112      * Indicates whether or not this parser is configured to
113      * understand namespaces.
114      *
115      * @return <code>true</code> if this parser is configured to
116      *         understand namespaces; <code>false</code> otherwise.
117      */
118     
119     public abstract boolean isNamespaceAware();
120 
121 
122     /**
123      * Indicates whether or not this parser is configured to validate
124      * XML documents.
125      *
126      * @return <code>true</code> if this parser is configured to validate XML
127      *          documents; <code>false</code> otherwise.
128      */
129 
130     public abstract boolean isValidating();

132 }
