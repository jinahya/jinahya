42  /** 
43   * Abstracts a single stream of media data.
44   *
45   * <h2>Stream Controls</h2>
46   *
47   * A <code>SourceStream</code> might support an operation that
48   * is not part of the <code>SourceStream</code> definition.
49   * For example a stream might support seeking to a particular byte
50   * in the stream. Some operations are dependent on the stream
51   * data, and support cannot be determined until the stream is in
52   * use.
53   * <p>
54   *
55   * To obtain all of the objects that provide control over a stream
56   * use <code>getControls</code>. To determine if a particular
57   * kind of control is available, and obtain the object that
58   * implements the control use <code>getControl</code>.
59   *
60   * 
61   * @see DataSource
62   * @see PushSourceStream
63   * @see PullSourceStream
64   * @see Seekable
65   *
66   * @version 1.12, 97/08/28.
67   */
68  public interface SourceStream extends Controls
69  {
70      public static final long LENGTH_UNKNOWN = -1L;
71  
72      /** 
73       * Get the current content type for this stream.
74       *
75       * @return The current <CODE>ContentDescriptor</CODE> for this stream.
76       */
77      public ContentDescriptor getContentDescriptor();
78  
79      /** 
80       * Get the size, in bytes, of the content on this stream.
81       * LENGTH_UNKNOWN is returned if the length is not known.
82       *
83       * @return The content length in bytes.
84       */
85      public long getContentLength();
86  
87      /** 
88       * Find out if the end of the stream has been reached.
89       *
90       * @return Returns <CODE>true</CODE> if there is no more data.
91       */
92      public boolean endOfStream();
93  }
