38  package javax.tv.xlet;

40  /** 
41   * An interface that provides methods allowing an Xlet to discover
42   * information about its environment. An XletContext is passed
43   * to an Xlet when the Xlet is initialized. It provides an Xlet with a
44   * mechanism to retrieve properties, as well as a way to signal
45   * internal state changes.
46   * <p>
47   * Critical resources (such as an Xlet's parent container and service
48   * context) can be obtained by means of an <code>XletContext</code>
49   * instance.  Therefore, an Xlet's <code>XletContext</code> instance
50   * should only be accessible to other code that is highly trusted.
51   *
52   * @see javax.tv.xlet.Xlet
53   * @see javax.tv.graphics.TVContainer
54   * @see javax.tv.service.selection.ServiceContextFactory#getServiceContext
55   */
56  public interface XletContext
57  {
58      /** 
59       * The property key used to obtain initialization arguments for the
60       * Xlet.  The call
61       * <code>XletContext.getXletProperty(XletContext.ARGS)</code> will
62       * return the arguments as an array of Strings.  If there are
63       * no arguments, then an array of length 0 will be returned.
64       *
65       * @see #getXletProperty
66       */
67      public static final String ARGS = "javax.tv.xlet.args";
68  
69      /** 
70       * Used by an application to notify its manager that it
71       * has entered into the
72       * <i>Destroyed</i> state.  The application manager will not
73       * call the Xlet's <code>destroy</code> method, and all resources
74       * held by the Xlet are considered eligible for immediate reclamation.  
75       * Before calling this method,
76       * the Xlet must perform the same operations
77       * (clean up, releasing of resources etc.) as it would in response
78       * to call to {@link Xlet#destroyXlet}.
79       * <p>
80       * If this method is called during the execution of one of the
81       * {@link Xlet} state transition methods, the Xlet will immediately
82       * transition into the <i>Destroyed</i> state and the Xlet state
83       * transition method is considered to have completed
84       * <em>unsuccesfully</em>.
85       */
86      public void notifyDestroyed();
87  
88      /** 
89       * Notifies the manager that the Xlet does not want to be active and has
90       * entered the <i>Paused</i> state.  Invoking this method will
91       * have no effect if the Xlet is destroyed, or if it has not
92       * yet been started.
93       * <p>
94       * If an Xlet calls <code>notifyPaused()</code>, in the
95       * future it may receive an <i>Xlet.startXlet()</i> call to request
96       * it to become active, or an <i>Xlet.destroyXlet()</i> call to request
97       * it to destroy itself.
98       * 
99       */
100     public void notifyPaused();
101 
102     /** 
103      * Provides an Xlet with a mechanism to retrieve named
104      * properties from the XletContext.
105      *
106      * @param key The name of the property.
107      *
108      * @return A reference to an object representing the property.
109      * <code>null</code> is returned if no value is available for key.
110      */
111     public Object getXletProperty(String key);
112 
113     /** 
114      * Provides the Xlet with a mechanism to indicate that it is
115      * interested in entering the <i>Active</i> state. Calls to this
116      * method can be used by an application manager to determine which
117      * Xlets to move to <i>Active</i> state.  Any subsequent call to
118      * <code>Xlet.startXlet()</code> as a result of this method will
119      * be made via a different thread than the one used to call
120      * <code>resumeRequest()</code>.
121      * 
122      * @see Xlet#startXlet
123      */
124     public void resumeRequest();
125 }
