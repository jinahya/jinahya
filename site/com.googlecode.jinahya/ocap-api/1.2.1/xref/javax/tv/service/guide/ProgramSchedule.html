48  /** 
49   * This interface represents a collection of program events for a given
50   * service ordered by time. It provides the current, next and future
51   * program events.<p>
52   *
53   * Note that all time values are in UTC time.
54   *
55   * @see java.util.Date java.util.Date
56   * @see javax.tv.service.guide.ProgramEvent
57   * @see javax.tv.service.ReadPermission
58   */
59  public interface ProgramSchedule
60  {
61  
62      /** 
63       * Retrieves the current <code>ProgramEvent</code>.  The resulting
64       * <code>ProgramEvent</code> is available for immediate viewing.<p>
65       *
66       * This method delivers its results asynchronously.  If the caller
67       * does not have
68       * <code>javax.tv.service.ReadPermission(pe.getLocator())</code>
69       * (where <code>pe</code> is the current program event), this
70       * method will result in an <code>SIRequestFailureType</code> of
71       * <code>DATA_UNAVAILABLE</code>.
72       *
73       * @param requestor The <code>SIRequestor</code> to be notified
74       * when this retrieval operation completes.
75       * 
76       * @return An <code>SIRequest</code> object identifying this
77       * asynchronous retrieval request.
78       *
79       * @see ProgramEvent
80       * @see javax.tv.service.ReadPermission
81       */
82      public SIRequest retrieveCurrentProgramEvent(javax.tv.service.SIRequestor
83          requestor);
84  
85      /** 
86       * Retrieves the program event for the specified time.  The
87       * specified time will fall between the resulting program event's
88       * start time (inclusive) and end time (exclusive).<p>
89       *
90       * This method delivers its results asynchronously.  If the caller
91       * does not have
92       * <code>javax.tv.service.ReadPermission(pe.getLocator())</code>
93       * (where <code>pe</code> is the program event at the specified
94       * time), this method will result in an
95       * <code>SIRequestFailureType</code> of
96       * <code>DATA_UNAVAILABLE</code>.
97       *
98       * @param time The time of the program event to be retrieved.
99       *	
100      * @param requestor The <code>SIRequestor</code> to be notified
101      * when this retrieval operation completes.
102      * 
103      * @return An <code>SIRequest</code> object identifying this
104      * asynchronous retrieval request.
105      *
106      * @throws SIException If <code>time</code> does not represent a
107      * future time value.
108      *
109      * @see ProgramEvent
110      * @see javax.tv.service.ReadPermission
111      */
112     public SIRequest retrieveFutureProgramEvent(Date time,
113         javax.tv.service.SIRequestor requestor) throws SIException;
114 
115     /** 
116      * Retrieves all known program events on this service for the
117      * specified time interval.  A program event <code>pe</code> is
118      * retrieved by this method if the time interval from
119      * <code>pe.getStartTime()</code> (inclusive) to
120      * <code>pe.getEndTime()</code> (exclusive) intersects the time
121      * interval from <code>begin</code> (inclusive) to <code>end</code>
122      * (exclusive) specified by the input parameters.<p>
123      *
124      * This method returns data asynchronously.  Only program events
125      * <code>pe</code> for which the caller has
126      * <code>javax.tv.service.ReadPermission(pe.getLocator())</code>
127      * will be retrieved.  If no program events meet this criteria,
128      * this method will result in an <code>SIRequestFailureType</code> of
129      * <code>DATA_UNAVAILABLE</code>.
130      *
131      * @param begin Time identifying the beginning of the interval.
132      *
133      * @param end Time identifying the end of the interval.
134      *
135      * @param requestor The <code>SIRequestor</code> to be notified
136      * when this retrieval operation completes.
137      * 
138      * @return An <code>SIRequest</code> object identifying this
139      * asynchronous retrieval request.
140      *
141      * @throws SIException If <code>end</code> represents a time value
142      * before <code>begin</code>, or if <code>end</code> does not
143      * represent a future time value.
144      *
145      * @see ProgramEvent
146      * @see javax.tv.service.ReadPermission
147      */
148     public SIRequest retrieveFutureProgramEvents(Date begin, Date end,
149         javax.tv.service.SIRequestor requestor) throws SIException;
150 
151     /** 
152      * Retrieves a event which follows the specified event.<p>
153      *
154      * This method delivers its results asynchronously.  If the caller
155      * does not have
156      * <code>javax.tv.service.ReadPermission(pe.getLocator())</code>
157      * (where <code>pe</code> is the next program event), this
158      * method will result in an <code>SIRequestFailureType</code> of
159      * <code>DATA_UNAVAILABLE</code>.
160      *
161      * @param event A reference event.
162      *
163      * @param requestor The <code>SIRequestor</code> to be notified
164      * when this retrieval operation completes.
165      * 
166      * @return An <code>SIRequest</code> object identifying this
167      * asynchronous retrieval request.
168      *
169      * @throws SIException If <code>event</code> does not belong to this
170      * <code>ProgramSchedule</code>.
171      *
172      * @see ProgramEvent
173      * @see javax.tv.service.ReadPermission
174      */
175     public SIRequest retrieveNextProgramEvent(ProgramEvent event,
176         javax.tv.service.SIRequestor requestor) throws SIException;
177 
178     /** 
179      * Retrieves a program event matching the locator. Note that
180      * the event must be part of this schedule.<p>
181      * 
182      * This method returns data asynchronously.
183      *
184      * @param locator Locator referencing the <code>ProgramEvent</code>
185      * of interest.
186      *
187      * @param requestor The <code>SIRequestor</code> to be notified
188      * when this retrieval operation completes.
189      * 
190      * @return An <code>SIRequest</code> object identifying this
191      * asynchronous retrieval request.
192      *
193      * @throws InvalidLocatorException If <code>locator</code> does not
194      * reference a valid <code>ProgramEvent</code> in this
195      * <code>ProgramSchedule</code>.
196      *
197      * @throws SecurityException If the caller does not have
198      * <code>javax.tv.service.ReadPermission(locator)</code>.
199      *
200      * @see ProgramEvent
201      * @see javax.tv.service.ReadPermission
202      */
203     public SIRequest retrieveProgramEvent(Locator locator,
204         javax.tv.service.SIRequestor requestor)
205         throws InvalidLocatorException, SecurityException;

207     /** 
208      * Registers a <code>ProgramScheduleListener</code> to be notified of
209      * changes to program events on this <code>ProgramSchedule</code>.
210      * Subsequent changes will be indicated through instances of
211      * <code>ProgramScheduleEvent</code>, with this
212      * <code>ProgramSchedule</code> as the event source and an
213      * <code>SIChangeType</code> of <code>ADD</code>,
214      * <code>REMOVE</code>, <code>MODIFY</code>, or
215      * <code>CURRENT_PROGRAM_EVENT</code>.  Only changes to

220      *
221      * This method is only a request for notification.  No guarantee is
222      * provided that the SI database will detect all, or even any,
223      * changes to the <code>ProgramSchedule</code>, or whether such changes
224      * will be detected in a timely fashion.<p>
225      * 
226      * If the specified <code>ProgramScheduleListener</code> is already
227      * registered, no action is performed.
228      *
229      * @param listener A <code>ProgramScheduleListener</code> to be notified of
230      * changes to program events on this <code>ProgramSchedule</code>.
231      *
232      * @see ProgramEvent
233      * @see ProgramScheduleEvent
234      * @see ProgramScheduleChangeType
235      * @see javax.tv.service.ReadPermission
236      */
237     public void addListener(ProgramScheduleListener listener);

239     /** 
240      * Unregisters a <code>ProgramScheduleListener</code>.  If the

242      * action is performed.
243      *
244      * @param listener A previously registered listener.
245      */
246     public void removeListener(ProgramScheduleListener listener);
