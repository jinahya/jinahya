40  /** 
41   * <code>PresentationTerminatedEvent</code> is generated when the
42   * presentation of a service terminates.  This includes both normal
43   * termination (e.g., due to an application calling the <code>stop()</code>
44   * method) and abnormal termination (e.g., due to some change in the
45   * environment). Examples of abnormal termination include:
46   *
47   * <ul>
48   * <li>a tuning operation making the service unavailable </li>
49   *
50   * <li>removal of fundamental resources required to present the
51   * service</li>
52   *
53   * <li>withdrawal of CA authorization</li>
54   * </ul>
55   *
56   * <code>PresentationTerminatedEvent</code> is also generated
57   * following a <code>SelectionFailedEvent</code> either if the service
58   * context was not previously in the <em>presenting</em> state or if
59   * recovery of what was being presented previously is not possible.
60   * <code>PresentationTerminatedEvent</code> is only generated when no
61   * components of the requested service can be presented.
62   * <p>
63   * When a <code>PresentationTerminatedEvent</code> is
64   * generated following a failed selection attempt, the
65   * reason code of the <code>PresentationTerminatedEvent</code>
66   * is derived from the reason code for the
67   * <code>SelectionFailedEvent</code> which preceded it, according
68   * to the table below.
69   * <p>
70   * <table align="center" width="80%" border="1" cellspacing="2" cellpadding="2">
71   * <tbody>
72   * <tr>
73   * <th align="center">SelectionFailedEvent reason code</th>
74   * <th align="center">PresentationTerminatedEvent reason code</th>
75   * </tr>
76   * 
77   * <tr>
78   * <td>CA_REFUSAL</td>
79   * <td>ACCESS_WITHDRAWN</td>
80   * </tr>
81   * 
82   * <tr>
83   * <td>CONTENT_NOT_FOUND</td>
84   * <td>SERVICE_VANISHED</td>
85   * </tr>
86   * 
87   * <tr>
88   * <td>INSUFFICIENT_RESOURCES</td>
89   * <td>RESOURCES_REMOVED</td>
90   * </tr>
91   * 
92   * <tr>
93   * <td>MISSING_HANDLER</td>
94   * <td>RESOURCES_REMOVED</td>
95   * </tr>
96   * 
97   * <tr>
98   * <td>TUNING_FAILURE</td>
99   * <td>TUNED_AWAY</td>
100  * </tr>
101  * 
102  * <tr>
103  * <td>OTHER</td>
104  * <td>OTHER</td>
105  * </tr>
106  * 
107  * </tbody>
108  * </table>
109  * <p>
110  * (No reason code corresponding to
111  * <code>SelectionFailedEvent.INTERRUPTED</code> is necessary, since this
112  * code signals that a new service selection operation is underway.)
113  * <p>
114  * Once a <code>PresentationTerminatedEvent</code> is generated, the
115  * <code>ServiceContext</code>
116  * will be in the <em>not presenting</em> state until a call to a
117  * <code>select()</code> method succeeds.  When this event is
118  * generated, all resources used for the presentation have been
119  * released, and <code>ServiceContentHandler</code> instances
120  * previously associated with the <code>ServiceContext</code> will
121  * have ceased presentation of their content.
122  *
123  * @see SelectionFailedEvent
124  */
125 public class PresentationTerminatedEvent extends ServiceContextEvent
126 {
127     /** 
128      * Reason code : The service vanished from the network.
129      */
130     public static final int SERVICE_VANISHED = 1;
131 
132     /** 
133      * Reason code : Tuning made the service unavailable.
134      */
135     public static final int TUNED_AWAY = 2;
136 
137     /** 
138      * Reason code : Resources needed to present the service have been removed.
139      */
140     public static final int RESOURCES_REMOVED = 3;
141 
142     /** 
143      * Reason code : Access to the service or some component of it has been
144      * withdrawn by the system. An example of this is the end of a free 
145      * preview period for IPPV content. 
146      */
147     public static final int ACCESS_WITHDRAWN = 4;
148 
149     /** 
150      * Reason code : The application or user requested that the
151      * presentation be stopped. 
152      */
153     public static final int USER_STOP = 5;
154 
155     /** 
156      * Reason code: The presentation was terminated due to an unknown reason
157      * or for multiple reasons.
158      */
159     public static final int OTHER = 255;
160 
161     /** 
162      * Constructs the event with a reason code.
163      *
164      * @param source The <code>ServiceContext</code> that generated the event.
165      * @param reason The reason for which the presentation was terminated.
166      *
167      */
168     public PresentationTerminatedEvent(ServiceContext source, int reason) { 
169         super(source);    
170     }
171 
172     /** 
173      * Reports the reason for which the presentation was terminated.
174      *
175      * @return A reason code for why the presentation was terminated.
176      */
177     public int getReason() {
178         return 0;
179     }
180 }
