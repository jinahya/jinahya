38  package javax.tv.service.guide;

40  import javax.tv.service.SIException;

42  /** 
43   * ContentRatingAdvisory indicates, for a given program event, ratings
44   * for any or all of the rating dimensions defined in the content
45   * rating system for the local rating region. A program event without
46   * a content advisory indicates that the rating value for any rating
47   * dimension is zero. The absence of ratings for a specific dimension
48   * is equivalent to having a zero-valued rating for such a
49   * dimension. The absence of ratings for a specific region implies the
50   * absence of ratings for all the dimensions in the region. <P>
51   *
52   * For example, this information may be obtained in the ATSC Content
53   * Advisory Descriptor or the DVB Parental Rating Descriptor. Note
54   * that the DVB rating system is based on age only. It can be easily
55   * mapped to this rating system as one of the dimensions.
56   *
57   * @see javax.tv.service.guide.ProgramEvent 
58   */
59  public interface ContentRatingAdvisory
60  {
61  
62      /** 
63       * Returns a list of names of all dimensions in this rating
64       * region by which the <code>ProgramEvent</code> is rated.
65       *
66       * @return An array of strings representing all rated dimensions in this
67       * rating region for the <code>ProgramEvent</code>.
68       *
69       * @see javax.tv.service.RatingDimension
70       */
71      public String[] getDimensionNames();
72  
73      /** 
74       * Returns a number representing the rating level in the specified
75       * <code>RatingDimension</code> associated with this rating region
76       * for the related <code>ProgramEvent</code>.
77       *
78       * @param dimensionName The name of the <code>RatingDimension</code>
79       * for which to obtain the rating level.
80       *
81       * @return A number representing the rating level. The meaning is
82       * dependent on the associated rating dimension.
83       *
84       * @throws SIException If <code>dimensionName</code> is not a valid
85       * name of a <code>RatingDimension</code> for the ProgramEvent.
86       *
87       * @see javax.tv.service.RatingDimension#getDimensionName 
88       */
89      public short getRatingLevel(String dimensionName) throws SIException;
90  
91      /** 
92       * Returns the rating level display string for the specified
93       * dimension.  The string is identical to
94       * <code>d.getRatingLevelDescription(getRatingLevel(dimensionName))[1]</code>,
95       * where <code>d</code> is the <code>RatingDimension</code> obtained
96       * by
97       * <code>javax.tv.service.SIManager.getRatingDimension(dimensionName)</code>.
98       *
99       * @param dimensionName The name of the <code>RatingDimension</code>
100      * for which to obtain the rating level text.
101      *
102      * @return A string representing the textual value of this rating level.
103      *
104      * @throws SIException If dimensionName is not a valid
105      * <code>RatingDimension</code> name for the <code>ProgramEvent</code>.
106      *
107      * @see javax.tv.service.RatingDimension#getDimensionName
108      * @see javax.tv.service.RatingDimension#getRatingLevelDescription
109      */
110     public String getRatingLevelText(String dimensionName) throws SIException;
111 
112     /** 
113      * Provides a single string representing textual rating values for all
114      * dimensions in which the program event is rated.
115      * The result will be a representation of the strings obtained via
116      * <code>d.getRatingLevelDescription(getRatingLevel(d.getDimensionName()))[0]</code>,
117      * for all dimensions <code>d</code> obtained through
118      * <code>javax.tv.service.SIManager.getRatingDimension(n)</code>,
119      * for all dimension names <code>n</code> obtained from
120      * <code>getDimensionNames()</code>.
121      * 
122      * @return A string representing the rating level values for all
123      * dimensions in which this program event is rated.  The format of
124      * the string may be implementation-specific.
125      *
126      * @see #getDimensionNames
127      * @see javax.tv.service.RatingDimension#getRatingLevelDescription
128      */
129     public String getDisplayText();
130 
131     /** 
132      * Compares the current rating value with the system rating
133      * ceiling.  The rating ceiling is set in a system-dependent manner.
134      * Content that exceeds the rating ceiling cannot be displayed.
135      *
136      * @return <code>true</code> if the rating exceeds the current
137      * system rating ceiling; <code>false</code> otherwise.  
138      */
139     public boolean exceeds();
140 }
