38  package javax.tv.media;

40  import java.awt.Rectangle;
41  import java.awt.Dimension;

43  /** 
44   * <code>AWTVideoSizeControl</code> allows setting clipping, scaling, and
45   * translation of a video stream in a simple, interoperable way.  Not
46   * all possible combinations of positioning will be supported, so this
47   * interface provides a mechanism to discover how closely the
48   * underlying platform will approximate a request for positioning.
49   * 
50   * <p> All interactions via AWTVideoSizeControl happen in the
51   * coordinate space of the screen.  For example, successfully setting
52   * the video's position to the location reported by
53   * <code>Component.getLocationOnScreen()</code> on the
54   * <code>Xlet</code>'s root container will cause the upper left-hand
55   * corner of the video and the root container to coincide.  <p> The
56   * screen, in the context of AWT, is the area into which graphics
57   * drawing operations are done.  Its size is given by
58   * java.awt.Toolkit.getScreenSize(), and locations reported by
59   * Component.getLocationOnScreen() are given in the screen's
60   * coordinate system.<p>
61   * 
62   * Instances of <code>AWTVideoSizeControl</code> may be obtained from
63   * a JMF <code>Player</code> via the methods
64   * <code>getControl(String)</code> and <code>getControls()</code>.
65   * Note that a Java TV API implementation may not always or ever
66   * support <code>AWTVideoSizeControl</code> for a given Player; in
67   * such a case, the failure modes specified by the two aforementioned
68   * methods will apply.
69   * 
70   * @version     1.16, 10/09/00
71   * @author      Bill Foote
72   *
73   * @see javax.tv.media.AWTVideoSize
74   * @see java.awt.Component#getLocationOnScreen java.awt.Component.getLocationOnScreen()
75   * @see javax.media.Player
76   */
77  public interface AWTVideoSizeControl extends javax.media.Control
78  {
79  
80      /** 
81       * Reports the <code>AWTVideoSize</code> at which the Player is
82       * currently operating.
83       *
84       * @return A copy of the JMF Player's current video size, in the AWT
85       * coordinate space.  
86       */
87      public AWTVideoSize getSize();
88  
89      /** 
90       * Reports the default <code>AWTVideoSize</code> for this control.
91       * For the background video plane, this will be the size that the
92       * video would be presented at if no program had manipulated the
93       * video size.
94       *
95       * @return The default <code>AWTVideoSize</code>.
96       */
97      public AWTVideoSize getDefaultSize();
98  
99      /** 
100      * Reports the size of the source video, in the screen's
101      * coordinate system.
102      *
103      * @return The size of the source video.
104      */
105     public Dimension getSourceVideoSize();
106 
107     /** 
108      * Sets the video size.  If the size provided cannot be supported
109      * by the underlying platform, this method does nothing and
110      * returns <code>false</code>.
111      *
112      * @param sz The desired video size, in the AWT coordinate space.
113      *
114      * @return <code>true</code> if the size was successfully changed;
115      * <code>false</code> if the platform is incapable of supporting
116      * the given size.
117      *
118      * @see #checkSize(AWTVideoSize) 
119      */
120     public boolean setSize(AWTVideoSize sz);
121 
122     /** 
123      * Reports how closely the underlying platform can approximate a
124      * desired video size.  If the underlying platform cannot support
125      * the given size, this method gives the closest approximation
126      * that the platform is capable of.
127      *
128      * @param sz The desired video size.
129      *
130      * @return The actual size that the platform would be able to set.  
131      */
132     public AWTVideoSize checkSize(AWTVideoSize sz);
133 }
