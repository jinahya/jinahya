47  package javax.net.ssl;
48  
49  import java.util.Enumeration;
50  
51  /** 
52   * A <code>SSLSessionContext</code> represents a set of
53   * <code>SSLSession</code>s associated with a single entity. For example,
54   * it could be associated with a server or client who participates in many
55   * sessions concurrently.
56   * <p>
57   * Not all environments will contain session contexts.
58   * <p>
59   * There are <code>SSLSessionContext</code> parameters that affect how
60   * sessions are stored:
61   * <UL>
62   *      <LI>Sessions can be set to expire after a specified
63   *      time limit.
64   *      <LI>The number of sessions that can be stored in context
65   *      can be limited.
66   * </UL>
67   * A session can be retrieved based on its session id, and all session id's
68   * in a <code>SSLSessionContext</code> can be listed.
69   *
70   * @see SSLSession
71   *
72   * @since 1.4
73   * @author Nathan Abramson
74   * @author David Brownell
75   * @version 1.12
76   */
77  public interface SSLSessionContext
78  {
79  
80      /** 
81       * Returns the <code>SSLSession</code> bound to the specified session id.
82       *
83       * @param sessionId the Session identifier
84       * @return the <code>SSLSession</code> or null if
85       * the specified session id does not refer to a valid SSLSession.
86       */
87      public SSLSession getSession(byte[] sessionId);
88  
89      /** 
90       * Returns an Enumeration of all session id's grouped under this
91       * <code>SSLSessionContext</code>.
92       *
93       * @return an enumeration of all the Session id's
94       */
95      public Enumeration getIds();
96  
97      /** 
98       * Sets the timeout limit for <code>SSLSession</code> objects grouped
99       * under this <code>SSLSessionContext</code>.
100      * <p>
101      * If the timeout limit is set to 't' seconds, a session exceeds the
102      * timeout limit 't' seconds after its creation time.
103      * When the timeout limit is exceeded for a session, the
104      * <code>SSLSession</code> object is invalidated and future connections
105      * cannot resume or rejoin the session.
106      * A check for sessions exceeding the timeout is made immediately whenever
107      * the timeout limit is changed for this <code>SSLSessionContext</code>.
108      *
109      * @param seconds the new session timeout limit in seconds; zero means
110      *		there is no limit.
111      *
112      * @exception IllegalArgumentException if the timeout specified is < 0.
113      * @see #getSessionTimeout
114      */
115     public void setSessionTimeout(int seconds) throws IllegalArgumentException;
116 
117     /** 
118      * Returns the timeout limit of <code>SSLSession</code> objects grouped
119      * under this <code>SSLSessionContext</code>.
120      * <p>
121      * If the timeout limit is set to 't' seconds, a session exceeds the
122      * timeout limit 't' seconds after its creation time.
123      * When the timeout limit is exceeded for a session, the
124      * <code>SSLSession</code> object is invalidated and future connections
125      * cannot resume or rejoin the session.
126      * A check for sessions exceeding the timeout limit is made immediately
127      * whenever the timeout limit is changed for this
128      * <code>SSLSessionContext</code>.
129      *
130      * @return the session timeout limit in seconds; zero means there is no
131      * limit.
132      * @see #setSessionTimeout
133      */
134     public int getSessionTimeout();
135 
136     /** 
137      * Sets the size of the cache used for storing
138      * <code>SSLSession</code> objects grouped under this
139      * <code>SSLSessionContext</code>.
140      *
141      * @param size the new session cache size limit; zero means there is no
142      * limit.
143      * @exception IllegalArgumentException if the specified size is < 0.
144      * @see #getSessionCacheSize
145      */
146     public void setSessionCacheSize(int size) throws IllegalArgumentException;
147 
148     /** 
149      * Returns the size of the cache used for storing
150      * <code>SSLSession</code> objects grouped under this
151      * <code>SSLSessionContext</code>.
152      *
153      * @return size of the session cache; zero means there is no size limit.
154      * @see #setSessionCacheSize
155      */
156     public int getSessionCacheSize();
157 }
