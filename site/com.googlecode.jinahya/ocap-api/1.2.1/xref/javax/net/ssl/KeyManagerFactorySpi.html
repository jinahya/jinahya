51  /** 
52   * This class defines the <i>Service Provider Interface</i> (<b>SPI</b>)
53   * for the <code>KeyManagerFactory</code> class.
54   *
55   * <p> All the abstract methods in this class must be implemented by each
56   * cryptographic service provider who wishes to supply the implementation
57   * of a particular key manager factory.
58   *
59   * @since 1.4
60   * @see KeyManagerFactory
61   * @see KeyManager
62   * @version 1.12
63   */
64  public abstract class KeyManagerFactorySpi
65  {
66  
67      public KeyManagerFactorySpi() { }
68  
69      /** 
70       * Initializes this factory with a source of key material.
71       *
72       * @param ks the key store or null
73       * @param password the password for recovering keys
74       * @throws KeyStoreException if this operation fails
75       * @throws NoSuchAlgorithmException if the specified algorithm is not
76       *		available from the specified provider.
77       * @throws UnrecoverableKeyException if the key cannot be recovered
78       * @see KeyManagerFactory#init(KeyStore, char[])
79       */
80      protected abstract void engineInit(KeyStore ks, char[] password)
81          throws KeyStoreException, NoSuchAlgorithmException,
82          UnrecoverableKeyException;
83  
84      /** 
85       * Initializes this factory with a source of key material.
86       * <P>
87       * In some cases, initialization parameters other than a keystore
88       * and password may be needed by a provider.  Users of that
89       * particular provider are expected to pass an implementation of
90       * the appropriate <CODE>ManagerFactoryParameters</CODE> as
91       * defined by the provider.  The provider can then call the
92       * specified methods in the ManagerFactoryParameters
93       * implementation to obtain the needed information.
94       *
95       * @param spec an implementation of a provider-specific parameter
96       *		specification
97       * @throws InvalidAlgorithmParameterException if there is problem
98       *		with the parameters
99       * @see KeyManagerFactory#init(ManagerFactoryParameters spec)
100      */
101     protected abstract void engineInit(ManagerFactoryParameters spec)
102         throws InvalidAlgorithmParameterException;
103 
104     /** 
105      * Returns one key manager for each type of key material.
106      *
107      * @return the key managers
108      */
109     protected abstract KeyManager[] engineGetKeyManagers();
