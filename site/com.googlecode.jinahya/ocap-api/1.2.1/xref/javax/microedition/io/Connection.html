37  package javax.microedition.io;
38  
39  import java.io.*;
40  
41  /** 
42   * This is the most basic type of generic connection. Only the
43   * <code>close</code> method is defined.  No open method is 
44   * defined here because opening is always done using the 
45   * <code>Connector.open()</code> methods.
46   *
47   * @author  Nik Shaylor
48   * @version 12/17/01 (CLDC 1.1)
49   * @since   CLDC 1.0
50   */
51  public interface Connection
52  {
53  
54      /** 
55       * Close the connection.
56       * <p>
57       * When a connection has been closed, access to any of its methods
58       * that involve an I/O operation will cause an <code>IOException</code>
59       * to be thrown.
60       * Closing an already closed connection has no effect. Streams 
61       * derived from the connection may be open when method is called.
62       * Any open streams will cause the connection to be held open
63       * until they themselves are closed. In this latter case access
64       * to the open streams is permitted, but access to the connection
65       * is not.
66       *
67       * @exception IOException  If an I/O error occurs
68       */
69      public void close() throws IOException;
70  }
