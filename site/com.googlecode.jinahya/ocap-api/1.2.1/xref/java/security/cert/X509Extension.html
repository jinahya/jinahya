41  /** 
42   * Interface for an X.509 extension.
43   *
44   * <p>The extensions defined for X.509 v3
45   * {@link X509Certificate Certificates} and v2
46   * {@link X509CRL CRLs} (Certificate Revocation
47   * Lists) provide methods
48   * for associating additional attributes with users or public keys,
49   * for managing the certification hierarchy, and for managing CRL
50   * distribution. The X.509 extensions format also allows communities
51   * to define private extensions to carry information unique to those
52   * communities.
53   *
54   * <p>Each extension in a certificate/CRL may be designated as
55   * critical or non-critical.  A certificate/CRL-using system (an application
56   * validating a certificate/CRL) must reject the certificate/CRL if it
57   * encounters a critical extension it does not recognize.  A non-critical
58   * extension may be ignored if it is not recognized.
59   * <p>
60   * The ASN.1 definition for this is:
61   * <pre>
62   * Extensions  ::=  SEQUENCE SIZE (1..MAX) OF Extension
63   *
64   * Extension  ::=  SEQUENCE  {
65   *     extnId        OBJECT IDENTIFIER,
66   *     critical      BOOLEAN DEFAULT FALSE,
67   *     extnValue     OCTET STRING
68   *                   -- contains a DER encoding of a value
69   *                   -- of the type registered for use with
70   *                   -- the extnId object identifier value
71   * }
72   * </pre>
73   * Since not all extensions are known, the <code>getExtensionValue</code>
74   * method returns the DER-encoded OCTET STRING of the
75   * extension value (i.e., the <code>extnValue</code>). This can then
76   * be handled by a <em>Class</em> that understands the extension.
77   *
78   * @author Hemma Prafullchandra
79   * @version 1.16 00/02/02
80   */
81  public interface X509Extension
82  {
83  
84      /** 
85       * Check if there is a critical extension that is not supported.
86       * 
87       * @return <tt>true</tt> if a critical extension is found that is
88       * not supported, otherwise <tt>false</tt>.
89       */
90      public boolean hasUnsupportedCriticalExtension();
91  
92      /** 
93       * Gets a Set of the OID strings for the extension(s) marked
94       * CRITICAL in the certificate/CRL managed by the object
95       * implementing this interface.
96       *
97       * Here is sample code to get a Set of critical extensions from an
98       * X509Certificate and print the OIDs:
99       * <pre><code>
100      * InputStream inStrm = new FileInputStream("DER-encoded-Cert");
101      * CertificateFactory cf = CertificateFactory.getInstance("X.509");
102      * X509Certificate cert = (X509Certificate)cf.generateCertificate(inStrm);
103      * inStrm.close();<p>
104      *
105      * Set critSet = cert.getCriticalExtensionOIDs();
106      * if (critSet != null && !critSet.isEmpty()) {
107      *     System.out.println("Set of critical extensions:");
108      *     for (Iterator i = critSet.iterator(); i.hasNext();) {
109      *         String oid = (String)i.next();
110      *         System.out.println(oid);
111      *     }
112      * }
113      * </code></pre>
114      * @return a Set (or an empty Set if none are marked critical) of
115      * the extension OID strings for extensions that are marked critical.
116      * If there are no extensions present at all, then this method returns
117      * null.
118      */
119     public Set getCriticalExtensionOIDs();
120 
121     /** 
122      * Gets a Set of the OID strings for the extension(s) marked
123      * NON-CRITICAL in the certificate/CRL managed by the object
124      * implementing this interface.
125      *
126      * Here is sample code to get a Set of non-critical extensions from an
127      * X509CRL revoked certificate entry and print the OIDs:
128      * <pre><code>
129      * InputStream inStrm = new FileInputStream("DER-encoded-CRL");
130      * CertificateFactory cf = CertificateFactory.getInstance("X.509");
131      * X509CRL crl = (X509CRL)cf.generateCRL(inStrm);
132      * inStrm.close();<p>
133      *
134      * byte[] certData = &lt;DER-encoded certificate data&gt;
135      * ByteArrayInputStream bais = new ByteArrayInputStream(certData);
136      * X509Certificate cert = (X509Certificate)cf.generateCertificate(bais);
137      * bais.close();
138      * X509CRLEntry badCert =
139      *              crl.getRevokedCertificate(cert.getSerialNumber());<p>
140      *
141      * if (badCert != null) {
142      *     Set nonCritSet = badCert.getNonCriticalExtensionOIDs();<p>
143      *     if (nonCritSet != null)
144      *         for (Iterator i = nonCritSet.iterator(); i.hasNext();) {
145      *             String oid = (String)i.next();
146      *             System.out.println(oid);
147      *         }
148      * }
149      * </code></pre>
150      *
151      * @return a Set (or an empty Set if none are marked non-critical) of
152      * the extension OID strings for extensions that are marked non-critical.
153      * If there are no extensions present at all, then this method returns
154      * null.
155      */
156     public Set getNonCriticalExtensionOIDs();

158     /** 
159      * Gets the DER-encoded OCTET string for the extension value

166      * <p>For example:<br>

181      * <tr><td>2.5.29.19</td>

183      * <tr><td>2.5.29.30</td>
184      * <td>NameConstraints</td></tr>
185      * <tr><td>2.5.29.33</td>
186      * <td>PolicyMappings</td></tr>
187      * <tr><td>2.5.29.35</td>
188      * <td>AuthorityKeyIdentifier</td></tr>
189      * <tr><td>2.5.29.36</td>
190      * <td>PolicyConstraints</td></tr>
191      * </table>
192      *
193      * @param oid the Object Identifier value for the extension.
194      * @return the DER-encoded octet string of the extension value or
195      * null if it is not present.
196      */
197     public byte[] getExtensionValue(String oid);
