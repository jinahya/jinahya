37  package java.security.interfaces;
38  
39  import java.security.*;

41  /** 
42   * An interface to an object capable of generating DSA key pairs. 
43   *
44   * <p>The <code>initialize</code> methods may each be called any number 
45   * of times. If no <code>initialize</code> method is called on a 
46   * DSAKeyPairGenerator, the default is to generate 1024-bit keys, using 
47   * precomputed p, q and g parameters and an instance of SecureRandom as 
48   * the random bit source.
49   * 
50   * <p>Users wishing to indicate DSA-specific parameters, and to generate a key 
51   * pair suitable for use with the DSA algorithm typically
52   * 
53   * <ol>
54   * 
55   * <li>Get a key pair generator for the DSA algorithm by calling the 
56   * KeyPairGenerator <code>getInstance</code> method with "DSA" 
57   * as its argument.<p> 
58   * 
59   * <li>Initialize the generator by casting the result to a DSAKeyPairGenerator
60   * and calling one of the 
61   * <code>initialize</code> methods from this DSAKeyPairGenerator interface.<p>
62   * 
63   * <li>Generate a key pair by calling the <code>generateKeyPair</code> 
64   * method from the KeyPairGenerator class.
65   * 
66   * </ol> 
67   *
68   * <p>Note: it is not always necessary to do do algorithm-specific
69   * initialization for a DSA key pair generator. That is, it is not always
70   * necessary to call an <code>initialize</code> method in this interface.
71   * Algorithm-independent initialization using the <code>initialize</code> method
72   * in the KeyPairGenerator
73   * interface is all that is needed when you accept defaults for algorithm-specific
74   * parameters.
75   * 
76   * @see java.security.KeyPairGenerator
77   */
78  public interface DSAKeyPairGenerator
79  {
80  
81      /** 
82       * Initializes the key pair generator using p, q and g, the DSA
83       * family parameters.
84       *
85       * @param params the parameters to use to generate the keys.
86       *
87       * @param random the random bit source to use to generate 
88       * key bits.
89       *
90       * @exception InvalidParameterException if the parameters passed are
91       * invalid or null.
92       */
93      public void initialize(DSAParams params, SecureRandom random)
94          throws InvalidParameterException;
95  
96      /** 
97       * Initializes the key pair generator for a given modulus length,
98       * without parameters. 
99       *
100      * <p>If <code>genParams</code> is true, this method will generate new 
101      * p, q and g parameters. If it is false, the method will use precomputed
102      * parameters for the modulus length requested. If there are no
103      * precomputed parameters for that modulus length, an exception will be 
104      * thrown. It is guaranteed that there will always be
105      * default parameters for modulus lengths of 512 and 1024 bits.
106      *
107      * @param modlen the modulus length, in bits. Valid values are any
108      * multiple of 8 between 512 and 1024, inclusive.
109      *
110      * @param random the random bit source to use to generate 
111      * key bits.
112      *
113      * @param genParams whether or not to generate new parameters for
114      * the modulus length requested.
115      *
116      * @exception InvalidParameterException if the modulus length is not
117      * between 512 and 1024, or if genParams is false and there are
118      * not precomputed parameters for the modulus length requested.  
119      */
120     public void initialize(int modlen, boolean genParams, SecureRandom random)
121         throws InvalidParameterException;
122 }
