37  package java.util;
38  
39  /** 
40   * The <code>Dictionary</code> class is the abstract parent of any 
41   * class, such as <code>Hashtable</code>, which maps keys to values. 
42   * Every key and every value is an object. In any one <tt>Dictionary</tt> 
43   * object, every key is associated with at most one value. Given a 
44   * <tt>Dictionary</tt> and a key, the associated element can be looked up. 
45   * Any non-<code>null</code> object can be used as a key and as a value.
46   * <p>
47   * As a rule, the <code>equals</code> method should be used by 
48   * implementations of this class to decide if two keys are the same. 
49   * <p>
50   * <strong>NOTE: This class is obsolete.  New implementations should
51   * implement the Map interface, rather than extending this class.</strong>
52   *
53   * @author  unascribed
54   * @version 1.15, 02/02/00
55   * @see	    java.util.Map
56   * @see     java.lang.Object#equals(java.lang.Object)
57   * @see     java.lang.Object#hashCode()
58   * @see     java.util.Hashtable
59   * @since   JDK1.0
60   */
61  public abstract class Dictionary
62  {
63  
64      /** 
65       * Sole constructor.  (For invocation by subclass constructors, typically
66       * implicit.)
67       */
68      public Dictionary() { }
69  
70      /** 
71       * Returns the number of entries (dinstint keys) in this dictionary.
72       *
73       * @return  the number of keys in this dictionary.
74       */
75      public abstract int size();
76  
77      /** 
78       * Tests if this dictionary maps no keys to value. The general contract 
79       * for the <tt>isEmpty</tt> method is that the result is true if and only 
80       * if this dictionary contains no entries. 
81       *
82       * @return  <code>true</code> if this dictionary maps no keys to values;
83       *          <code>false</code> otherwise.
84       */
85      public abstract boolean isEmpty();
86  
87      /** 
88       * Returns an enumeration of the keys in this dictionary. The general 
89       * contract for the keys method is that an <tt>Enumeration</tt> object 
90       * is returned that will generate all the keys for which this dictionary 
91       * contains entries. 
92       *
93       * @return  an enumeration of the keys in this dictionary.
94       * @see     java.util.Dictionary#elements()
95       * @see     java.util.Enumeration
96       */
97      public abstract Enumeration keys();
98  
99      /** 
100      * Returns an enumeration of the values in this dictionary. The general 
101      * contract for the <tt>elements</tt> method is that an 
102      * <tt>Enumeration</tt> is returned that will generate all the elements 
103      * contained in entries in this dictionary.
104      *
105      * @return  an enumeration of the values in this dictionary.
106      * @see     java.util.Dictionary#keys()
107      * @see     java.util.Enumeration
108      */
109     public abstract Enumeration elements();
110 
111     /** 
112      * Returns the value to which the key is mapped in this dictionary. 
113      * The general contract for the <tt>isEmpty</tt> method is that if this 
114      * dictionary contains an entry for the specified key, the associated 
115      * value is returned; otherwise, <tt>null</tt> is returned. 
116      *
117      * @return  the value to which the key is mapped in this dictionary;
118      * @param   key   a key in this dictionary.
119      *          <code>null</code> if the key is not mapped to any value in
120      *          this dictionary.
121      * @exception NullPointerException if the <tt>key</tt> is <tt>null</tt>.
122      * @see     java.util.Dictionary#put(java.lang.Object, java.lang.Object)
123      */
124     public abstract Object get(Object key);
125 
126     /** 
127      * Maps the specified <code>key</code> to the specified 
128      * <code>value</code> in this dictionary. Neither the key nor the 
129      * value can be <code>null</code>.
130      * <p>
131      * If this dictionary already contains an entry for the specified 
132      * <tt>key</tt>, the value already in this dictionary for that 
133      * <tt>key</tt> is returned, after modifying the entry to contain the
134      *  new element. <p>If this dictionary does not already have an entry 
135      *  for the specified <tt>key</tt>, an entry is created for the 
136      *  specified <tt>key</tt> and <tt>value</tt>, and <tt>null</tt> is 
137      *  returned.
138      * <p>
139      * The <code>value</code> can be retrieved by calling the 
140      * <code>get</code> method with a <code>key</code> that is equal to 
141      * the original <code>key</code>. 
142      *
143      * @param      key     the hashtable key.
144      * @param      value   the value.
145      * @return     the previous value to which the <code>key</code> was mapped
146      *             in this dictionary, or <code>null</code> if the key did not
147      *             have a previous mapping.
148      * @exception  NullPointerException  if the <code>key</code> or
149      *               <code>value</code> is <code>null</code>.
150      * @see        java.lang.Object#equals(java.lang.Object)
151      * @see        java.util.Dictionary#get(java.lang.Object)
152      */
153     public abstract Object put(Object key, Object value);
154 
155     /** 
156      * Removes the <code>key</code> (and its corresponding 
157      * <code>value</code>) from this dictionary. This method does nothing 
158      * if the <code>key</code> is not in this dictionary. 
159      *
160      * @param   key   the key that needs to be removed.
161      * @return  the value to which the <code>key</code> had been mapped in this
162      *          dictionary, or <code>null</code> if the key did not have a
163      *          mapping.
164      * @exception NullPointerException if <tt>key</tt> is <tt>null</tt>.
165      */
166     public abstract Object remove(Object key);
167 }
