37  package java.util;

39  /** 
40   * 
41   * An iterator for lists that allows the programmer 
42   * to traverse the list in either direction, modify 
43   * the list during iteration, and obtain the iterator's 
44   * current position in the list. A <TT>ListIterator</TT> 
45   * has no current element; its <I>cursor position</I> always 
46   * lies between the element that would be returned by a call 
47   * to <TT>previous()</TT> and the element that would be 
48   * returned by a call to <TT>next()</TT>. In a list of 
49   * length <TT>n</TT>, there are <TT>n+1</TT> valid 
50   * index values, from <TT>0</TT> to <TT>n</TT>, inclusive. 
51   * <PRE>
52   *
53   *          Element(0)   Element(1)   Element(2)   ... Element(n)   
54   *        ^            ^            ^            ^               ^
55   * Index: 0            1            2            3               n+1
56   *
57   * </PRE>
58   * <P>
59   * Note that the {@link #remove} and {@link #set(Object)} methods are
60   * <i>not</i> defined in terms of the cursor position;  they are defined to
61   * operate on the last element returned by a call to {@link #next} or {@link
62   * #previous()}.
63   * <P>
64   * This interface is a member of the 
65   * <a href="{@docRoot}/../guide/collections/index.html">
66   * Java Collections Framework</a>.
67   *
68   * @author  Josh Bloch
69   * @version 1.16, 02/02/00
70   * @see Collection
71   * @see List
72   * @see Iterator
73   * @see Enumeration
74   * @since   1.2
75   */
76  public interface ListIterator extends Iterator
77  {
78  
79      /** 
80       * Returns <tt>true</tt> if this list iterator has more elements when
81       * traversing the list in the forward direction. (In other words, returns
82       * <tt>true</tt> if <tt>next</tt> would return an element rather than
83       * throwing an exception.)
84       *
85       * @return <tt>true</tt> if the list iterator has more elements when
86       *		traversing the list in the forward direction.
87       */
88      public boolean hasNext();
89  
90      /** 
91       * Returns the next element in the list.  This method may be called
92       * repeatedly to iterate through the list, or intermixed with calls to
93       * <tt>previous</tt> to go back and forth.  (Note that alternating calls
94       * to <tt>next</tt> and <tt>previous</tt> will return the same element
95       * repeatedly.)
96       *
97       * @return the next element in the list.
98       * @exception NoSuchElementException if the iteration has no next element.
99       */
100     public Object next();
101 
102     /** 
103      * Returns <tt>true</tt> if this list iterator has more elements when
104      * traversing the list in the reverse direction.  (In other words, returns
105      * <tt>true</tt> if <tt>previous</tt> would return an element rather than
106      * throwing an exception.)
107      *
108      * @return <tt>true</tt> if the list iterator has more elements when
109      *	       traversing the list in the reverse direction.
110      */
111     public boolean hasPrevious();
112 
113     /** 
114      * Returns the previous element in the list.  This method may be called
115      * repeatedly to iterate through the list backwards, or intermixed with
116      * calls to <tt>next</tt> to go back and forth.  (Note that alternating
117      * calls to <tt>next</tt> and <tt>previous</tt> will return the same
118      * element repeatedly.)
119      *
120      * @return the previous element in the list.
121      * 
122      * @exception NoSuchElementException if the iteration has no previous
123      *            element.
124      */
125     public Object previous();
126 
127     /** 
128      * Returns the index of the element that would be returned by a subsequent
129      * call to <tt>next</tt>. (Returns list size if the list iterator is at the
130      * end of the list.)
131      *
132      * @return the index of the element that would be returned by a subsequent
133      * 	       call to <tt>next</tt>, or list size if list iterator is at end
134      *	       of list. 
135      */
136     public int nextIndex();
137 
138     /** 
139      * Returns the index of the element that would be returned by a subsequent
140      * call to <tt>previous</tt>. (Returns -1 if the list iterator is at the
141      * beginning of the list.)
142      *
143      * @return the index of the element that would be returned by a subsequent
144      * 	       call to <tt>previous</tt>, or -1 if list iterator is at
145      *	       beginning of list.
146      */
147     public int previousIndex();
148 
149     /** 
150      * Removes from the list the last element that was returned by
151      * <tt>next</tt> or <tt>previous</tt> (optional operation).  This call can
152      * only be made once per call to <tt>next</tt> or <tt>previous</tt>.  It
153      * can be made only if <tt>ListIterator.add</tt> has not been called after
154      * the last call to <tt>next</tt> or <tt>previous</tt>.
155      *
156      * @exception UnsupportedOperationException if the <tt>remove</tt>
157      * 		  operation is not supported by this list iterator.
158      * @exception IllegalStateException neither <tt>next</tt> nor
159      *		  <tt>previous</tt> have been called, or <tt>remove</tt> or
160      *		  <tt>add</tt> have been called after the last call to *
161      *		  <tt>next</tt> or <tt>previous</tt>.
162      */
163     public void remove();
164 
165     /** 
166      * Replaces the last element returned by <tt>next</tt> or
167      * <tt>previous</tt> with the specified element (optional operation).
168      * This call can be made only if neither <tt>ListIterator.remove</tt> nor
169      * <tt>ListIterator.add</tt> have been called after the last call to
170      * <tt>next</tt> or <tt>previous</tt>.
171      *
172      * @param o the element with which to replace the last element returned by
173      *          <tt>next</tt> or <tt>previous</tt>.
174      * @exception UnsupportedOperationException if the <tt>set</tt> operation
175      * 		  is not supported by this list iterator.
176      * @exception ClassCastException if the class of the specified element
177      * 		  prevents it from being added to this list.
178      * @exception IllegalArgumentException if some aspect of the specified
179      *		  element prevents it from being added to this list.
180      * @exception IllegalStateException if neither <tt>next</tt> nor
181      *	          <tt>previous</tt> have been called, or <tt>remove</tt> or
182      *		  <tt>add</tt> have been called after the last call to
183      * 		  <tt>next</tt> or <tt>previous</tt>.
184      */
185     public void set(Object o);
186 
187     /** 
188      * Inserts the specified element into the list (optional operation).  The
189      * element is inserted immediately before the next element that would be
190      * returned by <tt>next</tt>, if any, and after the next element that
191      * would be returned by <tt>previous</tt>, if any.  (If the list contains
192      * no elements, the new element becomes the sole element on the list.)
193      * The new element is inserted before the implicit cursor: a subsequent
194      * call to <tt>next</tt> would be unaffected, and a subsequent call to
195      * <tt>previous</tt> would return the new element.  (This call increases
196      * by one the value that would be returned by a call to <tt>nextIndex</tt>
197      * or <tt>previousIndex</tt>.)
198      *
199      * @param o the element to insert.
200      * @exception UnsupportedOperationException if the <tt>add</tt> method is
201      * 		  not supported by this list iterator.
202      * 
203      * @exception ClassCastException if the class of the specified element
204      * 		  prevents it from being added to this list.
205      * 
206      * @exception IllegalArgumentException if some aspect of this element
207      *            prevents it from being added to this list.
208      */
209     public void add(Object o);
