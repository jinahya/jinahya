36  package java.awt;

38  /** 
39   * Defines the interface for classes that know how to lay out 
40   * <code>Container</code>s.
41   *
42   * @see Container
43   *
44   * @version	1.24, 01/23/03
45   * @author 	Sami Shaio
46   * @author 	Arthur van Hoff
47   */
48  public interface LayoutManager
49  {
50  
51      /** 
52       * If the layout manager uses a per-component string,
53       * adds the component <code>comp</code> to the layout,
54       * associating it 
55       * with the string specified by <code>name</code>.
56       * 
57       * @param name the string to be associated with the component
58       * @param comp the component to be added
59       */
60      public void addLayoutComponent(String name, Component comp);
61  
62      /** 
63       * Removes the specified component from the layout.
64       * @param comp the component to be removed
65       */
66      public void removeLayoutComponent(Component comp);
67  
68      /** 
69       * Calculates the preferred size dimensions for the specified 
70       * container, given the components it contains.
71       * @param parent the container to be laid out
72       *  
73       * @see #minimumLayoutSize
74       */
75      public Dimension preferredLayoutSize(Container parent);
76  
77      /** 
78       * Calculates the minimum size dimensions for the specified 
79       * container, given the components it contains.
80       * @param parent the component to be laid out
81       * @see #preferredLayoutSize
82       */
83      public Dimension minimumLayoutSize(Container parent);
84  
85      /** 
86       * Lays out the specified container.
87       * @param parent the container to be laid out 
88       */
89      public void layoutContainer(Container parent);
90  }
