36  package java.awt.im;
37  
38  import java.awt.Rectangle;
39  import java.awt.font.TextHitInfo;
40  import java.text.AttributedCharacterIterator;
41  import java.text.AttributedCharacterIterator.Attribute;

43  /** 
44   * InputMethodRequests defines the requests that a text editing component
45   * has to handle in order to work with input methods. The component
46   * can implement this interface itself or use a separate object that
47   * implements it. The object implementing this interface must be returned
48   * from the component's getInputMethodRequests method.
49   *
50   * <p>
51   * The text editing component also has to provide an input method event
52   * listener.
53   *
54   * <p>
55   * The interface is designed to support one of two input user interfaces:
56   * <ul>
57   * <li><em>on-the-spot</em> input, where the composed text is displayed as part
58   *     of the text component's text body.
59   * <li><em>below-the-spot</em> input, where the composed text is displayed in
60   *     a separate composition window just below the insertion point where
61   *     the text will be inserted when it is committed. Note that, if text is
62   *     selected within the component's text body, this text will be replaced by
63   *     the committed text upon commitment; therefore it is not considered part
64   *     of the context that the text is input into.
65   * </ul>
66   *
67   * @see java.awt.Component#getInputMethodRequests
68   * @see java.awt.event.InputMethodListener
69   *
70   * @version 	1.17, 01/23/03
71   * @author JavaSoft Asia/Pacific
72   * @since 1.2
73   */
74  public interface InputMethodRequests
75  {
76  
77      /** 
78       * Gets the location of a specified offset in the current composed text,
79       * or of the selection in committed text.
80       * This information is, for example, used to position the candidate window
81       * near the composed text, or a composition window near the location
82       * where committed text will be inserted.
83       *
84       * <p>
85       * If the component has composed text (because the most recent
86       * InputMethodEvent sent to it contained composed text), then the offset is
87       * relative to the composed text - offset 0 indicates the first character
88       * in the composed text. The location returned should be for this character.
89       *
90       * <p>
91       * If the component doesn't have composed text, the offset should be ignored,
92       * and the location returned should reflect the beginning (in line
93       * direction) of the highlight in the last line containing selected text.
94       * For example, for horizontal left-to-right text (such as English), the
95       * location to the left of the left-most character on the last line
96       * containing selected text is returned. For vertical top-to-bottom text,
97       * with lines proceding from right to left, the location to the top of the
98       * left-most line containing selected text is returned.
99       *
100      * <p>
101      * The location is represented as a 0-thickness caret, that is, it has 0
102      * width if the text is drawn horizontally, and 0 height if the text is
103      * drawn vertically. Other text orientations need to be mapped to
104      * horizontal or vertical orientation. The rectangle uses absolute screen
105      * coordinates.
106      *
107      * @param offset the offset within the composed text, if there is composed
108      * text; null otherwise
109      * @return a rectangle representing the screen location of the offset
110      */
111     public Rectangle getTextLocation(TextHitInfo offset);
112 
113     /** 
114      * Gets the offset within the composed text for the specified absolute x
115      * and y coordinates on the screen. This information is used, for example
116      * to handle mouse clicks and the mouse cursor. The offset is relative to
117      * the composed text, so offset 0 indicates the beginning of the composed
118      * text.
119      *
120      * <p>
121      * Return null if the location is outside the area occupied by the composed
122      * text.
123      *
124      * @param x the absolute x coordinate on screen
125      * @param y the absolute y coordinate on screen
126      * @return a text hit info describing the offset in the composed text.
127      */
128     public TextHitInfo getLocationOffset(int x, int y);
129 
130     /** 
131      * Gets the offset of the insert position in the committed text contained
132      * in the text editing component. This is the offset at which characters
133      * entered through an input method are inserted. This information is used
134      * by an input method, for example, to examine the text surrounding the
135      * insert position.
136      * 
137      * @return the offset of the insert position
138      */
139     public int getInsertPositionOffset();
140 
141     /** 
142      * Gets an iterator providing access to the entire text and attributes
143      * contained in the text editing component except for uncommitted
144      * text. Uncommitted (composed) text should be ignored for index
145      * calculations and should not be made accessible through the iterator.
146      *
147      * <p>
148      * The input method may provide a list of attributes that it is
149      * interested in. In that case, information about other attributes that
150      * the implementor may have need not be made accessible through the
151      * iterator. If the list is null, all available attribute information
152      * should be made accessible.
153      *
154      * @param beginIndex the index of the first character
155      * @param endIndex the index of the character following the last character
156      * @param attributes a list of attributes that the input method is
157      * interested in
158      * @return an iterator providing access to the text and its attributes
159      */
160     public AttributedCharacterIterator getCommittedText(int beginIndex, int
161         endIndex, java.text.AttributedCharacterIterator.Attribute[] attributes);
162 
163     /** 
164      * Gets the length of the entire text contained in the text
165      * editing component except for uncommitted (composed) text.
166      *
167      * @return the length of the text except for uncommitted text
168      */
169     public int getCommittedTextLength();
170 
171     /** 
172      * Gets the latest committed text from the text editing component and
173      * removes it from the component's text body.
174      * This is used for the "Undo Commit" feature in some input methods, where
175      * the committed text reverts to its previous composed state. The composed
176      * text will be sent to the component using an InputMethodEvent.
177      *
178      * <p>
179      * Generally, this feature should only be supported immediately after the
180      * text was committed, not after the user performed other operations on the
181      * text. When the feature is not supported, return null.
182      *
183      * <p>
184      * The input method may provide a list of attributes that it is
185      * interested in. In that case, information about other attributes that
186      * the implementor may have need not be made accessible through the
187      * iterator. If the list is null, all available attribute information
188      * should be made accessible.
189      *
190      * @param attributes a list of attributes that the input method is
191      * interested in
192      * @return the latest committed text, or null when the "Undo Commit"
193      * feature is not supported
194      */
195     public AttributedCharacterIterator
196         cancelLatestCommittedText(java.text.AttributedCharacterIterator.Attribute[] attributes);
197 
198     /** 
199      * Gets the currently selected text from the text editing component.
200      * This may be used for a variety of purposes.
201      * One of them is the "Reconvert" feature in some input methods.
202      * In this case, the input method will typically send an input method event
203      * to replace the selected text with composed text. Depending on the input
204      * method's capabilities, this may be the original composed text for the
205      * selected text, the latest composed text entered anywhere in the text, or
206      * a version of the text that's converted back from the selected text.
207      *
208      * <p>
209      * The input method may provide a list of attributes that it is
210      * interested in. In that case, information about other attributes that
211      * the implementor may have need not be made accessible through the
212      * iterator. If the list is null, all available attribute information
213      * should be made accessible.
214      *
215      * @param attributes a list of attributes that the input method is
216      * interested in
217      * @return the currently selected text
218      */
219     public AttributedCharacterIterator
220         getSelectedText(java.text.AttributedCharacterIterator.Attribute[]
221         attributes);
