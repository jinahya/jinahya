36  package java.awt.image;
37  
38  import java.awt.Image;
39  
40  /** 
41   * An asynchronous update interface for receiving notifications about
42   * Image information as the Image is constructed.
43   *
44   * @version 	1.26 01/23/03
45   * @author 	Jim Graham
46   */
47  public interface ImageObserver
48  {
49      /** 
50       * This flag in the infoflags argument to imageUpdate indicates that 
51       * the width of the base image is now available and can be taken
52       * from the width argument to the imageUpdate callback method.
53       * @see Image#getWidth
54       * @see #imageUpdate
55       */
56      public static final int WIDTH = 1;
57  
58      /** 
59       * This flag in the infoflags argument to imageUpdate indicates that 
60       * the height of the base image is now available and can be taken
61       * from the height argument to the imageUpdate callback method.
62       * @see Image#getHeight
63       * @see #imageUpdate
64       */
65      public static final int HEIGHT = 2;
66  
67      /** 
68       * This flag in the infoflags argument to imageUpdate indicates that 
69       * the properties of the image are now available.
70       * @see Image#getProperty
71       * @see #imageUpdate
72       */
73      public static final int PROPERTIES = 4;
74  
75      /** 
76       * This flag in the infoflags argument to imageUpdate indicates that 
77       * more pixels needed for drawing a scaled variation of the image
78       * are available.  The bounding box of the new pixels can be taken
79       * from the x, y, width, and height arguments to the imageUpdate
80       * callback method.
81       * @see java.awt.Graphics#drawImage
82       * @see #imageUpdate
83       */
84      public static final int SOMEBITS = 8;
85  
86      /** 
87       * This flag in the infoflags argument to imageUpdate indicates that 
88       * another complete frame of a multi-frame image which was previously
89       * drawn is now available to be drawn again.  The x, y, width, and height
90       * arguments to the imageUpdate callback method should be ignored.
91       * @see java.awt.Graphics#drawImage
92       * @see #imageUpdate
93       */
94      public static final int FRAMEBITS = 16;
95  
96      /** 
97       * This flag in the infoflags argument to imageUpdate indicates that 
98       * a static image which was previously drawn is now complete and can
99       * be drawn again in its final form.  The x, y, width, and height
100      * arguments to the imageUpdate callback method should be ignored.
101      * @see java.awt.Graphics#drawImage
102      * @see #imageUpdate
103      */
104     public static final int ALLBITS = 32;
105 
106     /** 
107      * This flag in the infoflags argument to imageUpdate indicates that 
108      * an image which was being tracked asynchronously has encountered
109      * an error.  No further information will become available and
110      * drawing the image will fail.
111      * As a convenience, the ABORT flag will be indicated at the same
112      * time to indicate that the image production was aborted.
113      * @see #imageUpdate
114      */
115     public static final int ERROR = 64;
116 
117     /** 
118      * This flag in the infoflags argument to imageUpdate indicates that 
119      * an image which was being tracked asynchronously was aborted before
120      * production was complete.  No more information will become available
121      * without further action to trigger another image production sequence.
122      * If the ERROR flag was not also set in this image update, then
123      * accessing any of the data in the image will restart the production
124      * again, probably from the beginning.
125      * @see #imageUpdate
126      */
127     public static final int ABORT = 128;
128 
129     /** 
130      * This method is called when information about an image which was
131      * previously requested using an asynchronous interface becomes
132      * available.  Asynchronous interfaces are method calls such as
133      * getWidth(ImageObserver) and drawImage(img, x, y, ImageObserver)
134      * which take an ImageObserver object as an argument.  Those methods
135      * register the caller as interested either in information about
136      * the overall image itself (in the case of getWidth(ImageObserver))
137      * or about an output version of an image (in the case of the
138      * drawImage(img, x, y, [w, h,] ImageObserver) call).  
139      *
140      * <p>This method
141      * should return true if further updates are needed or false if the
142      * required information has been acquired.  The image which was being
143      * tracked is passed in using the img argument.  Various constants
144      * are combined to form the infoflags argument which indicates what
145      * information about the image is now available.  The interpretation
146      * of the x, y, width, and height arguments depends on the contents
147      * of the infoflags argument.
148      * <p>
149      * The <code>infoflags</code> argument should be the bitwise inclusive 
150      * <b>OR</b> of the following flags: <code>WIDTH</code>, 
151      * <code>HEIGHT</code>, <code>PROPERTIES</code>, <code>SOMEBITS</code>, 
152      * <code>FRAMEBITS</code>, <code>ALLBITS</code>, <code>ERROR</code>, 
153      * <code>ABORT</code>.
154      *
155      * @param     img   the image being observed.
156      * @param     infoflags   the bitwise inclusive OR of the following 
157      *               flags:  <code>WIDTH</code>, <code>HEIGHT</code>, 
158      *               <code>PROPERTIES</code>, <code>SOMEBITS</code>,
159      *               <code>FRAMEBITS</code>, <code>ALLBITS</code>, 
160      *               <code>ERROR</code>, <code>ABORT</code>. 
161      * @param     x   the <i>x</i> coordinate.
162      * @param     y   the <i>y</i> coordinate.
163      * @param     width    the width.
164      * @param     height   the height.
165      * @return    <code>false</code> if the infoflags indicate that the
166      *            image is completely loaded; <code>true</code> otherwise.
167      * 
168      * @see #WIDTH
169      * @see #HEIGHT
170      * @see #PROPERTIES
171      * @see #SOMEBITS
172      * @see #FRAMEBITS
173      * @see #ALLBITS
174      * @see #ERROR
175      * @see #ABORT
176      * @see Image#getWidth
177      * @see Image#getHeight
178      * @see java.awt.Graphics#drawImage
179      */
180     public boolean imageUpdate(Image img, int infoflags, int x, int y, int
181         width, int height);
182 }
