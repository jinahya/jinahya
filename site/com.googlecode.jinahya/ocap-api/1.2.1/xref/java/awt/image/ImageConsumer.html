40  /** 
41   * The interface for objects expressing interest in image data through
42   * the ImageProducer interfaces.  When a consumer is added to an image
43   * producer, the producer delivers all of the data about the image
44   * using the method calls defined in this interface.
45   *
46   * @see ImageProducer
47   *
48   * @version	1.20 01/23/03
49   * @author 	Jim Graham
50   */
51  public interface ImageConsumer
52  {
53      /** 
54       * The pixels will be delivered in a random order.  This tells the
55       * ImageConsumer not to use any optimizations that depend on the
56       * order of pixel delivery, which should be the default assumption
57       * in the absence of any call to the setHints method.
58       * @see #setHints
59       */
60      public static final int RANDOMPIXELORDER = 1;
61  
62      /** 
63       * The pixels will be delivered in top-down, left-to-right order.
64       * @see #setHints
65       */
66      public static final int TOPDOWNLEFTRIGHT = 2;
67  
68      /** 
69       * The pixels will be delivered in (multiples of) complete scanlines
70       * at a time.
71       * @see #setHints
72       */
73      public static final int COMPLETESCANLINES = 4;
74  
75      /** 
76       * The pixels will be delivered in a single pass.  Each pixel will
77       * appear in only one call to any of the setPixels methods.  An
78       * example of an image format which does not meet this criterion
79       * is a progressive JPEG image which defines pixels in multiple
80       * passes, each more refined than the previous.
81       * @see #setHints
82       */
83      public static final int SINGLEPASS = 8;
84  
85      /** 
86       * The image contain a single static image.  The pixels will be defined
87       * in calls to the setPixels methods and then the imageComplete method
88       * will be called with the STATICIMAGEDONE flag after which no more
89       * image data will be delivered.  An example of an image type which
90       * would not meet these criteria would be the output of a video feed,
91       * or the representation of a 3D rendering being manipulated
92       * by the user.  The end of each frame in those types of images will
93       * be indicated by calling imageComplete with the SINGLEFRAMEDONE flag.
94       * @see #setHints
95       * @see #imageComplete
96       */
97      public static final int SINGLEFRAME = 16;
98  
99      /** 
100      * An error was encountered while producing the image.
101      * @see #imageComplete
102      */
103     public static final int IMAGEERROR = 1;
104 
105     /** 
106      * One frame of the image is complete but there are more frames
107      * to be delivered.
108      * @see #imageComplete
109      */
110     public static final int SINGLEFRAMEDONE = 2;
111 
112     /** 
113      * The image is complete and there are no more pixels or frames
114      * to be delivered.
115      * @see #imageComplete
116      */
117     public static final int STATICIMAGEDONE = 3;
118 
119     /** 
120      * The image creation process was deliberately aborted.
121      * @see #imageComplete
122      */
123     public static final int IMAGEABORTED = 4;
124 
125     /** 
126      * The dimensions of the source image are reported using the
127      * setDimensions method call.
128      * @param width the width of the source image
129      * @param height the height of the source image
130      */
131     public void setDimensions(int width, int height);
132 
133     /** 
134      * Sets the extensible list of properties associated with this image.
135      * @param props the list of properties to be associated with this
136      *        image
137      */
138     public void setProperties(Hashtable props);
139 
140     /** 
141      * Sets the ColorModel object used for the majority of
142      * the pixels reported using the setPixels method
143      * calls.  Note that each set of pixels delivered using setPixels
144      * contains its own ColorModel object, so no assumption should
145      * be made that this model will be the only one used in delivering
146      * pixel values.  A notable case where multiple ColorModel objects
147      * may be seen is a filtered image when for each set of pixels
148      * that it filters, the filter
149      * determines  whether the
150      * pixels can be sent on untouched, using the original ColorModel,
151      * or whether the pixels should be modified (filtered) and passed
152      * on using a ColorModel more convenient for the filtering process.
153      * @param model the specified <code>ColorModel</code> 
154      * @see ColorModel
155      */
156     public void setColorModel(ColorModel model);
157 
158     /** 
159      * Sets the hints that the ImageConsumer uses to process the
160      * pixels delivered by the ImageProducer.
161      * The ImageProducer can deliver the pixels in any order, but
162      * the ImageConsumer may be able to scale or convert the pixels
163      * to the destination ColorModel more efficiently or with higher
164      * quality if it knows some information about how the pixels will
165      * be delivered up front.  The setHints method should be called
166      * before any calls to any of the setPixels methods with a bit mask
167      * of hints about the manner in which the pixels will be delivered.
168      * If the ImageProducer does not follow the guidelines for the
169      * indicated hint, the results are undefined.
170      * @param hintflags a set of hints that the ImageConsumer uses to
171      *        process the pixels
172      */
173     public void setHints(int hintflags);
174 
175     /** 
176      * Delivers the pixels of the image with one or more calls
177      * to this method.  Each call specifies the location and
178      * size of the rectangle of source pixels that are contained in
179      * the array of pixels.  The specified ColorModel object should
180      * be used to convert the pixels into their corresponding color
181      * and alpha components.  Pixel (m,n) is stored in the pixels array
182      * at index (n * scansize + m + off).  The pixels delivered using
183      * this method are all stored as bytes.
184      * @param x,&nbsp;y the coordinates of the upper-left corner of the 
185      *        area of pixels to be set
186      * @param w the width of the area of pixels
187      * @param h the height of the area of pixels
188      * @param model the specified <code>ColorModel</code>
189      * @param pixels the array of pixels
190      * @param off the offset into the <code>pixels</code> array
191      * @param scansize the distance from one row of pixels to the next in
192      * the <code>pixels</code> array
193      * @see ColorModel
194      */
195     public void setPixels(int x, int y, int w, int h, ColorModel model, byte[]
196         pixels, int off, int scansize);
197 
198     /** 
199      * The pixels of the image are delivered using one or more calls
200      * to the setPixels method.  Each call specifies the location and
201      * size of the rectangle of source pixels that are contained in
202      * the array of pixels.  The specified ColorModel object should
203      * be used to convert the pixels into their corresponding color
204      * and alpha components.  Pixel (m,n) is stored in the pixels array
205      * at index (n * scansize + m + off).  The pixels delivered using
206      * this method are all stored as ints.
207      * this method are all stored as ints.
208      * @param x,&nbsp;y the coordinates of the upper-left corner of the 
209      *        area of pixels to be set
210      * @param w the width of the area of pixels
211      * @param h the height of the area of pixels
212      * @param model the specified <code>ColorModel</code>
213      * @param pixels the array of pixels
214      * @param off the offset into the <code>pixels</code> array
215      * @param scansize the distance from one row of pixels to the next in
216      * the <code>pixels</code> array
217      * @see ColorModel
218      */
219     public void setPixels(int x, int y, int w, int h, ColorModel model, int[]
220         pixels, int off, int scansize);
221 
222     /** 
223      * The imageComplete method is called when the ImageProducer is
224      * finished delivering all of the pixels that the source image
225      * contains, or when a single frame of a multi-frame animation has
226      * been completed, or when an error in loading or producing the
227      * image has occured.  The ImageConsumer should remove itself from the
228      * list of consumers registered with the ImageProducer at this time,
229      * unless it is interested in successive frames.
230      * @param status the status of image loading
231      * @see ImageProducer#removeConsumer
232      */
233     public void imageComplete(int status);
