36  package java.awt;

38  import java.awt.event.*;

40  /** 
41   * The interface for objects which have an adjustable numeric value
42   * contained within a bounded range of values.
43   *
44   * @version 1.15 01/23/03
45   * @author Amy Fowler
46   * @author Tim Prinzing
47   */
48  public interface Adjustable
49  {
50      /** 
51       * Indicates that the <code>Adjustable</code> has horizontal orientation.  
52       */
53      public static final int HORIZONTAL = 0;
54  
55      /** 
56       * Indicates that the <code>Adjustable</code> has vertical orientation.  
57       */
58      public static final int VERTICAL = 1;
59  
60     /** 
61       * Indicates that the <code>Adjustable</code> has no orientation.  
62       */
63      public static final int NO_ORIENTATION = 2;
64  
65      /** 
66       * Gets the orientation of the adjustable object.
67       * @return the orientation of the adjustable object;
68       *   either <code>HORIZONTAL</code>, <code>VERTICAL</code>,
69       *   or <code>NO_ORIENTATION</code>
70       */
71      public int getOrientation();
72  
73      /** 
74       * Sets the minimum value of the adjustable object.
75       * @param min the minimum value
76       */
77      public void setMinimum(int min);
78  
79      /** 
80       * Gets the minimum value of the adjustable object.
81       * @return the minimum value of the adjustable object
82       */
83      public int getMinimum();
84  
85      /** 
86       * Sets the maximum value of the adjustable object.
87       * @param max the maximum value
88       */
89      public void setMaximum(int max);
90  
91      /** 
92       * Gets the maximum value of the adjustable object.
93       * @return the maximum value of the adjustable object
94       */
95      public int getMaximum();
96  
97      /** 
98       * Sets the unit value increment for the adjustable object.
99       * @param u the unit increment
100      */
101     public void setUnitIncrement(int u);
102 
103     /** 
104      * Gets the unit value increment for the adjustable object.
105      * @return the unit value increment for the adjustable object
106      */
107     public int getUnitIncrement();
108 
109     /** 
110      * Sets the block value increment for the adjustable object.
111      * @param b the block increment
112      */
113     public void setBlockIncrement(int b);
114 
115     /** 
116      * Gets the block value increment for the adjustable object.
117      * @return the block value increment for the adjustable object
118      */
119     public int getBlockIncrement();
120 
121     /** 
122      * Sets the length of the proportional indicator of the
123      * adjustable object.
124      * @param v the length of the indicator
125      */
126     public void setVisibleAmount(int v);
127 
128     /** 
129      * Gets the length of the proportional indicator.
130      * @return the length of the proportional indicator
131      */
132     public int getVisibleAmount();
133 
134     /** 
135      * Sets the current value of the adjustable object. This
136      * value must be within the range defined by the minimum and
137      * maximum values for this object.
138      * @param v the current value 
139      */
140     public void setValue(int v);
141 
142     /** 
143      * Gets the current value of the adjustable object.
144      * @return the current value of the adjustable object
145      */
146     public int getValue();
147 
148     /** 
149      * Adds a listener to receive adjustment events when the value of
150      * the adjustable object changes.
151      * @param l the listener to receive events
152      * @see AdjustmentEvent
153      */
154     public void addAdjustmentListener(AdjustmentListener l);
155 
156     /** 
157      * Removes an adjustment listener.
158      * @param l the listener being removed
159      * @see AdjustmentEvent
160      */
161     public void removeAdjustmentListener(AdjustmentListener l);
162 }
