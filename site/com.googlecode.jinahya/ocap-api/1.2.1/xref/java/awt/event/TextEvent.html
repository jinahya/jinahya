41  /** 
42   * A semantic event which indicates that an object's text changed.
43   * This high-level event is generated by an object (such as a TextComponent)
44   * when its text changes. The event is passed to
45   * every <code>TextListener</code> object which registered to receive such
46   * events using the component's <code>addTextListener</code> method. 
47   * <P>
48   * The object that implements the <code>TextListener</code> interface gets
49   * this <code>TextEvent</code> when the event occurs. The listener is
50   * spared the details of processing individual mouse movements and key strokes
51   * Instead, it can process a "meaningful" (semantic) event like "text changed". 
52   *
53   * @author Georges Saab
54   * @version 1.14 01/23/03
55   *
56  <!-- PBP/PP -->
57   * 
58   * @see TextListener
59   * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/textlistener.html">Tutorial: Writing a Text Listener</a>
60   * @see <a href="http://www.awl.com/cp/javaseries/jcl1_2.html">Reference: The Java Class Libraries (update file)</a>
61   *
62   * @since 1.1
63   */
