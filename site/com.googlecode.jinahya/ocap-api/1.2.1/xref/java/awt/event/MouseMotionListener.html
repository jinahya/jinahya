36  package java.awt.event;
37  
38  import java.util.EventListener;
39  
40  /** 
41   * The listener interface for receiving mouse motion events on a component.
42   * (For clicks and other mouse events, use the <code>MouseListener</code>.)
43   * <P>
44   * The class that is interested in processing a mouse motion event
45   * either implements this interface (and all the methods it
46   * contains) or extends the abstract <code>MouseMotionAdapter</code> class
47   * (overriding only the methods of interest).
48   * <P>
49   * The listener object created from that class is then registered with a
50   * component using the component's <code>addMouseMotionListener</code> 
51   * method. A mouse motion event is generated when the mouse is moved
52   * or dragged. (Many such events will be generated). When a mouse motion event
53   * occurs, the relevant method in the listener object is invoked, and 
54   * the <code>MouseEvent</code> is passed to it.
55   *
56   * @author Amy Fowler
57   * @version 1.14, 01/23/03
58   *
59   * @see MouseMotionAdapter
60   * @see MouseEvent
61   * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/mousemotionlistener.html">Tutorial: Writing a Mouse Motion Listener</a>
62   * @see <a href="http://www.awl.com/cp/javaseries/jcl1_2.html">Reference: The Java Class Libraries (update file)</a>
63   *
64   * @since 1.1
65   */
66  public interface MouseMotionListener extends EventListener
67  {
68  
69      /** 
70       * Invoked when a mouse button is pressed on a component and then 
71       * dragged.  <code>MOUSE_DRAGGED</code> events will continue to be 
72       * delivered to the component where the drag originated until the 
73       * mouse button is released (regardless of whether the mouse position 
74       * is within the bounds of the component).
75       * <p> 
76       * Due to platform-dependent Drag&Drop implementations, 
77       * <code>MOUSE_DRAGGED</code> events may not be delivered during a native 
78       * Drag&Drop operation.  
79       */
80      public void mouseDragged(MouseEvent e);
81  
82      /** 
83       * Invoked when the mouse cursor has been moved onto a component
84       * but no buttons have been pushed.
85       */
86      public void mouseMoved(MouseEvent e);
87  }
