39  /** 
40   * The class implements a buffered output stream. By setting up such 
41   * an output stream, an application can write bytes to the underlying 
42   * output stream without necessarily causing a call to the underlying 
43   * system for each byte written.
44   *
45   * @author  Arthur van Hoff
46   * @version 1.27, 02/02/00
47   * @since   JDK1.0
48   */
49  public class BufferedOutputStream extends FilterOutputStream
50  {
51      /** 
52       * The internal buffer where data is stored. 
53       */
54      protected byte[] buf;
55  
56      /** 
57       * The number of valid bytes in the buffer. This value is always 
58       * in the range <tt>0</tt> through <tt>buf.length</tt>; elements 
59       * <tt>buf[0]</tt> through <tt>buf[count-1]</tt> contain valid 
60       * byte data.
61       */
62      protected int count;
63  
64      /** 
65       * Creates a new buffered output stream to write data to the 
66       * specified underlying output stream with a default 512-byte 
67       * buffer size.
68       *
69       * @param   out   the underlying output stream.
70       */
71      public BufferedOutputStream(OutputStream out) { 
72          super(out);
73      }
74  
75      /** 
76       * Creates a new buffered output stream to write data to the 
77       * specified underlying output stream with the specified buffer 
78       * size. 
79       *
80       * @param   out    the underlying output stream.
81       * @param   size   the buffer size.
82       * @exception IllegalArgumentException if size &lt;= 0.
83       */
84      public BufferedOutputStream(OutputStream out, int size) { 
85          super(out);
86      }
87  
88      /** 
89       * Writes the specified byte to this buffered output stream. 
90       *
91       * @param      b   the byte to be written.
92       * @exception  IOException  if an I/O error occurs.
93       */
94      public synchronized void write(int b) throws IOException { }
95  
96      /** 
97       * Writes <code>len</code> bytes from the specified byte array 
98       * starting at offset <code>off</code> to this buffered output stream.
99       *
100      * <p> Ordinarily this method stores bytes from the given array into this
101      * stream's buffer, flushing the buffer to the underlying output stream as
102      * needed.  If the requested length is at least as large as this stream's
103      * buffer, however, then this method will flush the buffer and write the
104      * bytes directly to the underlying output stream.  Thus redundant
105      * <code>BufferedOutputStream</code>s will not copy data unnecessarily.
106      *
107      * @param      b     the data.
108      * @param      off   the start offset in the data.
109      * @param      len   the number of bytes to write.
110      * @exception  IOException  if an I/O error occurs.
111      */
112     public synchronized void write(byte[] b, int off, int len)
113         throws IOException
114     { }
115 
116     /** 
117      * Flushes this buffered output stream. This forces any buffered 
118      * output bytes to be written out to the underlying output stream. 
119      *
120      * @exception  IOException  if an I/O error occurs.
121      * @see        java.io.FilterOutputStream#out
122      */
123     public synchronized void flush() throws IOException { }
