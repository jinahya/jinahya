37  package java.io;
38  
39  /** 
40   * ObjectInput extends the DataInput interface to include the reading of
41   * objects. DataInput includes methods for the input of primitive types,
42   * ObjectInput extends that interface to include objects, arrays, and Strings.
43   *
44   * @author  unascribed
45   * @version 1.16, 02/02/00
46   * @see java.io.InputStream
47   * @see java.io.ObjectOutputStream
48   * @see java.io.ObjectInputStream
49   * @since   JDK1.1
50   */
51  public interface ObjectInput extends DataInput
52  {
53  
54      /** 
55       * Read and return an object. The class that implements this interface
56       * defines where the object is "read" from.
57       *
58       * @return the object read from the stream
59       * @exception java.lang.ClassNotFoundException If the class of a serialized
60       *      object cannot be found.
61       * @exception IOException If any of the usual Input/Output
62       * related exceptions occur.
63       */
64      public Object readObject() throws ClassNotFoundException, IOException;
65  
66      /** 
67       * Reads a byte of data. This method will block if no input is
68       * available.
69       * @return 	the byte read, or -1 if the end of the
70       *		stream is reached.
71       * @exception IOException If an I/O error has occurred.
72       */
73      public int read() throws IOException;
74  
75      /** 
76       * Reads into an array of bytes.  This method will
77       * block until some input is available.
78       * @param b	the buffer into which the data is read
79       * @return  the actual number of bytes read, -1 is
80       * 		returned when the end of the stream is reached.
81       * @exception IOException If an I/O error has occurred.
82       */
83      public int read(byte[] b) throws IOException;
84  
85      /** 
86       * Reads into an array of bytes.  This method will
87       * block until some input is available.
88       * @param b	the buffer into which the data is read
89       * @param off the start offset of the data
90       * @param len the maximum number of bytes read
91       * @return  the actual number of bytes read, -1 is
92       * 		returned when the end of the stream is reached.
93       * @exception IOException If an I/O error has occurred.
94       */
95      public int read(byte[] b, int off, int len) throws IOException;
96  
97      /** 
98       * Skips n bytes of input.
99       * @param n the number of bytes to be skipped
100      * @return	the actual number of bytes skipped.
101      * @exception IOException If an I/O error has occurred.
102      */
103     public long skip(long n) throws IOException;
104 
105     /** 
106      * Returns the number of bytes that can be read
107      * without blocking.
108      * @return the number of available bytes.
109      * @exception IOException If an I/O error has occurred.
110      */
111     public int available() throws IOException;
112 
113     /** 
114      * Closes the input stream. Must be called
115      * to release any resources associated with
116      * the stream.
117      * @exception IOException If an I/O error has occurred.
118      */
119     public void close() throws IOException;
120 }
