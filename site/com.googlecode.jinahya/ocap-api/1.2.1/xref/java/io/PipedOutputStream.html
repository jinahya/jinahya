37  package java.io;

39  import java.io.*;

41  /** 
42   * A piped output stream can be connected to a piped input stream 
43   * to create a communications pipe. The piped output stream is the 
44   * sending end of the pipe. Typically, data is written to a 
45   * <code>PipedOutputStream</code> object by one thread and data is 
46   * read from the connected <code>PipedInputStream</code> by some 
47   * other thread. Attempting to use both objects from a single thread 
48   * is not recommended as it may deadlock the thread.
49   *
50   * @author  James Gosling
51   * @version 1.23, 02/02/00
52   * @see     java.io.PipedInputStream
53   * @since   JDK1.0
54   */
55  public class PipedOutputStream extends java.io.OutputStream
56  {
57  
58      /** 
59       * Creates a piped output stream connected to the specified piped 
60       * input stream. Data bytes written to this stream will then be 
61       * available as input from <code>snk</code>.
62       *
63       * @param      snk   The piped input stream to connect to.
64       * @exception  IOException  if an I/O error occurs.
65       */
66      public PipedOutputStream(java.io.PipedInputStream snk)
67          throws java.io.IOException
68      { }
69  
70      /** 
71       * Creates a piped output stream that is not yet connected to a 
72       * piped input stream. It must be connected to a piped input stream, 
73       * either by the receiver or the sender, before being used. 
74       *
75       * @see     java.io.PipedInputStream#connect(java.io.PipedOutputStream)
76       * @see     java.io.PipedOutputStream#connect(java.io.PipedInputStream)
77       */
78      public PipedOutputStream() { }
79  
80      /** 
81       * Connects this piped output stream to a receiver. If this object
82       * is already connected to some other piped input stream, an 
83       * <code>IOException</code> is thrown.
84       * <p>
85       * If <code>snk</code> is an unconnected piped input stream and 
86       * <code>src</code> is an unconnected piped output stream, they may 
87       * be connected by either the call:
88       * <blockquote><pre>
89       * src.connect(snk)</pre></blockquote>
90       * or the call:
91       * <blockquote><pre>
92       * snk.connect(src)</pre></blockquote>
93       * The two calls have the same effect.
94       *
95       * @param      snk   the piped input stream to connect to.
96       * @exception  IOException  if an I/O error occurs.
97       */
98      public synchronized void connect(java.io.PipedInputStream snk)
99          throws java.io.IOException
100     { }
101 
102     /** 
103      * Writes the specified <code>byte</code> to the piped output stream. 
104      * If a thread was reading data bytes from the connected piped input 
105      * stream, but the thread is no longer alive, then an 
106      * <code>IOException</code> is thrown.
107      * <p>
108      * Implements the <code>write</code> method of <code>OutputStream</code>.
109      *
110      * @param      b   the <code>byte</code> to be written.
111      * @exception  IOException  if an I/O error occurs.
112      */
113     public void write(int b) throws java.io.IOException { }
114 
115     /** 
116      * Writes <code>len</code> bytes from the specified byte array 
117      * starting at offset <code>off</code> to this piped output stream. 
118      * If a thread was reading data bytes from the connected piped input 
119      * stream, but the thread is no longer alive, then an 
120      * <code>IOException</code> is thrown.
121      *
122      * @param      b     the data.
123      * @param      off   the start offset in the data.
124      * @param      len   the number of bytes to write.
125      * @exception  IOException  if an I/O error occurs.
126      */
127     public void write(byte[] b, int off, int len) throws java.io.IOException { }
128 
129     /** 
130      * Flushes this output stream and forces any buffered output bytes 
131      * to be written out. 
132      * This will notify any readers that bytes are waiting in the pipe.
133      *
134      * @exception IOException if an I/O error occurs.
135      */
136     public synchronized void flush() throws java.io.IOException { }
137 
138     /** 
139      * Closes this piped output stream and releases any system resources 
140      * associated with this stream. This stream may no longer be used for 
141      * writing bytes.
142      *
143      * @exception  IOException  if an I/O error occurs.
144      */
145     public void close() throws java.io.IOException { }
