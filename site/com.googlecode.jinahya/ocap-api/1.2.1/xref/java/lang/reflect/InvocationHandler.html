39  /** 
40   * <code>InvocationHandler</code> is the interface implemented by
41   * the <i>invocation handler</i> of a proxy instance.
42   *
43   * <p>Each proxy instance has an associated invocation handler.
44   * When a method is invoked on a proxy instance, the method
45   * invocation is encoded and dispatched to the <code>invoke</code>
46   * method of its invocation handler.
47   *
48   * @author	Peter Jones
49   * @version	1.5, 00/02/02
50   * @see		Proxy
51   * @since	JDK1.3
52   */
53  public interface InvocationHandler
54  {
55  
56      /** 
57       * Processes a method invocation on a proxy instance and returns
58       * the result.  This method will be invoked on an invocation handler
59       * when a method is invoked on a proxy instance that it is
60       * associated with.
61       *
62       * @param	proxy the proxy instance that the method was invoked on
63       *
64       * @param	method the <code>Method</code> instance corresponding to
65       * the interface method invoked on the proxy instance.  The declaring
66       * class of the <code>Method</code> object will be the interface that
67       * the method was declared in, which may be a superinterface of the
68       * proxy interface that the proxy class inherits the method through.
69       *
70       * @param	args an array of objects containing the values of the
71       * arguments passed in the method invocation on the proxy instance,
72       * or <code>null</code> if interface method takes no arguments.
73       * Arguments of primitive types are wrapped in instances of the
74       * appropriate primitive wrapper class, such as
75       * <code>java.lang.Integer</code> or <code>java.lang.Boolean</code>.
76       *
77       * @return	the value to return from the method invocation on the
78       * proxy instance.  If the declared return type of the interface
79       * method is a primitive type, then the value returned by
80       * this method must be an instance of the corresponding primitive
81       * wrapper class; otherwise, it must be a type assignable to the
82       * declared return type.  If the value returned by this method is
83       * <code>null</code> and the interface method's return type is
84       * primitive, then a <code>NullPointerException</code> will be
85       * thrown by the method invocation on the proxy instance.  If the
86       * value returned by this method is otherwise not compatible with
87       * the interface method's declared return type as described above,
88       * a <code>ClassCastException</code> will be thrown by the method
89       * invocation on the proxy instance.
90       *
91       * @throws	Throwable the exception to throw from the method
92       * invocation on the proxy instance.  The exception's type must be
93       * assignable either to any of the exception types declared in the
94       * <code>throws</code> clause of the interface method or to the
95       * unchecked exception types <code>java.lang.RuntimeException</code>
96       * or <code>java.lang.Error</code>.  If a checked exception is
97       * thrown by this method that is not assignable to any of the
98       * exception types declared in the <code>throws</code> clause of
99       * the interface method, then an
100      * {@link UndeclaredThrowableException} containing the
101      * exception that was thrown by this method will be thrown by the
102      * method invocation on the proxy instance.
103      *
104      * @see	UndeclaredThrowableException
105      */
106     public Object invoke(Object proxy, Method method, Object[] args)
107         throws Throwable;
