39  /** 
40   * An <code>Error</code> is a subclass of <code>Throwable</code> 
41   * that indicates serious problems that a reasonable application 
42   * should not try to catch. Most such errors are abnormal conditions. 
43   * The <code>ThreadDeath</code> error, though a "normal" condition,
44   * is also a subclass of <code>Error</code> because most applications
45   * should not try to catch it. 
46   * <p>
47   * A method is not required to declare in its <code>throws</code> 
48   * clause any subclasses of <code>Error</code> that might be thrown 
49   * during the execution of the method but not caught, since these 
50   * errors are abnormal conditions that should never occur. 
51   *
52   * @author  Frank Yellin
53   * @version 1.12, 02/02/00
54   * @see     java.lang.ThreadDeath
55   * @since   JDK1.0
56   */
57  public class Error extends java.lang.Throwable
58  {
59      static final long serialVersionUID = 4980196508277280342L;
60  
61      /** 
62       * Constructs a new error with <code>null</code> as its detail message.
63       * The cause is not initialized, and may subsequently be initialized by a
64       * call to {@link #initCause}.
65       */
66      public Error() { }
67  
68      /** 
69       * Constructs a new error with the specified detail message.  The
70       * cause is not initialized, and may subsequently be initialized by
71       * a call to {@link #initCause}.
72       *
73       * @param   message   the detail message. The detail message is saved for 
74       *          later retrieval by the {@link #getMessage()} method.
75       */
76      public Error(java.lang.String message) { }
77  
78      /** 
79       * Constructs a new error with the specified detail message and
80       * cause.  <p>Note that the detail message associated with
81       * <code>cause</code> is <i>not</i> automatically incorporated in
82       * this error's detail message.
83       *
84       * @param  message the detail message (which is saved for later retrieval
85       *         by the {@link #getMessage()} method).
86       * @param  cause the cause (which is saved for later retrieval by the
87       *         {@link #getCause()} method).  (A <tt>null</tt> value is
88       *         permitted, and indicates that the cause is nonexistent or
89       *         unknown.)
90       * @since  1.4
91       */
92      public Error(java.lang.String message, java.lang.Throwable cause) { }
93  
94      /** 
95       * Constructs a new error with the specified cause and a detail
96       * message of <tt>(cause==null ? null : cause.toString())</tt> (which
97       * typically contains the class and detail message of <tt>cause</tt>).
98       * This constructor is useful for errors that are little more than
99       * wrappers for other throwables.
100      *
101      * @param  cause the cause (which is saved for later retrieval by the
102      *         {@link #getCause()} method).  (A <tt>null</tt> value is
103      *         permitted, and indicates that the cause is nonexistent or
104      *         unknown.)
105      * @since  1.4
106      */
107     public Error(java.lang.Throwable cause) { }
