36  package java.rmi.registry;
37  
38  import java.rmi.AccessException;
39  import java.rmi.AlreadyBoundException;
40  import java.rmi.NotBoundException;
41  import java.rmi.Remote;
42  import java.rmi.RemoteException;
43  
44  // PBP/PP
45  /** 
46   * <code>Registry</code> is a remote interface to a simple remote
47   * object registry that provides methods for storing and retrieving
48   * remote object references bound with arbitrary string names.  The
49   * <code>bind</code>, <code>unbind</code>, and <code>rebind</code>
50   * methods are used to alter the name bindings in the registry, and
51   * the <code>lookup</code> and <code>list</code> methods are used to
52   * query the current name bindings.
53   *
54  
55   *
56   * <p>A <code>Registry</code> implementation may choose to restrict
57   * access to some or all of its methods (for example, methods that
58   * mutate the registry's bindings may be restricted to calls
59   * originating from the local host).  If a <code>Registry</code>
60   * method chooses to deny access for a given invocation, its
61   * implementation may throw {@link java.rmi.AccessException}.
62   *
63   * <p>The names used for bindings in a <code>Registry</code> are pure
64   * strings, not parsed.  A service which stores its remote reference
65   * in a <code>Registry</code> may wish to use a package name as a
66   * prefix in the name binding to reduce the likelihood of name
67   * collisions in the registry.
68   *
69   * @author	Ann Wollrath
70   * @author	Peter Jones
71   * @version	1.17, 03/01/23
72   * @since	JDK1.1
73   * @see		
74   */
75  public interface Registry extends Remote
76  {
77      /** Well known port for registry.  */
78      public static final int REGISTRY_PORT = 1099;
79  
80      /** 
81       * Returns the remote reference bound to the specified
82       * <code>name</code> in this registry.
83       *
84       * @param	name the name for the remote reference to look up
85       *
86       * @return	a reference to a remote object
87       *
88       * @throws	NotBoundException if <code>name</code> is not currently bound
89       *
90       * @throws	RemoteException if remote communication with the
91       * registry failed
92       *
93       * @throws	AccessException if this registry is local and it denies
94       * the caller access to perform this operation
95       *
96       * @throws	NullPointerException if <code>name</code> is <code>null</code>
97       */
98      public Remote lookup(String name)
99          throws java.rmi.RemoteException, NotBoundException, AccessException;
100 
101     /** 
102      * Binds a remote reference to the specified <code>name</code> in
103      * this registry.
104      *
105      * @param	name the name to associate with the remote reference
106      * @param	obj a reference to a remote object (usually a stub)
107      *
108      * @throws	AlreadyBoundException if <code>name</code> is already bound
109      *
110      * @throws	RemoteException if remote communication with the
111      * registry failed
112      *
113      * @throws	AccessException if this registry is local and it denies
114      * the caller access to perform this operation
115      *
116      * @throws	NullPointerException if <code>name</code> is
117      * <code>null</code>, or if <code>obj</code> is <code>null</code>
118      */
119     public void bind(String name, Remote obj)
120         throws java.rmi.RemoteException, AlreadyBoundException, AccessException;
121 
122     /** 
123      * Removes the binding for the specified <code>name</code> in
124      * this registry.
125      *
126      * @param	name the name of the binding to remove
127      *
128      * @throws	NotBoundException if <code>name</code> is not currently bound
129      *
130      * @throws	RemoteException if remote communication with the
131      * registry failed
132      *
133      * @throws	AccessException if this registry is local and it denies
134      * the caller access to perform this operation
135      *
136      * @throws	NullPointerException if <code>name</code> is <code>null</code>
137      */
138     public void unbind(String name)
139         throws java.rmi.RemoteException, NotBoundException, AccessException;
140 
141     /** 
142      * Replaces the binding for the specified <code>name</code> in
143      * this registry with the supplied remote reference.  If there is
144      * an existing binding for the specified <code>name</code>, it is
145      * discarded.
146      *
147      * @param	name the name to associate with the remote reference
148      * @param	obj a reference to a remote object (usually a stub)
149      *
150      * @throws	RemoteException if remote communication with the
151      * registry failed
152      *
153      * @throws	AccessException if this registry is local and it denies
154      * the caller access to perform this operation
155      *
156      * @throws	NullPointerException if <code>name</code> is
157      * <code>null</code>, or if <code>obj</code> is <code>null</code>
158      */
159     public void rebind(String name, Remote obj)
160         throws java.rmi.RemoteException, AccessException;
161 
162     /** 
163      * Returns an array of the names bound in this registry.  The
164      * array will contain a snapshot of the names bound in this
165      * registry at the time of the given invocation of this method.
166      *
167      * @return	an array of the names bound in this registry
168      *
169      * @throws	RemoteException if remote communication with the
170      * registry failed
171      *
172      * @throws	AccessException if this registry is local and it denies
173      * the caller access to perform this operation
174      */
175     public String[] list() throws java.rmi.RemoteException, AccessException;
176 }
