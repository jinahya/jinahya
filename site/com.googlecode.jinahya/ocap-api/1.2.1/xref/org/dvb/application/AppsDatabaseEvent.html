3   /**
4    * The <code>AppsDatabaseEvent</code> class indicates either an entry
5    * in the application database has changed, or so many changes have occurred.
6    * that the database should be considered totally new.
7    * An event with event_id NEW_DATABASE shall always be sent after switching to a new service.
8    * After such an event, the contents of the database (both the set of applications and 
9    * their attributes) shall reflect the new database contents. All former contents of 
10   * the database shall be discarded except for running externally authorised applications.
11   * It is platform dependant if and when a new database event is thrown 
12   * while tuned to the same service except that a NEW_DATABASE event shall not 
13   * be sent when only one application has changed within a service.
14   * <p>The APP_ADDED, APP_CHANGED and APP_DELETED events shall not be 
15   * generated in response to the same database change as caused a 
16   * NEW_DATABASE event to be generated.
17   * <p>
18   * @since   MHP1.0
19   */

21      /**

23       */
24      static public final int NEW_DATABASE = 0;
25      /**
26       * The changed event id. The APP_CHANGED event is generated whenever
27       * any of the information about an application changes. It is NOT generated
28       * when the entry is added to or removed from the AppsDatabase. In such cases, 
29       * the APP_ADDED or APP_DELETED events will be generated instead.
30       */
31      static public final int APP_CHANGED = 1;
32      /**
33       * The addition event id. The APP_ADDED event is generated whenever
34       * an entry is added to the AppsDatabase. It is NOT generated
35       * when the entry already in the AppsDatabase changes. 
36       */
37      static public final int APP_ADDED = 2;
38      /**
39       * The deletion event id. The APP_DELETED event is generated whenever
40       * an entry is removed from the AppsDatabase.

42      static public final int APP_DELETED = 3;
