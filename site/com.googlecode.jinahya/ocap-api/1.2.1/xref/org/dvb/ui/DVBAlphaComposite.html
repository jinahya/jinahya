5   /**
6    * This <code>DVBAlphaComposite</code> class implements the basic alpha
7    * compositing rules for combining source and destination pixels to achieve
8    * blending and transparency effects with graphics, images and video.
9    * The rules implemented by this class are a subset of the Porter-Duff
10   * rules described in
11   * T. Porter and T. Duff, "Compositing Digital Images", SIGGRAPH 84,
12   * 253-259.
13   *<p>
14   * If any input does not have an alpha channel, an alpha value of 1,0, 
15   * which is completely opaque, is assumed for all pixels. A constant 
16   * alpha value can also be specified to be multiplied with the alpha 
17   * value of the source pixels.
18   * <p>
19   * The following abbreviations are used in the description of the rules:
20   * <ul>
21   * <li>Cs = one of the color components of the source pixel without alpha.
22   * <li>cs = color component of a source pixel premultimlied with alpha (cs = As*Ar*Cs)
23   * <li>Cd = one of the color components of the destination pixel without alpha.
24   * <li>cd = color component of a destination pixel premultimlied with alpha
25   * <li>Cn = the new constructed color without alpha.
26   * <li>cn = the new constructed color pre-multiplied with alpha
27   * <li>As = alpha component of the source pixel.
28   * <li>Ad = alpha component of the destination pixel.
29   * <li>An = the new alpha after compositing
30   * <li>Ar = alpha, specified by getInstance(int Rule, float Ar). Unless otherwise specified Ar = 1,0f
31   * <li>Fs = fraction of the source pixel that contributes to the output.
32   * <li>Fd = fraction of the input destination pixel that contributes to the
33   * output.
34   * </ul>
35   *<p>
36   * The color and alpha components produced by the compositing operation are
37   * calculated as follows:
38   *<pre>
39   * 	cn = (As*Ar)*Cs*Fs + Ad*Cd*Fd
40   * 	An = (As*Ar)*Fs + Ad*Fd
41   *  Cn = cn/An
42   *</pre>
43   * where Fs and Fd are specified by each rule. 
44   *<p>
45   * The alpha resulting from the compositing operation is stored
46   * in the destination if the destination has an alpha channel.
47   * Otherwise, the resulting color is divided by the resulting
48   * alpha before being stored in the destination and the alpha is discarded.
49   * If the alpha value is 0,0, the color values are set to 0,0.
50   * @version 10 Feb 1997
51   * @see java.awt.AlphaComposite
52   */
53

57  	 * Porter-Duff Clear rule.

59  	 * Neither the source nor the destination is used as input.
60  	 *<p>
61  	 * Fs = 0 and Fd = 0, thus:
62  	 *<pre>
63  	 * 	cn = 0
64  	 * 	An = 0
65  	 *  Cn = 0
66  	 *</pre>

68  	 * <p> <b>Note that this operation is a fast drawing operation</b>
69  	 * This operation is the same as using a source with alpha= 0 and the SRC rule
70  	 * 
71  	 */

74  	/**
75  	 * Porter-Duff Source rule.
76  	 * The source is copied to the destination.
77  	 * The destination is not used as input.
78  	 *<p>
79  	 * Fs = 1 and Fd = 0, thus:
80  	 *<pre>
81  	 * 	cn = (As*Ar)*Cs
82  	 * 	An = As*Ar
83  	 *  Cn = Cs
84  	 *</pre>
85  	 * <img src="SRC.jpg">
86  	 * <p> <b>Note that this is a fast drawing routine</b>
87  	 */
88  	public static final int	SRC		= 2;
89  
90  	/**
91  	 * Porter-Duff Source Over Destination rule.
92  	 * The source is composited over the destination.
93  	 *<p>
94  	 * Fs = 1 and Fd = (1-(As*Ar)), thus:
95  	 *<pre>
96  	 * 	cn = (As*Ar)*Cs + Ad*Cd*(1-(As*Ar))
97  	 * 	An = (As*Ar) + Ad*(1-(As*Ar))
98  	 *</pre>
99  	 * <img src="SRC_OVER.jpg">
100 	 * <p> <b>Note that this can be a very slow drawing operation</b>
101 	 */
102 	public static final int	SRC_OVER	= 3;
103 
104 	/**
105 	 * Porter-Duff Destination Over Source rule.
106 	 * The destination is composited over the source and
107 	 * the result replaces the destination.
108 	 *<p>
109 	 * Fs = (1-Ad) and Fd = 1, thus:
110 	 *<pre>
111 	 * 	cn = (As*Ar)*Cs*(1-Ad) + Ad*Cd
112 	 * 	An = (As*Ar)*(1-Ad) + Ad
113 	 *</pre>
114 	 * <img src="DST_OVER.jpg">
115 	 * <p> <b>Note that this can be a very slow drawing operation</b>
116 	 */
117 	public static final int	DST_OVER	= 4;
118 
119 	/**
120 	 * Porter-Duff Source In Destination rule.
121 	 * The part of the source lying inside of the destination replaces
122 	 * the destination.
123 	 *<p>
124 	 * Fs = Ad and Fd = 0, thus:
125 	 *<pre>
126 	 * 	cn = (As*Ar)*Cs*Ad
127 	 * 	An = (As*Ar)*Ad
128 	 *  Cn = Cs
129 	 *</pre>
130 	 * <img src="SRC_IN.jpg">
131 	 * <p> <b>Note that this operation is faster than e.g. SRC_OVER but slower than SRC</b>
132 	 */
133 	public static final int	SRC_IN		= 5;

137 	 * The part of the destination lying inside of the source
138 	 * replaces the destination.
139 	 *<p>
140 	 * Fs = 0 and Fd = (As*Ar), thus:
141 	 *<pre>
142 	 * 	cn = Ad*Cd*(As*Ar)
143 	 * 	An = Ad*(As*Ar)
144 	 *  Cn = Cd
145 	 *</pre>
146 	 * <img src="DST_IN.jpg">
147 	 * <p> <b>Note that this operation is faster than e.g. SRC_OVER but slower than SRC</b>
148 	 */
