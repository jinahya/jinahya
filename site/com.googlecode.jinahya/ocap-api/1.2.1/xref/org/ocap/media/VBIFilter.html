9   import org.davic.mpeg.*;

11  /**
12   * <p>
13   * This class represents a VBI filter. 
14   * VBIFilter instances are created by a VBIFilterGroup based on the OCAP 
15   * resource management. 
16   * Line numbers and a data format to be filtered are specified when the filter 
17   * is created. 
18   * </p><p>
19   * The startFiltering() method starts filtering of the specified data format 
20   * in the specified VBI line and stores data units in an internal buffer. 
21   * When the first single data unit is filtered, a VBIFilterEvent with 
22   * EVENT_CODE_FIRST_VBI_DATA_AVAILABLE is issued only once. The VBIFilter 
23   * continues filtering. 
24   * </p><p>
25   * VBI filtering stops in the following cases: <br>
26   * <ul>
27   * <li> If a stopFiltering() method is called, a VBIFilterEvent with 
28   * EVENT_CODE_FORCIBLE_TERMINATED notifies it. <br>
29   * <li> If an internal buffer is full, a VBIFilterEvent with 
30   * EVENT_CODE_BUFFER_FULL notifies it. <br>
31   * <li> If a time out (specified by a timeout value in a 
32   * {@link VBIFilter#setTimeOut}) occurs, a VBIFilterEvent with 
33   * EVENT_CODE_TIMEOUT notifies it. <br>
34   * </ul></p><p>
35   * VBI filtering continues in the following cases: <br>
36   * <ul>
37   * <li> Timer notification by a VBIFilterEvent with 
38   * EVENT_CODE_TIME_NOTIFICATION. <br>
39   * <li> Cyclic notification for every specified number of data units by a 
40   * VBIFilterEvent with EVENT_CODE_UNITS_NOTIFICATION. <br>
41   * </ul></p><p>
42   * See also the {@link VBIFilterGroup}. 
43   * </p>
44   *
45   * @author  Shigeaki Watanabe (Panasonic)
46   */
47  public interface VBIFilter {
48      /**
49       * Initiate filtering of VBI data for the specified line and the specified 
50       * data format by a VBIFilterGroup. Filtering starts only after the 
51       * {@link VBIFilterGroup#attach} method is called. 
52       * 
53       * @param appData  application specific data. This data is notified 
54       *              to the application with a SectionFilterEvent. Null is 
55       *              possible. 
56       *
57       */
58      void startFiltering(Object appData);
59  
60  
61      /**
62       * Initiate filtering of VBI data for the specified line and the specified 
63       * data format by a VBIFilterGroup. Only data unit(s) matching with a 
64       * specified filter parameters are retrieved. Filtering starts only after 
65       * the {@link VBIFilterGroup#attach} method is called. 
66       *
67       * @param appData  application specific data. This data is notified 
68       *              to the application with a SectionFilterEvent. Null is 
69       *              possible. 
70       *
71       * @param offset  defines a number of offset bytes that the specified 
72       *              matching bits and masking bits are applied. Value 0 
73       *              means no offset. Value 1 means that the matching/masking 
74       *              bit is applied from the second byte. 
75       *
76       * @param posFilterDef  defines values to match for bits in a single data 
77       *              unit. Only data unit that has matching bytes with this 
78       *              posFilterDef are retrieved. 
79       *              Maximum length is 36 bytes. 
80       *
81       * @param posFilterMask  defines which bits in the data unit are to be 
82       *              compared against the posFilterDef bytes. 
83       *              Matching calculation of negFilterDef and negFilterMask 
84       *              obeys E.8.1 of DAVIC 1.4.1 Part 9. 
85       *              Maximum length is 36 bytes. 
86       *
87       * @param negFilterDef  defines values to match for bits in a single data 
88       *              unit. Only data unit that has matching bytes with this 
89       *              negFilterDef are retrieved. 
90       *              Maximum length is 36 bytes. 
91       *
92       * @param negFilterMask  defines which bits in the data unit are to be 
93       *              compared against the negFilterDef bytes. 
94       *              Matching calculation of negFilterDef and negFilterMask 
95       *              obeys E.8.1 of DAVIC 1.4.1 Part 9. 
96       *              Maximum length is 36 bytes. 
97       */
98      void startFiltering(Object appData, int offset, 
99              byte posFilterDef[], byte posFilterMask[], 
100             byte negFilterDef[], byte negFilterMask[]);
101 
102 
103     /**
104      * Stop current filtering of this VBI filter. Note that the VBIFilterGroup 
105      * holding this VBI filter doesn't detach. 
106      */
107     void stopFiltering();
108 
109 
110     /**
111      * Set a timeout value. If no VBI data unit is retrieved after calling 
112      * the startFiltering() method within the timeout value, the filtering 
113      * stops automatically and SectionFilterEvent with EVENT_CODE_TIMEOUT 
114      * notifies a timeout occurred. 
115      * 
116      * @param milliseconds  a timeout value in milli seconds. A default 
117      *              value is -1 that indicates infinite. 
118      */
119     void setTimeOut(long milliseconds);
120 
121 
122     /**
123      * Set a notification time. By setting a notification time, the 
124      * OCAP implementation notifies a VBIFilterEvent with 
125      * EVENT_CODE_TIME_NOTIFICATION when the specified time-period has 
126      * elapsed after receiving the first byte of the data unit. The event 
127      * shall be sent even if the data received does not form a complete 
128      * data unit. The event is sent only once. 
129      * The filter continues filtering after sending the event. 
130      * 
131      * @param milliseconds  a time-period value in milli seconds. A default 
132      *              value is -1 that indicates infinite. 
133      */
134     void setNotificationByTime(long milliseconds);
135 
136 
137     /**
138      * Set the number of data units to receive a cyclic notification. 
139      * By setting the number of data units, the OCAP implementation notifies 
140      * a VBIFilterEvent with EVENT_CODE_UNITS_NOTIFICATION cyclically 
141      * every time when the specified number of new data units are filtered 
142      * and stored in a buffer. 
143      * The filter continues filtering after sending the event. 
144      * 
145      * @param numberOfDataUnits  the number of data units to be notified. 
146      *              A default value is 0 that indicates no notification. 
147      *              Note that if a small number of data units is specified, 
148      *              the notification may be delayed and affect the host 
149      *              performance. For example, if 1 is specified for UNKNOWN 
150      *              data unit that comes every field (i.e., 1/60 seconds), 
151      *              the host has to notify every 1/60 seconds and makes an 
152      *              over load. 
153      *
154      * @throws IllegalArgumentException  if the numberOfDataUnit is larger 
155      *              than the bufferSize specified by a 
156      *              {@link VBIFilterGroup#newVBIFilter} method. 
157      */
158     void setNotificationByDataUnits(int numberOfDataUnits);
159 
160 
161     /**
162      * Add a new VBIFilterListener instance to this VBI filter. 
163      * If the same instance that exists currently is specified, this method 
164      * does nothing and no exception is thrown. 
165      *
166      * @param listener  a VBIFilterListener instance to be notified 
167      *              a VBI filtering events. 
168      */
169     void addVBIFilterListener(VBIFilterListener listener);
170 
171 
172     /**
173      * Remove an existing VBIFilterListener instance from this VBI filter. 
174      * If the specified instance has not been added, this method does nothing 
175      * and no exception is thrown. 
176      */
177     void removeVBIFilterListener(VBIFilterListener listener);
178 
179 
180     /**
181      * This method returns multiple VBI data unit bytes. The data unit 
182      * format is defined in a description of a {@link VBIFilter} interface. 
183      * The returned bytes is a simple concatenated VBI data at the moment. 
184      * Note that the return value is not aligned by a complete VBI data unit, 
185      * i.e., an incomplete data unit may return. 
186      * When this method is called, the internal buffer is cleared once, i.e., 
187      * the next call returns the next byte of retrieved VBI data. 
188      *
189      * @return  a concatenated VBI data of the form of specified by a 
190      *             {@link VBIFilterGroup#newVBIFilter} method. 
191      */
192     public byte[] getVBIData();
193 
194 
195     /**
196      * Clear an internal buffer to store retrieved VBI data. An application 
197      * shall call this method before the data buffer is full. 
198      */
199      public void clearBuffer();
