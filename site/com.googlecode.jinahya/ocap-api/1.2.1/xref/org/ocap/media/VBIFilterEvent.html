9   /**
10   * <p>
11   * This class represents a VBI filter event. When a specific event happens, 
12   * the {@link VBIFilterListener#filterUpdate} method is called with an event 
13   * that has a proper event code to indicate the event. 
14   * </p>
15   *
16   * @author  Shigeaki Watanabe (Panasonic)
17   */
18  public class VBIFilterEvent {
19      /**
20       * Indicates that the first VBI data unit is available. This event code 
21       * is issued only once after calling {@link VBIFilter#startFiltering} 
22       * method even if multiple lines/fields is specified to the filter. 
23       * Filtering continues. 
24       */
25      public static final int EVENT_CODE_FIRST_VBI_DATA_AVAILABLE = 1;
26  
27      /**
28       * Indicates current filtering is terminated forcibly for any reason 
29       * except other EVENT_CODE_ constants. 
30       * E.g., a {@link VBIFilter#stopFiltering} is called. 
31       */
32      public static final int EVENT_CODE_FORCIBLE_TERMINATED = 2;
33  
34      /**
35       * Indicates the current video for VBI data unit filtering has changed. 
36       * Note that current filtering doesn't stop even if this event happens. 
37       * An application may stop filtering and then restart to retrieve valid 
38       * data units. 
39       */
40      public static final int EVENT_CODE_VIDEO_SOURCE_CHANGED = 3;
41  
42      /**
43       * Indicates descrambling is unavailable for current video. 
44       * Note that current filtering doesn't stop even if this event happens. 
45       * Continues filtering until timeout. 
46       */
47      public static final int EVENT_CODE_FAILED_TO_DESCRAMBLE = 4;
48  
49      /**
50       * Indicates a timeout (specified by {@link VBIFilter#setTimeOut}) 
51       * occurred, i.e., this event code indicates no data unit is available. 
52       */
53      public static final int EVENT_CODE_TIMEOUT = 5;
54  
55      /**
56       * Indicates an internal buffer is full. Filtering stops automatically. 
57       */
58      public static final int EVENT_CODE_BUFFER_FULL = 6;
59  
60      /**
61       * Indicates that a specified time-period elapsed after receiving the 
62       * first byte of a data unit. 
63       */
64      public static final int EVENT_CODE_TIME_NOTIFICATION = 7;
65  
66      /**
67       * Indicates that the specified number of new data units are filtered 
68       * and stored in a buffer cyclically. 
69       */
70      public static final int EVENT_CODE_UNITS_NOTIFICATION= 8;
