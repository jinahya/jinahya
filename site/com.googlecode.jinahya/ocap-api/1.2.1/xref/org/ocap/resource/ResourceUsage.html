13   * This interface represents a grouping of resources specific to an function
14   * performed by an application.
15   */
16  public interface ResourceUsage

18      /**
19       * Gets the {@link AppID} of the application associated with the set
20       * of resources contained in the usage.
21       *
22       * @return AppID of the application.
23       */
24      public AppID getAppID();
25  
26      /**
27       * Gets the array of resource names associated with the resource reservation. 
28       *
29       * @return The array of qualified java class names for the resources used (or required) 
30       *    for this resource usage.
31       */
32      public java.lang.String [] getResourceNames();
33  
34      /**
35       * Gets the instance of {@link ResourceProxy} corresponding to a resource name returned
36       * by the getResourceNames method of this ResourceUsage. This method will return null if
37       * the resource is not yet reserved.  This method provides information to distinguish 
38       * which resources in the ResourceUsage have already been reserved. Since DAVIC resource 
39       * API reserves resource in one by one manner, the ResourceUsage may include both 
40       * reserved and unreserved resources.
41       *
42       * @param resourceName The fully qualified java name for a resource included in this
43       *    ResourceUsage.
44       *
45       * @return The instance of {@link ResourceProxy} corresponding to the resourceName.
46       *
47       * @throws IllegalArgumentException if the resourceName is not in the list of fully 
48       * qualified java class names returned by the method getResourceNames()
49       */
50      public ResourceProxy getResource(java.lang.String resourceName);
