6   /**
7    * This interface represents an MPEG-2 PMT Elementary Stream Info loop. 
8    * Each PMT will contain a loop of these blocks, as a block per an 
9    * elementary stream.
10   */
11  public interface PMTElementaryStreamInfo
12  {
13      /**
14       * Get the stream_type field.  Eight bit field specifying the type 
15       * of program element carried within the packets within the PID 
16       * returned by getElementaryPID().  See the StreamType interface for 
17       * defined values.
18       *
19       * @return The stream type.
20       */
21      public short getStreamType();
22  
23  
24      /**
25       * Get the elementary_PID field.  Thirteen bit field specifying the 
26       * PID of the associated elementary stream.
27       *
28       * @return The elementary PID.
29       */
30      public short getElementaryPID();
31  
32  
33      /**
34       * Get the descriptors associated with the elementary stream.
35       *
36       * @return The descriptor loop.
37       */
38      public Descriptor[] getDescriptorLoop();
39  
40  
41      /**
42       * Get the locator for the elementary stream.
43       * <p></p>
44       * For an Inband PMT, the returned OcapLocator
45       * corresponds to one of the following OCAP URL forms:
46       * <p></p>
47       * <tt>ocap://source_id.@&ltcomponent_tag&gt{&&ltcomponent_tag&gt}</tt></br>
48       * <tt>ocap://source_id.+PID</tt></br>
49       * <tt>ocap://f=frequency.program_number.@&ltcomponent_tag&gt{&&ltcomponent_tag&gt}</tt></br>
50       * <tt>ocap://f=frequency.program_number.+PID</tt>
51       * <p></p>
52       * The forms including the PID are returned if and only if no component
53       * tags are signaled.
54       * </br>
55       * The form returned (apart from the component tag and PID elements)
56       * must correspond to the form of the OCAP URL
57       * passed to the previous call to
58       * <tt>ProgramMapTableManager.retrieveInBand()</tt>
59       * or
60       * <tt>ProgramMapTableManager.addInBandChangeListener()</tt>.
61       * <p></p>
62       * For an OOB PMT, the returned OcapLocator
63       * corresponds to one of the following OCAP URL forms:
64       * <p></p>
65       * <tt>ocap://oobfdc.program_number.@&ltcomponent_tag&gt{&&ltcomponent_tag&gt}</tt></br>
66       * <tt>ocap://oobfdc.program_number.+PID</tt>
67       * <p></p>
68       * The form including the PID is returned if and only if no component tags
69       * are signaled.
70       * </br>
71       * <p></p>
72       * @return   The string which represents the URL of the elementary stream 
73       *           represented by this PMTElementaryStreamInfo. 
74       */
75      public java.lang.String getLocatorString();
76  }
