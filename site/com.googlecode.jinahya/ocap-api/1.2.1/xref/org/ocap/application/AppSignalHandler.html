3   package org.ocap.application;

5   /**
6    * <P>
7    * This interface represents a handler that can be registered in order to
8    * receive a notification whenever the XAIT version changes.
9    * An application which has a MonitorAppPermission("registrar") 
10   * may have a class implementing this interface, and may set it as
11   * the AppSignalHandler using the AppManagerProxy. 
12   * </P>
13   */
14  public interface AppSignalHandler
15  {
16      /**
17       * Notifies the registered handler that the version of the
18       * signaled XAIT has changed.  When an AppSignalHandler is
19       * registered the implementation SHALL call this method whenever
20       * the version of the XAIT changes or when an XAIT is received and a
21       * previous call to this method for the same version of the XAIT returned
22       * false.
23       * <p>
24       * When the return value is true the implementation SHALL process the
25       * corresponding XAIT as required by the OCAP specification.  When the 
26       * return value is false the implementation SHALL ignore the corresponding 
27       * XAIT and treat it as if it were never received. Subsequent XAIT receipt with
28       * the same version as the ignored table is treated as a new version and 
29       * notified using this method.
30       *
31       * @param newApps A list of instances of the OCAPAppAttributes class 
32       *       associated with all the applications whose details 
33       *       are listed in the new version of the XAIT.
34       * @return When true the XAIT is processed, otherwise the XAIT is ignored by 
35       *       the implementation.
36       */
37      public boolean notifyXAITUpdate(OcapAppAttributes[] newApps);
38  }
