31  package org.xml.sax;
32  
33  /**
34   * Interface for an XML filter.
35   *
36   * <p>An XML filter is like an XML reader, except that it obtains its
37   * events from another XML reader rather than a primary source like
38   * an XML document or database.  Filters can modify a stream of
39   * events as they pass on to the final application.</p>
40   *
41   * @since SAX 2.0
42   * @author David Megginson
43   * @see org.xml.sax.XMLReader
44   */
45  public interface XMLFilter extends XMLReader
46  {
47  
48      /**
49       * Set the parent reader.
50       *
51       * <p>This method allows the application to link the filter to
52       * a parent reader (which may be another filter).  The argument
53       * may not be null.</p>
54       *
55       * @param parent The parent reader.
56       */
57      public abstract void setParent (XMLReader parent);
58  
59  
60      /**
61       * Get the parent reader.
62       *
63       * <p>This method allows the application to query the parent
64       * reader (which may be another filter).  It is generally a
65       * bad idea to perform any operations on the parent reader
66       * directly: they should all pass through this filter.</p>
67       *
68       * @return The parent filter, or null if none has been set.
69       */
70      public abstract XMLReader getParent ();
71  
72  }
