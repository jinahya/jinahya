31  package org.xml.sax;
32  
33  /**
34   * Interface for a list of XML attributes.
35   *
36   * <p>This interface allows access to a list of attributes in
37   * three different ways:</p>
38   *
39   * <ol>
40   * <li>by attribute index;</li>
41   * <li>by Namespace-qualified name; or</li>
42   * <li>by qualified (prefixed) name.</li>
43   * </ol>
44   *
45   * <p>The list will not contain attributes that were declared
46   * #IMPLIED but not specified in the start tag.  It will also not
47   * contain attributes used as Namespace declarations (xmlns*) unless
48   * the <code>http://xml.org/sax/features/namespace-prefixes</code> 
49   * feature is set to <code>true</code> (it is <code>false</code> by 
50   * default).
51   * Because SAX2 conforms to the original "Namespaces in XML"
52   * recommendation, it normally does not
53   * give namespace declaration attributes a namespace URI.
54   * </p>
55   *
56   * <p>Some SAX2 parsers may support using an optional feature flag
57   * (<code>http://xml.org/sax/features/xmlns-uris</code>) to request
58   * that those attributes be given URIs, conforming to a later
59   * backwards-incompatible revision of that recommendation.  (The
60   * attribute's "local name" will be the prefix, or "xmlns" when
61   * defining a default element namespace.)  For portability, handler
62   * code should always resolve that conflict, rather than requiring
63   * parsers that can change the setting of that feature flag.  </p>
64   *
65   * <p>If the namespace-prefixes feature (see above) is
66   * <code>false</code>, access by qualified name may not be available; if
67   * the <code>http://xml.org/sax/features/namespaces</code> feature is
68   * <code>false</code>, access by Namespace-qualified names may not be
69   * available.</p>
70   *
71   * <p>The order of attributes in the list is unspecified, and will
72   * vary from implementation to implementation.</p>
73   *
74   * @since SAX 2.0
75   * @author David Megginson
76   * @see org.xml.sax.ext.Attributes2
77   */
78  public interface Attributes
79  {
80  
81      ////////////////////////////////////////////////////////////////////
82      // Indexed access.
83      ////////////////////////////////////////////////////////////////////
84  
85  
86      /**
87       * Return the number of attributes in the list.
88       *
89       * <p>Once you know the number of attributes, you can iterate
90       * through the list.</p>
91       *
92       * @return The number of attributes in the list.
93       * @see #getURI(int)
94       * @see #getLocalName(int)
95       * @see #getQName(int)
96       * @see #getType(int)
97       * @see #getValue(int)
98       */
99      public abstract int getLength ();
100 
101 
102     /**
103      * Look up an attribute's Namespace URI by index.
104      *
105      * @param index The attribute index (zero-based).
106      * @return The Namespace URI, or the empty string if none
107      *         is available, or null if the index is out of
108      *         range.
109      * @see #getLength
110      */
111     public abstract String getURI (int index);
112 
113 
114     /**
115      * Look up an attribute's local name by index.
116      *
117      * @param index The attribute index (zero-based).
118      * @return The local name, or the empty string if Namespace
119      *         processing is not being performed, or null
120      *         if the index is out of range.
121      * @see #getLength
122      */
123     public abstract String getLocalName (int index);
124 
125 
126     /**
127      * Look up an attribute's XML qualified (prefixed) name by index.
128      *
129      * @param index The attribute index (zero-based).
130      * @return The XML qualified name, or the empty string
131      *         if none is available, or null if the index
132      *         is out of range.
133      * @see #getLength
134      */
135     public abstract String getQName (int index);
136 
137 
138     /**
139      * Look up an attribute's type by index.
140      *
141      * <p>The attribute type is one of the strings "CDATA", "ID",
142      * "IDREF", "IDREFS", "NMTOKEN", "NMTOKENS", "ENTITY", "ENTITIES",
143      * or "NOTATION" (always in upper case).</p>
144      *
145      * <p>If the parser has not read a declaration for the attribute,
146      * or if the parser does not report attribute types, then it must
147      * return the value "CDATA" as stated in the XML 1.0 Recommendation
148      * (clause 3.3.3, "Attribute-Value Normalization").</p>
149      *
150      * <p>For an enumerated attribute that is not a notation, the
151      * parser will report the type as "NMTOKEN".</p>
152      *
153      * @param index The attribute index (zero-based).
154      * @return The attribute's type as a string, or null if the
155      *         index is out of range.
156      * @see #getLength
157      */
158     public abstract String getType (int index);
159 
160 
161     /**
162      * Look up an attribute's value by index.
163      *
164      * <p>If the attribute value is a list of tokens (IDREFS,
165      * ENTITIES, or NMTOKENS), the tokens will be concatenated
166      * into a single string with each token separated by a
167      * single space.</p>
168      *
169      * @param index The attribute index (zero-based).
170      * @return The attribute's value as a string, or null if the
171      *         index is out of range.
172      * @see #getLength
173      */
174     public abstract String getValue (int index);
175 
176 
177     ////////////////////////////////////////////////////////////////////
178     // Name-based query.
179     ////////////////////////////////////////////////////////////////////
180 
181 
182     /**
183      * Look up the index of an attribute by Namespace name.
184      *
185      * @param uri The Namespace URI, or the empty string if
186      *        the name has no Namespace URI.
187      * @param localName The attribute's local name.
188      * @return The index of the attribute, or -1 if it does not
189      *         appear in the list.
190      */
191     public int getIndex (String uri, String localName);
192 
193 
194     /**
195      * Look up the index of an attribute by XML qualified (prefixed) name.
196      *
197      * @param qName The qualified (prefixed) name.
198      * @return The index of the attribute, or -1 if it does not
199      *         appear in the list.
200      */
201     public int getIndex (String qName);
202 
203 
204     /**
205      * Look up an attribute's type by Namespace name.
206      *
207      * <p>See {@link #getType(int) getType(int)} for a description
208      * of the possible types.</p>
209      *
210      * @param uri The Namespace URI, or the empty String if the
211      *        name has no Namespace URI.
212      * @param localName The local name of the attribute.
213      * @return The attribute type as a string, or null if the
214      *         attribute is not in the list or if Namespace
215      *         processing is not being performed.
216      */
217     public abstract String getType (String uri, String localName);
218 
219 
220     /**
221      * Look up an attribute's type by XML qualified (prefixed) name.
222      *
223      * <p>See {@link #getType(int) getType(int)} for a description
224      * of the possible types.</p>
225      *
226      * @param qName The XML qualified name.
227      * @return The attribute type as a string, or null if the
228      *         attribute is not in the list or if qualified names
229      *         are not available.
230      */
231     public abstract String getType (String qName);
232 
233 
234     /**
235      * Look up an attribute's value by Namespace name.
236      *
237      * <p>See {@link #getValue(int) getValue(int)} for a description
238      * of the possible values.</p>
239      *
240      * @param uri The Namespace URI, or the empty String if the
241      *        name has no Namespace URI.
242      * @param localName The local name of the attribute.
243      * @return The attribute value as a string, or null if the
244      *         attribute is not in the list.
245      */
246     public abstract String getValue (String uri, String localName);
247 
248 
249     /**
250      * Look up an attribute's value by XML qualified (prefixed) name.
251      *
252      * <p>See {@link #getValue(int) getValue(int)} for a description
253      * of the possible values.</p>
254      *
255      * @param qName The XML qualified name.
256      * @return The attribute value as a string, or null if the
257      *         attribute is not in the list or if qualified names
258      *         are not available.
259      */
260     public abstract String getValue (String qName);
261 
262 }
