31  package org.xml.sax;
32  
33  import java.io.IOException;
34  
35  /**
36   * Basic interface for resolving entities.
37   *
38   * <p>If a SAX application needs to implement customized handling
39   * for external entities, it must implement this interface and
40   * register an instance with the SAX driver using the
41   * {@link org.xml.sax.XMLReader#setEntityResolver setEntityResolver}
42   * method.</p>
43   *
44   * <p>The XML reader will then allow the application to intercept any
45   * external entities (including the external DTD subset and external
46   * parameter entities, if any) before including them.</p>
47   *
48   * <p>Many SAX applications will not need to implement this interface,
49   * but it will be especially useful for applications that build
50   * XML documents from databases or other specialized input sources,
51   * or for applications that use URI types other than URLs.</p>
52   *
53   * <p>The following resolver would provide the application
54   * with a special character stream for the entity with the system
55   * identifier "http://www.myhost.com/today":</p>
56   *
57   * <pre>
58   * import org.xml.sax.EntityResolver;
59   * import org.xml.sax.InputSource;
60   *
61   * public class MyResolver implements EntityResolver {
62   *   public InputSource resolveEntity (String publicId, String systemId)
63   *   {
64   *     if (systemId.equals("http://www.myhost.com/today")) {
65   *              // return a special input source
66   *       MyReader reader = new MyReader();
67   *       return new InputSource(reader);
68   *     } else {
69   *              // use the default behaviour
70   *       return null;
71   *     }
72   *   }
73   * }
74   * </pre>
75   *
76   * <p>The application can also use this interface to redirect system
77   * identifiers to local URIs or to look up replacements in a catalog
78   * (possibly by using the public identifier).</p>
79   *
80   * @since SAX 1.0
81   * @author David Megginson
82   * @see org.xml.sax.XMLReader#setEntityResolver
83   * @see org.xml.sax.InputSource
84   */
85  public interface EntityResolver
86  {
87  
88      /**
89       * Allow the application to resolve external entities.
90       *
91       * <p>The parser will call this method before opening any external
92       * entity except the top-level document entity.  Such entities include
93       * the external DTD subset and external parameter entities referenced
94       * within the DTD (in either case, only if the parser reads external
95       * parameter entities), and external general entities referenced
96       * within the document element (if the parser reads external general
97       * entities).  The application may request that the parser locate
98       * the entity itself, that it use an alternative URI, or that it
99       * use data provided by the application (as a character or byte
100      * input stream).</p>
101      *
102      * <p>Application writers can use this method to redirect external
103      * system identifiers to secure and/or local URIs, to look up
104      * public identifiers in a catalogue, or to read an entity from a
105      * database or other input source (including, for example, a dialog
106      * box).  Neither XML nor SAX specifies a preferred policy for using
107      * public or system IDs to resolve resources.  However, SAX specifies
108      * how to interpret any InputSource returned by this method, and that
109      * if none is returned, then the system ID will be dereferenced as
110      * a URL.  </p>
111      *
112      * <p>If the system identifier is a URL, the SAX parser must
113      * resolve it fully before reporting it to the application.</p>
114      *
115      * @param publicId The public identifier of the external entity
116      *        being referenced, or null if none was supplied.
117      * @param systemId The system identifier of the external entity
118      *        being referenced.
119      * @return An InputSource object describing the new input source,
120      *         or null to request that the parser open a regular
121      *         URI connection to the system identifier.
122      * @exception org.xml.sax.SAXException Any SAX exception, possibly
123      *            wrapping another exception.
124      * @exception java.io.IOException A Java-specific IO exception,
125      *            possibly the result of creating a new InputStream
126      *            or Reader for the InputSource.
127      * @see org.xml.sax.InputSource
128      */
129     public abstract InputSource resolveEntity (String publicId, String systemId)
130         throws SAXException, IOException;
131 
132 }
