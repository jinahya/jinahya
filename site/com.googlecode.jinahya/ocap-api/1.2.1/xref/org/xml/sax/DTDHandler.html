31  package org.xml.sax;
32  
33  /**
34   * Receive notification of basic DTD-related events.
35   *
36   * <p>If a SAX application needs information about notations and
37   * unparsed entities, then the application implements this 
38   * interface and registers an instance with the SAX parser using 
39   * the parser's setDTDHandler method.  The parser uses the 
40   * instance to report notation and unparsed entity declarations to 
41   * the application.</p>
42   *
43   * <p>Note that this interface includes only those DTD events that
44   * the XML recommendation <em>requires</em> processors to report:
45   * notation and unparsed entity declarations.</p>
46   *
47   * <p>The SAX parser may report these events in any order, regardless
48   * of the order in which the notations and unparsed entities were
49   * declared; however, all DTD events must be reported after the
50   * document handler's startDocument event, and before the first
51   * startElement event.</p>
52   *
53   * <p>It is up to the application to store the information for 
54   * future use (perhaps in a hash table or object tree).
55   * If the application encounters attributes of type "NOTATION",
56   * "ENTITY", or "ENTITIES", it can use the information that it
57   * obtained through this interface to find the entity and/or
58   * notation corresponding with the attribute value.</p>
59   *
60   * @since SAX 1.0
61   * @author David Megginson
62   * @see org.xml.sax.XMLReader#setDTDHandler
63   */
64  public interface DTDHandler
65  {
66  
67      /**
68       * Receive notification of a notation declaration event.
69       *
70       * <p>It is up to the application to record the notation for later
71       * reference, if necessary;
72       * notations may appear as attribute values and in unparsed entity
73       * declarations, and are sometime used with processing instruction
74       * target names.</p>
75       *
76       * <p>At least one of publicId and systemId must be non-null.
77       * If a system identifier is present, and it is a URL, the SAX
78       * parser must resolve it fully before passing it to the
79       * application through this event.</p>
80       *
81       * <p>There is no guarantee that the notation declaration will be
82       * reported before any unparsed entities that use it.</p>
83       *
84       * @param name The notation name.
85       * @param publicId The notation's public identifier, or null if
86       *        none was given.
87       * @param systemId The notation's system identifier, or null if
88       *        none was given.
89       * @exception org.xml.sax.SAXException Any SAX exception, possibly
90       *            wrapping another exception.
91       * @see #unparsedEntityDecl
92       * @see org.xml.sax.Attributes
93       */
94      public abstract void notationDecl (String name, String publicId, String systemId)
95          throws SAXException;
96  
97  
98      /**
99       * Receive notification of an unparsed entity declaration event.
100      *
101      * <p>Note that the notation name corresponds to a notation
102      * reported by the {@link #notationDecl notationDecl} event.  
103      * It is up to the application to record the entity for later 
104      * reference, if necessary;
105      * unparsed entities may appear as attribute values. 
106      * </p>
107      *
108      * <p>If the system identifier is a URL, the parser must resolve it
109      * fully before passing it to the application.</p>
110      *
111      * @exception org.xml.sax.SAXException Any SAX exception, possibly
112      *            wrapping another exception.
113      * @param name The unparsed entity's name.
114      * @param publicId The entity's public identifier, or null if none
115      *        was given.
116      * @param systemId The entity's system identifier.
117      * @param notationName The name of the associated notation.
118      * @see #notationDecl
119      * @see org.xml.sax.Attributes
120      */
121     public abstract void unparsedEntityDecl (String name,
122         String publicId, String systemId, String notationName)
123         throws SAXException;
124 
125 }
