31  package org.xml.sax;
32  
33  /**
34   * Basic interface for SAX error handlers.
35   *
36   * <p>If a SAX application needs to implement customized error
37   * handling, it must implement this interface and then register an
38   * instance with the XML reader using the
39   * {@link org.xml.sax.XMLReader#setErrorHandler setErrorHandler}
40   * method.  The parser will then report all errors and warnings
41   * through this interface.</p>
42   *
43   * <p><strong>WARNING:</strong> If an application does <em>not</em>
44   * register an ErrorHandler, XML parsing errors will go unreported,
45   * except that <em>SAXParseException</em>s will be thrown for fatal errors.
46   * In order to detect validity errors, an ErrorHandler that does something
47   * with {@link #error error()} calls must be registered.</p>
48   *
49   * <p>For XML processing errors, a SAX driver must use this interface 
50   * in preference to throwing an exception: it is up to the application 
51   * to decide whether to throw an exception for different types of 
52   * errors and warnings.  Note, however, that there is no requirement that 
53   * the parser continue to report additional errors after a call to 
54   * {@link #fatalError fatalError}.  In other words, a SAX driver class 
55   * may throw an exception after reporting any fatalError.
56   * Also parsers may throw appropriate exceptions for non-XML errors.
57   * For example, {@link XMLReader#parse XMLReader.parse()} would throw
58   * an IOException for errors accessing entities or the document.</p>
59   *
60   * @since SAX 1.0
61   * @author David Megginson
62   * @see org.xml.sax.XMLReader#setErrorHandler
63   * @see org.xml.sax.SAXParseException 
64   */
65  public interface ErrorHandler
66  {
67  
68      /**
69       * Receive notification of a warning.
70       *
71       * <p>SAX parsers will use this method to report conditions that
72       * are not errors or fatal errors as defined by the XML
73       * recommendation.  The default behaviour is to take no
74       * action.</p>
75       *
76       * <p>The SAX parser must continue to provide normal parsing events
77       * after invoking this method: it should still be possible for the
78       * application to process the document through to the end.</p>
79       *
80       * <p>Filters may use this method to report other, non-XML warnings
81       * as well.</p>
82       *
83       * @param exception The warning information encapsulated in a
84       *                  SAX parse exception.
85       * @exception org.xml.sax.SAXException Any SAX exception, possibly
86       *            wrapping another exception.
87       * @see org.xml.sax.SAXParseException 
88       */
89      public abstract void warning (SAXParseException exception)
90          throws SAXException;
91  
92  
93      /**
94       * Receive notification of a recoverable error.
95       *
96       * <p>This corresponds to the definition of "error" in section 1.2
97       * of the W3C XML 1.0 Recommendation.  For example, a validating
98       * parser would use this callback to report the violation of a
99       * validity constraint.  The default behaviour is to take no
100      * action.</p>
101      *
102      * <p>The SAX parser must continue to provide normal parsing
103      * events after invoking this method: it should still be possible
104      * for the application to process the document through to the end.
105      * If the application cannot do so, then the parser should report
106      * a fatal error even if the XML recommendation does not require
107      * it to do so.</p>
108      *
109      * <p>Filters may use this method to report other, non-XML errors
110      * as well.</p>
111      *
112      * @param exception The error information encapsulated in a
113      *                  SAX parse exception.
114      * @exception org.xml.sax.SAXException Any SAX exception, possibly
115      *            wrapping another exception.
116      * @see org.xml.sax.SAXParseException 
117      */
118     public abstract void error (SAXParseException exception)
119         throws SAXException;
120 
121 
122     /**
123      * Receive notification of a non-recoverable error.
124      *
125      * <p><strong>There is an apparent contradiction between the
126      * documentation for this method and the documentation for {@link
127      * org.xml.sax.ContentHandler#endDocument}.  Until this ambiguity
128      * is resolved in a future major release, clients should make no
129      * assumptions about whether endDocument() will or will not be
130      * invoked when the parser has reported a fatalError() or thrown
131      * an exception.</strong></p>
132      *
133      * <p>This corresponds to the definition of "fatal error" in
134      * section 1.2 of the W3C XML 1.0 Recommendation.  For example, a
135      * parser would use this callback to report the violation of a
136      * well-formedness constraint.</p>
137      *
138      * <p>The application must assume that the document is unusable
139      * after the parser has invoked this method, and should continue
140      * (if at all) only for the sake of collecting additional error
141      * messages: in fact, SAX parsers are free to stop reporting any
142      * other events once this method has been invoked.</p>
143      *
144      * @param exception The error information encapsulated in a
145      *                  SAX parse exception.  
146      * @exception org.xml.sax.SAXException Any SAX exception, possibly
147      *            wrapping another exception.
148      * @see org.xml.sax.SAXParseException
149      */
150     public abstract void fatalError (SAXParseException exception)
151         throws SAXException;
152 
153 }
