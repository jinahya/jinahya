8   import java.awt.Graphics;

10  /**
11     The {@link org.havi.ui.HTextLayoutManager HTextLayoutManager} class
12     manages the layout and rendering on-screen of a
13     &quot;marked-up&quot; string.
14     <p>
15     Possible implementations of {@link org.havi.ui.HTextLayoutManager
16     HTextLayoutManager} could enable the following behaviors:
17     <ul>
18     <li> Interpreting basic markup, such as changing color or font, and
19     forced line breaks.
20     <li> Providing text alignment, as in the {@link
21     org.havi.ui.HDefaultTextLayoutManager HDefaultTextLayoutManager}.
22     <li> Providing text wrapping policies, such as word-wrap.
23     <li> Providing text orientations, such as right-to-left, or
24     top-to-bottom rendering.
25     <li> Providing specialized support for missing characters, or
26     fonts.
27     <li> Providing specific language support.
28     <li> Additional text styles, such as drop capitals or
29     &quot;shadow&quot; characters.
30     </ul>
31     <p>
32  
33     HTextLayoutManager supports passing a java.awt.Insets object as 
34     argument to the {@link
35     org.havi.ui.HTextLayoutManager#render render} method
36     to restrict the area in which text may be rendered. If the insets
37     are zero, the text is rendered into the area defined by the bounds
38     of the {@link org.havi.ui.HVisible HVisible} passed to the {@link
39     org.havi.ui.HTextLayoutManager#render render} method.
40  
41     <p> 
42     The clipping rectangle of the <code>Graphics</code> object passed
43     to the {@link org.havi.ui.HTextLayoutManager#render render} method
44     is <em>not</em> used to determine the area in which text is
45     rendered, as its size and position is not guaranteed to cover the
46     entire {@link org.havi.ui.HVisible HVisible} - for example, when
47     partial repainting of an {@link org.havi.ui.HVisible HVisible} is
48     performed. The diagram below shows a possible scenario:
49  
50     <p><table border>
51     <tr><td><img src="../../../images/HTextLayoutManager1.gif"></td></tr>
52     </table><p>
53  
54     The gray area shows the bounds of the {@link org.havi.ui.HVisible
55     HVisible}. The green area shows the clipping rectangle of the
56     <code>Graphics</code> context, and the dashed lines show the insets
57     passed to the {@link org.havi.ui.HTextLayoutManager#render render} 
58     method. The text is laid out into the rectangle
59     defined by the {@link org.havi.ui.HVisible HVisible} bounds after
60     subtracting the insets. However, only the part of the text covered
61     by the clipping rectangle is actually drawn to the screen, as shown
62     in the diagram below:
63  
64     <p><table border>
65     <tr><td><img src="../../../images/HTextLayoutManager2.gif"></td></tr>
66     </table><p>
67  
68     <p>
69     The behavior of the render method when the text to be rendered
70     does not fit in the current area specified is
71     implementation-specific.
72  
73  
74     @see org.havi.ui.HDefaultTextLayoutManager 
75  
76  */
77  
78  public interface HTextLayoutManager
79  {
80      /**
81       * Render the string. The {@link org.havi.ui.HTextLayoutManager
82       * HTextLayoutManager} should use the passed {@link
83       * org.havi.ui.HVisible HVisible} object to determine any
84       * additional information required to render the string,
85       * e.g. <code>Font</code>, <code>Color</code> etc.  
86       * <p>
87       * The text should be laid out in the layout area, which is
88       * defined by the bounds of the specified {@link
89       * org.havi.ui.HVisible HVisible}, after subtracting the 
90       * insets.  If the insets are <code>null</code> the full bounding
91       * rectangle is used as the area to render text into.
92       * <p> 
93       * The {@link org.havi.ui.HTextLayoutManager
94       * HTextLayoutManager} should not modify the clipping rectangle of
95       * the <code>Graphics</code> object.
96       * 
97       * @param markedUpString the string to render.
98       * @param g the graphics context, including a clipping rectangle
99       * which encapsulates the area within which rendering is
100      * permitted. If a valid insets value is passed to this method then
101      * text must only be rendered into the bounds of the widget after
102      * the insets are subtracted. If the insets value is <code>null</code>
103      * then text is rendered into the entire bounding area of the {@link
104      * org.havi.ui.HVisible HVisible}. It is implementation specific whether
105      * or not the renderer takes into account the intersection of the
106      * clipping rectangle in each case for optimization purposes.
107      * @param v the {@link org.havi.ui.HVisible HVisible} into which
108      * to render. 
109      * @param insets the insets  to
110      * determine the area in which to layout the text, or <code>null</code>.     
111      */
112     public void render(String markedUpString, java.awt.Graphics g, HVisible v, java.awt.Insets insets);
113 
114 }
