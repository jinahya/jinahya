9   /**
10   * The  {@link org.havi.ui.HAnimateEffect  HAnimateEffect}  interface defines effect constants and controls for time-varying animations. <p><hr><p> Implementations of HAnimateEffect should have the following default behaviors:  <ul> <li> By default the  {@link org.havi.ui.HAnimateEffect  HAnimateEffect}  should be stopped. Hence, to start an  {@link org.havi.ui.HAnimateEffect  HAnimateEffect}  the  {@link org.havi.ui.HAnimateEffect#start  start}  method must be explicitly invoked. This mechanism allows for animations that are programmatically controlled, e.g. via the  {@link org.havi.ui.HAnimateEffect#setPosition  setPosition}  method. <li> By default the position for rendering should be the first image in the sequence, i.e. 0. <li> By default the play mode should be  {@link org.havi.ui.HAnimateEffect#PLAY_REPEATING  PLAY_REPEATING} . <li> By default the repeat count should be  {@link org.havi.ui.HAnimateEffect#REPEAT_INFINITE  REPEAT_INFINITE} . <li> The default rendering should simply display the single image at the current position of the animation within the sequence. </ul>
11   */
12  
13  public interface HAnimateEffect
14  {
15      /**
16       * Indicates that the animation should be played forwards (in a
17       * repeating loop). <p> The images are rendered in the same order
18       * that they are present in the sequence (array):
19       * <p><center>0, 1, 2, 3, ... length-1</center><p>
20       * If the animation has not repeated sufficiently, then the
21       * rendering of the sequence is restarted from the first
22       * image, i.e. the images will continue to be rendered in the
23       * order:
24       * <p><center> 0, 1, 2, 3, ... length-1</center><p> 
25       * Each rendering of the sequence of images 0 to (length-1),
26       * should be considered as a single &quot;repeat&quot;.  
27       */
28      public static final int PLAY_REPEATING = 1;
29  
30      /**
31       * Indicates that the animation should be played in a repeating
32       * loop, alternating between the forward and reverse direction.
33       * <p> The images are rendered in the same order that they are
34       * present in the sequence (array):
35       * <p><center>0, 1, 2, 3, ... length-2, length-1 </center><p>
36       * If the animation has not repeated sufficiently, then the
37       * rendering of the sequence is reversed - i.e. the images are
38       * rendered in the order
39       * <p><center>length-2, length-3, ... 1, 0</center><p>
40       * If the animation has not repeated sufficiently, then the
41       * rendering of the sequence is reversed (again) back to a
42       * forwards direction. I.e. the images are rendered in the order
43       * <p><center>1, 2, 3, ... length-2, length-1</center></center><p>
44       * Each rendering of the sequence of images forwards or backwards,
45       * should be considered as a single &quot;repeat&quot;.
46       * <p>Note that when the sequence repeats, the last image (first
47       * image) is not rendered consecutively, i.e. twice.  
48       */
49      public static final int PLAY_ALTERNATING = 2;
50  
51      /**
52       * This value, when passed to <code>setRepeatCount</code>,
53       * indicates that the animation shall repeat until the {@link
54       * org.havi.ui.HAnimateEffect#stop stop} method is invoked.  
55       */
56      public static final int REPEAT_INFINITE = -1;
57  
58      /**
59       * This method starts this {@link org.havi.ui.HAnimateEffect
60       * HAnimateEffect} playing.  If <code>start</code> is called when
61       * the animation is already running it resets the animation
62       * according to the current play mode, as returned by {@link
63       * org.havi.ui.HAnimateEffect#getPlayMode getPlayMode}.
64       */
65      public void start();
66  
67      /**
68       * This method indicates that the running {@link
69       * org.havi.ui.HAnimateEffect HAnimateEffect} should be
70       * stopped. After calling this method, there is no guarantee that
71       * one or more frames will not be displayed before the animation
72       * actually stops playing. If the animation is already stopped
73       * further calls to <code>stop</code> have no effect.  
74       */
75      public void stop();
76  
77      /**
78       * This method indicates the animation (running) state of the
79       * {@link org.havi.ui.HAnimateEffect HAnimateEffect}.
80       * 
81       * @return <code>true</code> if this {@link org.havi.ui.HAnimateEffect
82       * HAnimateEffect} is running, i.e. the <code>start</code> method
83       * has been invoked - <code>false</code> otherwise.  
84       */
85      public boolean isAnimated();
86  
87      /**
88  	 * Set this  {@link org.havi.ui.HAnimateEffect  HAnimateEffect}  to display the content at the specified position. If the animation is already running a call to <code>setPosition</code> will change the current value and affect the animation immediately.
89  	 * @param position  an index into the content array which specifies  the next piece of content to be displayed. If  <code>position</code> is less than 0, then the array element at  index 0 is displayed, if <code>position</code> is greater than  or equal to the length of the content array, then the array  element at index [<code>length</code>-1] will be used.
90  	 * @uml.property  name="position"
91  	 */
92      public void setPosition(int position);
93  
94      /**
95  	 * Get the current index into the content array which this  {@link org.havi.ui.HAnimateEffect  HAnimateEffect}  is using to display content.
96  	 * @return  the index of the content currently being displayed, in  the range <code>0 <= index < length</code>
97  	 * @uml.property  name="position"
98  	 */
99      public int getPosition();

If the animation is already running a call to <code>setRepeatCount</code> will change the current value and reset the current number of repeats to 0, affecting the animation immediately.
103 	 * @param count  the number of times that an  {@link org.havi.ui.HAnimateEffect  HAnimateEffect}  should be  played. Valid values of the repeat count are one or more, and {@link org.havi.ui.HAnimateEffect#REPEAT_INFINITE  REPEAT_INFINITE}  .
104 	 * @uml.property  name="repeatCount"
105 	 */
106     public void setRepeatCount(int count);

108     /**
109 	 * Gets the number of times that this  {@link org.havi.ui.HAnimateEffect  HAnimateEffect}  is to be played. Note that this method does <em>not</em> return the number of repeats that are remaining to be played. <p> Except for <code>HAnimateEffect</code> implementations that specify a different default, <code>getRepeatCount()</code> returns <code>REPEAT_INFINITE</code> if no call to <code>setRepeatCount()</code> has previously been made.
110 	 * @return  the total number of times that an  {@link org.havi.ui.HAnimateEffect  HAnimateEffect}  is to be played. The  returned value shall be greater than zero, or  {@link org.havi.ui.HAnimateEffect#REPEAT_INFINITE  REPEAT_INFINITE}  .
111 	 * @uml.property  name="repeatCount"
112 	 */

117 	 * @param count  the content presentation delay in units of 0.1  seconds duration.  If count is less than one &quot;unit&quot;,  then it shall be treated as if it were a delay of one  &quot;unit&quot;, i.e. 0.1 seconds.

127     public int getDelay();
