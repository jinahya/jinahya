38  package org.w3c.dom.events;
39  
40  /**
41   * The <code>Event</code> interface is used to provide contextual information 
42   * about an event to the handler processing the event. An object which 
43   * implements the <code>Event</code> interface is generally passed as the 
44   * first parameter to an event handler. More specific context information is 
45   * passed to event handlers by deriving additional interfaces from 
46   * <code>Event</code> which contain information directly relating to the 
47   * type of event they accompany. These derived interfaces are also 
48   * implemented by the object passed to the event listener. 
49   * <p>See also the 
50   * <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Events-20001113'>
51   * Document Object Model (DOM) Level 2 Events Specification</a>.
52   * @since DOM Level 2
53   */
54  public interface Event {
55      // PhaseType
56      /**
57       * The current event phase is the capturing phase.
58       */
59      public static final short CAPTURING_PHASE           = 1;
60      /**
61       * The event is currently being evaluated at the target 
62       * <code>EventTarget</code>.
63       */
64      public static final short AT_TARGET                 = 2;
65      /**
66       * The current event phase is the bubbling phase.
67       */
68      public static final short BUBBLING_PHASE            = 3;
69  
70      /**
71       * The name of the event (case-sensitive). The name must be an XML name.
72       */
73      public String getType();
74  
75      /**
76       * Used to indicate the <code>EventTarget</code> to which the event was 
77       * originally dispatched. 
78       */
79      public EventTarget getTarget();
80  
81      /**
82       * Used to indicate the <code>EventTarget</code> whose 
83       * <code>EventListeners</code> are currently being processed. This is 
84       * particularly useful during capturing and bubbling. 
85       */
86      public EventTarget getCurrentTarget();
87  
88      /**
89       *  The namespace URI associated with this event at creation time, or 
90       * <code>null</code> if it is unspecified. 
91       * <br> For events initialized with a DOM Level 2 Events method 
92       * this is always <code>null</code>. 
93       * @since DOM Level 3
94       */
95      public String getNamespaceURI();
96  
97      /**
98       * Used to indicate which phase of event flow is currently being 
99       * evaluated. 
100      */
101     public short getEventPhase();
102 
103     /**
104      * Used to indicate whether or not an event is a bubbling event. If the 
105      * event can bubble the value is true, else the value is false. 
106      */
107     public boolean getBubbles();
108 
109     /**
110      * Used to indicate whether or not an event can have its default action 
111      * prevented. If the default action can be prevented the value is true, 
112      * else the value is false. 
113      */
114     public boolean getCancelable();
115 
116     /**
117      * Used to indicate whether <code>Event.preventDefault()</code> has been 
118      * called for this event. 
119      * <p><b>Note:</b>  Calling  <code>Event.preventDefault()</code> 
120      * for a non-cancelable event has no effect.
121      * @since DOM Level 3
122      */
123     public boolean getDefaultPrevented();
124 
125     /**
126      *  Used to specify the time (in milliseconds relative to the epoch) at 
127      * which the event was created. Due to the fact that some systems may 
128      * not provide this information the value of <code>timeStamp</code> may 
129      * be not available for all events. When not available, a value of 0 
130      * will be returned. Examples of epoch time are the time of the system 
131      * start or 0:0:0 UTC 1st January 1970. 
132      */
133     public long getTimeStamp();
134 
135     /**
136      * The <code>stopPropagation</code> method is used prevent further 
137      * propagation of an event during event flow. If this method is called 
138      * by any <code>EventListener</code> the event will cease propagating 
139      * through the tree. The event will complete dispatch to all listeners 
140      * on the current <code>EventTarget</code> before event flow stops. This 
141      * method may be used during any stage of event flow.
142      */
143     public void stopPropagation();
144 
145     /**
146      * If an event is cancelable, the <code>preventDefault</code> method is 
147      * used to signify that the event is to be canceled, meaning any default 
148      * action normally taken by the implementation as a result of the event 
149      * will not occur. If, during any stage of event flow, the 
150      * <code>preventDefault</code> method is called the event is canceled. 
151      * Any default action associated with the event will not occur. Calling 
152      * this method for a non-cancelable event has no effect. Once 
153      * <code>preventDefault</code> has been called it will remain in effect 
154      * throughout the remainder of the event's propagation. This method may 
155      * be used during any stage of event flow. 
156      */
157     public void preventDefault();
158 
159     /**
160      * The <code>initEvent</code> method is used to initialize the value of an 
161      * <code>Event</code> created through the <code>DocumentEvent</code> 
162      * interface. This method may only be called before the 
163      * <code>Event</code> has been dispatched via the 
164      * <code>dispatchEvent</code> method, though it may be called multiple 
165      * times during that phase if necessary. If called multiple times the 
166      * final invocation takes precedence. If called from a subclass of 
167      * <code>Event</code> interface only the values specified in the 
168      * <code>initEvent</code> method are modified, all other attributes are 
169      * left unchanged.
170      * @param eventTypeArg Specifies the event type. This type may be any 
171      *   event type currently defined in this specification or a new event 
172      *   type.. The string must be an XML name. Any new event type must not 
173      *   begin with any upper, lower, or mixed case version of the string 
174      *   "DOM". This prefix is reserved for future DOM event sets. It is 
175      *   also strongly recommended that third parties adding their own 
176      *   events use their own prefix to avoid confusion and lessen the 
177      *   probability of conflicts with other new events.
178      * @param canBubbleArg Specifies whether or not the event can bubble.
179      * @param cancelableArg Specifies whether or not the event's default 
180      *   action can be prevented.
181      */
182     public void initEvent(String eventTypeArg, 
183                           boolean canBubbleArg, 
184                           boolean cancelableArg);
185 
186     /**
187      *  The <code>initEventNS</code> method is used to initialize the value of 
188      * an <code>Event</code> object and has the same behavior as 
189      * <code>Event.initEvent()</code>. 
190      * @param namespaceURIArg  Specifies <code>Event.namespaceURI</code>, the 
191      *   namespace URI associated with this event, or <code>null</code> if 
192      *   no namespace. 
193      * @param eventTypeArg  Refer to the <code>Event.initEvent()</code> 
194      *   method for a description of this parameter. 
195      * @param canBubbleArg  Refer to the <code>Event.initEvent()</code> 
196      *   method for a description of this parameter. 
197      * @param cancelableArg  Refer to the <code>Event.initEvent()</code> 
198      *   method for a description of this parameter.
199      *
200      * @since DOM Level 3
201      */
202     public void initEventNS(String namespaceURIArg, 
203                             String eventTypeArg, 
204                             boolean canBubbleArg, 
205                             boolean cancelableArg);
206 }
