38  package org.w3c.dom.events;
39  
40  import org.w3c.dom.views.AbstractView;
41  
42  /**
43   *  The <code>KeyboardEvent</code> interface provides specific contextual 
44   * information associated with keyboard devices. Each keyboard event 
45   * references a key using an identifier. Keyboard events are commonly 
46   * directed at the element that has the focus. 
47   * <p> The <code>KeyboardEvent</code> interface provides convenient attributes 
48   * for some common modifiers keys: <code>KeyboardEvent.ctrlKey</code>, 
49   * <code>KeyboardEvent.shiftKey</code>, <code>KeyboardEvent.altKey</code>, 
50   * <code>KeyboardEvent.metaKey</code>.  
51   * <p> To create an instance of the <code>KeyboardEvent</code> interface, use 
52   * the <code>DocumentEvent.createEvent("KeyboardEvent")</code> method call. 
53   * <p>See also the 
54   * <a href='http://www.w3.org/TR/2006/WD-DOM-Level-3-Events-20060413'>
55   * Document Object Model (DOM) Level 3 Events Specification</a>.
56   *
57   * @since DOM Level 3
58   */
59  public interface KeyboardEvent extends UIEvent {
60  
61      /**
62       *  The key activation is not distinguished as the left or right version 
63       * of the key, and did not originate from the numeric keypad (or did not 
64       * originate with a virtual key corresponding to the numeric keypad). 
65       * Example: the 'Q' key on a PC 101 Key US keyboard. 
66       */
67      public static final int DOM_KEY_LOCATION_STANDARD = 0x00;
68  
69      /**
70       *  The key activated is in the left key location (there is more than one 
71       * possible location for this key). Example: the left Shift key on a PC 
72       * 101 Key US keyboard. 
73       */
74      public static final int DOM_KEY_LOCATION_LEFT     = 0x01;
75  
76      /**
77       *  The key activation is in the right key location (there is more than 
78       * one possible location for this key). Example: the right Shift key on 
79       * a PC 101 Key US keyboard. 
80       */
81      public static final int DOM_KEY_LOCATION_RIGHT    = 0x02;
82  
83      /**
84       *  The key activation originated on the numeric keypad or with a virtual 
85       * key corresponding to the numeric keypad. Example: the '1' key on a PC 
86       * 101 Key US keyboard located on the numeric pad. 
87       */
88      public static final int DOM_KEY_LOCATION_NUMPAD   = 0x03;
89  
90      /**
91       *  <code>keyIdentifier</code> holds the identifier of the key. The key 
92       * identifiers are defined in Appendix A.2 <a 
93       * href="http://www.w3.org/TR/DOM-Level-3-Events/keyset.html#KeySet-Set">
94       * "Key identifiers set"</a>. Implementations that are 
95       * unable to identify a key must use the key identifier 
96       * <code>"Unidentified"</code>. 
97       */
98      public String getKeyIdentifier();
99  
100     /**
101      *  The <code>keyLocation</code> attribute contains an indication of the 
102      * location of they key on the device, as described in <a 
103      * href="http://www.w3.org/TR/DOM-Level-3-Events/events.html#ID-KeyboardEvent-KeyLocationCode">
104      * Keyboard event types</a>. 
105      */
106     public int getKeyLocation();
107 
108     /**
109      *  <code>true</code> if the control (Ctrl) key modifier is activated. 
110      */
111     public boolean getCtrlKey();
112 
113     /**
114      *  <code>true</code> if the shift (Shift) key modifier is activated. 
115      */
116     public boolean getShiftKey();
117 
118     /**
119      *  <code>true</code> if the alternative (Alt) key modifier is activated. 
120      * <p ><b>Note:</b>  The Option key modifier on Macintosh systems must be 
121      * represented using this key modifier. 
122      */
123     public boolean getAltKey();
124 
125     /**
126      *  <code>true</code> if the meta (Meta) key modifier is activated. 
127      * <p ><b>Note:</b>  The Command key modifier on Macintosh systems must be 
128      * represented using this key modifier. 
129      */
130     public boolean getMetaKey();
131 
132     /**
133      *  The <code>initKeyboardEvent</code> method is used to initialize the 
134      * value of a <code>KeyboardEvent</code> object and has the same 
135      * behavior as <code>UIEvent.initUIEvent()</code>. The value of 
136      * <code>UIEvent.detail</code> remains undefined. 
137      * @param typeArg  Refer to the <code>UIEvent.initUIEvent()</code> method 
138      *   for a description of this parameter. 
139      * @param canBubbleArg  Refer to the <code>UIEvent.initUIEvent()</code> 
140      *   method for a description of this parameter. 
141      * @param cancelableArg  Refer to the <code>UIEvent.initUIEvent()</code> 
142      *   method for a description of this parameter. 
143      * @param viewArg  Refer to the <code>UIEvent.initUIEvent()</code> method 
144      *   for a description of this parameter. 
145      * @param keyIdentifierArg  Specifies 
146      *   <code>KeyboardEvent.keyIdentifier</code>. 
147      * @param keyLocationArg  Specifies <code>KeyboardEvent.keyLocation</code>
148      *   . 
149      * @param modifiersList  A 
150      *   <a href='http://www.w3.org/TR/2004/REC-xml-20040204/#NT-S'>white space
151      *   </a> separated list of modifier key identifiers to be activated on 
152      *   this object. 
153      */
154     public void initKeyboardEvent(String typeArg, 
155                                   boolean canBubbleArg, 
156                                   boolean cancelableArg, 
157                                   AbstractView viewArg, 
158                                   String keyIdentifierArg, 
159                                   int keyLocationArg, 
160                                   String modifiersList);
161 
162     /**
163      *  The <code>initKeyboardEventNS</code> method is used to initialize the 
164      * value of a <code>KeyboardEvent</code> object and has the same 
165      * behavior as <code>UIEvent.initUIEventNS()</code>. The value of 
166      * <code>UIEvent.detail</code> remains undefined. 
167      * @param namespaceURIArg  Refer to the <code>UIEvent.initUIEventNS()</code> 
168      *   method for a description of this parameter. 
169      * @param typeArg  Refer to the <code>UIEvent.initUIEventNS()</code> 
170      *   method for a description of this parameter. 
171      * @param canBubbleArg  Refer to the <code>UIEvent.initUIEventNS()</code> 
172      *   method for a description of this parameter. 
173      * @param cancelableArg  Refer to the <code>UIEvent.initUIEventNS()</code>
174      *   method for a description of this parameter. 
175      * @param viewArg  Refer to the <code>UIEvent.initUIEventNS()</code> 
176      *   method for a description of this parameter. 
177      * @param keyIdentifierArg  Refer to the 
178      *   <code>KeyboardEvent.initKeyboardEvent()</code> method for a 
179      *   description of this parameter. 
180      * @param keyLocationArg  Refer to the 
181      *   <code>KeyboardEvent.initKeyboardEvent()</code> method for a 
182      *   description of this parameter. 
183      * @param modifiersList  A 
184      *   <a href='http://www.w3.org/TR/2004/REC-xml-20040204/#NT-S'>white space
185      *   </a> separated list of modifier key identifiers to be activated on 
186      *   this object. As an example, <code>"Control Alt"</code> will activated 
187      *   the control and alt modifiers. 
188      */
189     public void initKeyboardEventNS(String namespaceURIArg, 
190                                     String typeArg, 
191                                     boolean canBubbleArg, 
192                                     boolean cancelableArg, 
193                                     AbstractView viewArg, 
194                                     String keyIdentifierArg, 
195                                     int keyLocationArg, 
196                                     String modifiersList);
197 }
