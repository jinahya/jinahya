47  package org.w3c.dom;
48  
49  /**
50   * Each <code>Document</code> has a <code>doctype</code> attribute whose value 
51   * is either <code>null</code> or a <code>DocumentType</code> object. The 
52   * <code>DocumentType</code> interface in the DOM Core provides an interface 
53   * to the list of entities that are defined for the document, and little 
54   * else because the effect of namespaces and the various XML schema efforts 
55   * on DTD representation are not clearly understood as of this writing.
56   * <br><b>Note:</b> If a parser expands entity references, the information
57   * contained in the corresponding Entity node becomes redundant. In the interest
58   * of footprint and performance, JSR 280 allows implementations to exclude
59   * Entity nodes corresponding to expanded entity references from
60   * the DOM tree if those references have been fully expanded by the parser. 
61   * <p>The DOM Level 2 and Level 3 do not support editing <code>DocumentType</code> nodes.
62   * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
63   */
64  public interface DocumentType extends Node {
65      /**
66       * The name of DTD; i.e., the name immediately following the 
67       * <code>DOCTYPE</code> keyword.
68       * @return the name of the DTD
69       */
70      public String getName();
71  
72      /**
73       * A <code>NamedNodeMap</code> containing the general entities, both 
74       * external and internal, declared in the DTD. Parameter entities are 
75       * not contained. Duplicates are discarded. For example in: 
76       * <pre>&lt;!DOCTYPE 
77       * ex SYSTEM "ex.dtd" [ &lt;!ENTITY foo "foo"&gt; &lt;!ENTITY bar 
78       * "bar"&gt; &lt;!ENTITY bar "bar2"&gt; &lt;!ENTITY % baz "baz"&gt; 
79       * ]&gt; &lt;ex/&gt;</pre>
80       *  the interface provides access to <code>foo</code> 
81       * and the first declaration of <code>bar</code> but not the second 
82       * declaration of <code>bar</code> or <code>baz</code>. Every node in 
83       * this map also implements the <code>Entity</code> interface.
84       * <br>The DOM Level 2 does not support editing entities, therefore 
85       * <code>entities</code> cannot be altered in any way.
86       * @return A <code>NamedNodeMap</code> containing the general entities 
87       * contained in the DTD
88       */
89      public NamedNodeMap getEntities();
90  
91      /**
92       * A <code>NamedNodeMap</code> containing the notations declared in the 
93       * DTD. Duplicates are discarded. Every node in this map also implements 
94       * the <code>Notation</code> interface.
95       * <br>The DOM Level 2 does not support editing notations, therefore 
96       * <code>notations</code> cannot be altered in any way.
97       * @return A <code>NamedNodeMap</code> containing the notations declared in the DTD
98       */
99      public NamedNodeMap getNotations();
100 
101     /**
102      * The public identifier of the external subset.
103      * @since DOM Level 2
104      * @return the public identifier of the external subset
105      */
106     public String getPublicId();
107 
108     /**
109      * The system identifier of the external subset.
110      * @since DOM Level 2
111      * @return The system identifier of the external subset
112      */
113     public String getSystemId();
114 
115     /**
116      * The internal subset as a string, or <code>null</code> if there is none.
117      * <p><b>Note:</b> The actual content returned depends on how much
118      * information is available to the implementation. This may vary
119      * depending on various parameters, including the XML processor used to
120      * build the document.
121      * @since DOM Level 2
122      * @return a String containing a representation of the internal subset
123      */
124     public String getInternalSubset();
125 
126 }
