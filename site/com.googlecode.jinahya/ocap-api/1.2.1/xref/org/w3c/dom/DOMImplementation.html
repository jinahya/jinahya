38  package org.w3c.dom;
39  
40  /**
41   * The <code>DOMImplementation</code> interface provides a number of methods 
42   * for performing operations that are independent of any particular instance 
43   * of the document object model.
44   * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
45   */
46  public interface DOMImplementation {
47      /**
48       * Test if the DOM implementation implements a specific feature.
49       * @param feature The name of the feature to test (case-insensitive). The 
50       *   values used by DOM features are defined throughout the DOM Level 2 
51       *   specifications and listed in the  section. The name must be an XML 
52       *   name. To avoid possible conflicts, as a convention, names referring 
53       *   to features defined outside the DOM specification should be made 
54       *   unique by reversing the name of the Internet domain name of the 
55       *   person (or the organization that the person belongs to) who defines 
56       *   the feature, component by component, and using this as a prefix. 
57       *   For instance, the W3C SVG Working Group defines the feature 
58       *   "org.w3c.dom.svg".
59       * @param version This is the version number of the feature to test. In 
60       *   Level 2, the string can be either "2.0" or "1.0". If the version is 
61       *   not specified (parameter is either null or the empty string, as per 
62       *   http://www.w3.org/TR/DOM-Level-3-Events/events.html#Conformance), 
63       *   supporting any version of the feature causes the 
64       *   method to return <code>true</code>.
65       * @return <code>true</code> if the feature is implemented in the 
66       *   specified version, <code>false</code> otherwise.
67       */
68      public boolean hasFeature(String feature, 
69                                String version);
70  
71      /**
72       * Creates an empty <code>DocumentType</code> node. Entity declarations 
73       * and notations are not made available. Entity reference expansions and 
74       * default attribute additions do not occur. It is expected that a 
75       * future version of the DOM will provide a way for populating a 
76       * <code>DocumentType</code>.
77       * @param qualifiedName The qualified name of the document type to be 
78       *   created. 
79       * @param publicId The external subset public identifier.
80       * @param systemId The external subset system identifier.
81       * @return A new <code>DocumentType</code> node with 
82       *   <code>Node.ownerDocument</code> set to <code>null</code>.
83       * @exception DOMException
84       *   INVALID_CHARACTER_ERR: Raised if the specified qualified name 
85       *   contains an illegal character.
86       *   <br>NAMESPACE_ERR: Raised if the <code>qualifiedName</code> is 
87       *   malformed.
88       * @since DOM Level 2
89       */
90      public DocumentType createDocumentType(String qualifiedName, 
91                                             String publicId, 
92                                             String systemId)
93                                             throws DOMException;
94  
95      /**
96       * Creates an XML <code>Document</code> object of the specified type with 
97       * its document element. 
98       * @param namespaceURI The namespace URI of the document element to create.
99       * @param qualifiedName The qualified name of the document element to be 
100      *   created.
101      * @param doctype The type of document to be created or <code>null</code>.
102      *   When <code>doctype</code> is not <code>null</code>, its 
103      *   <code>Node.ownerDocument</code> attribute is set to the document 
104      *   being created.
105      * @return A new <code>Document</code> object.
106      * @exception DOMException
107      *   INVALID_CHARACTER_ERR: Raised if the specified qualified name 
108      *   contains an illegal character.
109      *   <br>NAMESPACE_ERR: Raised if the <code>qualifiedName</code> is 
110      *   malformed, if the <code>qualifiedName</code> has a prefix and the 
111      *   <code>namespaceURI</code> is <code>null</code>, or if the 
112      *   <code>qualifiedName</code> has a prefix that is "xml" and the 
113      *   <code>namespaceURI</code> is different from "
114      *   http://www.w3.org/XML/1998/namespace" .
115      *   <br>WRONG_DOCUMENT_ERR: Raised if <code>doctype</code> has already 
116      *   been used with a different document or was created from a different 
117      *   implementation.
118      * @since DOM Level 2
119      */
120     public Document createDocument(String namespaceURI, 
121                                    String qualifiedName, 
122                                    DocumentType doctype)
123                                    throws DOMException;
124 
125     /**
126      *  This method returns a specialized object which implements the 
127      * specialized APIs of the specified feature and version, as specified 
128      * in <a href="http://www.w3.org/TR/DOM-Level-3-Core/core.html#DOMFeatures"
129      * >DOM Features</a>. The specialized object may also be obtained by using 
130      * binding-specific casting methods but is not necessarily expected to, 
131      * as discussed in 
132      * <a href="http://www.w3.org/TR/DOM-Level-3-Core/core.html#Embedded-DOM"
133      * >Mixed DOM Implementations</a>.
134      * This method also allow the implementation to 
135      * provide specialized objects which do not support the 
136      * <code>DOMImplementation</code> interface. 
137      * <p><b>Note:</b> when using the methods that take a feature and a 
138      * version as parameters, applications can use null or empty string 
139      * for the version parameter if they don't wish to specify a particular
140      * version for the specified feature.
141      * @param feature  The name of the feature requested. Note that any plus 
142      *   sign "+" prepended to the name of the feature will be ignored since 
143      *   it is not significant in the context of this method. 
144      * @param version  This is the version number of the feature to test. 
145      * @return  Returns an object which implements the specialized APIs of 
146      *   the specified feature and version, if any, or <code>null</code> if 
147      *   there is no object which implements interfaces associated with that 
148      *   feature. If the <code>DOMObject</code> returned by this method 
149      *   implements the <code>DOMImplementation</code> interface, it must 
150      *   delegate to the primary core <code>DOMImplementation</code> and not 
151      *   return results inconsistent with the primary core 
152      *   <code>DOMImplementation</code> such as <code>hasFeature</code>, 
153      *   <code>getFeature</code>, etc. 
154      * @since DOM Level 3
155      */
156     public Object getFeature(String feature, 
157                              String version);
158 }
