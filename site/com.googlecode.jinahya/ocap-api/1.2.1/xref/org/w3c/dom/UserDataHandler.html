38  package org.w3c.dom;
39  
40  /**
41   * When associating an object to a key on a node using 
42   * <code>Node.setUserData()</code> the application can provide a handler 
43   * that gets called when the node the object is associated to is being 
44   * cloned, imported, or renamed. This can be used by the application to 
45   * implement various behaviors regarding the data it associates to the DOM 
46   * nodes. This interface defines that handler. 
47   * <p>See also the <a href='http://www.w3.org/TR/2004/REC-DOM-Level-3-Core-20040407'>Document Object Model (DOM) Level 3 Core Specification</a>.
48   * @since DOM Level 3
49   */
50  public interface UserDataHandler {
51      // OperationType
52      /**
53       * The node is cloned, using <code>Node.cloneNode()</code>.
54       */
55      public static final short NODE_CLONED               = 1;
56      /**
57       * The node is imported, using <code>Document.importNode()</code>.
58       */
59      public static final short NODE_IMPORTED             = 2;
60      /**
61       * The node is deleted.
62       * <p ><b>Note:</b> This may not be supported or may not be reliable in 
63       * certain environments, such as Java, where the implementation has no 
64       * real control over when objects are actually deleted.
65       */
66      public static final short NODE_DELETED              = 3;
67      /**
68       * The node is adopted, using <code>Document.adoptNode()</code>.
69       */
70      public static final short NODE_ADOPTED              = 5;
71  
72      /**
73       * This method is called whenever the node for which this handler is 
74       * registered is imported or cloned.
75       * <br> DOM applications must not raise exceptions in a 
76       * <code>UserDataHandler</code>. The effect of throwing exceptions from 
77       * the handler is DOM implementation dependent. 
78       * @param operation Specifies the type of operation that is being 
79       *   performed on the node.
80       * @param key Specifies the key for which this handler is being called. 
81       * @param data Specifies the data for which this handler is being called. 
82       * @param src Specifies the node being cloned, adopted, imported, or 
83       *   renamed. This is <code>null</code> when the node is being deleted.
84       * @param dst Specifies the node newly created if any, or 
85       *   <code>null</code>.
86       */
87      public void handle(short operation, 
88                         String key, 
89                         Object data, 
90                         Node src, 
91                         Node dst);
92  
93  }
