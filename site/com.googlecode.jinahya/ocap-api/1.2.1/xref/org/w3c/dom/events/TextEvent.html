38  package org.w3c.dom.events;
39  
40  import org.w3c.dom.views.AbstractView;
41  
42  /**
43   *  The <code>TextEvent</code> interface provides specific contextual 
44   * information associated with Text Events. 
45   * <p> To create an instance of the <code>TextEvent</code> interface, use the 
46   * <code>DocumentEvent.createEvent("TextEvent")</code> method call. 
47   * <p>See also the 
48   * <a href='http://www.w3.org/TR/2006/WD-DOM-Level-3-Events-20060413'>
49   * Document Object Model (DOM) Level 3 Events Specification</a>.
50   *
51   * @since DOM Level 3
52   */
53  
54  public interface TextEvent extends UIEvent {
55  
56      /**
57       *  <code>data</code> holds the value of the characters generated by the 
58       * character device. This may be a single Unicode character or a 
59       * non-empty sequence of Unicode characters [Unicode]. Characters should be 
60       * normalized as defined by the Unicode normalization form NFC, defined in 
61       * [<a href='http://www.unicode.org/reports/tr15/'>UAX #15</a>]. This 
62       * attribute cannot be null or contain the empty string. 
63       */
64      public String getData();
65  
66      /**
67       *  The <code>initTextEvent</code> method is used to initialize the value 
68       * of a <code>TextEvent</code> object and has the same behavior as 
69       * <code>UIEvent.initUIEvent()</code>. The value of 
70       * <code>UIEvent.detail</code> remains undefined. 
71       * @param typeArg  Refer to the <code>UIEvent.initUIEvent()</code> method 
72       *   for a description of this parameter. 
73       * @param canBubbleArg  Refer to the <code>UIEvent.initUIEvent()</code> 
74       *   method for a description of this parameter. 
75       * @param cancelableArg  Refer to the <code>UIEvent.initUIEvent()</code> 
76       *   method for a description of this parameter. 
77       * @param viewArg  Refer to the <code>UIEvent.initUIEvent()</code> method 
78       *   for a description of this parameter. 
79       * @param dataArg  Specifies <code>TextEvent.data</code>. 
80       */
81      public void initTextEvent(String typeArg, 
82                                boolean canBubbleArg, 
83                                boolean cancelableArg, 
84                                AbstractView viewArg, 
85                                String dataArg);
86  
87      /**
88       *  The <code>initTextEventNS</code> method is used to initialize the 
89       * value of a <code>TextEvent</code> object and has the same behavior as 
90       * <code>UIEvent.initUIEventNS()</code>. The value of 
91       * <code>UIEvent.detail</code> remains undefined. 
92       * @param namespaceURIArg  Refer to the <code>UIEvent.initUIEventNS()</code> 
93       *   method for a description of this parameter. 
94       * @param typeArg  Refer to the <code>UIEvent.initUIEventNS()</code> method 
95       *   for a description of this parameter. 
96       * @param canBubbleArg  Refer to the <code>UIEvent.initUIEventNS()</code> 
97       *   method for a description of this parameter. 
98       * @param cancelableArg  Refer to the <code>UIEvent.initUIEventNS()</code>
99       *   method for a description of this parameter. 
100      * @param viewArg  Refer to the <code>UIEvent.initUIEventNS()</code> 
101      *   method for a description of this parameter. 
102      * @param dataArg  Refer to the <code>TextEvent.initTextEvent()</code> 
103      *   method for a description of this parameter. 
104      */
105     public void initTextEventNS(String namespaceURIArg, 
106                                 String typeArg, 
107                                 boolean canBubbleArg, 
108                                 boolean cancelableArg, 
109                                 AbstractView viewArg, 
110                                 String dataArg);
111 }
