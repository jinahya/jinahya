38  package org.w3c.dom.events;
39  
40  /**
41   *  The CustomEvent interface is the recommended interface for 
42   * application-specific event types. Unlike the <code>Event</code> 
43   * interface, it allows applications to provide contextual information about 
44   * the event type. Application-specific event types should have an 
45   * associated namespace to avoid clashes with future general-purpose event 
46   * types. 
47   * <p> To create an instance of the <code>CustomEvent</code> interface, use 
48   * the <code>DocumentEvent.createEvent("CustomEvent")</code> method call. 
49   * <p>See also the 
50   * <a href='http://www.w3.org/TR/2006/WD-DOM-Level-3-Events-20060413'>
51   * Document Object Model (DOM) Level 3 Events Specification</a>.
52   *
53   * @since DOM Level 3
54   */
55  
56  public interface CustomEvent extends Event {
57  
58      /**
59       *  Specifies some detail information about the <code>Event</code>. 
60       */
61      public Object getDetail();
62  
63      /**
64       *  The <code>initCustomEventNS</code> method is used to initialize the 
65       * value of a <code>CustomEvent</code> object and has the same behavior 
66       * as <code>Event.initEventNS()</code>. 
67       * @param namespaceURIArg  Refer to the <code>Event.initEventNS()</code> 
68       *   method for a description of this parameter. 
69       * @param typeArg  Refer to the <code>Event.initEventNS()</code> method 
70       *   for a description of this parameter. 
71       * @param canBubbleArg  Refer to the <code>Event.initEventNS()</code> 
72       *   method for a description of this parameter. 
73       * @param cancelableArg  Refer to the <code>Event.initEventNS()</code> 
74       *   method for a description of this parameter. 
75       * @param detailArg  Specifies <code>CustomEvent.detail</code>. This 
76       *   value may be <code>null</code>.   
77       */
78      public void initCustomEventNS(String namespaceURIArg, 
79                                    String typeArg, 
80                                    boolean canBubbleArg, 
81                                    boolean cancelableArg, 
82                                    Object detailArg);
83  }
