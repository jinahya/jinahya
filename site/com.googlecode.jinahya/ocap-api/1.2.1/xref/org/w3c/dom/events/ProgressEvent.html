38  package org.w3c.dom.events;
39  
40  /**
41   * <p>A progress event occurs when the user agent makes progress in some data
42   * transfer operation, such as loading a resource from the web via
43   * XMLHttpRequest. 
44   *
45   * <p><b>Note: </b> This interface has not yet been accepted for
46   * inclusion in the W3C DOM 3 Event specification. It may be removed from the
47   * JSR 280 specification or added to a different namespace in a
48   * subsequent version of the specification.</p>
49   *
50   * Other specifications which have a
51   * use case for these events <em>should</em> define when ProgressEvent events
52   * are dispatched.</p>
53   *
54   * <ul>
55   * <li>The user agent <em>must</em> dispatch a <code>start</code> event when a
56   *     relevant operation has begun. </li>
57   * <li>The user agent <span class="term"><em>may</em></span> dispatch one or
58   *     more <code>progress</code> events while a network operation is taking
59   *     place.</li>
60   * <li>If the operation fails to complete the user agent <em>must</em>
61   *     dispatch either an <code>error</code> event, if the failure to complete
62   *     was due to an error (such as a timeout, or network error), or an
63   *     <code>abort</code> event if the operation was deliberately cancelled
64   *     (e.g. by user interaction or through a script call).</li>
65   * <li>If the operation successfully completes, the user agent <em>must</em>
66   *     dispatch a <code>load</code> event. </li>
67   * </ul>
68   * <p>In short, there must be at least one <code>start</code> event, followed by
69   * zero or more <code>progress</code> events, followed by one event which may be
70   * any of <code>error</code>, <code>abort</code> or <code>load</code>, according
71   * to the outcome of the operation.</p>
72   *
73   * <h3 id="Event">Event definitions</h3>
74   *
75   * <p>The following events are defined in this specification</p>
76   *
77   * <table border="1">
78   *  <caption></caption>
79   *  <tbody>
80   *    <tr>
81   *      <th>Name</th>
82   *      <th>Description</th>
83   *      <th>How many?</th>
84   *      <th>When?</th>
85   *    </tr>
86   *    <tr>
87   *      <td><code>loadstart</code></td>
88   *      <td>The operation has begun</td>
89   *      <td>once</td>
90   *      <td>Must be dispatched first</td>
91   *    </tr>
92   *    <tr>
93   *      <td><code>progress</code></td>
94   *      <td>The operation is in progress</td>
95   *      <td>zero or more</td>
96   *      <td>May be dispatched zero or more times after a <code>loadstart</code>
97   *        event, before any of error, abort or load events are dispatched</td>
98   *    </tr>
99   *    <tr>
100  *      <td><code>error</code></td>
101  *      <td>The operation failed to complete, e.g. as a result of a network
102  *        error</td>
103  *      <td>never or once</td>
104  *      <td rowspan="3" valign="middle">Exactly one of these must be
105  *      dispatched</td>
106  *    </tr>
107  *    <tr>
108  *      <td><code>abort</code></td>
109  *      <td>The operation was cancelled, e.g. as a result of user
110  *      interaction</td>
111  *      <td>never or once</td>
112  *    </tr>
113  *    <tr>
114  *      <td><code>load</code></td>
115  *      <td>The operation successfully completed</td>
116  *      <td>never or once</td>
117  *    </tr>
118  *  </tbody>
119  * </table>
120  *
121  * <p>These events <em>must not</em> bubble, and <em>must not</em> be 
122  * cancelable.</p> 
123  * 
124  * <p>These events trigger event listeners attached on <code>Element</code>
125  * nodes for that event and on the capture and target phases. 
126  *
127  * <p>No default action is defined for these events. </p>
128  *
129  * <p>These events are in the <code>null</code> namespace. Two kinds of 
130  * initialization methods are provided: one in which the namespace is 
131  * required (and must be <code>null</code>) and one which assigns the 
132  * <code>null</code> namespace automatically, This specification does
133  * not recommend use of one method over the other, and authors may choose
134  * whichever method suits them better for any given usage.
135  *
136  */
137 public interface ProgressEvent extends Event {
138 
139     /**
140      * Specifies whether the total size of the transfer is known.
141      */
142     public boolean getLengthComputable();
143 
144     /**
145      * Specifies the number of bytes downloaded since the beginning of the 
146      * download. This refers to the content, excluding headers and overhead 
147      * from the transaction,
148      * and where there is a content-encoding or transfer-encoding refers
149      * to the number of bytes to be transferred, i.e. with the relevant
150      * encodings applied. For more details on HTTP see [RFC2616].
151      */
152     public int getLoaded();
153 
154     /**
155      * Specifies the expected total number of bytes of the content 
156      * transferred in the operation. Where the size of the transfer is for
157      * some reason unknown, the value of this attribute <em>must</em> be zero.
158      */
159     public int getTotal();
160 
161     /**
162      * The initProgressEvent method is used to initialize the value of a 
163      * progress event created through the DocumentEvent interface. 
164      * If this method is called multiple times, the final invocation takes 
165      * precedence .
166      *
167      * @param typeArg  
168      *   This <em>must</em> be one of <code>loadstart</code>, 
169      *   <code>progress</code>, <code>error</code>, <code>abort</code>,
170      *   <code>load</code>. If it is not one of those values then this
171      *   specification does not define the resulting event.
172      * @param canBubbleArg
173      *   This <em>must</em> be <var>false</var>. Where a
174      *   value of <var>true</var> is passed, implementations
175      *   <em>must</em> override that and change the value to
176      *   <var>false</var>.
177      * @param cancelableArg
178      *   This <em>must</em> be <var>false</var>. Where a
179      *   value of <var>true</var> is passed, implementations 
180      *   <em>must</em> override that and change the value to
181      *   <var>false</var>.
182      * @param lengthComputableArg
183      *   If the implementation has reliable information about
184      *   the value of <code>total</code>, then this should be <var>true</var>. 
185      *   If the implementation does not have reliable information about
186      *   the value of <code>total</code>, this should be <var>false</var>.
187      * @param loadedArg
188      *   Specifies the total number of bytes already loaded. If this value 
189      *   is not a non-negative number, 
190      *   the implementation <em>must</em> change it to zero.
191      * @param totalArg
192      *   Specifies the total number of bytes to be
193      *   loaded. If <code>lengthComputable</code> is <var>false</var>, 
194      *   this <em>must</em> be zero. If any other parameter is passed, and
195      *   <code>lengthComputable</code> is <var>false</var>, the implementation 
196      *   <em>must</em> override this and set the value to zero. If
197      *   <code>lengthComputable</code> is <var>true</var>, and the value 
198      *   of this parameter is not a non-negative number, the implementation 
199      *   <em>must</em> set <code>lengthComputable</code> to <var>false</var>
200      *   and the value of <code>total</code> to zero. 
201      */
202     public void initProgressEvent(String typeArg,
203                                   boolean canBubbleArg,
204                                   boolean cancelableArg,
205                                   boolean lengthComputableArg,
206                                   int loadedArg, 
207                                   int totalArg);
208

210     /**
211      * The initProgressEventNS method is used to initialize the value of a

217      *
218      * @param namespaceURIArg 
219      *   Specifies the URI for the namespace of the event.
220      *   For all events defined in this specification, the
221      *   value of this parameter is <code>null</code>.
222      * @param typeArg
223      *   This must be one of <code>loadstart</code>, 
224      *   <code>progress</code>, <code>error</code>,
225      *   <code>abort</code>, <code>load</code>. If it is not one
226      *   of those values then this specification does not define 
227      *   the resulting event.
228      * @param canBubbleArg
229      *   This <em>must</em> be <var>false</var>. Where a
230      *   value of <var>true</var> is passed, implementations
231      *   <em>must</em> override that and change the value to
232      *   <var>false</var>.
233      * @param cancelableArg
234      *   This <em>must</em> be <var>false</var>. Where a
235      *   value of <var>true</var> is passed, implementations
236      *   <em>must</em> override that and change the value to
237      *   <var>false</var>.
238      * @param lengthComputableArg
239      *   If the implementation has reliable information about
240      *   the value of total, then this should be <var>true</var>. If the
241      *   implementation does not have reliable information about
242      *   the value of total, this should be <var>false</var>.
243      * @param loadedArg
244      *   This parameter specifies the total number of bytes
245      *   already loaded.
246      *   If this value is not a non-negative number, the implementation 
247      *   <em>must</em> change it to zero.
248      * @param totalArg
249      *   This specifies the total number of bytes to be
250      *   loaded. If <code>lengthComputable</code> is <var>false</var>, 
251      *   this <em>must</em> be zero. If any other parameter is passed,
252      *   and <code>lengthComputable</code> is <var>false</var>, the
253      *   implementation <em>must</em> override this and set the value to
254      *   zero. If <code>lengthComputable</code> is <var>true</var>, and 
255      *   the value of this parameter is not a non-negative number, the 
256      *   implementation <em>must</em> set <code>lengthComputable</code> 
257      *   to <var>false</var> and the value of <code>total</code> to zero.
258      *
259      */
260     public void initProgressEventNS(String namespaceURIArg, 
261                                     String typeArg,
262                                     boolean canBubbleArg, 
263                                     boolean cancelableArg,
264                                     boolean lengthComputableArg,
265                                     int loadedArg, 
266                                     int totalArg);
