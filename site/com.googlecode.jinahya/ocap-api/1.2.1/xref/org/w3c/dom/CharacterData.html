38  package org.w3c.dom;
39  
40  /**
41   * The <code>CharacterData</code> interface extends Node with a set of 
42   * attributes and methods for accessing character data in the DOM. For 
43   * clarity this set is defined here rather than on each object that uses 
44   * these attributes and methods. No DOM objects correspond directly to 
45   * <code>CharacterData</code>, though <code>Text</code> and others do 
46   * inherit the interface from it. All <code>offsets</code> in this interface 
47   * start from <code>0</code>.
48   * <p>As explained in the <code>DOMString</code> interface, text strings in 
49   * the DOM are represented in UTF-16, i.e. as a sequence of 16-bit units. In 
50   * the following, the term 16-bit units is used whenever necessary to 
51   * indicate that indexing on CharacterData is done in 16-bit units.
52   * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
53   */
54  public interface CharacterData extends Node {
55      /**
56       * The character data of the node that implements this interface. The DOM 
57       * implementation may not put arbitrary limits on the amount of data 
58       * that may be stored in a <code>CharacterData</code> node. However, 
59       * implementation limits may mean that the entirety of a node's data may 
60       * not fit into a single <code>DOMString</code>. In such cases, the user 
61       * may call <code>substringData</code> to retrieve the data in 
62       * appropriately sized pieces.
63       * @return The specified substring.
64       * @exception DOMException
65       *   DOMSTRING_SIZE_ERR: Raised when it would return more characters than 
66       *   fit in a <code>DOMString</code> variable on the implementation 
67       *   platform.
68       */
69      public String getData()
70                              throws DOMException;
71  
72      /**
73       * The character data of the node that implements this interface. The DOM 
74       * implementation may not put arbitrary limits on the amount of data 
75       * that may be stored in a <code>CharacterData</code> node. However, 
76       * implementation limits may mean that the entirety of a node's data may 
77       * not fit into a single <code>DOMString</code>. In such cases, the user 
78       * may call <code>substringData</code> to retrieve the data in 
79       * appropriately sized pieces.
80       * @param data character data to add to the node, may not be null
81       * @exception DOMException
82       *   NO_MODIFICATION_ALLOWED_ERR: Raised when the node is readonly.
83       *   SYNTAX_ERR: Raised if an invalid or illegal string is specified.
84       */
85      public void setData(String data)
86                              throws DOMException;
87  
88      /**
89       * The number of 16-bit units that are available through <code>data</code> 
90       * and the <code>substringData</code> method below. This may have the 
91       * value zero, i.e., <code>CharacterData</code> nodes may be empty.
92       * @return the number of 16-bit units available in the CharacterData
93       */
94      public int getLength();
95  
96      /**
97       * Extracts a range of data from the node.
98       * @param offset Start offset of substring to extract.
99       * @param count The number of 16-bit units to extract.
100      * @return The specified substring. If the sum of <code>offset</code> and 
101      *   <code>count</code> exceeds the <code>length</code>, then all 16-bit 
102      *   units to the end of the data are returned.
103      * @exception DOMException
104      *   INDEX_SIZE_ERR: Raised if the specified <code>offset</code> is 
105      *   negative or greater than the number of 16-bit units in 
106      *   <code>data</code>, or if the specified <code>count</code> is 
107      *   negative.
108      *   <br>DOMSTRING_SIZE_ERR: Raised if the specified range of text does 
109      *   not fit into a <code>DOMString</code>.
110      */
111     public String substringData(int offset, 
112                                 int count)
113                                 throws DOMException;
114 
115     /**
116      * Append the string to the end of the character data of the node. Upon 
117      * success, <code>data</code> provides access to the concatenation of 
118      * <code>data</code> and the <code>DOMString</code> specified.
119      * @param arg The <code>DOMString</code> to append.
120      * @exception DOMException
121      *   NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
122      */
123     public void appendData(String arg)
124                            throws DOMException;
125 
126     /**
127      * Insert a string at the specified 16-bit unit offset.
128      * @param offset The character offset at which to insert.
129      * @param arg The <code>DOMString</code> to insert.
130      * @exception DOMException
131      *   INDEX_SIZE_ERR: Raised if the specified <code>offset</code> is 
132      *   negative or greater than the number of 16-bit units in 
133      *   <code>data</code>.
134      *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
135      */
136     public void insertData(int offset, 
137                            String arg)
138                            throws DOMException;
139 
140     /**
141      * Remove a range of 16-bit units from the node. Upon success, 
142      * <code>data</code> and <code>length</code> reflect the change.
143      * @param offset The offset from which to start removing.
144      * @param count The number of 16-bit units to delete. If the sum of 
145      *   <code>offset</code> and <code>count</code> exceeds 
146      *   <code>length</code> then all 16-bit units from <code>offset</code> 
147      *   to the end of the data are deleted.
148      * @exception DOMException
149      *   INDEX_SIZE_ERR: Raised if the specified <code>offset</code> is 
150      *   negative or greater than the number of 16-bit units in 
151      *   <code>data</code>, or if the specified <code>count</code> is 
152      *   negative.
153      *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
154      */
155     public void deleteData(int offset, 
156                            int count)
157                            throws DOMException;
158 
159     /**
160      * Replace the characters starting at the specified 16-bit unit offset 
161      * with the specified string.
162      * @param offset The offset from which to start replacing.
163      * @param count The number of 16-bit units to replace. If the sum of 
164      *   <code>offset</code> and <code>count</code> exceeds 
165      *   <code>length</code>, then all 16-bit units to the end of the data 
166      *   are replaced; (i.e., the effect is the same as a <code>remove</code>
167      *   method call with the same range, followed by an <code>append</code>
168      *   method invocation).
169      * @param arg The <code>DOMString</code> with which the range must be 
170      *   replaced.
171      * @exception DOMException
172      *   INDEX_SIZE_ERR: Raised if the specified <code>offset</code> is 
173      *   negative or greater than the number of 16-bit units in 
174      *   <code>data</code>, or if the specified <code>count</code> is 
175      *   negative.
176      *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
177      */
178     public void replaceData(int offset, 
179                             int count, 
180                             String arg)
181                             throws DOMException;
182 
183 }
