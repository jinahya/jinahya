8   /**
9    * The callback interface to be implemented by application classes that wish
10   * to receive notification of changes to the status of a <code>VideoOutputPort</code>.
11   * 
12   * @author Aaron Kamienski
13   */
14  public interface VideoOutputPortListener extends java.util.EventListener
15  {
16      /**
17       * Method to be invoked when a display device is connected or disconnected.
18       * <p>
19       * Note that this method will not be invoked where such information cannot
20       * be known by the Host device.
21       * 
22       * @param source the <code>VideoOutputPort</code> whose status has changed
23       * @param status <code>true</code> when a display device is connected;
24       *               <code>false</code> when a display device is disconnected
25       *               
26       * @see VideoOutputSettings#isDisplayConnected
27       */
28      public void connectionStatusChanged(VideoOutputPort source, boolean status);
29  
30      /**
31       * Method to be invoked when the <code>VideoOutputPort</code> is enabled or
32       * disabled.
33       * 
34       * @param source the <code>VideoOutputPort</code> whose status has changed
35       * @param status <code>true</code> when the video output port is enabled;
36       *               <code>false</code> when the video output port is disabled
37       *               
38       * @see VideoOutputPort#enable()
39       * @see VideoOutputPort#disable()
40       * @see VideoOutputPort#status()
41       */
42      public void enabledStatusChanged(VideoOutputPort source, boolean status);
43  
44      /**
45       * Method to be invoked when the configuration of a <code>VideoOutputPort</code> changes.
46       * 
47       * @param source the <code>VideoOutputPort</code> whose status has changed
48       * @param oldConfig the previous configuration
49       * @param newConfig the new configuration
50       */
51      public void configurationChanged(VideoOutputPort source, 
52                                       VideoOutputConfiguration oldConfig, VideoOutputConfiguration newConfig);
53  }
