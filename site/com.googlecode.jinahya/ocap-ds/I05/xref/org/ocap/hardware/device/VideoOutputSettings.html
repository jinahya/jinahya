11  /**
12   * The <code>VideoOutputSettings</code> interface extends the functionality of the 
13   * video outputs to support configuration of video output resolution. 
14   * In Host devices supporting this specification, all instances of <code>VideoOutputPort</code>
15   * SHALL implement this interface.
16   * <p>
17   * An application MAY query the set of output configurations supported by a given
18   * video output port by calling {@link #getSupportedConfigurations()}.
19   * The supported configurations MAY be used to configure the given video output port
20   * by calling {@link #setOutputConfiguration}.
21   * 
22   * @see VideoOutputPort
23   */
24  public interface VideoOutputSettings
25  {
26      /**
27       * Get the fixed set of supported output configurations for this video output port. 
28       * This returns all supported output resolutions, regardless of input resolution, as
29       * an array of {@link VideoOutputConfiguration} instances.
30       * <p>
31       * The returned set of configurations SHALL be the intersection of those supported
32       * by the output port of the Host device and those supported by the connected
33       * display device, where such information is known by the Host.
34       *
35       * @return The set of all supported output resolutions as an array.
36       */
37      public VideoOutputConfiguration[] getSupportedConfigurations();
38      
39      /**
40       * Determine if this video output port supports dynamic output configuration based upon
41       * input video attributes.
42       * 
43       * @return <code>true</code> if instances of <code>DynamicVideoOutputConfiguration</code>
44       * may be set via {@link #setOutputConfiguration}
45       */
46      public boolean isDynamicConfigurationSupported();
47      
48      /**
49       * Get the current output configuration for this video output port.
50       * Changes to a returned configuration instance SHALL have no effect on 
51       * the current configuration unless applied via invocation of
52       * <code>setOutputConfiguration</code>.
53       * <p>
54       * If the returned <code>VideoOutputConfiguration</code> corresponds to a
55       * configuration returned by {@link #getSupportedConfigurations}, that same
56       * configuration SHALL be returned.
57       *
58       * @return The current output configuration.
59       *
60       * @see #setOutputConfiguration
61       */
62      public VideoOutputConfiguration getOutputConfiguration();
63      
64      /**
65       * Set the output configuration for this video output port.
66       * Only <code>VideoOutputConfiguration</code>s instances corresponding to 
67       * supported video configurations SHALL be accpeted; anything else results
68       * in an exception.
69       * <p>
70       * The output resolution configuration is applied at method invocation
71       * time.  Changes to a configuration instance SHALL have no effect on the
72       * current configuration unless applied via invocation of this method
73       * specifying the given configuration.
74       * <p> 
75       * Changing this setting on the "main" video output port for an <code>HScreen</code>
76       * MAY affect the configuration of the <code>HScreenDevice</code>s of that
77       * <code>HScreen</code> to maintain consistent display aspect ratios
78       * as described in the body of this specification. 
79       *
80       * @param config The new output configuration.
81       *    
82       * @throws IllegalArgumentException if the output resolution specified
83       * does not correspond to one of the supported configurations
84       * @throws FeatureNotSupportedException if the given configuration
85       * corresponds to one of the supported resolutions, but cannot be satisfied
86       *  
87       * @throws SecurityException  if the caller does not have 
88       * <code>MonitorAppPermission("deviceController")</code>
89       * 
90       * @see #getOutputConfiguration
91       */
92      public void setOutputConfiguration(VideoOutputConfiguration config) throws FeatureNotSupportedException;
93      
94      /**
95       * Get the <code>AudioOutputPort</code> object which can be used to control
96       * the audio output port associated with this video output port.
97       * 
98       * @return the <code>AudioOutputPort</code> for this video output port
99       */
100     public AudioOutputPort getAudioOutputPort();
101     
102     /**
103      * Add  the given listener to monitor this video output port for status changes.
104      * 
105      * @param l the listener to add
106      * 
107      * @see #removeListener(VideoOutputPortListener)
108      */
109     public void addListener(VideoOutputPortListener l);
110     
111     /**
112      * Remove the given listener from further notification of status changes 
113      * for this video output port.  
114      * 
115      * @param l the listener to remove
116      * 
117      * @see #addListener(VideoOutputPortListener)
118      */
119     public void removeListener(VideoOutputPortListener l);
120     
121     /**
122      * Get the set of attributes describing the display currently connected to
123      * this <code>VideoOutputPort</code>.
124      * The display attributes information is returned in the form of a <code>Hashtable</code>,
125      * mapping attribute names (specified as <code>String</code>s) to attribute
126      * values (specified as attribute-specific <code>Object</code>s).
127      * <p> 
128      * For example, where EEDID (Enhanced Extended Display Identification Data)
129      * retrieval is supported, this information SHALL be accessible as attributes
130      * of the connected display.  In case of EEDID, the following table describes
131      * the attribute mappings that SHALL be supported and used where appropriate.
132      * <table border>
133      * <p>
134      * <tr><th>Attribute name</th> <th>Attribute value</th></tr>
135      * <tr><td>"Manufacturer Name"</td> <td>3-character EISA manufacturer ID as a {@link java.lang.String}</code></td></tr>
136      * <tr><td>"Product Code"</td> <td>Vendor assigned product code as a {@link java.lang.Short}</td></tr>
137      * <tr><td>"Serial Number"</td> <td>Vendor assigned serial number as a {@link java.lang.Integer}</td></tr>
138      * <tr><td>"Manufacture Week"</td> <td>Week of manufacture as a {@link java.lang.Byte}</td></tr>
139      * <tr><td>"Manufacture Year"</td> <td>Year of manufacture (offset from 1990) as a {@link java.lang.Byte}</td></tr>  
140      * </table>
141      *  
142      * @return a <code>Hashtable</code> describing known display attributes;
143      *         <code>null</code> is returned if {@link #isDisplayConnected()} 
144      *         would return <code>false</code>
145      */
146     public Hashtable getDisplayAttributes();
147     
148     /**
149      * Get the connection status of this video output port.
150      * <p>
151      * If the Host device is incapable of determining connection status (e.g., 
152      * for {@link VideoOutputPort#AV_OUTPUT_PORT_TYPE_COMPONENT_VIDEO component video}),
153      * then <code>false</code> SHALL be returned.
154      * 
155      * @return <code>true</code> if a display is known to be connected;
156      *         <code>false</code> otherwise
157      */
158     public boolean isDisplayConnected();
159     
160     /**
161      * Get the protection status of content on this video output port.
162      * <p>
163      * This SHALL return <code>true</code> if both this port and the connected
164      * display support content protection and content is protected; <code>false</code>
165      * SHALL be returned otherwise.
166      * 
167      * @return whether content is currently protected on this video output port
168      */
169     public boolean isContentProtected();
170     
171     /**
172      * Get the aspect ratio of the display connected to this video output port.
173      *  
174      * @return one of {@link VideoFormatControl#DAR_4_3},
175      *                {@link VideoFormatControl#DAR_16_9},
176      *             or -1 if unknown
177      */
178     public int getDisplayAspectRatio();
