61      /**
62       * The magic number identifying the class file format; it has the value
63       * <code>0xCAFEBABE</code>.
64       */
65      public static final int MAGIC = 0xCAFEBABE;
66  
67  
68      @Override
69      public void read(final DataInput input) throws IOException {
70  
71          assert input.readInt() == MAGIC;
72  
73          //minorVersion = input.readUnsignedShort();
74  
75          //majorVersion = input.readUnsignedShort();
76  
77          version = ClassFileVersion.valueOf(input.readUnsignedShort(),
78                                    input.readUnsignedShort());
79  
80          // -------------------------------------------------------- constantPool
81          final int constantPoolCount = input.readUnsignedShort();
82          for (int i = 0; i < constantPoolCount - 1; i++) {
83              final int tag = input.readUnsignedByte();
84              final Constant constant = Tag.valueOf(tag).newConstant();
85              System.out.println("constant[" + (i + 1) + "].tag: " + constant.getTag());
86              constant.setIndex(i);
87              constant.read(input);
88              getConstants().add(constant);
89          }
90  
91          accessFlags = input.readUnsignedShort();
92  
93          thisClass = input.readUnsignedShort();
94  
95          superClass = input.readUnsignedShort();
96  
97          // ----------------------------------------------------------- interface
98          final int interfacesCount = input.readUnsignedShort();
99          for (int i = 0; i < interfacesCount; i++) {
100             getInterfaces().add(input.readUnsignedShort());
101         }
102 
103         // -------------------------------------------------------------- fields
104         final int fieldsCount = input.readUnsignedShort();
105         getFields().clear();
106         for (int i = 0; i < fieldsCount; i++) {
107             final Field field = new Field();
108             field.setClassFile(this);
109             field.read(input);
110             getFields().add(field);
111         }
112 
113         // ------------------------------------------------------------- methods
114         final int methodsCount = input.readUnsignedShort();
115         System.out.println("methodsCount: " + methodsCount);
116         getMethods().clear();
117         for (int i = 0; i < methodsCount; i++) {
118             final Method method = new Method();
119             method.setClassFile(this);
120             method.read(input);
121             System.out.println("method[" + i + "]: " + method);
122             getMethods().add(method);
123         }
124 
125         // ---------------------------------------------------------- attributes
126         final int attributesCount = input.readUnsignedShort();
127         getAttributes().clear();
128         final AttributeInfo info = new AttributeInfo();
129         for (int i = 0; i < attributesCount; i++) {
130             info.read(input);
131             getAttributes().add(Attribute.newInstance(info, this));
132         }
133     }

136     @Override
137     public void write(final DataOutput output) throws IOException {
138 
139         output.writeInt(MAGIC);
140 
141         //output.writeShort(minorVersion);
142 
143         //output.writeShort(majorVersion);
144 
145         output.writeShort(version.minor);
146 
147         output.writeShort(version.major);
148 
149         // -------------------------------------------------------- constantPool
150         output.writeShort(getConstants().size() + 1);
151         for (Constant constant : getConstants()) {
152             output.write(constant.getTag().getValue());
153             constant.write(output);
154         }
155 
156         output.writeShort(accessFlags);
157 
158         output.writeShort(thisClass);
159 
160         output.writeShort(superClass);
161 
162         // ---------------------------------------------------------- interfaces
163         output.writeShort(getInterfaces().size());
164         for (Iterator<Integer> i = getInterfaces().iterator(); i.hasNext();) {
165             output.writeShort(i.next());
166         }
167 
168         // -------------------------------------------------------------- fields
169         output.writeShort(getFields().size());
170         for (Field field : fields) {
171             field.write(output);
172         }
173 
174         // ------------------------------------------------------------- methods
175         output.writeShort(getMethods().size());
176         for (Method method : methods) {
177             method.write(output);
178         }
179 
180         // ---------------------------------------------------------- attributes
181         output.writeShort(getAttributes().size());
182         final AttributeInfo info = new AttributeInfo();
183         for (Attribute attribute : getAttributes()) {
184             info.parse(attribute);
185             info.write(output);
186         }
187     }
