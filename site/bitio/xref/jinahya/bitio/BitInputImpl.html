76      /**
77       *
78       *
79       * @param length
80       * @throws IOException if an I/O error occurs
81       * @throws IllegalArgumentException
82       */
83      protected int readUnsignedByte(int length) throws IOException {
84  
85          if (length < 1 || length >= 8) {
86              throw new IllegalArgumentException("illegal length: " + length);
87          }
88  
89          if (index == 0x08) {
90              if ((octet = input.readByte()) == -1) {
91                  throw new EOFException("EOF");
92              }
93              index = 0x00;
94          }
95  
96          int available = 0x08 - index;
97  
98          if (available >= length) {
99              index += length;
100             count += length;
101             return (octet >>> (available - length)) & powers[length -1];
102         }
103 
104         int required = length - available;
105         return ((readUnsignedByte(available) << required) |
106                 readUnsignedByte(required));
107     }

117     protected int readUnsignedShort(int length) throws IOException {
118 
119         if (length < 1 || length >= 16) {
120             throw new IllegalArgumentException("illegal length: " + length);
121         }
122 
123         int value = 0x00;
124 
125         int quotient = length / 7;
126         for (int i = 0; i < quotient; i++) {
127             value <<= 7;
128             value |= readUnsignedByte(7);
129         }
130 
131         int remainder = length % 7;
132         if (remainder > 0) {
133             value <<= remainder;
134             value |= readUnsignedByte(remainder);
135         }
136 
137         return value;

164     /** {@inheritDoc} */
165     public int readInt(int length) throws IOException {
166 
167         if (length <= 1 || length > 32) {
168             throw new IllegalArgumentException("illegal length: " + length);
169         }
170 
171         int value = readUnsignedByte(1);
172         value <<= (length - 1);
173         for (int i = 0; i < 32 - (length - 1); i++) {
174             value |= (value <<= 1);
175         }
176 
177         return value | readUnsignedInt(length -1);
178     }

207     public long readLong(int length) throws IOException {
208 
209         if (length <= 1 || length > 64) {
210             throw new IllegalArgumentException("illegal length: " + length);
211         }
212 
213         long value = readUnsignedByte(1); // sign bit
214         value <<= (length - 1); // zero padding
215         for (int i = 0; i < 64 - (length - 1); i++) {
216             value |= (value <<= 1); // copy shift
217         }

219         return value | readUnsignedLong(length -1);
