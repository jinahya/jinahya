63      private static final int[] powers  = new int[] {
64          0x01, // 1   0000 0001
65          0x03, // 3   0000 0011
66          0x07, // 7   0000 0111
67          0x0F, // 15  0000 1111
68          0x1F, // 31  0001 1111
69          0x3F, // 63  0011 1111
70          0x7F, // 127 0111 1111
71          0xFF  // 255 1111 1111

75      /**
76       * Writes an unsigned byte.
77       *
78       * @param length 0 < length < 8
79       * @param value 2^0 < value < 2^8
80       * @throws IOException if an I/O error occurs
81       */
82      protected void writeUnsignedByte(int length, int value) throws IOException {
83  
84          if (length < 1 || length >= 8) {
85              throw new IllegalArgumentException("illegal length: " + length);
86          }
87  
88          int available = 0x08 - index;
89          if (available >= length) {
90              octet |= ((value & powers[length - 1]) << (available - length));
91              count += length;
92              if ((index += length) == 0x08) {
93                  output.writeByte(octet);
94                  index = 0x00;
95                  octet = 0x00;
96              }
97              return;
98          }
99          int required = length - available;
100         writeUnsignedByte(available, value >> required); // high available bits
101         writeUnsignedByte(required, value); // low required bits
102     }

105     /**
106      * Writes an unsigned short.
107      *
108      * @param length 0 < length < 16
109      * @param value 2^0 < value < 2^16
110      * @throws IOException if an I/O error occurs
111      */
112     protected void writeUnsignedShort(int length, int value)
113         throws IOException {
114 
115         if (length < 1 || length >= 16) {
116             throw new IllegalArgumentException("illegal length: " + length);
117         }
118 
119         int quotient = length / 7;
120         int remainder = length % 7;
121         if (remainder > 0) {
122             writeUnsignedByte(remainder, value >> (quotient * 7));
123         }
124         for (int i = quotient - 1; i >= 0; i--) {
125             writeUnsignedByte(7, value >> (i * 7));
126         }
127     }

149     public void writeInt(int length, int value) throws IOException {
150 
151         if (length <= 1 || length > 32) {
152             throw new IllegalArgumentException("illegal length: " + length);
153         }
154 
155         writeUnsignedByte(1, (value < 0 ? 0x01 : 0x00)); // sign bit
156         //writeUnsignedByte(1, value >>> (length - 1)); // sign bit
157         writeUnsignedInt(length - 1, value);

166         }

169         int remainder = length % 31;
170         if (remainder > 0) {
171             writeUnsignedInt(remainder, (int)(value >> (quotient * 31)));
172         }
173         for (int i = quotient - 1; i >= 0; i--) {
174             writeUnsignedInt(31, (int)(value >> (i * 31)));
