117     /**
118      *
119      *
120      * @throws IOException if an I/O error occurs
121      */
122     public void encode() throws IOException {
123         int bitsPerChar = RFC4648Utils.bitsPerChar(alphabet);
124         int bytesPerWord = RFC4648Utils.bytesPerWord(bitsPerChar);
125         int charsPerWord = RFC4648Utils.charsPerWord(bytesPerWord, bitsPerChar);
126         while (true) {
127             for (int i = 0; i < charsPerWord; i++) {
128                 int available = 8 - ((bitsPerChar * i) % 8);
129                 if (available >= bitsPerChar) {
130                     try {
131                         int unsigned = input.readUnsignedInt(bitsPerChar);
132                         output.write(alphabet.charAt(unsigned));
133                     } catch (EOFException eofe) {
134                         if (i != 0) {
135                             throw new IOException("this couldn't happen");
136                         }
137                         // i == 0
138                         return;
139                     }
140                 } else { // need next octet
141                     int required = bitsPerChar - available;
142                     int unsigned =
143                         (input.readUnsignedInt(available) << required);
144                     try {
145                         unsigned |= input.readUnsignedInt(required);
146                         output.write(alphabet.charAt(unsigned));
147                     } catch (EOFException eofe) {
148                         output.write(alphabet.charAt(unsigned));
149                         for (int j = i + 1; j < charsPerWord; j++) {
150                             output.write(RFC4648Constants.pad);
151                         }
152                         return;
153                     }
154                 }
155             }
156         }
157     }
