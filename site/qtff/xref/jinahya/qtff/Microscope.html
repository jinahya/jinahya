38      private Spectrum ROOT() throws IOException {
39          if (root == null) {
40              long offset = -8L;
41              long length = file.length() + 8L;
42              if (length >= 4294967296L) { // 2^32
43                  offset -= 8L;
44                  length += 8L;
45              }
46              Spectrum root = new Spectrum();
47              root.setParent(null);
48              root.setOffset(offset);

68      public Microscope getChildren(Spectrum parent,
69                                    Collection<Spectrum> spectrums)
70          throws IOException {
71  
72          if (parent == null) {
73              return getChildren(ROOT(), spectrums);
74          }
75  
76          long childOffset = parent.getDataOffset();
77          while ((childOffset + 8L) <= parent.getLimit()) {
78              Spectrum child = read(childOffset);
79  
80              if (child == null) {
81                  break;
82              }
83  
84              if (child.getLimit() > parent.getLimit()) {
85                  break;
86              }
87  
88              if (parent.getType() != _ROOT_SPECTRUM_TYPE) {
89                  child.setParent(parent);
90              }
91  
92              spectrums.add(child);
93  
94              childOffset = child.getLimit();
95          }
96  
97          return this;
98      }

108     private Spectrum read(long offset) throws IOException {
109         file.seek(offset);
110 
111         long length = (file.readInt() & 0xFFFFFFFFL);
112         if (length < 8L) {
113             return null;
114         }
115         int type = file.readInt();
116         if (length == 1L) { // extended size
117             length = file.readLong();
118             if (length < 16L) {
119                 return null;
120             }
121         }
122 
123         Spectrum spectrum = new Spectrum();
124         spectrum.setOffset(offset);
125         spectrum.setLength(length);
126         spectrum.setType(type);
127         return spectrum;
128     }
